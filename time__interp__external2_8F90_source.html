823  nyw = jec-jsc+1

830  nyw = je_in - js_in + 1
831  isc = isc + is_in - 1
832  iec = isc + ie_in - is_in
833  jsc = jsc + js_in - 1
834  jec = jsc + je_in - js_in
835  endif

838  jsw = (ny-nyw)/2+1; jew = jsw+nyw-1

918  field(index)%data(isc:iec,jsc:jec,:,i2)*w2

926  field(index)%data(isc:iec,jsc:jec,:,i2)*w2
927  end where
928  endif
929  if(PRESENT(mask_out)) &
930  mask_out(isw:iew,jsw:jew,:) = &
931  field(index)%mask(isc:iec,jsc:jec,:,i1).and.&
932  field(index)%mask(isc:iec,jsc:jec,:,i2)
