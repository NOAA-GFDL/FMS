19 !> @defgroup horiz_interp_conserve_mod horiz_interp_conserve_mod
20 !> @ingroup horiz_interp
21 !> @brief Performs spatial interpolation between grids using conservative interpolation
22 !!
23 !> @author Bruce Wyman, Zhi Liang
24 !!
25 !> This module can conservatively interpolate data from any logically rectangular grid
26 !! to any rectangular grid. The interpolation scheme is area-averaging
27 !! conservative scheme. There is an optional mask field for missing input data in both
28 !! horiz_interp__conserveinit and horiz_interp_conserve. For efficiency purpose, mask should only be
29 !! kept in horiz_interp_init (will remove the mask in horiz_interp in the future).
30 !! There are 1-D and 2-D version of horiz_interp_conserve_init for 1-D and 2-D grid.
31 !! There is a optional argument mask in horiz_interp_conserve_init_2d and no mask should
32 !! to passed into horiz_interp_conserv. optional argument mask will not be passed into
33 !! horiz_interp_conserve_init_1d and optional argument mask may be passed into
34 !! horiz_interp_conserve (For the purpose of reproduce Memphis??? results).
35 !! An optional output mask field may be used in conjunction with the input mask to show
36 !! where output data exists.
37

130  subroutine horiz_interp_conserve_new_1dx1d ( Interp, lon_in, lat_in, lon_out, lat_out, verbose)
131  type(horiz_interp_type), intent(inout) :: interp
132  real, intent(in), dimension(:) :: lon_in , lat_in
133  real, intent(in), dimension(:) :: lon_out, lat_out
134  integer, intent(in), optional :: verbose
135 
136  !-----------------------------------------------------------------------
137  real, dimension(size(lat_out(:))-1,2) :: sph
138  real, dimension(size(lon_out(:))-1,2) :: theta
139  real, dimension(size(lat_in(:))) :: slat_in
140  real, dimension(size(lon_in(:))-1) :: dlon_in
141  real, dimension(size(lat_in(:))-1) :: dsph_in
142  real, dimension(size(lon_out(:))-1) :: dlon_out
143  real, dimension(size(lat_out(:))-1) :: dsph_out
144  real :: blon, fac, hpi, tpi, eps
145  integer :: num_iters = 4
146  integer :: i, j, m, n, nlon_in, nlat_in, nlon_out, nlat_out, &
147  iverbose, m2, n2, iter
148  logical :: s2n
149  character(len=64) :: mesg
150 
151  if(.not. module_is_initialized) call mpp_error(fatal, &
152  'horiz_interp_conserve_new_1dx1d: horiz_interp_conserve_init is not called')
153 
154  if(great_circle_algorithm) call mpp_error(fatal, &
155  'horiz_interp_conserve_new_1dx1d: great_circle_algorithm is not implemented, contact developer')
156  !-----------------------------------------------------------------------
157  iverbose = 0; if (present(verbose)) iverbose = verbose
158 
159  pe = mpp_pe()
160  root_pe = mpp_root_pe()
161  !-----------------------------------------------------------------------
162  hpi = 0.5*pi
163  tpi = 4.*hpi
164  interp%version = 1
165  nlon_in = size(lon_in(:))-1; nlat_in = size(lat_in(:))-1
166  nlon_out = size(lon_out(:))-1; nlat_out = size(lat_out(:))-1
167 
168  allocate ( interp % facj (nlat_out,2), interp % jlat (nlat_out,2), &
169  interp % faci (nlon_out,2), interp % ilon (nlon_out,2), &
170  interp % area_src (nlon_in, nlat_in), &
171  interp % area_dst (nlon_out, nlat_out) )
172 
173  !-----------------------------------------------------------------------
174  ! --- set-up for input grid boxes ---
175 
176  do j = 1, nlat_in+1
177  slat_in(j) = sin(lat_in(j))
178  enddo
179 
180  do j = 1, nlat_in
181  dsph_in(j) = abs(slat_in(j+1)-slat_in(j))
182  enddo
183 
184  do i = 1,nlon_in
185  dlon_in(i) = abs(lon_in(i+1)-lon_in(i))
186  enddo
187 
188  ! set south to north flag
189  s2n = .true.
190  if (lat_in(1) > lat_in(nlat_in+1)) s2n = .false.
191 
192  !-----------------------------------------------------------------------
193  ! --- set-up for output grid boxes ---
194 
195  do n = 1, nlat_out
196  dsph_out(n) = abs(sin(lat_out(n+1))-sin(lat_out(n)))
197  enddo
198 
199  do m = 1,nlon_out
200  theta(m,1) = lon_out(m)
201  theta(m,2) = lon_out(m+1)
202  dlon_out(m) = abs(lon_out(m+1)-lon_out(m))
203  enddo
204 
205  interp%nlon_src = nlon_in; interp%nlat_src = nlat_in
206  interp%nlon_dst = nlon_out; interp%nlat_dst = nlat_out
207  !***********************************************************************
208 
209  !------ set up latitudinal indexing ------
210  !------ make sure output grid goes south to north ------
211 
212  do n = 1, nlat_out
213  if (lat_out(n) < lat_out(n+1)) then
214  sph(n,1) = sin(lat_out(n))
215  sph(n,2) = sin(lat_out(n+1))
216  else
217  sph(n,1) = sin(lat_out(n+1))
218  sph(n,2) = sin(lat_out(n))
219  endif
220  enddo
221 
222  interp%jlat = 0
223  do n2 = 1, 2 ! looping on grid box edges
224  do n = 1, nlat_out ! looping on output latitudes
225  eps = 0.0
226  do iter=1,num_iters
227  ! find indices from input latitudes
228  do j = 1, nlat_in
229  if ( (s2n .and. (slat_in(j)-sph(n,n2)) <= eps .and. &
230  (sph(n,n2)-slat_in(j+1)) <= eps) .or. &
231  (.not.s2n .and. (slat_in(j+1)-sph(n,n2)) <= eps .and. &
232  (sph(n,n2)-slat_in(j)) <= eps) ) then
233  interp%jlat(n,n2) = j
234  ! weight with sin(lat) to exactly conserve area-integral
235  fac = (sph(n,n2)-slat_in(j))/(slat_in(j+1)-slat_in(j))
236  if (s2n) then
237  if (n2 == 1) interp%facj(n,n2) = 1.0 - fac
238  if (n2 == 2) interp%facj(n,n2) = fac
239  else
240  if (n2 == 1) interp%facj(n,n2) = fac
241  if (n2 == 2) interp%facj(n,n2) = 1.0 - fac
242  endif
243  exit
244  endif
245  enddo
246  if ( interp%jlat(n,n2) /= 0 ) exit
247  ! did not find this output grid edge in the input grid
248  ! increase tolerance for multiple passes
249  eps = epsilon(sph)*real(10**iter)
250  enddo
251  ! no match
252  if ( interp%jlat(n,n2) == 0 ) then
253  write (mesg,710) n,sph(n,n2)
254 710 format (': n,sph=',i3,f14.7,40x)
255  call mpp_error(fatal, 'horiz_interp_conserve_mod:no latitude index found'//trim(mesg))
256  endif
257  enddo
258  enddo
259 
260  !------ set up longitudinal indexing ------
261 
262  interp%ilon = 0
263  do m2 = 1, 2 ! looping on grid box edges
264  do m = 1, nlon_out ! looping on output longitudes
265  blon = theta(m,m2)
266  if ( blon < lon_in(1) ) blon = blon + tpi
267  if ( blon > lon_in(nlon_in+1) ) blon = blon - tpi
268  eps = 0.0
269  do iter=1,num_iters
270  ! find indices from input longitudes
271  do i = 1, nlon_in
272  if ( (lon_in(i)-blon) <= eps .and. &
273  (blon-lon_in(i+1)) <= eps ) then
274  interp%ilon(m,m2) = i
275  fac = (blon-lon_in(i))/(lon_in(i+1)-lon_in(i))
276  if (m2 == 1) interp%faci(m,m2) = 1.0 - fac
277  if (m2 == 2) interp%faci(m,m2) = fac
278  exit
279  endif
280  enddo
281  if ( interp%ilon(m,m2) /= 0 ) exit
282  ! did not find this output grid edge in the input grid
283  ! increase tolerance for multiple passes
284  eps = epsilon(blon)*real(10**iter)
285  enddo
286  ! no match
287  if ( interp%ilon(m,m2) == 0 ) then
288  print *, 'lon_out,blon,blon_in,eps=', &
289  theta(m,m2),blon,lon_in(1),lon_in(nlon_in+1),eps
290  call mpp_error(fatal, 'horiz_interp_conserve_mod: no longitude index found')
291  endif
292  enddo
293  enddo
294 
295  ! --- area of input grid boxes ---
296 
297  do j = 1,nlat_in
298  do i = 1,nlon_in
299  interp%area_src(i,j) = dlon_in(i) * dsph_in(j)
300  enddo
301  enddo
302 
303  ! --- area of output grid boxes ---
304 
305  do n = 1, nlat_out
306  do m = 1, nlon_out
307  interp%area_dst(m,n) = dlon_out(m) * dsph_out(n)
308  enddo
309  enddo
310 
311  !-----------------------------------------------------------------------
312  ! this output may be quite lengthy and is not recommended
313  ! when using more than one processor
314  if (iverbose > 2) then
315  write (*,801) (i,interp%ilon(i,1),interp%ilon(i,2), &
316  interp%faci(i,1),interp%faci(i,2),i=1,nlon_out)
317  write (*,802) (j,interp%jlat(j,1),interp%jlat(j,2), &
318  interp%facj(j,1),interp%facj(j,2),j=1,nlat_out)
319 801 format (/,2x,'i',4x,'is',5x,'ie',4x,'facis',4x,'facie', &
320  /,(i4,2i7,2f10.5))
321 802 format (/,2x,'j',4x,'js',5x,'je',4x,'facjs',4x,'facje', &
322  /,(i4,2i7,2f10.5))
323  endif
324  !-----------------------------------------------------------------------
325 
326  end subroutine horiz_interp_conserve_new_1dx1d

330  subroutine horiz_interp_conserve_new_1dx2d ( Interp, lon_in, lat_in, lon_out, lat_out, &
331  mask_in, mask_out, verbose)
332  type(horiz_interp_type), intent(inout) :: interp
333  real, intent(in), dimension(:) :: lon_in , lat_in
334  real, intent(in), dimension(:,:) :: lon_out, lat_out
335  real, intent(in), optional, dimension(:,:) :: mask_in
336  real, intent(inout), optional, dimension(:,:) :: mask_out
337  integer, intent(in), optional :: verbose
338 
339 
340  integer :: create_xgrid_1dx2d_order1, get_maxxgrid, maxxgrid
341  integer :: create_xgrid_great_circle
342  integer :: nlon_in, nlat_in, nlon_out, nlat_out, nxgrid, i, j
343  real(double_kind), dimension(size(lon_in(:))-1, size(lat_in(:))-1) :: mask_src
344  integer, allocatable, dimension(:) :: i_src, j_src, i_dst, j_dst
345  real(double_kind), allocatable, dimension(:) :: xgrid_area, clon, clat
346  real(double_kind), allocatable, dimension(:,:) :: dst_area, lon_src, lat_src
347  real(double_kind), allocatable, dimension(:) :: lat_in_flip
348  real(double_kind), allocatable, dimension(:,:) :: mask_src_flip
349  real(double_kind), allocatable, dimension(:) :: lon_in_r8, lat_in_r8
350  real(double_kind), allocatable, dimension(:,:) :: lon_out_r8, lat_out_r8
351 
352  integer :: nincrease, ndecrease
353  logical :: flip_lat
354  integer :: wordsz
355  integer(kind=1) :: one_byte(8)
356 
357  if(.not. module_is_initialized) call mpp_error(fatal, &
358  'horiz_interp_conserve_new_1dx2d: horiz_interp_conserve_init is not called')
359 
360  wordsz=size(transfer(lon_in(1), one_byte))
361  if(wordsz .NE. 4 .AND. wordsz .NE. 8) call mpp_error(fatal, &
362  'horiz_interp_conserve_new_1dx2d: wordsz should be 4 or 8')
363 
364  if( (size(lon_out,1) .NE. size(lat_out,1)) .OR. (size(lon_out,2) .NE. size(lat_out,2)) ) &
365  call mpp_error(fatal, 'horiz_interp_conserve_mod: size mismatch between lon_out and lat_out')
366  nlon_in = size(lon_in(:)) - 1; nlat_in = size(lat_in(:)) - 1
367  nlon_out = size(lon_out,1) - 1; nlat_out = size(lon_out,2) - 1
368 
369  mask_src = 1.
370  if(present(mask_in)) then
371  if( (size(mask_in,1) .NE. nlon_in) .OR. (size(mask_in,2) .NE. nlat_in)) call mpp_error(fatal, &
372  'horiz_interp_conserve_mod: size mismatch between mask_in and lon_in/lat_in')
373  mask_src = mask_in
374  end if
375 
376  maxxgrid = get_maxxgrid()
377  allocate(i_src(maxxgrid), j_src(maxxgrid), i_dst(maxxgrid), j_dst(maxxgrid) )
378  allocate( xgrid_area(maxxgrid), dst_area(nlon_out, nlat_out) )
379 
380  !--- check if source latitude is flipped
381  nincrease = 0
382  ndecrease = 0
383  do j = 1, nlat_in
384  if( lat_in(j+1) > lat_in(j) ) then
385  nincrease = nincrease + 1
386  else if ( lat_in(j+1) < lat_in(j) ) then
387  ndecrease = ndecrease + 1
388  endif
389  enddo
390 
391  if(nincrease == nlat_in) then
392  flip_lat = .false.
393  else if(ndecrease == nlat_in) then
394  flip_lat = .true.
395  else
396  call mpp_error(fatal, 'horiz_interp_conserve_mod: nlat_in should be equal to nincreaase or ndecrease')
397  endif
398 
399  allocate(lon_out_r8(size(lon_out,1),size(lon_out,2)))
400  allocate(lat_out_r8(size(lat_out,1),size(lat_out,2)))
401  lon_out_r8 = lon_out
402  lat_out_r8 = lat_out
403 
404  if( .not. great_circle_algorithm ) then
405  if(flip_lat) then
406  allocate(lat_in_flip(nlat_in+1), mask_src_flip(nlon_in,nlat_in))
407  do j = 1, nlat_in+1
408  lat_in_flip(j) = lat_in(nlat_in+2-j)
409  enddo
410  do j = 1, nlat_in
411  mask_src_flip(:,j) = mask_src(:,nlat_in+1-j)
412  enddo
413  allocate(lon_in_r8(size(lon_in)))
414  lon_in_r8 = lon_in
415  nxgrid = create_xgrid_1dx2d_order1(nlon_in, nlat_in, nlon_out, nlat_out, lon_in_r8, lat_in_flip, &
416  lon_out_r8, lat_out_r8, mask_src_flip, i_src, j_src, i_dst, j_dst, xgrid_area)
417  deallocate(lon_in_r8, lat_in_flip, mask_src_flip)
418  else
419  allocate(lon_in_r8(size(lon_in)))
420  allocate(lat_in_r8(size(lat_in)))
421  lon_in_r8 = lon_in
422  lat_in_r8 = lat_in
423  nxgrid = create_xgrid_1dx2d_order1(nlon_in, nlat_in, nlon_out, nlat_out, lon_in_r8, lat_in_r8, lon_out_r8, &
424  & lat_out_r8, mask_src, i_src, j_src, i_dst, j_dst, xgrid_area)
425  deallocate(lon_in_r8,lat_in_r8)
426  endif
427  else
428  allocate(lon_src(nlon_in+1,nlat_in+1), lat_src(nlon_in+1,nlat_in+1))
429  allocate(clon(maxxgrid), clat(maxxgrid))
430  if(flip_lat) then
431  allocate(mask_src_flip(nlon_in,nlat_in))
432  do j = 1, nlat_in+1
433  do i = 1, nlon_in+1
434  lon_src(i,j) = lon_in(i)
435  lat_src(i,j) = lat_in(nlat_in+2-j)
436  enddo
437  enddo
438  do j = 1, nlat_in
439  mask_src_flip(:,j) = mask_src(:,nlat_in+1-j)
440  enddo
441  nxgrid = create_xgrid_great_circle(nlon_in, nlat_in, nlon_out, nlat_out, lon_src, lat_src, lon_out_r8, &
442  & lat_out_r8, mask_src_flip, i_src, j_src, i_dst, j_dst, xgrid_area, clon, clat)
443  deallocate(mask_src_flip)
444  else
445  do j = 1, nlat_in+1
446  do i = 1, nlon_in+1
447  lon_src(i,j) = lon_in(i)
448  lat_src(i,j) = lat_in(j)
449  enddo
450  enddo
451  nxgrid = create_xgrid_great_circle(nlon_in, nlat_in, nlon_out, nlat_out, lon_src, lat_src, lon_out_r8, &
452  & lat_out_r8, mask_src, i_src, j_src, i_dst, j_dst, xgrid_area, clon, clat)
453  endif
454  deallocate(lon_src, lat_src, clon, clat)
455  endif
456 
457  deallocate(lon_out_r8, lat_out_r8)
458 
459  allocate(interp%i_src(nxgrid), interp%j_src(nxgrid) )
460  allocate(interp%i_dst(nxgrid), interp%j_dst(nxgrid) )
461  allocate(interp%area_frac_dst(nxgrid) )
462  interp%version = 2
463  interp%nxgrid = nxgrid
464  interp%i_src = i_src(1:nxgrid)+1 ! in C, the starting index is 0
465  interp%j_src = j_src(1:nxgrid)+1
466  if(flip_lat) interp%j_src = nlat_in+1-interp%j_src
467  interp%i_dst = i_dst(1:nxgrid)+1
468  interp%j_dst = j_dst(1:nxgrid)+1
469 
470  ! sum over exchange grid area to get destination grid area
471  dst_area = 0.
472  do i = 1, nxgrid
473  dst_area(interp%i_dst(i), interp%j_dst(i)) = dst_area(interp%i_dst(i), interp%j_dst(i)) + xgrid_area(i)
474  end do
475 
476  do i = 1, nxgrid
477  interp%area_frac_dst(i) = xgrid_area(i)/dst_area(interp%i_dst(i), interp%j_dst(i) )
478  end do
479  interp%nlon_src = nlon_in; interp%nlat_src = nlat_in
480  interp%nlon_dst = nlon_out; interp%nlat_dst = nlat_out
481  if(present(mask_out)) then
482  if( (size(mask_out,1) .NE. nlon_out) .OR. (size(mask_out,2) .NE. nlat_out) ) call mpp_error(fatal, &
483  'horiz_interp_conserve_mod: size mismatch between mask_out and lon_out/lat_out')
484  mask_out = 0.0
485  do i = 1, nxgrid
486  mask_out(interp%i_dst(i),interp%j_dst(i)) = mask_out(interp%i_dst(i), &
487  & interp%j_dst(i)) + interp%area_frac_dst(i)
488  end do
489  end if
490 
491  deallocate(i_src, j_src, i_dst, j_dst, xgrid_area, dst_area )
492 
493  end subroutine horiz_interp_conserve_new_1dx2d

497  subroutine horiz_interp_conserve_new_2dx1d ( Interp, lon_in, lat_in, lon_out, lat_out, &
498  mask_in, mask_out, verbose)
499  type(horiz_interp_type), intent(inout) :: interp
500  real, intent(in), dimension(:,:) :: lon_in , lat_in
501  real, intent(in), dimension(:) :: lon_out, lat_out

750  subroutine horiz_interp_conserve_version1 ( Interp, data_in, data_out, verbose, &
751  mask_in, mask_out)
752  !-----------------------------------------------------------------------
753  type (horiz_interp_type), intent(in) :: interp
754  real, intent(in), dimension(:,:) :: data_in
755  real, intent(out), dimension(:,:) :: data_out
756  integer, intent(in), optional :: verbose
757  real, intent(in), dimension(:,:), optional :: mask_in
758  real, intent(out), dimension(:,:), optional :: mask_out
759  !----------local variables----------------------------------------------------
760  integer :: m, n, nlon_in, nlat_in, nlon_out, nlat_out, &
761  miss_in, miss_out, is, ie, js, je, &
762  np, npass, iverbose
763  real :: dsum, wsum, avg_in, min_in, max_in, &
764  avg_out, min_out, max_out, eps, asum, &
765  dwtsum, wtsum, arsum, fis, fie, fjs, fje
766  !-----------------------------------------------------------------------
767  iverbose = 0; if (present(verbose)) iverbose = verbose
768 
769  eps = epsilon(wtsum)
770 
771  nlon_in = interp%nlon_src; nlat_in = interp%nlat_src
772  nlon_out = interp%nlon_dst; nlat_out = interp%nlat_dst
773 
774  if (present(mask_in)) then
775  if ( count(mask_in < -.0001 .or. mask_in > 1.0001) > 0 ) &
776  call mpp_error(fatal, 'horiz_interp_conserve_mod: input mask not between 0,1')
777  endif
778 
779  !-----------------------------------------------------------------------
780  !---- loop through output grid boxes ----
781 
782  data_out = 0.0
783  do n = 1, nlat_out
784  ! latitude window
785  ! setup ascending latitude indices and weights
786  if (interp%jlat(n,1) <= interp%jlat(n,2)) then
787  js = interp%jlat(n,1); je = interp%jlat(n,2)
788  fjs = interp%facj(n,1); fje = interp%facj(n,2)
789  else
790  js = interp%jlat(n,2); je = interp%jlat(n,1)
791  fjs = interp%facj(n,2); fje = interp%facj(n,1)
792  endif
793 
794  do m = 1, nlon_out
795  ! longitude window
796  is = interp%ilon(m,1); ie = interp%ilon(m,2)
797  fis = interp%faci(m,1); fie = interp%faci(m,2)
798  npass = 1
799  dwtsum = 0.
800  wtsum = 0.
801  arsum = 0.
802 
803  ! wrap-around on input grid
804  ! sum using 2 passes (pass 1: end of input grid)
805  if ( ie < is ) then
806  ie = nlon_in
807  fie = 1.0
808  npass = 2
809  endif
810 
811  do np = 1, npass
812  ! pass 2: beginning of input grid
813  if ( np == 2 ) then
814  is = 1
815  fis = 1.0
816  ie = interp%ilon(m,2)
817  fie = interp%faci(m,2)
818  endif
819 
820  ! summing data*weight and weight for single grid point
821  if (present(mask_in)) then
822  call data_sum ( data_in(is:ie,js:je), interp%area_src(is:ie,js:je), &
823  fis, fie, fjs,fje, dwtsum, wtsum, arsum, mask_in(is:ie,js:je) )
824  else if( allocated(interp%mask_in) ) then
825  call data_sum ( data_in(is:ie,js:je), interp%area_src(is:ie,js:je), &
826  fis, fie, fjs,fje, dwtsum, wtsum, arsum, interp%mask_in(is:ie,js:je) )
827  else
828  call data_sum ( data_in(is:ie,js:je), interp%area_src(is:ie,js:je), &
829  fis, fie, fjs,fje, dwtsum, wtsum, arsum )
830  endif
831  enddo
832 
833  if (wtsum > eps) then
834  data_out(m,n) = dwtsum/wtsum
835  if (present(mask_out)) mask_out(m,n) = wtsum/arsum
836  else
837  data_out(m,n) = 0.
838  if (present(mask_out)) mask_out(m,n) = 0.0
839  endif
840 
841  enddo
842  enddo
843 
844  !***********************************************************************
845  ! compute statistics: minimum, maximum, and mean
846  !-----------------------------------------------------------------------
847 
848  if (iverbose > 0) then
849 
850  ! compute statistics of input data
851 
852  call stats(data_in, interp%area_src, asum, dsum, wsum, min_in, max_in, miss_in, mask_in)
853  ! diagnostic messages
854  ! on the root_pe, we can calculate the global mean, minimum and maximum.
855  if(pe == root_pe) then
856  if (wsum > 0.0) then
857  avg_in=dsum/wsum
858  else
859  print *, 'horiz_interp stats: input area equals zero '
860  avg_in=0.0
861  endif
862  if (iverbose > 1) print '(2f16.11)', 'global sum area_in = ', asum, wsum
863  endif
864 
865  ! compute statistics of output data
866  call stats(data_out, interp%area_dst, asum, dsum, wsum, min_out, max_out, miss_out, mask_out)
867  ! diagnostic messages
868  if(pe == root_pe) then
869  if (wsum > 0.0) then
870  avg_out=dsum/wsum
871  else
872  print *, 'horiz_interp stats: output area equals zero '
873  avg_out=0.0
874  endif
875  if (iverbose > 1) print '(2f16.11)', 'global sum area_out = ', asum, wsum
876  endif
877  !---- output statistics ----
878  ! the global mean, min and max are calculated on the root pe.
879  if(pe == root_pe) then
880  write (*,900)
881  write (*,901) min_in ,max_in ,avg_in
882  if (present(mask_in)) write (*,903) miss_in
883  write (*,902) min_out,max_out,avg_out
884  if (present(mask_out)) write (*,903) miss_out
885  endif
886 
887 900 format (/,1x,10('-'),' output from horiz_interp ',10('-'))
888 901 format (' input: min=',f16.9,' max=',f16.9,' avg=',f22.15)
889 902 format (' output: min=',f16.9,' max=',f16.9,' avg=',f22.15)
890 903 format (' number of missing points = ',i6)
891 
892  endif
893 
894  !-----------------------------------------------------------------------
895  end subroutine horiz_interp_conserve_version1

898  subroutine horiz_interp_conserve_version2 ( Interp, data_in, data_out, verbose )
899  !-----------------------------------------------------------------------
900  type (horiz_interp_type), intent(in) :: interp
901  real, intent(in), dimension(:,:) :: data_in
902  real, intent(out), dimension(:,:) :: data_out
903  integer, intent(in), optional :: verbose
