21 subroutine drifters_set_field_XXX(self, index_field, x, y, &
22 #if _DIMS >= 3
23  & z, &
24 #endif
25  & data, ermesg)
26  use cloud_interpolator_mod
27  type(drifters_type) :: self
28  ! field index must be consistent with field_names from input file
29  integer, intent(in) :: index_field
30  real, intent(in) :: x(:)
31  real, intent(in) :: y(:)
32 #if _DIMS == 2
33  real, intent(in) :: data(:,:)
34 #endif
35 #if _DIMS == 3
36  real, intent(in) :: z(:)
37  real, intent(in) :: data(:,:,:)
38 #endif
39  character(len=*), intent(out) :: ermesg
40 
41  integer i, j, ip, ier, ij(self%core%nd), nsizes(self%core%nd), nf
42  real fvals(2**self%core%nd), ts(self%core%nd)
43 
44  ermesg = ''
45  ! only interpolate field if RK step is complete
46  if(self%rk4_step > 1) return
47 
48  ! interpolate onto new positions
49  nsizes(1) = size(x)
50  nsizes(2) = size(y)
51 #if _DIMS >= 3
52  nsizes(3) = size(z)
53 #endif
54 
55  if(nsizes(1) /= size(data, 1) .or. nsizes(2) /= size(data, 2)) then
56  ermesg = 'drifters_set_field_XXX: ERROR size mismatch between data and x or y'
57  return
58  end if
59 #if _DIMS >=3
60  if(nsizes(3) /= size(data, 3)) then
61  ermesg = 'drifters_set_field_XXX: ERROR size mismatch between data and z'
62  return
63  endif
64 #endif
65 
66  if(size(self%fields, 2) < self%core%np) then
67  ! resize
68  deallocate(self%fields, stat=ier)
69  nf = size(self%input%field_names)
70  allocate(self%fields(nf, self%core%npdim))
71  self%fields = -huge(1.)
72  endif
73 
74  do ip = 1, self%core%np
75  call cld_ntrp_locate_cell(x, self%core%positions(1,ip), i, ier)
76  ij(1) = i
77 #ifdef _DEBUG
78  if(i<1) then
79  print *,'drifters_set_field_XXX::OUT OF BOUND ERROR. xmin, x, xmax=', minval(x), self%core%positions(1,ip), maxval(x)
80  endif
81 #endif
82  ts(1) = (self%core%positions(1,ip) - x(i))/(x(i+1) - x(i))
83 
84  call cld_ntrp_locate_cell(y, self%core%positions(2,ip), j, ier)
85  ij(2) = j
86 #ifdef _DEBUG
87  if(j<1) then
88  print *,'drifters_set_field_XXX::OUT OF BOUND ERROR. ymin, y, ymax=', minval(y), self%core%positions(2,ip), maxval(y)
89  endif
90 #endif
91  ts(2) = (self%core%positions(2,ip) - y(j))/(y(j+1) - y(j))
92 
93 #if _DIMS >= 3
94  call cld_ntrp_locate_cell(z, self%core%positions(3,ip), j, ier)
95  ij(3) = j
96 #ifdef _DEBUG
97  if(j<1) then
98  print *,'drifters_set_field_XXX::OUT OF BOUND ERROR. ymin, y, ymax=', minval(y), self%core%positions(2,ip), maxval(y)
99  endif
100 #endif
101  ts(3) = (self%core%positions(3,ip) - z(j))/(z(j+1) - z(j))
102 #endif
103 
104  call cld_ntrp_get_cell_values(nsizes, _FLATTEN(data), ij, fvals, ier)
105  call cld_ntrp_linear_cell_interp(fvals, ts, self%fields(index_field, ip), ier)
106  enddo
107 
108 end subroutine drifters_set_field_XXX
