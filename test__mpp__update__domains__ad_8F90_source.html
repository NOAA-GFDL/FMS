68  !> test calling mpp_halo_update_ad on a 3D 64-bit real data array
69  subroutine test_halo_update_ad_r8( test_type )
70  character(len=*), intent(in) :: test_type
71  ! local
72  type(domain2D) :: domain
73  integer :: shift, i, j, k
74  logical :: is_symmetry
75  integer :: is, ie, js, je, isd, ied, jsd, jed, pe
76  real(kind=r8_kind), allocatable, dimension(:,:,:) :: x_ad, y_ad, x_fd, y_fd, x_save, y_save
77  real(kind=r8_kind) :: ad_sum, fd_sum, sum_diff
78 
79  if(index(test_type, 'symmetry') == 0) then
80  is_symmetry = .false.
81  else
82  is_symmetry = .true.
83  end if
84  select case(test_type)
85  case( 'Simple', 'Simple symmetry' )
86  call mpp_define_layout( (/1,nx,1,ny/), npes, layout )
87  call mpp_define_domains( (/1,nx,1,ny/), layout, domain, whalo=whalo, ehalo=ehalo, &
88  shalo=shalo, nhalo=nhalo, name=test_type, symmetry = is_symmetry )
89  case default
90  call mpp_error( fatal, 'test_mpp_update_domains_ad_r8: '//test_type//' is not a valid test.')
91  end select
92 
93 !set up x array
94  call mpp_get_compute_domain( domain, is, ie, js, je )
95  call mpp_get_data_domain( domain, isd, ied, jsd, jed )
96 
97  shift=1
98 !---test 3d single fields----------------------------------------------------------
99  allocate( x_fd(isd:ied,jsd:jed,nz) )
100  allocate( x_ad(isd:ied,jsd:jed,nz) )
101  allocate( x_save(isd:ied,jsd:jed,nz) )
102  x_fd = 0.0; x_ad = 0.0; x_save = 0.0
103 
104  do k = 1,nz
105  do j = js,je
106  do i = is,ie
107  x_fd(i,j,k) = i*j
108  end do
109  end do
110  end do
111  x_save = x_fd
112 
113  ! full update
114  call mpp_update_domains( x_fd, domain )
115 
116  fd_sum = 0.0
117  do k = 1,nz
118  do j = jsd,jed
119  do i = isd,ied
120  fd_sum = fd_sum + x_fd(i,j,k)*x_fd(i,j,k)
121  end do
122  end do
123  end do
124  call mpp_sum( fd_sum )
125 
126  x_ad = x_fd
127  call mpp_update_domains_ad( x_ad, domain )
128 
129  ad_sum = 0.0
130  do k = 1,nz
131  do j = jsd,jed
132  do i = isd,ied
133  ad_sum = ad_sum + x_ad(i,j,k)*x_save(i,j,k)
134  end do
135  end do
136  end do
137  call mpp_sum( ad_sum )
138  call mpp_sync()
139  pe = mpp_pe()
140  sum_diff = 0.0
141  sum_diff = abs(ad_sum-fd_sum)/fd_sum
142 
143  if( pe.EQ.mpp_root_pe() ) then
144  if (sum_diff .lt. 1e-7) then
145  call mpp_error(note, "Passed Adjoint Dot Test: mpp_update_domains_ad_r8(single 3D field)")
146  else
147  call mpp_error(fatal, "FAILED Adjoint Dot Test: mpp_update_domains_ad_r8(single 3D field)")
148  endif
149  endif
150 
151  deallocate (x_ad, x_fd, x_save)
152 
153  ! test 3d vector fields
154  allocate( x_ad(isd:ied+shift,jsd:jed ,nz) )
155  allocate( x_fd(isd:ied+shift,jsd:jed ,nz) )
156  allocate( x_save(isd:ied+shift,jsd:jed ,nz) )
157  allocate( y_ad(isd:ied ,jsd:jed+shift,nz) )
158  allocate( y_fd(isd:ied ,jsd:jed+shift,nz) )
159  allocate( y_save(isd:ied ,jsd:jed+shift,nz) )
160 
161  x_fd=0; y_fd=0
162  do k = 1,nz
163  do j = js,je
164  do i = is,ie
165  x_fd(i,j,k)=i*j
166  y_fd(i,j,k)=i*j
167  end do
168  end do
169  end do
170 
171  call mpp_update_domains( x_fd, y_fd, domain, gridtype=cgrid_ne)
172  x_save=x_fd
173  y_save=y_fd
174 
175  fd_sum = 0.
176  do k = 1,nz
177  do j = jsd,jed
178  do i = isd,ied+shift
179  fd_sum = fd_sum + x_fd(i,j,k)*x_fd(i,j,k)
180  end do
181  end do
182  end do
183  do k = 1,nz
184  do j = jsd,jed+shift
185  do i = isd,ied
186  fd_sum = fd_sum + y_fd(i,j,k)*y_fd(i,j,k)
187  end do
188  end do
189  end do
190  call mpp_sum( fd_sum )
191 
192  x_ad = x_fd
193  y_ad = y_fd
194  call mpp_update_domains_ad( x_ad, y_ad, domain, gridtype=cgrid_ne)
195 
196  ad_sum = 0.0
197  do k = 1,nz
198  do j = jsd,jed
199  do i = isd,ied+shift
200  ad_sum = ad_sum + x_ad(i,j,k)*x_save(i,j,k)
201  end do
202  end do
203  end do
204  do k = 1,nz
205  do j = jsd,jed+shift
206  do i = isd,ied
207  ad_sum = ad_sum + y_ad(i,j,k)*y_save(i,j,k)
208  end do
209  end do
210  end do
211  call mpp_sum( ad_sum )
212  call mpp_sync()
213  pe = mpp_pe()
214  sum_diff = 0.0
215  sum_diff = abs(ad_sum-fd_sum)/fd_sum
216 
217  if ( pe.EQ.mpp_root_pe() ) then
218  if (sum_diff .lt. 1e-7) then
219  call mpp_error(note, "Passed Adjoint Dot Test: mpp_update_domains_ad_r8(vector 3D fields)")
220  else
221  call mpp_error(fatal,"FAILED Adjoint Dot Test: mpp_update_domains_ad_r8(vector 3D fields)")
222  endif
223  endif
224  deallocate (x_ad, y_ad, x_fd, y_fd, x_save, y_save)
225 
226  end subroutine test_halo_update_ad_r8
