107 !> @brief Get exchange grid information from mosaic xgrid file.
108 !> Example usage:
109 !!
110 !! call get_mosaic_xgrid(fileobj, nxgrid, i1, j1, i2, j2, area)
111 !!
112  subroutine get_mosaic_xgrid(fileobj, i1, j1, i2, j2, area, ibegin, iend)
113  type(fmsnetcdffile_t), intent(in) :: fileobj !> The file that contains exchange grid information.
114  integer, intent(inout) :: i1(:), j1(:), i2(:), j2(:) !> i and j indices for grids 1 and 2
115  class(*), intent(inout) :: area(:) !> area of the exchange grid. The area is scaled to
116  !! represent unit earth area
117  integer, optional, intent(in) :: ibegin, iend
118 
119  integer :: start(4), nread(4), istart
120  real, dimension(2, size(i1(:))) :: tile1_cell, tile2_cell
121  integer :: nxgrid, n
122  real :: garea
123  real :: get_global_area
124 
125  garea = get_global_area()
126 
127  ! When start and nread present, make sure nread(1) is the same as the size of the data
128  if(present(ibegin) .and. present(iend)) then
129  istart = ibegin
130  nxgrid = iend - ibegin + 1
131  if(nxgrid .NE. size(i1(:))) call mpp_error(fatal, .NE."get_mosaic_xgrid: nxgrid size(i1(:))")
132  if(nxgrid .NE. size(j1(:))) call mpp_error(fatal, .NE."get_mosaic_xgrid: nxgrid size(j1(:))")
133  if(nxgrid .NE. size(i2(:))) call mpp_error(fatal, .NE."get_mosaic_xgrid: nxgrid size(i2(:))")
134  if(nxgrid .NE. size(j2(:))) call mpp_error(fatal, .NE."get_mosaic_xgrid: nxgrid size(j2(:))")
135  if(nxgrid .NE. size(area(:))) call mpp_error(fatal, .NE."get_mosaic_xgrid: nxgrid size(area(:))")
136  else
137  istart = 1
138  nxgrid = size(i1(:))
139  endif
140 
141  start = 1; nread = 1
142  start(1) = istart; nread(1) = nxgrid
143 
144  select type(area)
145  type is (real(r4_kind))
146  call read_data(fileobj, 'xgrid_area', area, corner=start, edge_lengths=nread)
147  type is (real(r8_kind))
148  call read_data(fileobj, 'xgrid_area', area, corner=start, edge_lengths=nread)
149  class default
150  call mpp_error(fatal,"get_mosaic_xgrid: invalid data type for area, must be real(r4_kind) or real(r8_kind)")
151  end select
152 
153  start = 1; nread = 1
154  nread(1) = 2
155  start(2) = istart; nread(2) = nxgrid
156 
157  select type(area)
158  type is (real(r4_kind))
159  call read_data(fileobj, 'tile1_cell', tile1_cell, corner=start, edge_lengths=nread)
160  call read_data(fileobj, 'tile2_cell', tile2_cell, corner=start, edge_lengths=nread)
161  type is (real(r8_kind))
162  call read_data(fileobj, 'tile1_cell', tile1_cell, corner=start, edge_lengths=nread)
163  call read_data(fileobj, 'tile2_cell', tile2_cell, corner=start, edge_lengths=nread)
164  end select
165

167  i1(n) = int(tile1_cell(1,n))
168  j1(n) = int(tile1_cell(2,n))
169  i2(n) = int(tile2_cell(1,n))
170  j2(n) = int(tile2_cell(2,n))
171  select type(area)
172  type is (real(r4_kind))
173  area(n) = real(area(n)/garea, r4_kind)
174  type is (real(r8_kind))
175  area(n) = real(area(n)/garea, r8_kind)
176  end select
177  end do
178 
179  return
180 
181  end subroutine get_mosaic_xgrid

261  !###############################################################################
262  !> Get contact information from mosaic_file
263  !> <br>Example usage:
264  !! call get_mosaic_contact(mosaic_file, tile1, tile2, istart1, iend1, jstart1, jend1,
265  !! istart2, iend2, jstart2, jend2)
266  !> @param fileobj mosaic file object
267  !> @param[inout] tile1 list of tile numbers in tile 1 of each contact
268  !> @param[inout] tile2 list of tile numbers in tile 2 of each contact
269  !> @param[inout] istart1 list of starting i-index in tile 1 of each contact
270  !> @param[inout] iend1 list of ending i-index in tile 1 of each contact
271  !> @param[inout] jstart1 list of starting j-index in tile 1 of each contact
272  !> @param[inout] jend1 list of ending j-index in tile 1 of each contact
273  !> @param[inout] istart2 list of starting i-index in tile 2 of each contact
274  !> @param[inout] iend2 list of ending i-index in tile 2 of each contact
275  !> @param[inout] jstart2 list of starting j-index in tile 2 of each contact
276  !> @param[inout] jend2 list of ending j-index in tile 2 of each contact
277  subroutine get_mosaic_contact( fileobj, tile1, tile2, istart1, iend1, jstart1, jend1, &
278  istart2, iend2, jstart2, jend2)
279  type(fmsnetcdffile_t), intent(in) :: fileobj
280  integer, dimension(:), intent(inout) :: tile1, tile2
281  integer, dimension(:), intent(inout) :: istart1, iend1, jstart1, jend1
282  integer, dimension(:), intent(inout) :: istart2, iend2, jstart2, jend2
283  character(len=MAX_NAME), allocatable :: gridtiles(:)
284  character(len=MAX_NAME), allocatable :: contacts(:)
285  character(len=MAX_NAME), allocatable :: contacts_index(:)
286  character(len=MAX_NAME) :: strlist(8)
287  integer :: ntiles, n, m, ncontacts, nstr, ios
288  integer :: i1_type, j1_type, i2_type, j2_type
289  logical :: found
290 
291  ntiles = get_mosaic_ntiles(fileobj)
292  allocate(gridtiles(ntiles))
293  call read_data(fileobj, 'gridtiles', gridtiles)
294 
295  ncontacts = get_mosaic_ncontacts(fileobj)
296  if(ncontacts>0) then
297  allocate(contacts(ncontacts), contacts_index(ncontacts))
298  call read_data(fileobj, "contacts", contacts)
299  call read_data(fileobj, "contact_index", contacts_index)
300  endif
301  do n = 1, ncontacts
302  nstr = parse_string(contacts(n), ":", strlist)
303  if(nstr .NE. 4) call mpp_error(fatal, &
304  "mosaic_mod(get_mosaic_contact): number of elements in contact seperated by :/:: should be 4")
305  found = .false.
306  do m = 1, ntiles
307  if(trim(gridtiles(m)) == trim(strlist(2)) ) then !found the tile name
308  found = .true.
309  tile1(n) = m
310  exit
311  endif
312  enddo
313 
314  if(.not.found) call mpp_error(fatal, &
315  "mosaic_mod(get_mosaic_contact):the first tile name specified in contact is not found in tile list")
316 
317  found = .false.
318  do m = 1, ntiles
319  if(trim(gridtiles(m)) == trim(strlist(4)) ) then !found the tile name
320  found = .true.
321  tile2(n) = m
322  exit
323  endif
324  enddo
325 
326  if(.not.found) call mpp_error(fatal, &
327  "mosaic_mod(get_mosaic_contact):the second tile name specified in contact is not found in tile list")
328 
329  nstr = parse_string(contacts_index(n), ":,", strlist)
330  if(nstr .NE. 8) then
331  if(mpp_pe()==mpp_root_pe()) then
332  print*, "nstr is ", nstr
333  print*, "contacts is ", contacts_index(n)
334  do m = 1, nstr
335  print*, "strlist is ", trim(strlist(m))
336  enddo
337  endif
338  call mpp_error(fatal, &
339  "mosaic_mod(get_mosaic_contact): number of elements in contact_index seperated by :/, should be 8")
340  endif
341  read(strlist(1), *, iostat=ios) istart1(n)
342  if(ios .NE. 0) call mpp_error(fatal, &
343  "mosaic_mod(get_mosaic_contact): Error in reading istart1")
344  read(strlist(2), *, iostat=ios) iend1(n)
345  if(ios .NE. 0) call mpp_error(fatal, &
346  "mosaic_mod(get_mosaic_contact): Error in reading iend1")
347  read(strlist(3), *, iostat=ios) jstart1(n)
348  if(ios .NE. 0) call mpp_error(fatal, &
349  "mosaic_mod(get_mosaic_contact): Error in reading jstart1")
350  read(strlist(4), *, iostat=ios) jend1(n)
351  if(ios .NE. 0) call mpp_error(fatal, &
352  "mosaic_mod(get_mosaic_contact): Error in reading jend1")
353  read(strlist(5), *, iostat=ios) istart2(n)
354  if(ios .NE. 0) call mpp_error(fatal, &
355  "mosaic_mod(get_mosaic_contact): Error in reading istart2")
356  read(strlist(6), *, iostat=ios) iend2(n)
357  if(ios .NE. 0) call mpp_error(fatal, &
358  "mosaic_mod(get_mosaic_contact): Error in reading iend2")
359  read(strlist(7), *, iostat=ios) jstart2(n)
360  if(ios .NE. 0) call mpp_error(fatal, &
361  "mosaic_mod(get_mosaic_contact): Error in reading jstart2")
362  read(strlist(8), *, iostat=ios) jend2(n)
363  if(ios .NE. 0) call mpp_error(fatal, &
364  "mosaic_mod(get_mosaic_contact): Error in reading jend2")
365 
366  i1_type = transfer_to_model_index(istart1(n), iend1(n), x_refine)
367  j1_type = transfer_to_model_index(jstart1(n), jend1(n), y_refine)
368  i2_type = transfer_to_model_index(istart2(n), iend2(n), x_refine)
369  j2_type = transfer_to_model_index(jstart2(n), jend2(n), y_refine)
370 
371  if( i1_type == 0 .AND. j1_type == 0 ) call mpp_error(fatal, &
372  "mosaic_mod(get_mosaic_contact): istart1==iend1 and jstart1==jend1")
373  if( i2_type == 0 .AND. j2_type == 0 ) call mpp_error(fatal, &
374  "mosaic_mod(get_mosaic_contact): istart2==iend2 and jstart2==jend2")
375  if( i1_type + j1_type .NE. i2_type + j2_type ) call mpp_error(fatal, &
376  "mosaic_mod(get_mosaic_contact): It is not a line or overlap contact")
377 
378  enddo
379 
380  deallocate(gridtiles)
381  if(ncontacts>0) deallocate(contacts, contacts_index)
382 
383  end subroutine get_mosaic_contact

386 function transfer_to_model_index(istart, iend, refine_ratio)
387  integer, intent(inout) :: istart, iend
388  integer :: refine_ratio
389  integer :: transfer_to_model_index
390  integer :: istart_in, iend_in
391 
392  istart_in = istart
393  iend_in = iend
394 
395  if( istart_in == iend_in ) then
396  transfer_to_model_index = 0
397  istart = (istart_in + 1)/refine_ratio
398  iend = istart
399  else
400  transfer_to_model_index = 1
401  if( iend_in > istart_in ) then
402  istart = istart_in + 1
403  iend = iend_in
404  else
405  istart = istart_in
406  iend = iend_in + 1
407  endif
408  if( mod(istart, refine_ratio) .NE. 0 .OR. mod(iend,refine_ratio) .NE. 0) call mpp_error(fatal, &
409  "mosaic_mod(transfer_to_model_index): mismatch between refine_ratio and istart/iend")
410  istart = istart/refine_ratio
411  iend = iend/refine_ratio
412 
413  endif
414 
415  return
416 
417 end function transfer_to_model_index
