21 !! @brief Routines to get data in a file whose path is described in a user-provided data_table
22 !! and do spatial and temporal interpolation if necessary to convert data to model's grid and time.
23 !! @author Z. Liang, M.J. Harrison, M. Winton
24 !!
25 !! Before using @ref data_override a data_table must be created with the following entries:
26 !! gridname, fieldname_code, fieldname_file, file_name, ongrid, factor.
27 !!
28 !! More explainations about data_table entries can be found in the source code (defining data_type)
29 !!
30 !! If user wants to override fieldname_code with a const, set fieldname_file in data_table = ""
31 !! and factor = const
32 !!
33 !! If user wants to override fieldname_code with data from a file, set fieldname_file = name in
34 !! the netCDF data file, factor then will be for unit conversion (=1 if no conversion required)
35 !!
36 !! A field can be overriden globally (by default) or users can specify one or two regions in which
37 !! data_override will take place, field values outside the region will not be affected.

187  type (domain2d), intent(in), optional :: ocean_domain_in, ice_domain_in
188  type (domain2d), intent(in), optional :: land_domain_in
189  type(domainug) , intent(in), optional :: land_domainug_in
190 
191  character(len=128) :: grid_file = 'INPUT/grid_spec.nc'

195  type(fmsnetcdffile_t) :: fileobj

199  read (input_nml_file, data_override_nml, iostat=io_status)
200  ierr = check_nml_error(io_status, 'data_override_nml')
201  unit = stdlog()
202  write(unit, data_override_nml)
203 
204 ! if(module_is_initialized) return
205 
206  atm_on = PRESENT(atm_domain_in)
207  ocn_on = PRESENT(ocean_domain_in)
208  lnd_on = PRESENT(land_domain_in)
209  ice_on = PRESENT(ice_domain_in)
210  lndug_on = PRESENT(land_domainug_in)
211  if(.not. module_is_initialized) then
212  atm_domain = null_domain2d
213  ocn_domain = null_domain2d
214  lnd_domain = null_domain2d
215  ice_domain = null_domain2d
216  lnd_domainug = null_domainug
217  end if
218  if (atm_on) atm_domain = atm_domain_in
219  if (ocn_on) ocn_domain = ocean_domain_in
220  if (lnd_on) lnd_domain = land_domain_in
221  if (ice_on) ice_domain = ice_domain_in
222  if (lndug_on) lnd_domainug = land_domainug_in
223 
224  if(.not. module_is_initialized) then
225  call horiz_interp_init
226  call write_version_number("DATA_OVERRIDE_MOD", version)
227 
228 ! Initialize user-provided data table
229  default_table%gridname = 'non'
230  default_table%fieldname_code = 'none'
231  default_table%fieldname_file = 'none'
232  default_table%file_name = 'none'
233  default_table%factor = 1.
234  default_table%interpol_method = 'bilinear'
235 
236 #ifdef use_yaml
237  call read_table_yaml(data_table)
238 #else
239  do i = 1,max_table
240  data_table(i) = default_table
241  enddo
242  call read_table(data_table)
243 #endif
244 
245 ! Initialize override array
246  default_array%gridname = 'NONE'
247  default_array%fieldname = 'NONE'
248  default_array%t_index = -1
249  default_array%dims = -1
250  default_array%comp_domain = -1
251  do i = 1, max_array
252  override_array(i) = default_array
253  enddo
254  call time_interp_external_init
255  end if
256 
257  module_is_initialized = .true.
258 
259  if ( .NOT. (atm_on .or. ocn_on .or. lnd_on .or. ice_on .or. lndug_on)) return
260  if (table_size .eq. 0) then
261  call mpp_error(note, "data_table is empty, not doing any data_overrides")
262  return
263  endif
264  call fms2_io_init
265 
266 ! Test if grid_file is already opened
267  inquire (file=trim(grid_file), opened=file_open)
268  if(file_open) call mpp_error(fatal, trim(grid_file)//' already opened')
269 
270  if(.not. open_file(fileobj, grid_file, 'read' )) then
271  call mpp_error(fatal, 'data_override_mod: Error in opening file '//trim(grid_file))
272  endif
273 
274  if(variable_exists(fileobj, "x_T" ) .OR. variable_exists(fileobj, "geolon_t" ) ) then
275  use_get_grid_version = 1
276  call close_file(fileobj)
277  else if(variable_exists(fileobj, "ocn_mosaic_file" ) .OR. variable_exists(fileobj, "gridfiles" ) ) then
278  use_get_grid_version = 2
279  if(variable_exists(fileobj, "gridfiles" ) ) then
280  if(count_ne_1((ocn_on .OR. ice_on), lnd_on, atm_on)) call mpp_error(fatal, 'data_override_mod: the grid file ' //&
281  'is a solo mosaic, one and only one of atm_on, lnd_on or ice_on/ocn_on should be true')
282  end if
283  else
284  call mpp_error(fatal, 'data_override_mod: none of x_T, geolon_t, ocn_mosaic_file or gridfiles exist in '// &
285  & trim(grid_file))
286  endif
287 
288  if(use_get_grid_version .EQ. 1) then
289  if (atm_on .and. .not. allocated(lon_local_atm) ) then
290  call mpp_get_compute_domain( atm_domain,is,ie,js,je)
291  allocate(lon_local_atm(is:ie,js:je), lat_local_atm(is:ie,js:je))

293  min_glo_lon_atm, max_glo_lon_atm, grid_center_bug )
294  endif
295  if (ocn_on .and. .not. allocated(lon_local_ocn) ) then
296  call mpp_get_compute_domain( ocn_domain,is,ie,js,je)
297  allocate(lon_local_ocn(is:ie,js:je), lat_local_ocn(is:ie,js:je))
298  call get_grid_version_1(grid_file, 'ocn', ocn_domain, is, ie, js, je, lon_local_ocn, lat_local_ocn, &
299  min_glo_lon_ocn, max_glo_lon_ocn, grid_center_bug )
300  endif
301 
302  if (lnd_on .and. .not. allocated(lon_local_lnd) ) then
303  call mpp_get_compute_domain( lnd_domain,is,ie,js,je)
304  allocate(lon_local_lnd(is:ie,js:je), lat_local_lnd(is:ie,js:je))
305  call get_grid_version_1(grid_file, 'lnd', lnd_domain, is, ie, js, je, lon_local_lnd, lat_local_lnd, &
306  min_glo_lon_lnd, max_glo_lon_lnd, grid_center_bug )
307  endif
308 
309  if (ice_on .and. .not. allocated(lon_local_ice) ) then
310  call mpp_get_compute_domain( ice_domain,is,ie,js,je)
311  allocate(lon_local_ice(is:ie,js:je), lat_local_ice(is:ie,js:je))
312  call get_grid_version_1(grid_file, 'ice', ice_domain, is, ie, js, je, lon_local_ice, lat_local_ice, &
313  min_glo_lon_ice, max_glo_lon_ice, grid_center_bug )
314  endif
315  else
316  if (atm_on .and. .not. allocated(lon_local_atm) ) then
317  call mpp_get_compute_domain(atm_domain,is,ie,js,je)
318  allocate(lon_local_atm(is:ie,js:je), lat_local_atm(is:ie,js:je))
319  call get_grid_version_2(fileobj, 'atm', atm_domain, is, ie, js, je, lon_local_atm, lat_local_atm, &
320  min_glo_lon_atm, max_glo_lon_atm )
321  endif
322 
323  if (ocn_on .and. .not. allocated(lon_local_ocn) ) then
324  call mpp_get_compute_domain( ocn_domain,is,ie,js,je)
325  allocate(lon_local_ocn(is:ie,js:je), lat_local_ocn(is:ie,js:je))
326  call get_grid_version_2(fileobj, 'ocn', ocn_domain, is, ie, js, je, lon_local_ocn, lat_local_ocn, &
327  min_glo_lon_ocn, max_glo_lon_ocn )
328  endif
329 
330  if (lnd_on .and. .not. allocated(lon_local_lnd) ) then
331  call mpp_get_compute_domain( lnd_domain,is,ie,js,je)
332  allocate(lon_local_lnd(is:ie,js:je), lat_local_lnd(is:ie,js:je))
333  call get_grid_version_2(fileobj, 'lnd', lnd_domain, is, ie, js, je, lon_local_lnd, lat_local_lnd, &
334  min_glo_lon_lnd, max_glo_lon_lnd )
335  endif
336 
337  if (ice_on .and. .not. allocated(lon_local_ice) ) then
338  call mpp_get_compute_domain( ice_domain,is,ie,js,je)
339  allocate(lon_local_ice(is:ie,js:je), lat_local_ice(is:ie,js:je))
340  call get_grid_version_2(fileobj, 'ocn', ice_domain, is, ie, js, je, lon_local_ice, lat_local_ice, &
341  min_glo_lon_ice, max_glo_lon_ice )
342  endif
343  end if
344  if(use_get_grid_version .EQ. 2) then
345  call close_file(fileobj)
346  end if
347 
348 end subroutine data_override_init

350 #ifndef use_yaml
351 subroutine read_table(data_table)
352  type(data_type), dimension(max_table), intent(inout) :: data_table
353 
354  integer :: ntable
355  integer :: ntable_lima
356  integer :: ntable_new
357 
358  integer :: iunit
359  integer :: io_status
360  character(len=256) :: record
361  type(data_type) :: data_entry
362 
363  logical :: ongrid
364  logical :: table_exists !< Flag indicating existence of data_table
365  character(len=128) :: region, region_type
366 
367  integer :: sunit
368 
369 ! Read coupler_table
370  inquire(file='data_table', exist=table_exists)
371  if (.not. table_exists) then
372  call mpp_error(note, 'data_override_mod: File data_table does not exist.')
373  table_size = 0
374  return
375  end if
376 
377  open(newunit=iunit, file='data_table', action='READ', iostat=io_status)
378  if(io_status/=0) call mpp_error(fatal, 'data_override_mod: Error in opening file data_table.')
379 
380  ntable = 0
381  ntable_lima = 0
382  ntable_new = 0
383 
384  do while (ntable <= max_table)
385  read(iunit,'(a)',end=100) record
386  if (record(1:1) == '#') cycle
387  if (record(1:10) == ' ') cycle
388  ntable=ntable+1
389  if(index(lowercase(record), "inside_region") .ne. 0 .or. index(lowercase(record), "outside_region") .ne. 0) then
390  if(index(lowercase(record), ".false.") .ne. 0 .or. index(lowercase(record), ".true.") .ne. 0 ) then
391  ntable_lima = ntable_lima + 1
392  read(record,*,err=99) data_entry%gridname, data_entry%fieldname_code, data_entry%fieldname_file, &
393  data_entry%file_name, ongrid, data_entry%factor, region, region_type
394  if(ongrid) then
395  data_entry%interpol_method = 'none'
396  else
397  data_entry%interpol_method = 'bilinear'
398  endif
399  else
400  ntable_new=ntable_new+1
401  read(record,*,err=99) data_entry%gridname, data_entry%fieldname_code, data_entry%fieldname_file, &
402  data_entry%file_name, data_entry%interpol_method, data_entry%factor, region, &
403  & region_type
404  if (data_entry%interpol_method == 'default') then
405  data_entry%interpol_method = default_table%interpol_method
406  endif
407  if (.not.(data_entry%interpol_method == 'default' .or. &
408  data_entry%interpol_method == 'bicubic' .or. &
409  data_entry%interpol_method == 'bilinear' .or. &
410  data_entry%interpol_method == 'none')) then
411  sunit = stdout()
412  write(sunit,*)" gridname is ", trim(data_entry%gridname)
413  write(sunit,*)" fieldname_code is ", trim(data_entry%fieldname_code)
414  write(sunit,*)" fieldname_file is ", trim(data_entry%fieldname_file)
415  write(sunit,*)" file_name is ", trim(data_entry%file_name)
416  write(sunit,*)" factor is ", data_entry%factor
417  write(sunit,*)" interpol_method is ", trim(data_entry%interpol_method)
418  call mpp_error(fatal, 'data_override_mod: invalid last entry in data_override_table, ' &
419  //'its value should be "default", "bicubic", "bilinear" or "none" ')
420  endif
421  endif
422  if( trim(region_type) == "inside_region" ) then
423  data_entry%region_type = inside_region
424  else if( trim(region_type) == "outside_region" ) then
425  data_entry%region_type = outside_region
426  else
427  call mpp_error(fatal, 'data_override_mod: region type should be inside_region or outside_region')
428  endif
429  if (data_entry%file_name == "") call mpp_error(fatal, &
430  "data_override: filename not given in data_table when region_type is not NO_REGION")
431  if(data_entry%fieldname_file == "") call mpp_error(fatal, &
432  "data_override: fieldname_file must be specified in data_table when region_type is not NO_REGION")
433  if( trim(data_entry%interpol_method) == 'none') call mpp_error(fatal, &
434  "data_override(data_override_init): ongrid must be false when region_type is not NO_REGION")
435  read(region,*) data_entry%lon_start, data_entry%lon_end, data_entry%lat_start, data_entry%lat_end
436  !--- make sure data_entry%lon_end > data_entry%lon_start and data_entry%lat_end > data_entry%lat_start
437  if(data_entry%lon_end .LE. data_entry%lon_start) call mpp_error(fatal, &
438  "data_override: lon_end should be greater than lon_start")
439  if(data_entry%lat_end .LE. data_entry%lat_start) call mpp_error(fatal, &
440  "data_override: lat_end should be greater than lat_start")
441  ! old format
442  else if (index(lowercase(record), ".false.") .ne. 0 .or. index(lowercase(record), ".true.") .ne. 0 ) then
443  ntable_lima = ntable_lima + 1
444  read(record,*,err=99) data_entry%gridname, data_entry%fieldname_code, data_entry%fieldname_file, &
445  data_entry%file_name, ongrid, data_entry%factor
446  if(ongrid) then
447  data_entry%interpol_method = 'none'
448  else
449  data_entry%interpol_method = 'bilinear'
450  endif
451  data_entry%lon_start = 0.0
452  data_entry%lon_end = -1.0
453  data_entry%lat_start = 0.0
454  data_entry%lat_end = -1.0
455  data_entry%region_type = no_region
456  else ! new format
457  ntable_new=ntable_new+1
458  read(record,*,err=99) data_entry%gridname, data_entry%fieldname_code, data_entry%fieldname_file, &
459  data_entry%file_name, data_entry%interpol_method, data_entry%factor
460  if (data_entry%interpol_method == 'default') then
461  data_entry%interpol_method = default_table%interpol_method
462  endif
463  if (.not.(data_entry%interpol_method == 'default' .or. &
464  data_entry%interpol_method == 'bicubic' .or. &
465  data_entry%interpol_method == 'bilinear' .or. &
466  data_entry%interpol_method == 'none')) then
467  sunit = stdout()
468  write(sunit,*)" gridname is ", trim(data_entry%gridname)
469  write(sunit,*)" fieldname_code is ", trim(data_entry%fieldname_code)

683  character(len=3), intent(in) :: gridname !< model grid ID
684  character(len=*), intent(in) :: fieldname_code !< field name as used in the model
685  logical, optional, intent(out) :: override !< true if the field has been overriden succesfully
686  type(time_type), intent(in) :: time !< (target) model time
687  integer, optional, intent(in) :: data_index

689  integer, optional, intent(in) :: is_in, ie_in, js_in, je_in
690  logical, dimension(:,:,:), allocatable :: mask_out

693  character(len=512) :: filename2 !< file containing source data
694  character(len=128) :: fieldname !< fieldname used in the data file
695  integer :: i,j

698  integer :: id_time !< index for time interp in override array
699  integer :: axis_sizes(4)
700  character(len=32) :: axis_names(4)
701  real, dimension(:,:), pointer :: lon_local =>null() !< of output (target) grid cells
702  real, dimension(:,:), pointer :: lat_local =>null() !< of output (target) grid cells
703  real, dimension(:), allocatable :: lon_tmp, lat_tmp
704 
705  logical :: data_file_is_2D = .false. !< data in netCDF file is 2D
706  logical :: ongrid, use_comp_domain
707  type(domain2d) :: domain
708  integer :: curr_position !< position of the field currently processed in override_array
709  real :: factor

711  integer :: nxd, nyd, nxc, nyc, nwindows
712  integer :: nwindows_x, ipos, jpos, window_size(2)

717  real :: lat_min, lat_max
718  integer :: is_src, ie_src, js_src, je_src
719  logical :: exists
720  type(fmsnetcdffile_t) :: fileobj
721  integer :: startingi !< Starting x index for the compute domain relative to the input buffer
722  integer :: endingi !< Ending x index for the compute domain relative to the input buffer
723  integer :: startingj !< Starting y index for the compute domain relative to the input buffer
724  integer :: endingj !< Ending y index for the compute domain relative to the input buffer
725  integer :: nhalox !< Number of halos in the x direction
726  integer :: nhaloy !< Number of halos in the y direction
727 
728  use_comp_domain = .false.
729  if(.not.module_is_initialized) &
730  call mpp_error(fatal,'Error: need to call data_override_init first')
731 
732 !1 Look for the data file in data_table
733  if(PRESENT(override)) override = .false.
734  if (present(data_index)) then
735  index1 = data_index
736  else
737  index1 = -1
738  do i = 1, table_size
739  if( trim(gridname) /= trim(data_table(i)%gridname)) cycle
740  if( trim(fieldname_code) /= trim(data_table(i)%fieldname_code)) cycle
741  index1 = i ! field found
742  exit
743  enddo
744  if(index1 .eq. -1) then
745  if(debug_data_override) &
746  call mpp_error(warning,'this field is NOT found in data_table: '//trim(fieldname_code))
747  return ! NO override was performed
748  endif
749  endif
750 
751  fieldname = data_table(index1)%fieldname_file ! fieldname in netCDF data file
752  factor = data_table(index1)%factor
753 
754  if(fieldname == "") then
755  data = factor
756  if(PRESENT(override)) override = .true.
757  return
758  else
759  filename = data_table(index1)%file_name
760  if (filename == "") call mpp_error(fatal,'data_override: filename not given in data_table')
761  endif
762 
763  ongrid = (data_table(index1)%interpol_method == 'none')
764 
765 !3 Check if fieldname has been previously processed
766 !$OMP CRITICAL
767  curr_position = -1
768  if(num_fields > 0 ) then
769  do i = 1, num_fields
770  if(trim(override_array(i)%gridname) /= trim(gridname)) cycle
771  if(trim(override_array(i)%fieldname) /= trim(fieldname_code)) cycle
772  curr_position = i
773  exit
774  enddo
775  endif
776 
777  if(curr_position < 0) then ! the field has not been processed previously
778  num_fields = num_fields + 1
779  curr_position = num_fields
780 
781 ! Get working domain from model's gridname
782  call get_domain(gridname,domain,comp_domain)
783  call mpp_get_data_domain(domain, xsize=nxd, ysize=nyd)
784  nxc = comp_domain(2)-comp_domain(1) + 1
785  nyc = comp_domain(4)-comp_domain(3) + 1
786 
787 ! record fieldname, gridname in override_array
788  override_array(curr_position)%fieldname = fieldname_code
789  override_array(curr_position)%gridname = gridname
790  override_array(curr_position)%comp_domain = comp_domain
791 ! get number of threads
792  override_array(curr_position)%numthreads = 1
793 #if defined(_OPENMP)
794  override_array(curr_position)%numthreads = omp_get_num_threads()
795 #endif
796 !--- data_override may be called from physics windows. The following are possible situations
797 !--- 1. size(data,1) == nxd and size(data,2) == nyd ( on data domain and there is only one window).
798 !--- 2. nxc is divisible by size(data,1), nyc is divisible by size(data,2),
799 !--- nwindow = (nxc/size(data(1))*(nyc/size(data,2)), also we require nwindows is divisible by nthreads.
800 !--- The another restrition is that size(data,1) == ie_in - is_in + 1,
801 !--- size(data,2) == je_in - js_in + 1
802  nwindows = 1
803  if( nxd == size(data,1) .AND. nyd == size(data,2) ) then !
804  use_comp_domain = .false.
805  else if ( mod(nxc, size(data,1)) ==0 .AND. mod(nyc, size(data,2)) ==0 ) then
806  use_comp_domain = .true.
807  nwindows = (nxc/size(data,1))*(nyc/size(data,2))
808  else
809  call mpp_error(fatal, &
810  & "data_override: data is not on data domain and compute domain is not divisible by size(data)")
811  endif
812  override_array(curr_position)%window_size(1) = size(data,1)
813  override_array(curr_position)%window_size(2) = size(data,2)
814 
815  window_size = override_array(curr_position)%window_size
816  override_array(curr_position)%numwindows = nwindows
817  if( mod(nwindows, override_array(curr_position)%numthreads) .NE. 0 ) then
818  call mpp_error(fatal, "data_override: nwindow is not divisible by nthreads")
819  endif
820  allocate(override_array(curr_position)%need_compute(nwindows))
821  override_array(curr_position)%need_compute = .true.
822 
823 !4 get index for time interp
824  if(ongrid) then
825  if( data_table(index1)%region_type .NE. no_region ) then
826  call mpp_error(fatal,.NE.'data_override: ongrid must be false when region_type NO_REGION')
827  endif
828 
829 ! Allow on-grid data_overrides on cubed sphere grid
830  inquire(file=trim(filename),exist=exists)
831  if (.not. exists) then
832  call get_mosaic_tile_file(filename,filename2,.false.,domain)
833  filename = filename2
834  endif

837  id_time = init_external_field(filename,fieldname,domain=domain,verbose=.false., &
838  use_comp_domain=use_comp_domain, nwindows=nwindows, ongrid=ongrid)
839  dims = get_external_field_size(id_time)
840  override_array(curr_position)%dims = dims
841  if(id_time<0) call mpp_error(fatal,'data_override:field not found in init_external_field 1')
842  override_array(curr_position)%t_index = id_time
843  else !ongrid=false
844  id_time = init_external_field(filename,fieldname,domain=domain, axis_names=axis_names,&
845  axis_sizes=axis_sizes, verbose=.false.,override=.true.,use_comp_domain=use_comp_domain, &
846  nwindows = nwindows)
847  dims = get_external_field_size(id_time)
848  override_array(curr_position)%dims = dims
849  if(id_time<0) call mpp_error(fatal,'data_override:field not found in init_external_field 2')
850  override_array(curr_position)%t_index = id_time
851 
852  ! get lon and lat of the input (source) grid, assuming that axis%data contains
853  ! lat and lon of the input grid (in degrees)
854 
855  allocate(override_array(curr_position)%horz_interp(nwindows))
856  allocate(override_array(curr_position)%lon_in(axis_sizes(1)+1))
857  allocate(override_array(curr_position)%lat_in(axis_sizes(2)+1))

859  call axis_edges(fileobj, axis_names(1), override_array(curr_position)%lon_in, &
860  reproduce_null_char_bug_flag=reproduce_null_char_bug)
861  call axis_edges(fileobj, axis_names(2), override_array(curr_position)%lat_in, &
862  reproduce_null_char_bug_flag=reproduce_null_char_bug)
863  else
864  call mpp_error(fatal,'data_override: file '//trim(filename)//' is not opened in time_interp_external')
865  end if
866 ! convert lon_in and lat_in from deg to radian
867  override_array(curr_position)%lon_in = override_array(curr_position)%lon_in * deg_to_radian
868  override_array(curr_position)%lat_in = override_array(curr_position)%lat_in * deg_to_radian
869 
870  !--- find the region of the source grid that cover the local model grid.
871  !--- currently we only find the index range for j-direction because
872  !--- of the cyclic condition in i-direction. The purpose of this is to
873  !--- decrease the memory usage and increase the IO performance.
874  select case(gridname)
875  case('OCN')
876  lon_local => lon_local_ocn; lat_local => lat_local_ocn
877  case('ICE')
878  lon_local => lon_local_ice; lat_local => lat_local_ice
879  case('ATM')
880  lon_local => lon_local_atm; lat_local => lat_local_atm
881  case('LND')
882  lon_local => lon_local_lnd; lat_local => lat_local_lnd
883  case default
884  call mpp_error(fatal,'error: gridname not recognized in data_override')
885  end select
886 
887  lat_min = minval(lat_local)
888  lat_max = maxval(lat_local)
889  is_src = 1
890  ie_src = axis_sizes(1)
891  js_src = 1
892  je_src = axis_sizes(2)
893  do j = 1, axis_sizes(2)+1
894  if( override_array(curr_position)%lat_in(j) > lat_min ) exit
895  js_src = j
896  enddo
897  do j = 1, axis_sizes(2)+1
898  je_src = j
899  if( override_array(curr_position)%lat_in(j) > lat_max ) exit
900  enddo
901 
902  !--- bicubic interpolation need one extra point in each direction. Also add
903  !--- one more point for because lat_in is in the corner but the interpolation
904  !--- use center points.
905  select case (data_table(index1)%interpol_method)
906  case ('bilinear')
907  js_src = max(1, js_src-1)
908  je_src = min(axis_sizes(2), je_src+1)
909  case ('bicubic')
910  js_src = max(1, js_src-2)
911  je_src = min(axis_sizes(2), je_src+2)
912  end select
913  override_array(curr_position)%is_src = is_src
914  override_array(curr_position)%ie_src = ie_src
915  override_array(curr_position)%js_src = js_src
916  override_array(curr_position)%je_src = je_src
917  call reset_src_data_region(id_time, is_src, ie_src, js_src, je_src)
918

923  call read_data(fileobj, axis_names(2), lat_tmp)

925  ! lat_start, lat_end are inside lat_in
926  if(data_table(index1)%lon_start < lon_tmp(1) .OR. data_table(index1)%lon_start .GT. lon_tmp(axis_sizes(1)))&
927  call mpp_error(fatal, "data_override: lon_start is outside lon_T")
928  if( data_table(index1)%lon_end < lon_tmp(1) .OR. data_table(index1)%lon_end .GT. lon_tmp(axis_sizes(1))) &
929  call mpp_error(fatal, "data_override: lon_end is outside lon_T")
