20 program parser_demo
21 !> @brief This programs demostrates how to use the parser
22 
23 #ifdef use_yaml
24 use fms_mod, only: fms_init, fms_end
25 use yaml_parser_mod
26 use platform_mod
27 
28 implicit none
29 
30 integer :: diag_yaml_id !< Id for the diag_table yaml
31 integer :: nfiles !< Number of files in the diag_table yaml
32 integer, allocatable :: file_ids(:) !< Ids of the files in the diag_table yaml
33 integer :: nvariables !< Number of variables in the diag_table yaml
34 integer, allocatable :: var_ids(:) !< Ids of the variables in the diag_table yaml
35 integer :: i, j !< For do loops
36 character(len=255) :: string_buffer !< Buffer to read strings to
37 integer :: int_buffer !< Buffer to read integers to
38 real(kind=r8_kind) :: r8_buffer !< Buffer to read r8 to
39 
40 call fms_init
41 call fms_end
42 
43 diag_yaml_id = open_and_parse_file("diag_table.yaml")
44 print *, ""
45 
46 call get_value_from_key(diag_yaml_id, 0, "title", string_buffer)
47 print *, "title:", trim(string_buffer)
48 
49 call get_value_from_key(diag_yaml_id, 0, "baseDate", string_buffer)
50 print *, "baseDate:", trim(string_buffer)
51 
52 nfiles = get_num_blocks(diag_yaml_id, "diag_files")
53 allocate(file_ids(nfiles))
54 call get_block_ids(diag_yaml_id, "diag_files", file_ids)
55 print *, ""
56 
57 do i = 1, nfiles
58  print *, "File number:", i
59 
60  call get_value_from_key(diag_yaml_id, file_ids(i), "fileName", string_buffer)
61  print *, "fileName:", trim(string_buffer)
62 
63  call get_value_from_key(diag_yaml_id, file_ids(i), "freq", int_buffer)
64  print *, "freq:", int_buffer
65 
66  call get_value_from_key(diag_yaml_id, file_ids(i), "frequnit", string_buffer)
67  print *, "frequnit:", trim(string_buffer)
68 
69  call get_value_from_key(diag_yaml_id, file_ids(i), "timeunit", string_buffer)
70  print *, "timeunit:", trim(string_buffer)
71 
72  call get_value_from_key(diag_yaml_id, file_ids(i), "unlimdim", string_buffer)
73  print *, "unlimdim:", trim(string_buffer)
74 
75  !< The number of variables that are part of the current file
76  nvariables = get_num_blocks(diag_yaml_id, "varlist", parent_block_id=file_ids(i))
77  allocate(var_ids(nvariables))
78  call get_block_ids(diag_yaml_id, "varlist", var_ids, parent_block_id=file_ids(i))
79 
80  do j = 1, nvariables
81  print *, " Variable number:", j
82 
83  call get_value_from_key(diag_yaml_id, var_ids(j), "varName", string_buffer)
84  print *, " varName:", trim(string_buffer)
85 
86  string_buffer = ""
87  call get_value_from_key(diag_yaml_id, var_ids(j), "reduction", string_buffer)
88  print *, " reduction:", trim(string_buffer)
89 
90  string_buffer = ""
91  call get_value_from_key(diag_yaml_id, var_ids(j), "module", string_buffer)
92  print *, " module:", trim(string_buffer)
93 
94  r8_buffer = 0.
95  call get_value_from_key(diag_yaml_id, var_ids(j), "fill_value", r8_buffer, is_optional=.true.)
96  print *, " fill_value:", r8_buffer
97 
98  string_buffer = ""
99  call get_value_from_key(diag_yaml_id, var_ids(j), "outName", string_buffer, is_optional=.true.)
100  print *, " outName:", trim(string_buffer)
101 
102  string_buffer = ""
103  call get_value_from_key(diag_yaml_id, var_ids(j), "kind", string_buffer, is_optional=.true.)
104  print *, " kind:", trim(string_buffer)
105 
106  int_buffer = 0.
107  call get_value_from_key(diag_yaml_id, var_ids(j), "mullions", int_buffer, is_optional=.true.)
108  print *, " mullions:", int_buffer
109 
110  print *, ""
111  end do
112  deallocate(var_ids)
113  print *, ""
114 enddo
115 deallocate(file_ids)
116 
117 #endif
118 
119 end program parser_demo
