19 !> @defgroup tracer_manager_mod tracer_manager_mod
20 !> @ingroup tracer_manager
21 !> @brief Code to manage the simple addition of tracers to the FMS code.
22 !! This code keeps track of the numbers and names of tracers included
23 !! in a tracer table.
24 !!
25 !> @author William Cooke
26 !!
27 !> This code is a grouping of calls which will allow the simple
28 !! introduction of tracers into the FMS framework. It is designed to
29 !! allow users of a variety of component models interact easily with
30 !! the dynamical core of the model.
31 !!
32 !! In calling the tracer manager routines the user must provide a
33 !! parameter identifying the model that the user is working with. This
34 !! parameter is defined within field_manager as MODEL_X
35 !! where X is one of [ATMOS, OCEAN, LAND, ICE].
36 !!
37 !! In many of these calls the argument list includes model and tracer_index. These
38 !! are the parameter corresponding to the component model and the tracer_index N is
39 !! the Nth tracer within the component model. Therefore a call with MODEL_ATMOS and 5
40 !! is different from a call with MODEL_OCEAN and 5.

101 !> @brief Function which returns the number assigned to the tracer name.
102 !!
103 !> This is a function which returns the index, as implied within the component model.
104 !! There are two overloaded interfaces: one of type integer, one logical.
105 !!
106 !! @param model A integer parameter to identify which model is being used
107 !! @param name The name of the tracer (as assigned in the field table).
108 !! @param indices An array indices. When present, the returned index will limit the search for the
109 !! tracer to thos tracers whose indices are among those in array 'indices'. This would be useful
110 !! when it is desired to limit the search to a subset of the tracers. Such a subset might be the
111 !! diagnostic or prognostic tracers.(note that @ref get_tracer_indices returns these subsets)
112 !! @param verbose Optional flag for debugging
113 !! @param[out] index Holds the returned index if given
114 !! @returns The integer function returns the index of the tracer named "name". If no tracer by that
115 !! name exists then the returned value is NO_TRACER. The logical function returns false if no
116 !! tracer by the given name exists and true otherwise, and stores the returned value in index.
117 !!
118 !! <br>Example usage:
119 !! @code{.F90}
120 !! integer:
121 !! index = get_tracer_index(model, name, indices, verbose)
122 !! logical:
123 !! if ( get_tracer_index(model, name, index, indices, verbose) ) then
124 !! @endcode

203 !> @brief Read in tracer table and store tracer information associated with "model" in "tracers"
204 !! array.
205 subroutine get_tracer_meta_data(model, num_tracers,num_prog,num_diag)
206 
207 integer, intent(in) :: model !< model being used
208 integer, intent(out) :: num_tracers, num_prog, num_diag
209 character(len=256) :: warnmesg
210 
211 character(len=32) :: name_type, type, name
212 integer :: n, m, mod, num_tracer_methods, nfields, swop
213 integer :: j, log_unit, num_methods
214 logical :: flag_type
215 type(method_type), dimension(MAX_TRACER_METHOD) :: methods
216 integer :: instances, siz_inst,i
217 character(len = 32) :: digit,suffnam
218 
219 character(len=128) :: list_name , control
220 integer :: index_list_name
221 logical :: fm_success
222 
223 ! <ERROR MSG="invalid model type" STATUS="FATAL">
224 ! The index for the model type is invalid.
225 ! </ERROR>
226 if (model .ne. model_atmos .and. model .ne. model_land .and. &
227  model .ne. model_ocean .and. model .ne. model_ice .and. &
228  model .ne. model_coupler) call mpp_error(fatal,'tracer_manager_init : invalid model type')
229 
230 ! One should only call get_tracer_meta_data once for each model type
231 ! Therefore need to set up an array to stop the subroutine being
232 ! unnecssarily called multiple times.
233 
234 if ( model_registered(model) ) then
235 ! This routine has already been called for the component model.
236 ! Fill in the values from the previous registration and return.
237  num_tracers = total_tracers(model)
238  num_prog = prog_tracers(model)
239  num_diag = diag_tracers(model)
240  return
241 endif
242 
243 ! Initialize the number of tracers to zero.
244 num_tracers = 0; num_prog = 0; num_diag = 0
245 
246 call field_manager_init(nfields=nfields)
247 
248 ! <ERROR MSG="No tracers are available to be registered." STATUS="NOTE">
249 ! No tracers are available to be registered. This means that the field
250 ! table does not exist or is empty.
251 ! </ERROR>
252 if (nfields == 0 ) then
253 if (mpp_pe() == mpp_root_pe()) &
254  call mpp_error(note,'tracer_manager_init : No tracers are available to be registered.')
255  return
256 endif
257 
258 ! search through field entries for model tracers
259 total_tracers(model) = 0
260 
261 do n=1,nfields
262  call get_field_info(n,type,name,mod,num_methods)
263 
264  if (mod == model .and. type == 'tracer') then
265  num_tracer_fields = num_tracer_fields + 1
266  total_tracers(model) = total_tracers(model) + 1
267 ! <ERROR MSG="MAX_TRACER_FIELDS exceeded" STATUS="FATAL">
268 ! The maximum number of tracer fields has been exceeded.
269 ! </ERROR>
270  if(num_tracer_fields > max_tracer_fields) call mpp_error(fatal, &
271  & 'tracer_manager_init: MAX_TRACER_FIELDS exceeded')
272  tracer_array(model,total_tracers(model)) = num_tracer_fields
273  tracers(num_tracer_fields)%model = model
274  tracers(num_tracer_fields)%tracer_name = name
275  tracers(num_tracer_fields)%tracer_units = 'none'
276  tracers(num_tracer_fields)%tracer_longname = tracers(num_tracer_fields)%tracer_name
277  tracers(num_tracer_fields)%instances_set = .false.
278 ! By default, tracers need mass and positive definite adjustments.
279 ! We hardwire exceptions for compatibility with existing field_tables
280 ! This should ideally be cleaned up.
281  tracers(num_tracer_fields)%needs_mass_adjust = .true.
282  tracers(num_tracer_fields)%needs_positive_adjust = .true.
283  if (name == 'cld_amt') then
284  tracers(num_tracer_fields)%needs_mass_adjust = .false.
285  endif
286  if (name == 'cld_amt' .or. name == 'liq_wat' .or. name == 'ice_wat') then
287  tracers(num_tracer_fields)%needs_positive_adjust = .false.
288  endif
289 
290  num_tracer_methods = 0
291  methods = default_method ! initialize methods array
292  call get_field_methods(n,methods)
293  do j=1,num_methods
294  select case (methods(j)%method_type)
295  case ('units')
296  tracers(num_tracer_fields)%tracer_units = methods(j)%method_name
297  case ('longname')
298  tracers(num_tracer_fields)%tracer_longname = methods(j)%method_name
299  case ('instances')
300 ! tracers(num_tracer_fields)%instances = methods(j)%method_name
301  siz_inst = parse(methods(j)%method_name,"",instances)
302  tracers(num_tracer_fields)%instances = instances
303  tracers(num_tracer_fields)%instances_set = .true.
304  case ('adjust_mass')
305  if (methods(j)%method_name == "false") then
306  tracers(num_tracer_fields)%needs_mass_adjust = .false.
307  endif
308  case ('adjust_positive_def')
309  if (methods(j)%method_name == "false") then
310  tracers(num_tracer_fields)%needs_positive_adjust = .false.
311  endif
312  case default
313  num_tracer_methods = num_tracer_methods+1
314 ! tracers(num_tracer_fields)%methods(num_tracer_methods) = methods(j)
315  end select
316  enddo
317  tracers(num_tracer_fields)%num_methods = num_tracer_methods
318  tracers(num_tracer_fields)%needs_init = .false.
319  flag_type = query_method('tracer_type',model,total_tracers(model),name_type)
320  if (flag_type .and. name_type == 'diagnostic') then
321  tracers(num_tracer_fields)%is_prognostic = .false.
322  else
323  tracers(num_tracer_fields)%is_prognostic = .true.
324  endif
325  if (tracers(num_tracer_fields)%is_prognostic) then
326  num_prog = num_prog+1
327  else
328  num_diag = num_diag+1
329  endif
330  endif
331 enddo
332 
333 ! Now cycle through the tracers and add additional instances of the tracers.
334 
335 do n = 1, num_tracer_fields !{
336 ! call get_field_info(n,type,name,mod,num_methods)
337 
338  if ( model == tracers(n)%model .and. tracers(n)%instances_set ) then !{ We have multiple instances of this tracer
339 
340  if ( num_tracer_fields + tracers(n)%instances > max_tracer_fields ) then
341  write(warnmesg, '("tracer_manager_init: Number of tracers will exceed MAX_TRACER_FIELDS with &
342  &multiple (",I3," instances) setup of tracer ",A)') tracers(n)%instances,tracers(n)%tracer_name
343  call mpp_error(fatal, warnmesg)
344  endif
345 
346  do i = 2, tracers(n)%instances !{
347  num_tracer_fields = num_tracer_fields + 1
348  total_tracers(model) = total_tracers(model) + 1
349  tracer_array(model,total_tracers(model)) = num_tracer_fields
350  ! Copy the original tracer type to the multiple instances.
351  tracers(num_tracer_fields) = tracers(n)
352  if ( query_method('instances', model,model_tracer_number(model,n),name, control)) then !{
353 
354  if (i .lt. 10) then !{
355  write (suffnam,'(''suffix'',i1)') i
356  siz_inst = parse(control, suffnam,digit)
357  if (siz_inst == 0 ) then
358  write (digit,'(''_'',i1)') i
359  else
360  digit = "_"//trim(digit)
361  endif
362  elseif (i .lt. 100) then !}{
363  write (suffnam,'(''suffix'',i2)') i
364  siz_inst = parse(control, suffnam,digit)
365  if (siz_inst == 0 ) then
366  write (digit,'(''_'',i2)') i
367  else
368  digit = "_"//trim(digit)
369  endif
370  else !}{
371  call mpp_error(fatal, 'tracer_manager_init: MULTIPLE_TRACER_SET_UP exceeds 100 for '// &
372  & tracers(n)%tracer_name )
373  endif !}
374 
375  select case(model)
376  case (model_coupler)
377  list_name = "/coupler_mod/tracer/"//trim(tracers(num_tracer_fields)%tracer_name)
378  case (model_atmos)
379  list_name = "/atmos_mod/tracer/"//trim(tracers(num_tracer_fields)%tracer_name)
380  case (model_ocean)
381  list_name = "/ocean_mod/tracer/"//trim(tracers(num_tracer_fields)%tracer_name)
382  case (model_ice )
383  list_name = "/ice_mod/tracer/"//trim(tracers(num_tracer_fields)%tracer_name)
384  case (model_land )
385  list_name = "/land_mod/tracer/"//trim(tracers(num_tracer_fields)%tracer_name)
386  case default
387  list_name = "/default/tracer/"//trim(tracers(num_tracer_fields)%tracer_name)
388  end select
389 
390  if (mpp_pe() == mpp_root_pe() ) write (*,*) "Creating list name = ",trim(list_name)//trim(digit)
391 
392  index_list_name = fm_copy_list(trim(list_name),digit, create = .true.)
393  tracers(num_tracer_fields)%tracer_name = trim(tracers(num_tracer_fields)%tracer_name)//trim(digit)
394  endif !}
395 
396  if (tracers(num_tracer_fields)%is_prognostic) then !{
397  num_prog = num_prog+1
398  else !}{
399  num_diag = num_diag+1
400  endif !}
401  enddo !}
402  ! Multiple instances of tracers were found so need to rename the original tracer.
403  digit = "_1"
404  siz_inst = parse(control, "suffix1",digit)
405  if (siz_inst > 0 ) then !{
406  digit = "_"//trim(digit)
407  endif !}
408  fm_success = fm_modify_name(trim(list_name), trim(tracers(n)%tracer_name)//trim(digit))
409  tracers(n)%tracer_name = trim(tracers(n)%tracer_name)//trim(digit)
410  endif !}
411 enddo !}
412 
413 ! Find any field entries with the instances keyword.
414 do n=1,nfields
415  call get_field_info(n,type,name,mod,num_methods)
416 
417  if ( mod == model .and. type == 'instances' ) then
418  call get_field_methods(n,methods)
419  do j=1,num_methods
420 
421  if (.not.get_tracer_index(mod,methods(j)%method_type,m)) then
422  call mpp_error(fatal,'tracer_manager_init: The instances keyword was found for undefined tracer '&
423  //trim(methods(j)%method_type))
424  else
425  if ( tracers(m)%instances_set ) &
426  call mpp_error(fatal,'tracer_manager_init: The instances keyword was found for '&
427  //trim(methods(j)%method_type)//' but has previously been defined in the tracer entry')
428  siz_inst = parse(methods(j)%method_name,"",instances)
429  tracers(m)%instances = instances
430  call mpp_error(note,'tracer_manager_init: '//trim(instantiations(j)%name)// &
431  ' will have '//trim(methods(j)%method_name)//' instances')
432  endif
433  if ( num_tracer_fields + instances > max_tracer_fields ) then
434  write(warnmesg, '("tracer_manager_init: Number of tracers will exceed MAX_TRACER_FIELDS with &
435  &multiple (",I3," instances) setup of tracer ",A)') tracers(m)%instances,tracers(m)%tracer_name
436  call mpp_error(fatal, warnmesg)
437  endif
438 ! We have found a valid tracer that has more than one instantiation.
439 ! We need to modify that tracer name to tracer_1 and add extra tracers for the extra instantiations.
440  if (instances .eq. 1) then
441  siz_inst = parse(methods(j)%method_control, 'suffix1',digit)
442  if (siz_inst == 0 ) then
443  digit = '_1'
444  else
445  digit = "_"//trim(digit)
446  endif
447  endif
448  do i = 2, instances
449  num_tracer_fields = num_tracer_fields + 1
450  total_tracers(model) = total_tracers(model) + 1
451  tracer_array(model,total_tracers(model)) = num_tracer_fields
452  tracers(num_tracer_fields) = tracers(m)
453 
454  if (i .lt. 10) then !{
455  write (suffnam,'(''suffix'',i1)') i
456  siz_inst = parse(methods(j)%method_control, suffnam,digit)
457  if (siz_inst == 0 ) then
458  write (digit,'(''_'',i1)') i
459  else
460  digit = "_"//trim(digit)
461  endif
462  elseif (i .lt. 100) then !}{
463  write (suffnam,'(''suffix'',i2)') i
464  siz_inst = parse(methods(j)%method_control, suffnam,digit)
465  if (siz_inst == 0 ) then
466  write (digit,'(''_'',i2)') i
467  else
468  digit = "_"//trim(digit)
469  endif
470  else !}{
471  call mpp_error(fatal, 'tracer_manager_init: MULTIPLE_TRACER_SET_UP exceeds 100 for '&
472  //tracers(num_tracer_fields)%tracer_name )
473  endif !}
474 
475  select case(model)
476  case (model_coupler)
477  list_name = "/coupler_mod/tracer/"//trim(tracers(num_tracer_fields)%tracer_name)
478  case (model_atmos)
479  list_name = "/atmos_mod/tracer/"//trim(tracers(num_tracer_fields)%tracer_name)
480  case (model_ocean)
481  list_name = "/ocean_mod/tracer/"//trim(tracers(num_tracer_fields)%tracer_name)
482  case (model_ice )
483  list_name = "/ice_mod/tracer/"//trim(tracers(num_tracer_fields)%tracer_name)
484  case (model_land )
485  list_name = "/land_mod/tracer/"//trim(tracers(num_tracer_fields)%tracer_name)
486  case default
487  list_name = "/default/tracer/"//trim(tracers(num_tracer_fields)%tracer_name)
488  end select
489 
490  if (mpp_pe() == mpp_root_pe() ) write (*,*) "Creating list name = ",trim(list_name)
491 
492  index_list_name = fm_copy_list(trim(list_name),digit, create = .true.)
493 
494  tracers(num_tracer_fields)%tracer_name = trim(tracers(num_tracer_fields)%tracer_name)//digit
495  if (tracers(num_tracer_fields)%is_prognostic) then
496  num_prog = num_prog+1
497  else
498  num_diag = num_diag+1
499  endif
500  enddo
501 !Now rename the original tracer to tracer_1 (or if suffix1 present to tracer_'value_of_suffix1')
502  siz_inst = parse(methods(j)%method_control, 'suffix1',digit)
503  if (siz_inst == 0 ) then
504  digit = '_1'
505  else
506  digit = "_"//trim(digit)
507  endif
508  fm_success = fm_modify_name(trim(list_name), trim(tracers(m)%tracer_name)//trim(digit))
509  tracers(m)%tracer_name = trim(tracers(m)%tracer_name)//trim(digit)
510  enddo
511  endif
512 enddo
513 
514 num_tracers = num_prog + num_diag
515 ! Make the number of tracers available publicly.
516 total_tracers(model) = num_tracers
517 prog_tracers(model) = num_prog
518 diag_tracers(model) = num_diag
519 model_registered(model) = .true.
520 
521 ! Now sort through the tracer fields and sort them so that the
522 ! prognostic tracers are first.
523 
524 do n=1, num_tracers
525  if (.not.check_if_prognostic(model,n) .and. n.le.num_prog) then
526  ! This is a diagnostic tracer so find a prognostic tracer to swop with
527  do m = n, num_tracers
528  if (check_if_prognostic(model,m) .and. .not.check_if_prognostic(model,n)) then
529  swop = tracer_array(model,n)
530  tracer_array(model,n) = tracer_array(model,m)
531  tracer_array(model,m) = swop
532  cycle
533  endif
534  enddo
535  endif
536 enddo
537 
538 do n=1, num_tracer_fields
539  call print_tracer_info(model,n)
540 enddo
541 
542 log_unit = stdlog()
543 if ( mpp_pe() == mpp_root_pe() ) then
544  write (log_unit,15) trim(model_names(model)),total_tracers(model)
545 endif
546 
547 15 format ('Number of tracers in field table for ',a,' model = ',i4)
548 
549 end subroutine get_tracer_meta_data

551 function model_tracer_number(model,n)
552 integer, intent(in) :: model, n
553 integer model_tracer_number
554 
555 integer :: i
556

1011 !> @brief Subroutine to set the tracer field to the wanted profile.
1012 !!
1013 !> If the profile type is 'fixed' then the tracer field values are set
1014 !! equal to the surface value.
1015 !! If the profile type is 'profile' then the top/bottom of model and
1016 !! surface values are read and an exponential profile is calculated,
1017 !! with the profile being dependent on the number of levels in the
1018 !! component model. This should be called from the part of the dynamical
1019 !! core where tracer restarts are called in the event that a tracer
1020 !! restart file does not exist.
1021 !!
1022 !! This can be activated by adding a method to the field_table
1023 !! e.g.
1024 !! @verbose "profile_type","fixed","surface_value = 1e-12" @endverbose
1025 !! would return values of surf_value = 1e-12 and a multiplier of 1.0
1026 !! One can use these to initialize the entire field with a value of 1e-12.
1027 !!
1028 !! "profile_type","profile","surface_value = 1e-12, top_value = 1e-15"
1029 !! In a 15 layer model this would return values of surf_value = 1e-12 and
1030 !! multiplier = 0.6309573 i.e 1e-15 = 1e-12*(0.6309573^15)
1031 !! In this case the model should be MODEL_ATMOS as you have a "top" value.
1032 !!
1033 !! If you wish to initialize the ocean model, one can use bottom_value instead
1034 !! of top_value.
1035 subroutine set_tracer_profile(model, n, tracer, err_msg)
1036 
1037 integer, intent(in) :: model !< Parameter representing component model in use
1038 integer, intent(in) :: n !< Tracer number
1039 class(*), intent(inout), dimension(:,:,:) :: tracer !< Initialized tracer array
1040 character(len=*), intent(out), optional :: err_msg
1041 
1042 real :: surf_value, multiplier
1043 integer :: numlevels, k, n1, flag
1044 real :: top_value, bottom_value
1045 character(len=80) :: scheme, control,profile_type
1046 character(len=128) :: err_msg_local
1047 character(len=11) :: chn
1048 
1049 if(.not.module_is_initialized) call tracer_manager_init
1050 
1051 if (n < 1 .or. n > total_tracers(model)) then
1052  write(chn, '(i11)') n
1053  err_msg_local = ' Invalid tracer index. Model name = '//trim(model_names(model))//', Index='//trim(chn)
1054  if(error_handler('set_tracer_profile', err_msg_local, err_msg)) return
1055 endif
1056 n1 = tracer_array(model,n)
1057 
1058 !default values
1059 profile_type = 'Fixed'
1060 surf_value = 0.0e+00
1061 top_value = surf_value
1062 bottom_value = surf_value
1063 multiplier = 1.0
1064 
1065 select type (tracer)
1066 type is (real(kind=r4_kind))
1067  tracer = surf_value
1068 type is (real(kind=r8_kind))
1069  tracer = surf_value
1070 class default
1071  call mpp_error(fatal,&
1072  &"set_tracer_profile : tracer is not one of the supported types of real(kind=4) or real(kind=8)")
1073 end select
1074 
1075 if ( query_method( 'profile_type',model,n,scheme,control)) then
1076 !Change the tracer_number to the tracer_manager version
1077 
1078  if(lowercase(trim(scheme(1:5))).eq.'fixed') then
1079  profile_type = 'Fixed'
1080  flag =parse(control,'surface_value',surf_value)
1081  multiplier = 1.0
1082  select type (tracer)
1083  type is (real(kind=r4_kind))
1084  tracer = surf_value
1085  type is (real(kind=r8_kind))
1086  tracer = surf_value
1087  class default
1088  call mpp_error(fatal,&
1089  &"set_tracer_profile : tracer is not one of the supported types of real(kind=4) or real(kind=8)")
1090  end select
1091  endif
1092 
1093  if(lowercase(trim(scheme(1:7))).eq.'profile') then
1094  profile_type = 'Profile'
1095  flag=parse(control,'surface_value',surf_value)
1096  if (surf_value .eq. 0.0) &
1097  call mpp_error(fatal,'set_tracer_profile : Cannot have a zero surface value for an exponential profile. Tracer '&
1098  //tracers(n1)%tracer_name//" "//control//" "//scheme)
1099  select case (tracers(n1)%model)
1100  case (model_atmos)
1101  flag=parse(control,'top_value',top_value)
1102  if(mpp_pe()==mpp_root_pe() .and. flag == 0) &
1103  call mpp_error(note,'set_tracer_profile : Parameter top_value needs to be defined for the tracer profile.')
