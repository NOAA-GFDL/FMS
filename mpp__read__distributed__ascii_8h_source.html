19 subroutine mpp_read_distributed_ascii_1d_ (unit,fmt,ssize,data,iostat)
20  integer, intent(in) :: unit
21  character(*), intent(in) :: fmt
22  integer, intent(in) :: ssize
23  mpp_type_, dimension(:), intent(inout) :: data
24  integer, intent(out) :: iostat
25 
26  integer, allocatable :: pelist(:)
27  logical :: is_ioroot=.false.
28 
29  if(.not.module_is_initialized) call mpp_error(fatal,'MPP_READ_DISTRIBUTED_ASCII_1D_: module not initialized')
30 
31  iostat = 0
32  call mpp_dist_io_pelist(ssize,pelist) ! ALLOCATE and create pelist if size of group > 1
33  if(.not. ALLOCATED(pelist)) &
34  call mpp_error(fatal,'MPP_READ_DISTRIBUTED_ASCII_1D_:: pelist allocation failed')
35  is_ioroot = mpp_is_dist_ioroot(ssize)
36  if(is_ioroot) then
37  if(trim(fmt)=='*')then
38  read(unit,*,iostat=iostat) data
39  else
40  read(unit,fmt=trim(fmt),iostat=iostat) data
41  endif
42  if(iostat /= 0) return ! Calling routine must handle error
43  endif
44 
45  call mpp_broadcast(data,size(data),pelist(1),pelist)
46  deallocate(pelist) ! Don't forget to deallocate pelist
47 end subroutine mpp_read_distributed_ascii_1d_
