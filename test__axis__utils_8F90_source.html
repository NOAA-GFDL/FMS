20 program test_axis_utils
21 
22 use fms_mod, only : fms_init, fms_end, check_nml_error

31 implicit none
32 
33 type data_type
34  real(kind=r8_kind) :: var(10) !< Axis data
35  real(kind=r8_kind) :: var_edges(2,10) !< The boundaries of the axis data
36  real(kind=r8_kind) :: answers(11) !< The expected result
37 end type data_type
38 
39 type(data_type) :: data_in !< Data used to create the netcdf file
40 integer, allocatable :: pes(:) !< List of pes
41 type(fmsnetcdffile_t) :: fileobj !< FMS2_io fileobj
42 
43 real(kind=r8_kind) :: answers(11) !< Results obtained from the axis_edges call
44 
45 call fms_init
46 
47 !< Get the current pelist
48 allocate(pes(mpp_npes()))
49 call mpp_get_current_pelist(pes)
50 
51 call set_data(data_in)
52 call create_input_files(data_in)
53 
54 !< Test calls to axis_edges
55 if ( .not. open_file(fileobj, "test_axis_utils.nc", "read", pelist=pes)) then
56  call mpp_error(fatal, "Error opening test_axis_utils.nc to read")
57 endif
58 
59 !< Case 1: Here the variable "axis" in the file does not have the attribute "bounds" or "edges", so
60 !! it calculates them from the data in "axis"
61 answers = 0.0
62 call axis_edges(fileobj, "axis", answers)
63 call compare_answers(answers, data_in%answers, "1")
64 
65 !< Case 2: Here the variable "axis_with_bounds" in the file has the attribute
66 !! "bounds", so the data is read from the variable "bounds"
67 answers = 0.0
68 call axis_edges(fileobj, "axis_with_bounds", answers)
69 call compare_answers(answers, data_in%answers, "2")
70 
71 !< Case 3: Here the variable "axis_with_edges" in the file has the attribute
72 !"edges", so the data is read from the variable "edges"
73 answers = 0.0
74 call axis_edges(fileobj, "axis_with_edges", answers)
75 call compare_answers(answers, data_in%answers, "3")
76 
77 !< Case 4: Here the flag "reproduce_null_char_bug_flag" is turned on, so the
78 !! edges are calculated from the data in axis because edges has a null character
79 !! in the end
80 answers = 0.0
81 call axis_edges(fileobj, "axis_with_edges", answers, reproduce_null_char_bug_flag=.true.)
82 call compare_answers(answers, data_in%answers, "4")
83 
84 call close_file(fileobj)
85 deallocate(pes)
86 
87 call fms_end
88 
89 contains

91 !> @brief Compares the values of two arrays
92 subroutine compare_answers(answers_in, answers_expected, test_case)
93 real(kind=r8_kind), intent(in) :: answers_in(:) !< Answer calculated
94 real(kind=r8_kind), intent(in) :: answers_expected(:) !< Answer expected
95 character(1), intent(in) :: test_case !< String indicating the case number
96 
97 integer :: i !< For do loop
98 
99 do i = 1, size(answers_expected,1)
100  if(answers_in(i) .ne. answers_expected(i)) then
101  print *, "i=", i, " Answer in: ", answers_in(i), " Answer expected ", answers_expected(i)
102  call mpp_error(fatal, "axis_edges case"//trim(test_case)//": Answers are not correct")
103  endif
104 enddo
105 end subroutine compare_answers
106 
107 !> @brief Sets the values of the data_type to be use to write the file, and to
108 !! compare answers
109 subroutine set_data(data_in)
110 type(data_type), intent(out) :: data_in !< data_type to set the expected values to
111 
112 integer :: i !< For do loop
113 
114 do i=1,10
115  data_in%var(i) = real(i, kind=r8_kind)-0.5_r8_kind
116 
117  data_in%var_edges(1,i) = real(i-1, kind=r8_kind)
118  data_in%var_edges(2,i) = real(i, kind=r8_kind)
119 
120  data_in%answers(i) = real(i-1, kind=r8_kind)
121 enddo
122 
123 data_in%answers(11) = real(10, kind=r8_kind)
124 
125 end subroutine
126 
127 !> @brief Creates a netcdf file to test the different test cases of
128 !!"axis_edges"
129 subroutine create_input_files(data_in)
130 type(data_type), intent(in) :: data_in !< data_type containing the values to be added to the file
131 
132 type(fmsnetcdffile_t) :: fileobj !< FMS2_io fileobj
133 
134 if (mpp_pe() .eq. mpp_root_pe()) then
135  if ( .not. open_file(fileobj, "test_axis_utils.nc", "overwrite")) then
136  call mpp_error(fatal, "Error opening test_axis_utils.nc to write")
137  endif
138 
139  call register_axis(fileobj, "dim1", 10)
140  call register_axis(fileobj, "dim2", 2)
141 
142  call register_field(fileobj, "axis", "double", dimensions=(/"dim1"/))
143 
144  call register_field(fileobj, "axis_with_bounds", "double", dimensions=(/"dim1"/))
145  call register_variable_attribute(fileobj, "axis_with_bounds", "bounds", "bounds", str_len=6)
146  call register_field(fileobj, "bounds", "double", dimensions=(/"dim2", "dim1"/))
147 
148  call register_field(fileobj, "axis_with_edges", "double", dimensions=(/"dim1"/))
149  call register_variable_attribute(fileobj, "axis_with_edges", "edges", "edges"//char(0), str_len=6)
150  call register_field(fileobj, "edges", "double", dimensions=(/"dim2", "dim1"/))
151 
152  call write_data(fileobj, "axis", data_in%var)
153  call write_data(fileobj, "axis_with_bounds", data_in%var)
154  call write_data(fileobj, "axis_with_edges", data_in%var)
155  call write_data(fileobj, "bounds", data_in%var_edges)
156  call write_data(fileobj, "edges", data_in%var_edges)
157 
158  call close_file(fileobj)
159 endif
160 
161 !< Wait for root_pe to catch up!
162 call mpp_sync()
163 
164 end subroutine create_input_files
165 
166 end program test_axis_utils
