20 !> @brief This programs tests calls to get_grid_version_1 used by data_override
21 
22 program test_get_grid_v1
23 
24 use netcdf, only: nf90_create, nf90_clobber, nf90_64bit_offset, nf90_double, &
25  nf90_def_dim, nf90_def_var, nf90_enddef, nf90_put_var, &
26  nf90_close
27 use mpp_mod, only: mpp_init, mpp_exit, mpp_root_pe, mpp_pe, mpp_error, fatal, mpp_sync
28 use mpp_domains_mod, only: mpp_define_domains, mpp_define_io_domain, mpp_get_compute_domain, &
29  domain2d
30 use fms2_io_mod, only: fms2_io_init
31 use platform_mod
32 
33 use get_grid_version_mod, only : get_grid_version_1, deg_to_radian
34 
35 implicit none
36 
37 type(domain2d) :: Domain !< 2D domain
38 integer :: is, ie, js, je !< Starting and ending compute
39  !! domain indices
40 integer :: nlon, nlat !< Number of lat, lon in grid
41 real :: min_lon, max_lon !< Maximum lat and lon
42 real, dimension(:,:), allocatable :: lon, lat !< Lat and lon
43 integer :: ncid, err !< Netcdf integers
44 integer :: dimid1, dimid2, dimid3, dimid4 !< Dimensions IDs
45 integer :: varid1, varid2, varid3, varid4, varid5 !< Variable IDs
46 real :: lat_in(1), lon_in(1) !< Lat and lon to be written to file
47 real, dimension(:,:,:), allocatable :: lat_vert_in, lon_vert_in !<Lat and lon vertices
48 
49 
50 lat_in = real(55.5, kind=r8_kind)
51 lon_in = real(44.5, kind=r8_kind)
52 
53 call mpp_init
54 call fms2_io_init
55 
56 !< Create a grid_spec file with information you need
57 if (mpp_pe() .eq. mpp_root_pe()) then
58  err = nf90_create("grid_spec.nc", ior(nf90_clobber, nf90_64bit_offset), ncid)
59  err = nf90_def_dim(ncid, 'xta', 1, dimid1)
60  err = nf90_def_var(ncid, 'xta', nf90_double, (/dimid1/), varid1)
61 
62  err = nf90_def_dim(ncid, 'yta', 1, dimid2)
63  err = nf90_def_var(ncid, 'yta', nf90_double, (/dimid2/), varid2)
64 
65  err = nf90_def_dim(ncid, 'gridlon_vert_t', 2, dimid3)
66  err = nf90_def_dim(ncid, 'gridlat_vert_t', 2, dimid4)
67 
68  err = nf90_def_var(ncid, 'geolon_vert_t', nf90_double, (/dimid3,dimid4/), varid3)
69  err = nf90_def_var(ncid, 'geolat_vert_t', nf90_double, (/dimid3,dimid4/), varid4)
70  err = nf90_def_var(ncid, 'geolon_t', nf90_double, (/dimid3,dimid4/), varid5)
71 
72  err = nf90_enddef(ncid)
73  err = nf90_put_var(ncid, varid1, lon_in(1))
74  err = nf90_put_var(ncid, varid2, lat_in(1))
75  err = nf90_close(ncid)
76 endif
77 
78 call mpp_sync()
79 
80 nlon = 1
81 nlat = 1
82 
83 !< Create a domain
84 call mpp_define_domains( (/1,nlon,1,nlat/), (/1, 1/), domain, name='Atm')
85 call mpp_define_io_domain(domain, (/1,1/))
86 call mpp_get_compute_domain(domain,is,ie,js,je)
87 
88 !< Call "get_grid_version_1" on a "atm" grid
89 allocate(lon(is:ie,js:je), lat(is:ie,js:je))
90 call get_grid_version_1("grid_spec.nc", "atm", domain, is, ie, js, je, lon, lat, &
91  min_lon, max_lon)
92 
93 !< Error checking:
94 if (lon(1,1) .ne. lon_in(1)*deg_to_radian) call mpp_error(fatal,'test_get_grid_v1: lon is not the expected result')
95 if (lat(1,1) .ne. lat_in(1)*deg_to_radian) call mpp_error(fatal,'test_get_grid_v1: lat is not the expected result')
96 
97 !< Try again with ocean
98 lat = 0.
99 lon = 0.
100 
101 call get_grid_version_1("grid_spec.nc", "ocn", domain, is, ie, js, je, lon, lat, &
102  min_lon, max_lon)
103 
104 !< Try again with ocean, "new_grid"
105 allocate(lat_vert_in(1,1,4), lon_vert_in(1,1,4))
106 lat_vert_in = real(55.5, kind=r8_kind)
107 lon_vert_in = real(65.5, kind=r8_kind)
108 
109 !< Create a new grid_spec file with "x_T" instead of "geolon_t"
110 if (mpp_pe() .eq. mpp_root_pe()) then
111  err = nf90_create("grid_spec.nc", ior(nf90_clobber, nf90_64bit_offset), ncid)
112  err = nf90_def_dim(ncid, 'nlat', 1, dimid1)
113  err = nf90_def_dim(ncid, 'nlon', 1, dimid2)
114  err = nf90_def_dim(ncid, 'nz', 4, dimid3)
115 
116  err = nf90_def_var(ncid, 'x_T', nf90_double, (/dimid1, dimid2, dimid3/), varid1)
117  err = nf90_def_var(ncid, 'x_vert_T', nf90_double, (/dimid1, dimid2, dimid3/), varid2)
118  err = nf90_def_var(ncid, 'y_vert_T', nf90_double, (/dimid1, dimid2, dimid3/), varid3)
119 
120  err = nf90_enddef(ncid)
121  err = nf90_put_var(ncid, varid2, lon_vert_in)
122  err = nf90_put_var(ncid, varid3, lat_vert_in)
123 
124  err = nf90_close(ncid)
125 endif
126 call mpp_sync()
127 
128 call get_grid_version_1("grid_spec.nc", "ocn", domain, is, ie, js, je, lon, lat, &
129  min_lon, max_lon)
130 
131 !< Error checking:
132 if (lon(1,1) .ne. sum(lon_vert_in)/4*deg_to_radian ) then
133  call mpp_error(fatal,'test_get_grid_v1: ocn, new grid, lon is not the expected result')
134 endif
135 
136 if (lat(1,1) .ne. sum(lat_vert_in)/4*deg_to_radian ) then
137  call mpp_error(fatal,'test_get_grid_v1: ocn, new grid, lat is not the expected result')
138 endif
139 
140 deallocate(lat_vert_in, lon_vert_in, lat, lon)
141 
142 call mpp_exit()
143 
144 end program test_get_grid_v1
