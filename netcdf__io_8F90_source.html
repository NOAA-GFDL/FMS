363  else
364  call error("unrecognized netcdf file format "//trim(netcdf_default_format)// &
365  '. The acceptable values are "64bit", "classic", "netcdf4". Check fms2_io_nml: netcdf_default_format')

386 !> @brief Switch to the correct netcdf mode.
387 !! @internal
388 subroutine set_netcdf_mode(ncid, mode)
389 
390  integer, intent(in) :: ncid !< Netcdf file id.
391  integer, intent(in) :: mode !< Netcdf file mode.
392 
393  integer :: err
394 
395  if (mode .eq. define_mode) then
396  err = nf90_redef(ncid)
397  if (err .eq. nf90_eindefine .or. err .eq. nf90_eperm) then
398  return
399  endif
400  elseif (mode .eq. data_mode) then
401  if (fms2_header_buffer_val == -1) call error("set_netcdf_mode: fms2_header_buffer_val not set, call fms2_io_init")
402  err = nf90_enddef(ncid, h_minfree=fms2_header_buffer_val)
403  if (err .eq. nf90_enotindefine .or. err .eq. nf90_eperm) then
404  return
405  endif
406  else
407  call error("mode must be either define_mode or data_mode.")
408  endif
409  call check_netcdf_code(err, "set_netcdf_mode")

1754 !> @brief Store the valid range for a variable.
1755 !! @return A ValidType_t object containing data about the valid
1756 !! range data for this variable can take.
1757 function get_valid(fileobj, variable_name) &
1758  result(valid)
1759 
1760  class(fmsnetcdffile_t), intent(in) :: fileobj !< File object.
1761  character(len=*), intent(in) :: variable_name !< Variable name.
1762  type(valid_t) :: valid
1763 
1764  integer :: varid
1765  real(kind=r8_kind) :: scale_factor
1766  real(kind=r8_kind) :: add_offset
1767  real(kind=r8_kind), dimension(2) :: buffer
1768  integer :: xtype
1769  character(len=200) :: append_error_msg !< Msg to be appended to FATAL error message
1770 
1771  append_error_msg = "get_valid: file:"//trim(fileobj%path)
1772  if (fileobj%is_root) then
1773  varid = get_variable_id(fileobj%ncid, variable_name, msg=append_error_msg)
1774  valid%has_max = .false.
1775  valid%has_min = .false.
1776  valid%has_fill = .false.
1777  valid%has_missing = .false.
1778  valid%has_range = .false.
1779 
1780  !This routine makes use of netcdf's automatic type conversion to
1781  !store all range information in double precision.
1782  if (attribute_exists(fileobj%ncid, varid, "scale_factor", msg=append_error_msg)) then
1783  call get_variable_attribute(fileobj, variable_name, "scale_factor", scale_factor, &
1784  broadcast=.false.)
1785  else
1786  scale_factor = 1._r8_kind
1787  endif
1788  if (attribute_exists(fileobj%ncid, varid, "add_offset", msg=append_error_msg)) then
1789  call get_variable_attribute(fileobj, variable_name, "add_offset", add_offset, &
1790  broadcast=.false.)
1791  else
1792  add_offset = 0._r8_kind
1793  endif
1794 
1795  !valid%max_val and valid%min_val are defined by the "valid_range", "valid_min", and
1796  !"valid_max" variable attributes if they are present in the file. If either the maximum value
1797  !or minimum value is defined, valid%has_range is set to .true. (i.e. open ended ranges
1798  !are valid and should be tested within the is_valid function).
1799  if (attribute_exists(fileobj%ncid, varid, "valid_range", msg=append_error_msg)) then
1800  call get_variable_attribute(fileobj, variable_name, "valid_range", buffer, &
1801  broadcast=.false.)
1802  valid%max_val = buffer(2)*scale_factor + add_offset
1803  valid%has_max = .true.
1804  valid%min_val = buffer(1)*scale_factor + add_offset
1805  valid%has_min = .true.
1806  else
1807  if (attribute_exists(fileobj%ncid, varid, "valid_max", msg=append_error_msg)) then
1808  call get_variable_attribute(fileobj, variable_name, "valid_max", buffer(1), &
1809  broadcast=.false.)
1810  valid%max_val = buffer(1)*scale_factor + add_offset
1811  valid%has_max = .true.
1812  endif
1813  if (attribute_exists(fileobj%ncid, varid, "valid_min", msg=append_error_msg)) then
1814  call get_variable_attribute(fileobj, variable_name, "valid_min", buffer(1), &
1815  broadcast=.false.)
1816  valid%min_val = buffer(1)*scale_factor + add_offset
1817  valid%has_min = .true.
1818  endif
1819  endif
1820  valid%has_range = valid%has_min .or. valid%has_max
1821 
1822 
1823  !Get the missing value from the file if it exists.
1824  if (attribute_exists(fileobj%ncid, varid, "missing_value", msg=append_error_msg)) then
1825  call get_variable_attribute(fileobj, variable_name, "missing_value", buffer(1), &
1826  broadcast=.false.)
1827  valid%missing_val = buffer(1)*scale_factor + add_offset
1828  valid%has_missing = .true.
1829  endif
1830 
1831  !Get the fill value from the file if it exists.
1832  !If the _FillValue attribute is present and the maximum or minimum value is not defined,
1833  !then the maximum or minimum value will be determined by the _FillValue according to the NUG convention.
1834  !The NUG convention states that a positive fill value will be the exclusive upper
1835  !bound (i.e. valid values are less than the fill value), while a
1836  !non-positive fill value will be the exclusive lower bound (i.e. valis
1837  !values are greater than the fill value). As before, valid%has_range is true
1838  !if either a maximum or minimum value is set.
1839  if (attribute_exists(fileobj%ncid, varid, "_FillValue", msg=append_error_msg)) then
1840  call get_variable_attribute(fileobj, variable_name, "_FillValue", buffer(1), &
1841  broadcast=.false.)
1842  valid%fill_val = buffer(1)*scale_factor + add_offset
1843  valid%has_fill = .true.
1844  xtype = get_variable_type(fileobj%ncid, varid, msg=append_error_msg)
1845 
1846  if (.not. valid%has_range) then
1847  if (xtype .eq. nf90_short .or. xtype .eq. nf90_int) then
1848  if (buffer(1) .gt. 0) then
1849  valid%max_val = (buffer(1) - 1._r8_kind)*scale_factor + add_offset
1850  valid%has_max = .true.
1851  else
1852  valid%min_val = (buffer(1) + 1._r8_kind)*scale_factor + add_offset
1853  valid%has_min = .true.
1854  endif
1855  elseif (xtype .eq. nf90_float .or. xtype .eq. nf90_double) then
1856  if (buffer(1) .gt. 0) then
1857  valid%max_val = (nearest(nearest(buffer(1), -1._r8_kind), -1._r8_kind)) &
1858  *scale_factor + add_offset
1859  valid%has_max = .true.
1860  else
1861  valid%min_val = (nearest(nearest(buffer(1), 1._r8_kind), 1._r8_kind)) &
1862  *scale_factor + add_offset
1863  valid%has_min = .true.
1864  endif
1865  else
1866  call error("Unsupported variable type:"//trim(append_error_msg))
1867  endif
1868  valid%has_range = .true.
1869  endif
1870  endif
1871 
1872  endif
1873 
1874  call mpp_broadcast(valid%has_min, fileobj%io_root, pelist=fileobj%pelist)
1875  if (valid%has_min) then
1876  call mpp_broadcast(valid%min_val, fileobj%io_root, pelist=fileobj%pelist)
1877  endif
1878  call mpp_broadcast(valid%has_max, fileobj%io_root, pelist=fileobj%pelist)
1879  if (valid%has_max) then
1880  call mpp_broadcast(valid%max_val, fileobj%io_root, pelist=fileobj%pelist)
1881  endif
1882  call mpp_broadcast(valid%has_range, fileobj%io_root, pelist=fileobj%pelist)
1883 
1884  call mpp_broadcast(valid%has_fill, fileobj%io_root, pelist=fileobj%pelist)
1885  if (valid%has_fill) then
1886  call mpp_broadcast(valid%fill_val, fileobj%io_root, pelist=fileobj%pelist)
1887  endif
1888 
1889  call mpp_broadcast(valid%has_missing, fileobj%io_root, pelist=fileobj%pelist)
1890  if (valid%has_missing) then
1891  call mpp_broadcast(valid%missing_val, fileobj%io_root, pelist=fileobj%pelist)
1892  endif
1893 
1894 end function get_valid

1927 !! @return A flag telling if the data element is "valid."
1928 elemental function check_if_valid(rdatum, validobj) &
1929  result(valid_data)
1930 
1931  real(kind=r8_kind), intent(in) :: rdatum !< packed data element.
1932  type(valid_t), intent(in) :: validobj !< Valid object.
1933  logical :: valid_data
1934

1937  ! range.
1938  if (validobj%has_range) then
1939  if (validobj%has_min .and. .not. validobj%has_max) then
1940  valid_data = rdatum .ge. validobj%min_val
1941  elseif (validobj%has_max .and. .not. validobj%has_min) then
1942  valid_data = rdatum .le. validobj%max_val
1943  else
1944  valid_data = .not. (rdatum .lt. validobj%min_val .or. rdatum .gt. validobj%max_val)
1945  endif
1946  endif
1947  ! If the variable has a fill value or missing value, valid values must not be
1948  ! equal to either.
1949  if (validobj%has_fill .or. validobj%has_missing) then
1950  if (validobj%has_fill .and. .not. validobj%has_missing) then
1951  valid_data = rdatum .ne. validobj%fill_val
1952  elseif (validobj%has_missing .and. .not. validobj%has_fill) then
1953  valid_data = rdatum .ne. validobj%missing_val
1954  else
1955  valid_data = .not. (rdatum .eq. validobj%missing_val .or. rdatum .eq. validobj%fill_val)
1956  endif
