24 subroutine mpp_global_sum_ad_( domain, field, gsum_, flags, position, tile_count, overflow_check)
25  mpp_type_ :: mpp_global_sum_
26  type(domain2D), intent(in) :: domain
27  mpp_type_, intent(inout) :: field(:,: mpp_extra_indices_ )
28  mpp_type_, intent(in) :: gsum_
29  integer, intent(in), optional :: flags
30  integer, intent(in), optional :: position
31  integer, intent(in), optional :: tile_count
32  logical, intent(in), optional :: overflow_check
33 
34  mpp_type_, dimension(:,:), allocatable :: field2d
35  mpp_type_, dimension(:,:), allocatable :: global2d
36  mpp_type_, dimension(MAX_TILES), save :: gsum, nbrgsum, mygsum
37 
38  integer :: i,j, ioff,joff, isc, iec, jsc, jec, is, ie, js, je, ishift, jshift, ioffset, joffset
39  integer :: gxsize, gysize
40  integer :: global_flag, tile, ntile, nlist, n, list, m
41 
42  if( domain%max_ntile_pe > max_tiles ) call mpp_error(fatal, "MPP_GLOBAL_SUM: number of tiles is exceed MAX_TILES")
43  ntile = size(domain%x(:))
44  nlist = size(domain%list(:))
45  tile = 1
46  if(present(tile_count)) tile = tile_count
47  global_flag = non_bitwise_exact_sum
48  if(present(flags)) global_flag = flags
49 
50  call mpp_get_domain_shift(domain, ishift, jshift, position)
51 
52  if( size(field,1).EQ.domain%x(tile)%compute%size+ishift .AND. &
53  & size(field,2).EQ.domain%y(tile)%compute%size+jshift )then
54 !field is on compute domain
55  ioff = -domain%x(tile)%compute%begin + 1
56  joff = -domain%y(tile)%compute%begin + 1
57  else if( size(field,1).EQ.domain%x(tile)%memory%size+ishift .AND. &
58  & size(field,2).EQ.domain%y(tile)%memory%size+jshift )then
59 !field is on data domain
60  ioff = -domain%x(tile)%data%begin + 1
61  joff = -domain%y(tile)%data%begin + 1
62  else
63  call mpp_error( fatal, &
64  & 'MPP_GLOBAL_SUM_: incoming field array must match either compute domain or data domain.' )
65  end if
66 
67  if(domain%ntiles > max_tiles) call mpp_error( fatal, &
68  'MPP_GLOBAL_SUM_: number of tiles on this mosaic is greater than MAXTILES')
69 
70  call mpp_get_compute_domain( domain, is, ie, js, je, tile_count = tile_count )
71  isc = is; iec = ie + ishift; jsc = js; jec = je + jshift
72 
73  call mpp_get_global_domain(domain, xsize = gxsize, ysize = gysize )
74  mpp_global_sum_ = 0
75  if( global_flag == bitwise_exact_sum )then
76  !this is bitwise exact across different PE counts.
77 
78  allocate( field2d(isc:iec,jsc:jec) )
79  allocate( global2d( gxsize+ishift, gysize+jshift ) )
80  field2d = 0.
81  ioffset = domain%x(tile)%goffset*ishift; joffset = domain%y(tile)%goffset*jshift
82  if( tile == ntile) then
83  if(domain%ntiles == 1 ) then
84  mpp_global_sum_ = gsum_
85  mygsum(tile) = mpp_global_sum_
86  else if( nlist == 1) then
87  mpp_global_sum_ = gsum_
88  mygsum(1:ntile) = mpp_global_sum_
89  else ! need to sum by the order of tile_count
90  mpp_global_sum_ = gsum_
91  gsum(1:domain%ntiles) = mpp_global_sum_
92  !--- receive data from root_pe of each tile
93  do list = 1, nlist - 1
94  m = mod( domain%pos+nlist-list, nlist )
95  if( domain%list(m)%pe == domain%list(m)%tile_root_pe ) then
96  do n = 1, size(domain%list(m)%x(:))
97  nbrgsum(n) = gsum(domain%list(m)%tile_id(n))
98  end do
99  call mpp_send( nbrgsum(1), plen=size(domain%list(m)%x(:)), to_pe=domain%list(m)%pe, tag=comm_tag_1)
100  end if
101  end do
102  !--- send the data to other pe if the current pe is the root pe of any tile
103  if( mpp_domain_is_tile_root_pe(domain) ) then
104  do list = 1, nlist - 1
105  m = mod( domain%pos+list, nlist )
106  call mpp_recv( mygsum(1), glen=ntile, from_pe=domain%list(m)%pe, tag=comm_tag_1 )
107  end do
108  end if
109  call mpp_sync_self()
110  ! first fill the global sum on current pe.
111  do n = 1, ntile
112  mygsum(n) = gsum(domain%tile_id(n))
113  end do
114 
115  end if
116  end if
117  global2d(1:gxsize+ioffset,1:gysize+joffset) = mygsum(tile)
118  if ( present( tile_count ) ) then
119  call mpp_global_field_ad( domain, field2d, global2d, position=position, tile_count=tile_count )
120  else
121  call mpp_global_field_ad( domain, field2d, global2d, position=position )
122  endif
123 
124  do j = jsc, jec
125  do i = isc, iec
126  field(i+ioff:i+ioff,j+joff:j+joff mpp_extra_indices_) = field2d(i,j)
127  end do
128  end do
129 
130  deallocate(global2d, field2d)
131  else if ( global_flag == bitwise_efp_sum )then
132 #ifdef DO_EFP_SUM_
133  !this is bitwise across different PE counts using EFP sum
134  if( ntile > 1 ) then
135  call mpp_error( fatal, 'MPP_GLOBAL_SUM_: multiple tile per pe is not supported for BITWISE_EFP_SUM')
136  endif
137 
138  do j = jsc, jec
139  do i = isc, iec
140  field(i+ioff:i+ioff,j+joff:j+joff mpp_extra_indices_) = gsum_
141  end do
142  end do
143 #else
144  call mpp_error( fatal, &
145  & 'MPP_GLOBAL_SUM_: BITWISE_EFP_SUM is only implemented for real number, contact developer')
146 #endif
147  else !this is not bitwise-exact across different PE counts
148  ioffset = domain%x(tile)%loffset*ishift; joffset = domain%y(tile)%loffset*jshift
149  mygsum(1:ntile) = 0
150  if(tile == ntile) then
151  mpp_global_sum_ = gsum_
152  call mpp_sum_ad( mpp_global_sum_, domain%list(:)%pe )
153  mygsum(1:ntile) = mpp_global_sum_
154  end if
155  field(is+ioff:ie+ioff+ioffset, js+joff:je+joff+joffset mpp_extra_indices_) = mygsum(tile)
156  end if
157 
158  return
159  end subroutine mpp_global_sum_ad_
160 !> @}
