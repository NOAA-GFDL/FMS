26 subroutine system_clock_mpi( count, count_rate, count_max )
27 ! There can be one ONE baseline count0 and this routine is
28 ! included in multiple places.
29 !mimics F90 SYSTEM_CLOCK intrinsic
30  integer(i8_kind), intent(out), optional :: count, count_rate, count_max
31 !count must return a number between 0 and count_max
32  integer(i8_kind), parameter :: maxtick=HUGE(count_max)
33  if(first_call_system_clock_mpi)then
34  first_call_system_clock_mpi=.false.
35  mpi_count0 = MPI_WTime()
36  mpi_tick_rate = real(1.d0/MPI_WTick(), r8_kind)
37  endif
38  if( PRESENT(count) )then
39  count = int((MPI_WTime()-mpi_count0)*mpi_tick_rate, i8_kind)
40  end if
41  if( PRESENT(count_rate) )then
42  count_rate = int(mpi_tick_rate, i8_kind)
43  end if
44  if( PRESENT(count_max) )then
45  count_max = maxtick-1
46  end if
47  return
48  end subroutine system_clock_mpi

53 !mimics F90 SYSTEM_CLOCK intrinsic
