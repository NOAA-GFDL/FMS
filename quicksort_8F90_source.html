30 !> Create array partitions for quicksort
31 function qksrt_partition(n, list, start, end) result(top)
32  implicit none
33  integer, intent(in) :: n
34  _typ, intent(inout) :: list(n)
35  integer, intent(in) :: start, end
36 
37  integer pivot, bottom, top
38  logical done
39 
40  pivot = list(end) ! Partition around the last value
41  bottom = start-1 ! Start outside the area to be partitioned
42  top = end ! Ditto
43 
44  done = .false.
45  do while (.not. done) ! Until all elements are partitioned...
46 
47  do while (.not. done) ! Until we find an out of place element...
48  bottom = bottom+1 ! ... move the bottom up.
49 
50  if(bottom == top) then ! If we hit the top...
51  done = .true. ! ... we are done.
52  exit
53  endif
54 
55  if(list(bottom) > pivot) then ! Is the bottom out of place?
56  list(top) = list(bottom) ! Then put it at the top...
57  exit ! ... and start searching from the top.
58  endif
59  enddo
60 
61  do while (.not. done) ! Until we find an out of place element...
62  top = top-1 ! ... move the top down.
63 
64  if(top == bottom) then ! If we hit the bottom...
65  done = .true. ! ... we are done.
66  exit
67  endif
68 
69  if(list(top) < pivot) then ! Is the top out of place?
70  list(bottom) = list(top) ! Then put it at the bottom...
71  exit ! ...and start searching from the bottom.
72  endif
73  enddo
74  enddo
75 
76  list(top) = pivot ! Put the pivot in its place.
77  ! Return the split point
78 
79 end function qksrt_partition
80 
81 !> quicksort a given list
82 recursive subroutine qksrt_quicksort(n, list, start, end)
83  implicit none
84  integer, intent(in) :: n
85  _typ, intent(inout) :: list(n)
86  integer, intent(in) :: start, end
87  integer :: split, qksrt_partition
88  external :: qksrt_partition
89  if(start < end) then ! If there are two or more elements...
90  split = qksrt_partition(n, list, start, end) ! ... partition the sublist...
91  call qksrt_quicksort(n, list, start, split-1) ! ... and sort both halves.
92  call qksrt_quicksort(n, list, split+1, end)
93  endif
94 end subroutine qksrt_quicksort
