20 !> @brief This programs tests the "register_global_attribute" and
21 !! "get_global_attribute" interfaces
22 program test_global_att
23 use fms2_io_mod
24 use mpp_mod
25 use platform_mod
26 
27 type(FmsNetcdfFile_t) :: fileobj !< fms2io netcd file obj
28 real(kind=r8_kind) :: buf_r8_kind !< r8_kind buffer
29 real(kind=r8_kind), dimension(2) :: buf_r8_kind_1d !< r8_kind 1D buffer
30 real(kind=r4_kind) :: buf_r4_kind !< r4_kind buffer
31 real(kind=r4_kind), dimension(2) :: buf_r4_kind_1d !< r4_kind 1D buffer
32 integer(kind=i4_kind) :: buf_i4_kind !< i4_kind buffer
33 integer(kind=i4_kind), dimension(2) :: buf_i4_kind_1d !< i4_kind 1D buffer
34 integer(kind=i8_kind) :: buf_i8_kind !< i8_kind buffer
35 integer(kind=i8_kind), dimension(2) :: buf_i8_kind_1d !< i8_kind 1D buffer
36 character (len = 120), dimension(3) :: my_format !< Array of formats to try.
37 
38 character(len=20) :: buf_str !< character buffer
39 integer :: i !< For Do loop
40 
41 call fms2_io_init
42 call mpp_init
43 
44 my_format(1) = '64bit'
45 my_format(2) = 'classic'
46 my_format(3) = 'netcdf4'
47 
48 do i = 1, size(my_format)
49 !< Write out the different possible global attributes to a netcdf file
50 if (open_file(fileobj, "test_global_att_"//trim(my_format(i))//".nc", "overwrite", nc_format=my_format(i))) then
51  call register_global_attribute(fileobj, "buf_r8_kind", real(7., kind=r8_kind))
52  call register_global_attribute(fileobj, "buf_r8_kind_1d", (/ real(7., kind=r8_kind), real(9., kind=r8_kind) /))
53 
54  call register_global_attribute(fileobj, "buf_r4_kind", real(4., kind=r4_kind))
55  call register_global_attribute(fileobj, "buf_r4_kind_1d", (/ real(4., kind=r4_kind), real(6., kind=r4_kind)/) )
56 
57  call register_global_attribute(fileobj, "buf_i4_kind", int(3, kind=i4_kind))
58  call register_global_attribute(fileobj, "buf_i4_kind_1d", (/ int(3, kind=i4_kind), int(5, kind=i4_kind) /) )
59 
60  !< int8 is only supported with the "netcdf4" type
61  if(i .eq. 3) then
62  call register_global_attribute(fileobj, "buf_i8_kind", int(2, kind=i8_kind))
63  call register_global_attribute(fileobj, "buf_i8_kind_1d", (/ int(2, kind=i8_kind), int(4, kind=i8_kind) /) )
64  endif
65 
66  call register_global_attribute(fileobj, "buf_str", "some text"//char(0), str_len=10)
67 
68  call close_file(fileobj)
69 else
70  call mpp_error(fatal, "test_global_att: error opening the file for writting")
71 endif
72 
73 !< Read the global attributes from the netcdf file
74 if (open_file(fileobj, "test_global_att_"//trim(my_format(i))//".nc", "read", nc_format=my_format(i))) then
75  call get_global_attribute(fileobj, "buf_r8_kind", buf_r8_kind)
76  call get_global_attribute(fileobj, "buf_r8_kind_1d", buf_r8_kind_1d)
77 
78  call get_global_attribute(fileobj, "buf_r4_kind", buf_r4_kind)
79  call get_global_attribute(fileobj, "buf_r4_kind_1d", buf_r4_kind_1d)
80 
81  call get_global_attribute(fileobj, "buf_i4_kind", buf_i4_kind)
82  call get_global_attribute(fileobj, "buf_i4_kind_1d", buf_i4_kind_1d)
83 
84  !< int8 is only supported with the "netcdf4" type
85  if(i .eq. 3) then
86  call get_global_attribute(fileobj, "buf_i8_kind", buf_i8_kind)
87  call get_global_attribute(fileobj, "buf_i8_kind_1d", buf_i8_kind_1d)
88  endif
89 
90  call get_global_attribute(fileobj, "buf_str", buf_str)
91 
92  call close_file(fileobj)
93 else
94  call mpp_error(fatal, "test_global_att: error opening the file for reading")
95 endif
96 
97 !< Compares the values read with the expected values
98 if (buf_r8_kind /= real(7., kind=r8_kind)) call mpp_error(fatal, "test_global_att-"//trim(my_format(i))//": error reading buf_r8_kind")
99 if (buf_r8_kind_1d(1) /= real(7., kind=r8_kind) .or. buf_r8_kind_1d(2) /= real(9., kind=r8_kind)) &
100  call mpp_error(fatal, "test_global_att-"//trim(my_format(i))//": error reading buf_r8_kind_1d")
101 
102 if (buf_r4_kind /= real(4., kind=r4_kind)) call mpp_error(fatal, "test_global_att-"//trim(my_format(i))//": error reading buf_r4_kind")
103 if (buf_r4_kind_1d(1) /= real(4., kind=r4_kind) .or. buf_r4_kind_1d(2) /= real(6., kind=r4_kind)) &
104  call mpp_error(fatal, "test_global_att-"//trim(my_format(i))//": error reading buf_r4_kind_1d")
105 
106 if (buf_i4_kind /= int(3, kind=i4_kind)) call mpp_error(fatal, "test_global_att-"//trim(my_format(i))//": error reading buf_i4_kind")
107 if (buf_i4_kind_1d(1) /= int(3, kind=i4_kind) .or. buf_i4_kind_1d(2) /= int(5, kind=i4_kind)) &
108  call mpp_error(fatal, "test_global_att-"//trim(my_format(i))//": error reading buf_i4_kind_1d")
109 
110 !< int8 is only supported with the "netcdf4" type
111 if(i .eq. 3) then
112  if (buf_i8_kind /= int(2, kind=i8_kind)) call mpp_error(fatal, "test_global_att-"//trim(my_format(i))//": error reading buf_i8_kind")
113  if (buf_i8_kind_1d(1) /= int(2, kind=i8_kind) .or. buf_i8_kind_1d(2) /= int(4, kind=i8_kind)) &
114  & call mpp_error(fatal, "test_global_att-"//trim(my_format(i))//": error reading buf_i8_kind_1d")
115 endif
116 
117 if (trim(buf_str) /= "some text") then
118  print *, "buf_str read in = ", trim(buf_str)
119  call mpp_error(fatal, "test_global_att-"//trim(my_format(i))//": error reading buf_str")
120 endif
121 
122 enddo !< do i = 1, size(my_format)
123 
124 call mpp_exit()
125 end program
