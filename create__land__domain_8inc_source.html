20 !> @brief Create an unstructured land-like domain.
21 subroutine create_land_domain(atmos_domain, nx, ny, ntiles, land_domain, npes_group)
22 
23  type(domain2d), intent(in) :: atmos_domain
24  integer, intent(in) :: nx
25  integer, intent(in) :: ny
26  integer, intent(in) :: ntiles
27  type(domainug), intent(inout) :: land_domain
28  integer, intent(in) :: npes_group
29 
30  integer :: isc
31  integer :: iec
32  integer :: jsc
33  integer :: jec
34  integer :: isd
35  integer :: ied
36  integer :: jsd
37  integer :: jed
38  logical, dimension(:,:,:), allocatable :: lmask
39  integer, dimension(:), allocatable :: npts_tile
40  real :: rmask
41  integer :: ntotal_land
42  integer, dimension(:), allocatable :: grid_index
43  integer, dimension(:), allocatable :: isl
44  integer, dimension(:), allocatable :: iel
45  integer, dimension(:), allocatable :: jsl
46  integer, dimension(:), allocatable :: jel
47  integer, dimension(:), allocatable :: ntiles_grid
48  integer :: i
49  integer :: j
50  integer :: l
51  integer :: n
52 
53  call mpp_get_compute_domain(atmos_domain, isc, iec, jsc, jec)
54  call mpp_get_data_domain(atmos_domain, isd, ied, jsd, jed)
55  allocate(npts_tile(ntiles))
56  if (mpp_pe() .eq. mpp_root_pe()) then
57  allocate(lmask(nx, ny, ntiles))
58  lmask = .false.
59  do n = 1, ntiles
60  do j = 1, ny
61  do i = 1, nx
62  call random_number(rmask)
63  if (rmask .gt. 0.5) then
64  lmask(i, j, n) = .true.
65  endif
66  enddo
67  enddo
68  npts_tile(n) = count(lmask(:, :, n))
69  enddo
70  ntotal_land = sum(npts_tile)
71  allocate(grid_index(ntotal_land))
72  l = 0
73  allocate(isl(0:mpp_npes()-1))
74  allocate(iel(0:mpp_npes()-1))
75  allocate(jsl(0:mpp_npes()-1))
76  allocate(jel(0:mpp_npes()-1))
77  call mpp_get_compute_domains(atmos_domain, xbegin=isl, xend=iel, ybegin=jsl, yend=jel)
78  do n = 1, ntiles
79  do j = 1, ny
80  do i = 1, nx
81  if (lmask(i, j, n)) then
82  l = l + 1
83  grid_index(l) = (j-1)*nx+i
84  endif
85  enddo
86  enddo
87  enddo
88  deallocate(lmask)
89  deallocate(isl)
90  deallocate(iel)
91  deallocate(jsl)
92  deallocate(jel)
93  endif
94  call mpp_broadcast(npts_tile, ntiles, mpp_root_pe())
95  if (mpp_pe() .ne. mpp_root_pe()) then
96  ntotal_land = sum(npts_tile)
97  allocate(grid_index(ntotal_land))
98  endif
99  call mpp_broadcast(grid_index, ntotal_land, mpp_root_pe())
100  allocate(ntiles_grid(ntotal_land))
101  ntiles_grid = 1
102  call mpp_define_unstruct_domain(land_domain, atmos_domain, npts_tile, ntiles_grid, &
103  mpp_npes(), npes_group, grid_index, name="Unstructured domain")
104  deallocate(npts_tile)
105  deallocate(grid_index)
106  deallocate(ntiles_grid)
107 end subroutine create_land_domain
