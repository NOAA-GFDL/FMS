42  !> Updates data domain of 3D field whose computational domains have been computed
43  subroutine mpp_update_domains_ad_3d_( field, domain, flags, complete, position, &
44  whalo, ehalo, shalo, nhalo, name, tile_count)
45  mpp_type_, intent(inout) :: field(:,:,:)
46  type(domain2d), intent(inout) :: domain
47  integer, intent(in), optional :: flags
48  logical, intent(in), optional :: complete
49  integer, intent(in), optional :: position
50  integer, intent(in), optional :: whalo, ehalo, shalo, nhalo ! specify halo region to be updated.
51  character(len=*), intent(in), optional :: name
52  integer, intent(in), optional :: tile_count
53 
54  integer :: update_position, update_whalo, update_ehalo, update_shalo, update_nhalo, ntile
55 
56  integer(i8_kind),dimension(MAX_DOMAIN_FIELDS, MAX_TILES),save :: f_addrs=-9999
57  integer :: tile, max_ntile
58  character(len=3) :: text
59  logical :: set_mismatch, is_complete
60  logical :: do_update
61  integer, save :: isize=0, jsize=0, ke=0, l_size=0, list=0
62  integer, save :: pos, whalosz, ehalosz, shalosz, nhalosz
63  mpp_type_ :: d_type
64  type(overlapspec), pointer :: update => null()
65  type(overlapspec), pointer :: check => null()
66 
67  if(present(whalo)) then
68  update_whalo = whalo
69  if(abs(update_whalo) > domain%whalo ) call mpp_error(fatal, "MPP_UPDATE_AD_3D: "// &
70  "optional argument whalo should not be larger than the whalo when define domain.")
71  else
72  update_whalo = domain%whalo
73  end if
74  if(present(ehalo)) then
75  update_ehalo = ehalo
76  if(abs(update_ehalo) > domain%ehalo ) call mpp_error(fatal, "MPP_UPDATE_AD_3D: "// &
77  "optional argument ehalo should not be larger than the ehalo when define domain.")
78  else
79  update_ehalo = domain%ehalo
80  end if
81  if(present(shalo)) then
82  update_shalo = shalo
83  if(abs(update_shalo) > domain%shalo ) call mpp_error(fatal, "MPP_UPDATE_AD_3D: "// &
84  "optional argument shalo should not be larger than the shalo when define domain.")
85  else
86  update_shalo = domain%shalo
87  end if
88  if(present(nhalo)) then
89  update_nhalo = nhalo
90  if(abs(update_nhalo) > domain%nhalo ) call mpp_error(fatal, "MPP_UPDATE_AD_3D: "// &
91  "optional argument nhalo should not be larger than the nhalo when define domain.")
92  else
93  update_nhalo = domain%nhalo
94  end if
95 
96  !--- when there is NINETY or MINUS_NINETY rotation for some contact, the salar data can not be on E or N-cell,
97  if(present(position)) then
98  if(domain%rotated_ninety .AND. ( position == east .OR. position == north ) ) &
99  call mpp_error(fatal, 'MPP_UPDATE_AD_3D: hen there is NINETY or MINUS_NINETY rotation, ' // &
100  'can not use scalar version update_domain for data on E or N-cell' )
101  end if
102 
103  max_ntile = domain%max_ntile_pe
104  ntile = size(domain%x(:))
105  is_complete = .true.
106  if(PRESENT(complete)) then
107  is_complete = complete
108  end if
109  tile = 1
110 
111  if(max_ntile>1) then
112  if(ntile>max_tiles) then
113  write( text,'(i2)' ) max_tiles
114  call mpp_error(fatal,'MPP_UPDATE_AD_3D: MAX_TILES='//text//' is less than number of tiles on this pe.' )
115  endif
116  if(.NOT. present(tile_count) ) call mpp_error(fatal, "MPP_UPDATE_AD_3D: "// &
117  "optional argument tile_count should be present when number of tiles on this pe is more than 1")
118  tile = tile_count
119  end if
120  do_update = (tile == ntile) .AND. is_complete
121  list = list+1
122  if(list > max_domain_fields)then
123  write( text,'(i2)' ) max_domain_fields
124  call mpp_error(fatal,'MPP_UPDATE_AD_3D: MAX_DOMAIN_FIELDS='//text//' exceeded for group update.' )
125  endif
126  f_addrs(list, tile) = loc(field)
127  update_position = center
128  if(present(position)) update_position = position
129  if(list == 1 .AND. tile == 1 )then
130  isize=size(field,1); jsize=size(field,2); ke = size(field,3); pos = update_position
131  whalosz = update_whalo; ehalosz = update_ehalo; shalosz = update_shalo; nhalosz = update_nhalo
132  else
133  set_mismatch = .false.
134  set_mismatch = set_mismatch .OR. (isize /= size(field,1))
135  set_mismatch = set_mismatch .OR. (jsize /= size(field,2))
136  set_mismatch = set_mismatch .OR. (ke /= size(field,3))
137  set_mismatch = set_mismatch .OR. (update_position /= pos)
138  set_mismatch = set_mismatch .OR. (update_whalo /= whalosz)
139  set_mismatch = set_mismatch .OR. (update_ehalo /= ehalosz)
140  set_mismatch = set_mismatch .OR. (update_shalo /= shalosz)
141  set_mismatch = set_mismatch .OR. (update_nhalo /= nhalosz)
142  if(set_mismatch)then
143  write( text,'(i2)' ) list
144  call mpp_error(fatal,'MPP_UPDATE_AD_3D: Incompatible field at count '//text//' for group update.' )
145  endif
146  endif
147  if(is_complete) then
148  l_size = list
149  list = 0
150  end if
151  if(do_update )then
152  if( domain_update_is_needed(domain, update_whalo, update_ehalo, update_shalo, update_nhalo) )then
153  if(debug_update_level .NE. no_check) then
154  check => search_check_overlap(domain, update_position)
155  if(ASSOCIATED(check) ) then
156  call mpp_do_check(f_addrs(1:l_size,1:ntile), domain, check, d_type, ke, flags, name )
157  endif
158  endif
159  update => search_update_overlap(domain, update_whalo, update_ehalo, update_shalo, update_nhalo, &
160  & update_position)
161 
162  !call mpp_do_update( f_addrs(1:l_size,1:ntile), domain, update, d_type, ke, &
163  ! b_addrs(1:l_size,1:ntile), bsize, flags)
164 
165  if ( PRESENT ( flags ) ) then
166  call mpp_do_update_ad( f_addrs(1:l_size,1:ntile), domain, update, d_type, ke, flags )
167  else
168  call mpp_do_update_ad( f_addrs(1:l_size,1:ntile), domain, update, d_type, ke )
169  endif
170 
171 
172  end if
173  l_size=0; f_addrs=-9999; isize=0; jsize=0; ke=0
174  endif
175  return
176 
177  end subroutine mpp_update_domains_ad_3d_
178

248  subroutine mpp_update_domains_ad_3d_v_( fieldx, fieldy, domain, flags, gridtype, complete, &
249  whalo, ehalo, shalo, nhalo, name, tile_count)
250 !updates data domain of 3D field whose computational domains have been computed
251  mpp_type_, intent(inout) :: fieldx(:,:,:), fieldy(:,:,:)
252  type(domain2d), intent(inout) :: domain
253  integer, intent(in), optional :: flags, gridtype
254  logical, intent(in), optional :: complete
255  integer, intent(in), optional :: whalo, ehalo, shalo, nhalo
256  character(len=*), intent(in), optional :: name
257  integer, intent(in), optional :: tile_count
258 
259  integer :: update_whalo, update_ehalo, update_shalo, update_nhalo, ntile
260  integer :: grid_offset_type
261  logical :: exchange_uv
262 
263  integer(i8_kind),dimension(MAX_DOMAIN_FIELDS, MAX_TILES),save :: f_addrsx=-9999, f_addrsy=-9999
264  logical :: do_update, is_complete
265  integer, save :: isize(2)=0,jsize(2)=0,ke=0,l_size=0, offset_type=0, list=0
266  integer, save :: whalosz, ehalosz, shalosz, nhalosz
267  integer :: tile, max_ntile
268  integer :: position_x, position_y
269  logical :: set_mismatch
270  character(len=3) :: text
271  mpp_type_ :: d_type
272  type(overlapspec), pointer :: updatex => null()
273  type(overlapspec), pointer :: updatey => null()
274  type(overlapspec), pointer :: checkx => null()
275  type(overlapspec), pointer :: checky => null()
276 
277  if(present(whalo)) then
278  update_whalo = whalo
279  if(abs(update_whalo) > domain%whalo ) call mpp_error(fatal, "MPP_UPDATE_AD_3D_V: "// &
280  "optional argument whalo should not be larger than the whalo when define domain.")
281  else
282  update_whalo = domain%whalo
283  end if
284  if(present(ehalo)) then
285  update_ehalo = ehalo
286  if(abs(update_ehalo) > domain%ehalo ) call mpp_error(fatal, "MPP_UPDATE_AD_3D_V: "// &
287  "optional argument ehalo should not be larger than the ehalo when define domain.")
288  else
289  update_ehalo = domain%ehalo
290  end if
291  if(present(shalo)) then
292  update_shalo = shalo
293  if(abs(update_shalo) > domain%shalo ) call mpp_error(fatal, "MPP_UPDATE_AD_3D_V: "// &
294  "optional argument shalo should not be larger than the shalo when define domain.")
295  else
296  update_shalo = domain%shalo
297  end if
298  if(present(nhalo)) then
299  update_nhalo = nhalo
300  if(abs(update_nhalo) > domain%nhalo ) call mpp_error(fatal, "MPP_UPDATE_AD_3D_V: "// &
301  "optional argument nhalo should not be larger than the nhalo when define domain.")
302  else
303  update_nhalo = domain%nhalo
304  end if
305 
306  grid_offset_type = agrid
307  if( PRESENT(gridtype) ) grid_offset_type = gridtype
308 
309  exchange_uv = .false.
310  if(grid_offset_type == dgrid_ne) then
311  exchange_uv = .true.
312  grid_offset_type = cgrid_ne
313  else if( grid_offset_type == dgrid_sw ) then
314  exchange_uv = .true.
315  grid_offset_type = cgrid_sw
316  end if
317 
318  max_ntile = domain%max_ntile_pe
319  ntile = size(domain%x(:))
320 
321  is_complete = .true.
322  if(PRESENT(complete)) then
323  is_complete = complete
324  end if
325  tile = 1
326 
327  if(max_ntile>1) then
328  if(ntile>max_tiles) then
329  write( text,'(i2)' ) max_tiles
330  call mpp_error(fatal,'MPP_UPDATE_AD_3D_V: MAX_TILES='//text//' is less than number of tiles on this pe.' )
331  endif
332  if(.NOT. present(tile_count) ) call mpp_error(fatal, "MPP_UPDATE_AD_3D_V: "// &
333  "optional argument tile_count should be present when number of tiles on some pe is more than 1")
334  tile = tile_count
335  end if
336 
337  do_update = (tile == ntile) .AND. is_complete
338  list = list+1
339  if(list > max_domain_fields)then
340  write( text,'(i2)' ) max_domain_fields
341  call mpp_error(fatal,'MPP_UPDATE_AD_3D_V: MAX_DOMAIN_FIELDS='//text//' exceeded for group update.' )
342  endif
343 
344  f_addrsx(list, tile) = loc(fieldx)
345  f_addrsy(list, tile) = loc(fieldy)
346 
347  if(list == 1 .AND. tile == 1)then
348  isize(1)=size(fieldx,1); jsize(1)=size(fieldx,2); ke = size(fieldx,3)
349  isize(2)=size(fieldy,1); jsize(2)=size(fieldy,2)
350  offset_type = grid_offset_type
351  whalosz = update_whalo; ehalosz = update_ehalo; shalosz = update_shalo; nhalosz = update_nhalo
352  else
353  set_mismatch = .false.
354  set_mismatch = set_mismatch .OR. (isize(1) /= size(fieldx,1))
355  set_mismatch = set_mismatch .OR. (jsize(1) /= size(fieldx,2))
356  set_mismatch = set_mismatch .OR. (ke /= size(fieldx,3))
357  set_mismatch = set_mismatch .OR. (isize(2) /= size(fieldy,1))
358  set_mismatch = set_mismatch .OR. (jsize(2) /= size(fieldy,2))
359  set_mismatch = set_mismatch .OR. (ke /= size(fieldy,3))
360  set_mismatch = set_mismatch .OR. (grid_offset_type /= offset_type)
361  set_mismatch = set_mismatch .OR. (update_whalo /= whalosz)
362  set_mismatch = set_mismatch .OR. (update_ehalo /= ehalosz)
363  set_mismatch = set_mismatch .OR. (update_shalo /= shalosz)
364  set_mismatch = set_mismatch .OR. (update_nhalo /= nhalosz)
365  if(set_mismatch)then
366  write( text,'(i2)' ) list
367  call mpp_error(fatal,'MPP_UPDATE_AD_3D_V: Incompatible field at count '// &
368  & text//' for group vector update.' )
369  end if
370  end if
371  if(is_complete) then
372  l_size = list
373  list = 0
374  end if
375  if(do_update)then
376  if( domain_update_is_needed(domain, update_whalo, update_ehalo, update_shalo, update_nhalo) )then
377  select case(grid_offset_type)
378  case (agrid)
379  position_x = center
380  position_y = center
381  case (bgrid_ne, bgrid_sw)
382  position_x = corner
383  position_y = corner
384  case (cgrid_ne, cgrid_sw)
385  position_x = east
386  position_y = north
387  case default
388  call mpp_error(fatal, "mpp_update_domains2D_ad.h: invalid value of grid_offset_type")
389  end select
390 
391  if(debug_update_level .NE. no_check) then
392  checkx => search_check_overlap(domain, position_x)
393  checky => search_check_overlap(domain, position_y)
394  if(ASSOCIATED(checkx)) then
395  if(exchange_uv) then
396  call mpp_do_check(f_addrsx(1:l_size,1:ntile),f_addrsy(1:l_size,1:ntile), domain, &
397  checky, checkx, d_type, ke, flags, name)
398  else
399  call mpp_do_check(f_addrsx(1:l_size,1:ntile),f_addrsy(1:l_size,1:ntile), domain, &
400  checkx, checky, d_type, ke, flags, name)
401  end if
402  endif
403  endif
404  updatex => search_update_overlap(domain, update_whalo, update_ehalo, update_shalo, update_nhalo, &
405  & position_x)
406  updatey => search_update_overlap(domain, update_whalo, update_ehalo, update_shalo, update_nhalo, &
407  & position_y)
408  if(exchange_uv) then
409  call mpp_do_update_ad(f_addrsx(1:l_size,1:ntile),f_addrsy(1:l_size,1:ntile), domain, updatey, updatex, &
410  d_type,ke, grid_offset_type, flags)
411  else
412  call mpp_do_update_ad(f_addrsx(1:l_size,1:ntile),f_addrsy(1:l_size,1:ntile), domain, updatex, updatey, &
413  d_type,ke,grid_offset_type, flags)
414  end if
415  end if
416  l_size=0; f_addrsx=-9999; f_addrsy=-9999; isize=0; jsize=0; ke=0
417  end if
418 
419  return
420  end subroutine mpp_update_domains_ad_3d_v_

search_update_overlap
