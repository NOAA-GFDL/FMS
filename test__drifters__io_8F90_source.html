20 program test_drifters_io
21 
22  use drifters_io_mod
23  use mpp_mod, only : mpp_error, fatal, stdout, mpp_init, mpp_exit
24 
25  implicit none
26  type(drifters_io_type) :: drfts_io
27  character(len=128) :: ermesg
28  character(len=31) :: filename
29  integer :: np, nd, nf, nt, i, j, k, npmax
30  real :: dt, time, xmin, xmax, ymin, ymax, u, v, dr, x, y
31  integer, allocatable :: ids(:)
32  real, allocatable :: positions(:,:), fields(:,:)
33 
34  call mpp_init()
35 
36  ! number of dimensions
37  nd = 3
38  ! number of fields
39  nf = 2
40  ! max number of dirfters
41  npmax = 20
42  ! number of time steps
43  nt = 50
44  ! starting time
45  time = 0.
46 
47  ! domain boundary. (drifters outside domain will not be written to file.)
48  xmin = 0.
49  ymin = 0.
50  xmax = 1.
51  ymax = 1.
52 
53  ! constant velocity
54  u = (xmax-xmin)*sqrt(2.)
55  v = (ymax-ymin)*sqrt(2.)
56  dt = 1/real(nt)
57 
58  ! open file
59 
60  filename = 'test_drifter_io.nc'
61  call drifters_io_new(drfts_io, filename, nd, nf, ermesg)
62  if(ermesg/='') then
63  print *,'ERROR after drifters_io_new: ', ermesg
64  call mpp_error(fatal, ermesg)
65  endif
66  ! set attributes
67 
68  call drifters_io_set_position_names(drfts_io, (/'x','y','z'/), ermesg)
69  if(ermesg/='') then
70  print *,'ERROR after drifters_io_position_names: ', ermesg
71  call mpp_error(fatal, ermesg)
72  endif
73 
74  ! note the trailing blanks in the first field, which are added here to
75  ! ensure that "salinity" will not be truncated (all names must have the
76  ! same length)
77  call drifters_io_set_field_names(drfts_io, (/'temp ','salinity'/), ermesg)
78  if(ermesg/='') then
79  print *,'ERROR after drifters_io_field_names: ', ermesg
80  call mpp_error(fatal, ermesg)
81  endif
82 
83  call drifters_io_set_position_units(drfts_io, (/'deg east ','deg north','meters '/), ermesg)
84  if(ermesg/='') then
85  print *,'ERROR after drifters_io_position_units: ', ermesg
86  call mpp_error(fatal, ermesg)
87  endif
88 
89  call drifters_io_set_field_units(drfts_io, (/'deg K','ppm '/), ermesg)
90  if(ermesg/='') then
91  print *,'ERROR after drifters_io_field_units: ', ermesg
92  call mpp_error(fatal, ermesg)
93  endif
94 
95  allocate(positions(nd, npmax), ids(npmax), fields(nf, npmax))
96  dr = sqrt( (xmax-xmin)**2 + (ymax-ymin)**2 )/real(npmax)
97 
98 
99  ! x
100  positions(1, :) = +(/ (i*dr,i=0,npmax-1) /)/sqrt(2.)
101  ! y
102  positions(2, :) = -(/ (i*dr,i=0,npmax-1) /)/sqrt(2.)
103  ! z
104  positions(3, :) = 0.
105 
106  ! drifters' identity array (can be any integer number)
107  ids = (/ (i, i=1, npmax) /)
108 
109  ! set fields as a function of space time
110  fields(1, :) = sqrt( (positions(1,:)-xmin)**2 + (positions(2,:)-ymin)**2 )
111  fields(2, :) = positions(1,:)-u*time + positions(2,:)-v*time ! invariant
112 
113  ! write to disk only drifters inside domain
114  do i = 1, npmax
115  x = positions(1,i)
116  y = positions(2,i)
117  if(x>=xmin .and. x<=xmax .and. y>=ymin .and. y<=ymax) then
118  call drifters_io_write(drfts_io, time, np=1, nd=nd, nf=nf, &
119  & ids=ids(i), positions=positions(:,i), fields=fields(:,i), ermesg=ermesg)
120  if(ermesg/='') then
121  print *,'ERROR after drifters_io_write: ', ermesg
122  call mpp_error(fatal, ermesg)
123  endif
124  endif
125  enddo
126 
127  ! advect
128 
129  do j = 1, nt
130  time = time + dt
131  positions(1, :) = positions(1, :) + u*dt
132  positions(2, :) = positions(2, :) + v*dt
133  fields(1, :) = sqrt( (positions(1,:)-xmin)**2 + (positions(2,:)-ymin)**2 )
134  fields(2, :) = positions(1,:)-u*time + positions(2,:)-v*time ! invariant
135 
136  do i = 1, npmax
137  x = positions(1,i)
138  y = positions(2,i)
139  if(x>=xmin .and. x<=xmax .and. y>=ymin .and. y<=ymax) then
140  call drifters_io_write(drfts_io, time, np=1, nd=nd, nf=nf, &
141  & ids=ids(i), positions=positions(:,i), fields=fields(:,i), ermesg=ermesg)
142  if(ermesg/='') then
143  print *,'ERROR after drifters_io_write: ', ermesg
144  call mpp_error(fatal, ermesg)
145  endif
146  endif
147  enddo
148 
149  enddo
150 
151  deallocate(positions, ids, fields)
152 
153  call drifters_io_del(drfts_io, ermesg)
154  if(ermesg/='') then
155  print *,'ERROR after drifters_io_del: ', ermesg
156  call mpp_error(fatal, ermesg)
157  endif
158  call mpp_exit()
159 end program test_drifters_io
