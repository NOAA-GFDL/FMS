242 !> @brief Safely copy a string from one buffer to another.
243 subroutine string_copy(dest, source, check_for_null)
244  character(len=*), intent(inout) :: dest !< Destination string.
245  character(len=*), intent(in) :: source !< Source string.
246  logical, intent(in), optional :: check_for_null !<Flag indicating to test for null character
247 
248  integer :: i
249  logical :: check_null
250 
251  check_null = .false.
252  if (present(check_for_null)) check_null = check_for_null
253 
254  i = 0
255  if (check_null) then
256  i = index(source, char(0)) - 1
257  endif
258 
259  if (i < 1 ) i = len_trim(source)
260 
261  if (len_trim(source(1:i)) .gt. len(dest)) then
262  call error("The input destination string is not big enough to" &
263  //" to hold the input source string.")
264  endif
265  dest = ""
266  dest = adjustl(trim(source(1:i)))
267 
268 end subroutine string_copy

271 !> @brief Compare strings.
272 !! @return Flag telling if the strings are the same.
273 function string_compare(string1, string2, ignore_case) &
274  result(same)
275 
276  character(len=*), intent(in) :: string1
277  character(len=*), intent(in) :: string2
278  logical,intent(in), optional :: ignore_case
279  logical :: same
280 
281  if (len_trim(string1) .ne. len_trim(string2)) then
282  same = .false.
283  return
284  endif
285  if (present(ignore_case)) then
286  if (ignore_case) then
287  same = trim(lowercase(string1)) .eq. trim(lowercase(string2))

290  endif

295 !> @brief Determine if a string exists in a character linked list.
296 !! @return Flag telling if the string is found in the list.
297 function is_in_list(list, string, ignore_case) &
298  result(in_list)
299 
300  type(char_linked_list), pointer, intent(in) :: list !< Linked list.
301  character(len=*), intent(in) :: string !< Input string.
302  logical, intent(in), optional :: ignore_case !< Flag to ignore case.
303  logical :: in_list
304 
305  type(char_linked_list), pointer :: p
306 
307  in_list = .false.
308  p => list
309  do while(associated(p))
310  if (string_compare(string, p%string, ignore_case)) then
311  in_list = .true.
312  return
313  else
314  p => p%head
315  endif
316  enddo
317 end function is_in_list

323  character(len=*), intent(in) :: string !< Input string.
324 
325  type(char_linked_list), pointer :: node

360 !> @brief Determine if the "domain tile string" (.tilex.) exists in the input filename.
361 !! @internal
362 function has_domain_tile_string(string) &
363  result(has_string)
364 
365  character(len=*), intent(in) :: string !< Input string.
366  logical :: has_string
367 
368  integer :: l
369  integer :: i, j
370 
371  has_string = .false.
372 ! Assigns i to the index where ".tile" starts
373  i = index(trim(string), ".tile", back=.true.)
374  if (i .ne. 0) then
375  l = len_trim(string)
376 ! Sets i to the index after .tile
377  i = i + 5
378  j = i
379  do while (i .le. l)
380 ! If the ith characters is a dot but i not equal to the index after .tile set has_string to true
381  if (verify(string(i:i), ".") .eq. 0 .and. j .ne. i) then
382  has_string = .true.
383  exit
384 ! If the ith characters is NOT a number exit function and has_string will stay as false
385  elseif (verify(string(i:i), "0123456789") .ne. 0) then
386  exit
387  endif
388  i = i + 1
389  enddo
390  endif
391 end function has_domain_tile_string

519 !> @brief Populate 2D maskmap from mask_table given a model
520 subroutine parse_mask_table_2d(mask_table, maskmap, modelname)
521 
522  character(len=*), intent(in) :: mask_table !< Mask table to be read in
523  logical, intent(out) :: maskmap(:,:) !< 2D Mask output
524  character(len=*), intent(in) :: modelname !< Model to which this applies
525 
526  integer :: nmask, layout(2)
527  integer, allocatable :: mask_list(:,:)
528  character(len=:), dimension(:), allocatable :: mask_table_contents
529  integer :: iocheck, n, stdoutunit, offset
530  character(len=128) :: record
531 
532  maskmap = .true.
533  nmask = 0
534  stdoutunit = stdout()
535  call ascii_read(mask_table, mask_table_contents)
536  if( mpp_pe() == mpp_root_pe() ) then
537  read(mask_table_contents(1), fmt=*, iostat=iocheck) nmask
538  if (iocheck > 0) then
539  call mpp_error(fatal, "fms2_io(parse_mask_table_2d): Error in reading nmask from file variable")
540  elseif (iocheck < 0) then
541  call mpp_error(fatal, "fms2_io(parse_mask_table_2d): Error: nmask not completely read from file variable")
542  endif
543  write(stdoutunit,*)"parse_mask_table: Number of domain regions masked in ", trim(modelname), " = ", nmask
544  if( nmask > 0 ) then
545  !--- read layout from mask_table and confirm it matches the shape of maskmap
546  read(mask_table_contents(2), fmt=*, iostat=iocheck) layout
547  if (iocheck > 0) then
548  call mpp_error(fatal, "fms2_io(parse_mask_table_2d): Error in reading layout from file variable")
549  elseif (iocheck < 0) then
550  call mpp_error(fatal, "fms2_io(parse_mask_table_2d): Error: layout not completely read from file variable")
551  endif
552  if( (layout(1) .NE. size(maskmap,1)) .OR. (layout(2) .NE. size(maskmap,2)) )then
553  write(stdoutunit,*)"layout=", layout, ", size(maskmap) = ", size(maskmap,1), size(maskmap,2)
554  call mpp_error(fatal, "fms2_io(parse_mask_table_2d): layout in file "//trim(mask_table)// &
555  "does not match size of maskmap for "//trim(modelname))
556  endif
557  !--- make sure mpp_npes() == layout(1)*layout(2) - nmask
558  if( mpp_npes() .NE. layout(1)*layout(2) - nmask ) call mpp_error(fatal, &
559  .NE."fms2_io(parse_mask_table_2d): mpp_npes() layout(1)*layout(2) - nmask for "//trim(modelname))
560  endif
561  endif
562 
563  call mpp_broadcast(nmask, mpp_root_pe())
564 
565  if(nmask==0) return
566 
567  allocate(mask_list(nmask,2))
568 
569  if( mpp_pe() == mpp_root_pe() ) then
570  n = 0
571  offset = 3
572  do while (offset + n < size(mask_table_contents)+1)
573  read(mask_table_contents(n+offset),'(a)',iostat=iocheck) record
574  if (iocheck > 0) then
575  call mpp_error(fatal, "fms2_io(parse_mask_table_2d): Error in reading record from file variable")
576  elseif (iocheck < 0) then
577  call mpp_error(fatal, "fms2_io(parse_mask_table_2d): Error: record not completely read from file variable")
578  endif
579  if (record(1:1) == '#') then
580  offset = offset + 1
581  cycle
582  elseif (record(1:10) == ' ') then
583  offset = offset + 1
584  cycle
585  endif
586  n = n + 1
587  if( n > nmask ) then
588  call mpp_error(fatal, "fms2_io(parse_mask_table_2d): number of mask_list entry "// &
589  "is greater than nmask in file "//trim(mask_table) )
590  endif
591  read(record,*,iostat=iocheck) mask_list(n,1), mask_list(n,2)
592  if (iocheck > 0) then
593  call mpp_error(fatal, "fms2_io(parse_mask_table_2d): Error in reading mask_list from record variable")
594  elseif (iocheck < 0) then
595  call mpp_error(fatal, "fms2_io(parse_mask_table_2d): Error: mask_list not completely read from record variable")
596  endif
597  enddo
598 
599  !--- make sure the number of entry for mask_list is nmask
600  if( n .NE. nmask) call mpp_error(fatal, &
601  "fms2_io(parse_mask_table_2d): number of mask_list entry does not match nmask in file "//trim(mask_table))
602  endif
603 
604  call mpp_broadcast(mask_list, 2*nmask, mpp_root_pe())
605  do n = 1, nmask
606  maskmap(mask_list(n,1),mask_list(n,2)) = .false.
607  enddo
608 
609  deallocate(mask_list)
610 
611 end subroutine parse_mask_table_2d

614 !> @brief Populate 3D maskmap from mask_table given a model
615 subroutine parse_mask_table_3d(mask_table, maskmap, modelname)
616 
617  character(len=*), intent(in) :: mask_table !< Mask table to be read in
618  logical, intent(out) :: maskmap(:,:,:) !< 2D Mask output
619  character(len=*), intent(in) :: modelname !< Model to which this applies
620 
621  integer :: nmask, layout(2)

631  if( mpp_pe() == mpp_root_pe() ) then
632  read(mask_table_contents(1), fmt=*, iostat=iocheck) nmask
633  if (iocheck > 0) then
634  call mpp_error(fatal, "fms2_io(parse_mask_table_3d): Error in reading nmask from file variable")
635  elseif (iocheck < 0) then
636  call mpp_error(fatal, "fms2_io(parse_mask_table_3d): Error: nmask not completely read from file variable")
637  endif
638  write(stdoutunit,*)"parse_mask_table: Number of domain regions masked in ", trim(modelname), " = ", nmask
639  if( nmask > 0 ) then
640  !--- read layout from mask_table and confirm it matches the shape of maskmap
641  read(mask_table_contents(2), fmt=*, iostat=iocheck) layout(1), layout(2), ntiles
642  if (iocheck > 0) then
643  call mpp_error(fatal, "fms2_io(parse_mask_table_3d): Error in reading layout from file variable")
644  elseif (iocheck < 0) then
645  call mpp_error(fatal, "fms2_io(parse_mask_table_3d): Error: layout not completely read from file variable")
646  endif

648  write(stdoutunit,*)"layout=", layout, ", size(maskmap) = ", size(maskmap,1), size(maskmap,2)
649  call mpp_error(fatal, "fms2_io(parse_mask_table_3d): layout in file "//trim(mask_table)// &
650  "does not match size of maskmap for "//trim(modelname))
651  endif
652  if( ntiles .NE. size(maskmap,3) ) then
653  write(stdoutunit,*)"ntiles=", ntiles, ", size(maskmap,3) = ", size(maskmap,3)
654  call mpp_error(fatal, "fms2_io(parse_mask_table_3d): ntiles in file "//trim(mask_table)// &
655  "does not match size of maskmap for "//trim(modelname))
656  endif
657  !--- make sure mpp_npes() == layout(1)*layout(2) - nmask
658  if( mpp_npes() .NE. layout(1)*layout(2)*ntiles - nmask ) then
659  print*, "layout=", layout, nmask, mpp_npes()
660  call mpp_error(fatal, &
661  .NE."fms2_io(parse_mask_table_3d): mpp_npes() layout(1)*layout(2) - nmask for "//trim(modelname))
662  endif
663  endif
664  endif
665 
666  call mpp_broadcast(nmask, mpp_root_pe())
667 
668  if(nmask==0) return
669 
670  allocate(mask_list(nmask,3))
671 
672  if( mpp_pe() == mpp_root_pe() ) then
673  n = 0
674  offset = 3
675  do while (offset + n < size(mask_table_contents)+1)
676  read(mask_table_contents(n+offset),'(a)',iostat=iocheck) record
677  if (iocheck > 0) then
678  call mpp_error(fatal, "fms2_io(parse_mask_table_3d): Error in reading record from file variable")
679  elseif (iocheck < 0) then
680  call mpp_error(fatal, "fms2_io(parse_mask_table_3d): Error: record not completely read from file variable")
681  endif
682  if (record(1:1) == '#') then
683  offset = offset + 1
684  cycle
685  elseif (record(1:10) == ' ') then
686  offset = offset + 1
687  cycle
688  endif
689  n = n + 1
690  if( n > nmask ) then
691  call mpp_error(fatal, "fms2_io(parse_mask_table_3d): number of mask_list entry "// &
692  "is greater than nmask in file "//trim(mask_table) )
693  endif
694  read(record,*,iostat=iocheck) mask_list(n,1), mask_list(n,2), mask_list(n,3)
695  if (iocheck > 0) then
696  call mpp_error(fatal, "fms2_io(parse_mask_table_3d): Error in reading mask_list from record variable")
697  elseif (iocheck < 0) then
698  call mpp_error(fatal, "fms2_io(parse_mask_table_3d): Error: mask_list not completely read from record variable")
699  endif
700  enddo
701 
702  !--- make sure the number of entry for mask_list is nmask
703  if( n .NE. nmask) call mpp_error(fatal, &
704  "fms2_io(parse_mask_table_3d): number of mask_list entry does not match nmask in file "//trim(mask_table))
705 ! call mpp_close(unit)
706  endif
707 
708  call mpp_broadcast(mask_list, 3*nmask, mpp_root_pe())
709  do n = 1, nmask
710  maskmap(mask_list(n,1),mask_list(n,2),mask_list(n,3)) = .false.
711  enddo
712 
713  deallocate(mask_list)
714 end subroutine parse_mask_table_3d
