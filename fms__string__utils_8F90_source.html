126  !> @brief Converts a character array to an array of c pointers!
127  !! @return An array of c pointers
128  function fms_array_to_pointer(my_array) &
129  result(my_pointer)
130  character(len=*), target :: my_array(:) !!< Array of strings to convert
131  type(c_ptr), allocatable :: my_pointer(:)
132 
133  integer :: i !< For do loops
134 
135  if (allocated(my_pointer)) call mpp_error(fatal, "The c pointer array is &
136  already allocated. Deallocated before calling fms_array_to_pointer")
137  allocate(my_pointer(size(my_array)))
138 
139  do i = 1, size(my_array)
140  my_pointer(i) = c_loc(my_array(i))
141  enddo
142  end function fms_array_to_pointer

163  !> @brief Searches through a SORTED array of pointers for a string
164  !! @return the indices where the array was found
165  !! If the string was not found, indices will be indices(1) = -999
166  !> <br>Example usage:
167  !! my_pointer = fms_array_to_pointer(my_array)
168  !! call fms_sort_this(my_pointer, n_array, indices)
169  !! ifind = fms_find_my_string(my_pointer, n_array, string_to_find)
170  function fms_find_my_string(my_pointer, narray, string_to_find) &
171  result(ifind)
172  type(c_ptr), intent(in) :: my_pointer(*) !< Array of c pointer
173  integer, intent(in) :: narray !< Length of the array
174  character(len=*), intent(in) :: string_to_find !< string to find
175  integer, allocatable :: ifind(:)
176 
177  integer :: nfind !< number of times the string was found
178  character(len=:), allocatable :: buffer !< buffer to read the indices into
179 
180  buffer = fms_c2f_string(&
181  fms_find_my_string_binding(my_pointer, narray, trim(string_to_find)//c_null_char, nfind))
182 
183  if (allocated(ifind)) call mpp_error(fatal, "The indices array is already allocated. &
184  Deallocate it before calling fms_find_my_string")
185 
186  if (nfind .gt. 0) then
187  allocate(ifind(nfind))
188  read(buffer,*) ifind
189  else
190  allocate(ifind(1))

192  endif

196  !> \brief Converts a C-string to a pointer and then to a Fortran string
197  function cstring_fortran_conversion (cstring) result(fstring)
198  character (kind=c_char), intent(in) :: cstring (*) !< Input C-string
199  character(len=:), allocatable :: fstring !< The fortran string returned
200  fstring = cpointer_fortran_conversion(fms_cstring2cpointer(cstring))

203  !> \brief Converts a C-string returned from a TYPE(C_PTR) function to
204  !! a fortran string with type character.
205  function cpointer_fortran_conversion (cstring) result(fstring)
206  type (c_ptr), intent(in) :: cstring !< Input C-pointer
207  character(len=:), allocatable :: fstring !< The fortran string returned
208  character(len=:,kind=c_char), pointer :: string_buffer !< A temporary pointer to between C and Fortran
209  integer(c_size_t) :: length !< The string length
210 
211  length = c_strlen(cstring)
212  allocate (character(len=length, kind=c_char) :: string_buffer)
213  block
214  character(len=length,kind=c_char), pointer :: s
215  call c_f_pointer(cstring,s) ! Recovers a view of the C string
216  string_buffer = s ! Copies the string contents
217  end block
218 
219  allocate(character(len=length) :: fstring) !> Set the length of fstring
220  fstring = string_buffer
221  deallocate(string_buffer)
222  end function cpointer_fortran_conversion

224 !> @brief Copies a Fortran string into a C string and puts c_null_char in any trailing spaces
225  subroutine fms_f2c_string (dest, str_in)
226  character (c_char), intent (out) :: dest (:) !< C String to be copied into
227  character (len=*), intent (in) :: str_in !< Fortran string to copy to C string
228  integer :: i !< for looping
229 !> Drop an error if the C string is not large enough to hold the input and the c_null_char at the end.
230  if (len(trim(str_in)) .ge. size(dest)) call mpp_error(fatal, &
231  "The string "//trim(str_in)//" is larger than the destination C string")
232 !> Copy c_null_char into each spot in dest
233  dest = c_null_char
234 !> Loop though and put each character of the Fortran string into the C string array
235  do i = 1, len(trim(str_in))
236  dest(i) = str_in(i:i)
237  enddo
238 end subroutine fms_f2c_string

266  !> @brief Safely copy a string from one buffer to another.
267  subroutine string_copy(dest, source, check_for_null)
268  character(len=*), intent(inout) :: dest !< Destination string.
269  character(len=*), intent(in) :: source !< Source string.
270  logical, intent(in), optional :: check_for_null !<Flag indicating to test for null character
271 
272  integer :: i
273  logical :: check_null
274 
275  check_null = .false.
276  if (present(check_for_null)) check_null = check_for_null
277 
278  i = 0
279  if (check_null) then
280  i = index(source, char(0)) - 1
281  endif
282 
283  if (i < 1 ) i = len_trim(source)
284 
285  if (len_trim(source(1:i)) .gt. len(dest)) then
286  call mpp_error(fatal, "The input destination string is not big enough to" &
287  //" to hold the input source string.")
288  endif
289  dest = ""
290  dest = adjustl(trim(source(1:i)))
291  end subroutine string_copy
