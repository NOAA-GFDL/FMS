22 subroutine mpp_type_create_(field, array_of_subsizes, array_of_starts, &
23  dtype_out)
24  mpp_type_, intent(in) :: field(:,:,:)
25  integer, intent(in) :: array_of_subsizes(:)
26  integer, intent(in) :: array_of_starts(:)
27  type(mpp_type), target, intent(out) :: dtype_out
28 
29  type(mpp_type), pointer :: dtype
30  integer :: newtype ! MPI datatype ID
31 
32  if (.NOT. module_is_initialized) &
33  call mpp_error(fatal, 'MPP_TYPE_CREATE_: You must first call mpp_init.')
34 
35  if (current_clock .NE. 0) &
36  call system_clock(start_tick)
37 
38  if (verbose) &
39  call mpp_error(note, 'MPP_TYPE_CREATE_: &
40  &using MPI_Type_create_subarray...')
41 
42  dtype => datatypes%head
43  ! TODO: Check mpp_byte
44 
45  ! Check if mpp_type already exists
46  do while (.not. associated(dtype))
47  dtype => dtype%next
48 
49  if (dtype%ndims /= rank(field)) cycle
50  if (any(dtype%sizes /= shape(field))) cycle
51  if (any(dtype%subsizes /= array_of_subsizes)) cycle
52  if (any(dtype%starts /= array_of_starts)) cycle
53  if (dtype%etype /= mpi_type_) cycle
54 
55  ! If all parameters match, then the datatype exists and return dtype
56  dtype%counter = dtype%counter + 1
57  dtype_out = dtype
58  return
59  end do
60 
61  ! The type does not exist; create a new internal type
62  call mpi_type_create_subarray( &
63  rank(field), &
64  shape(field), &
65  array_of_subsizes, &
66  array_of_starts, &
67  mpi_order_fortran, &
68  mpi_type_, &
69  newtype, &
70  error &
71  )
72 
73  ! Register on the MPI runtime
74  call mpi_type_commit(newtype, error)
75 
76  ! Create new entry
77  allocate(dtype)
78  allocate(dtype%sizes(rank(field)))
79  allocate(dtype%subsizes(rank(field)))
80  allocate(dtype%starts(rank(field)))
81 
82  ! Populate values
83  dtype%counter = 1
84  dtype%ndims = rank(field)
85  dtype%sizes = shape(field)
86  dtype%subsizes = array_of_subsizes
87  dtype%starts = array_of_starts
88  dtype%etype = mpi_type_
89  dtype%id = newtype
90 
91  ! Add dtype to the list
92  dtype%prev => datatypes%tail
93  dtype%prev%next => dtype
94  datatypes%tail => dtype
95  datatypes%length = datatypes%length + 1
96 
97  ! Copy dtype to output
98  dtype_out = dtype
99 
100  if (current_clock .NE. 0) &
101  call increment_current_clock(event_type_create, mpp_type_bytelen_)
102 
103 end subroutine mpp_type_create_
104 !> @}
