20 subroutine land_compressed_restart_file(nz, nt, debug_flag)
21 
22  integer, intent(in) :: nz
23  integer, intent(in) :: nt
24  logical, intent(in), optional :: debug_flag
25 
26  type(FmsNetcdfFile_t) :: fileobj
27  integer :: my_rank
28  integer :: num_ranks
29  integer, dimension(:), allocatable :: pelist
30  integer, dimension(:), allocatable :: npes_corner
31  integer, dimension(:), allocatable :: npes_nelems
32  integer :: nx
33  integer :: total_nx
34  real(kind=r8_kind), dimension(:,:), allocatable :: temperature_data
35  real(kind=r8_kind), dimension(:), allocatable :: cdim_data
36  real(kind=r8_kind), dimension(:), allocatable :: lev_data
37  integer(kind=i8_kind) :: out_chksum
38  integer :: ndims
39  integer, dimension(:), allocatable :: dim_sizes
40  integer(kind=i8_kind) :: in_chksum
41  logical :: debug
42  integer :: err
43  integer :: i
44  integer :: j
45  integer :: k
46  integer :: s
47 
48  !Tell which test is running.
49  call mpi_barrier(mpi_comm_world, err)
50  call mpi_check(err)
51  if (mpp_pe() .eq. 0) then
52  write(output_unit,'(/a)',advance="no") &
53  "Running land (compressed) restart file test ... "
54  endif
55  call mpi_barrier(mpi_comm_world, err)
56  call mpi_check(err)
57 
58  !Initialize.
59  if (present(debug_flag)) then
60  debug = debug_flag
61  else
62  debug = .false.
63  endif
64  call random_seed()
65 
66  !Initalize some data. Each rank will have (my_rank+1)*5
67  !data elements.
68  my_rank = mpp_pe()
69  num_ranks = mpp_npes()
70  allocate(pelist(num_ranks))
71  allocate(npes_corner(num_ranks))
72  allocate(npes_nelems(num_ranks))
73  nx = 1
74  do i = 1, npes
75  pelist(i) = i - 1
76  npes_corner(i) = nx
77  npes_nelems(i) = i*5
78  nx = nx + npes_nelems(i)
79  enddo
80  nx = npes_nelems(my_rank+1)
81  total_nx = sum(npes_nelems)
82  allocate(temperature_data(nx, nz))
83  do i = 1, nz
84  do j = 1, nx
85  call random_number(temperature_data(j, i))
86  temperature_data(j, i) = temperature_data(j, i) + 300.05
87  enddo
88  enddo
89 
90  !Write a restart file.
91  !Open a restart file and initialize the file object.
92  call open_check(open_file(fileobj, "land_compressed.nc", "overwrite", &
93  pelist=pelist, nc_format="netcdf4", is_restart=.true.))
94 
95  !Add dimensions and corresponding variables to the file.
96  !cdim (compressed dimension).
97  call register_axis(fileobj, "cdim", npes_corner=npes_corner, npes_nelems=npes_nelems)
98  call register_field(fileobj, "cdim", "double", (/"cdim"/))
99  allocate(cdim_data(nx))
100  do i = 1, nx
101  cdim_data(i) = 4.178 + 3.9567*i
102  enddo
103  call write_data(fileobj, "cdim", cdim_data)
104  call register_variable_attribute(fileobj, "cdim", "units", "compressed value", str_len=len_trim("compressed value"))
105 
106  !Height.
107  call register_axis(fileobj, "lev", nz)
108  call register_field(fileobj, "lev", "double", (/"lev"/))
109  allocate(lev_data(nz))
110  do i = 1, nz
111  lev_data(i) = 0.1111 + 20.236*i
112  enddo
113  call write_data(fileobj, "lev", lev_data)
114  call register_variable_attribute(fileobj, "lev", "units", "mb", str_len=len_trim("mb"))
115 
116  !Add a restart variable.
117  call register_restart_field(fileobj, "temperature", temperature_data, (/"cdim","lev "/))
118  call register_variable_attribute(fileobj, "temperature", "units", "K", str_len=len_trim("K"))
119 
120  !Perform a "simulation".
121  do i = 1, nt
122  do j = 1, nz
123  do k = 1, nx
124  temperature_data(k, j) = temperature_data(k, j) + 0.5*mpp_pe()
125  enddo
126  enddo
127  enddo
128 
129  !Calculate a checksum, that will be used to check that the data is
130  !read back in correctly.
131  out_chksum = mpp_chksum(temperature_data, pelist=(/mpp_pe()/))
132 
133  !Write out the restart file.
134  call write_restart(fileobj)
135 
136  !Close the file and destroy the file object.
137  call close_file(fileobj)
138  call mpi_barrier(mpi_comm_world, err)
139  call mpi_check(err)
140 
141  !Prepare to read the data back in.
142  deallocate(cdim_data)
143  deallocate(lev_data)
144  deallocate(temperature_data)
145 
146  !Read restart with an explicit axis register call.
147  !Re-open the restart file and re-initialize the file object.
148  call open_check(open_file(fileobj, "land_compressed.nc", "read", pelist=pelist, &
149  nc_format="netcdf4", is_restart=.true.))
150 
151  !Read the cdim variable back in and make sure it is the
152  !correct size.
153  ndims = get_variable_num_dimensions(fileobj, "cdim", broadcast=.true.)
154  allocate(dim_sizes(ndims))
155  call get_variable_size(fileobj, "cdim", dim_sizes, .true.)
156  if (dim_sizes(1) .ne. total_nx) then
157  call mpp_error(fatal, "compressed axis is not the correct size.")
158  endif
159  deallocate(dim_sizes)
160 
161  !Allocate a buffer and associate it with the
162  !temperature variable.
163  ndims = get_variable_num_dimensions(fileobj, "temperature", broadcast=.true.)
164  allocate(dim_sizes(ndims))
165  call get_variable_size(fileobj, "temperature", dim_sizes, broadcast=.true.)
166  allocate(temperature_data(dim_sizes(1), dim_sizes(2)))
167  deallocate(dim_sizes)
168  temperature_data(:, :) = 0.
169 
170  !Read in the restart variable and close the file.
171  call read_data(fileobj, "temperature", temperature_data)
172  call close_file(fileobj)
173  call mpi_barrier(mpi_comm_world, err)
174  call mpi_check(err)
175 
176  !Check to see if the read-in data matches the data that was
177  !written out.
178  s = npes_corner(my_rank+1)
179  in_chksum = mpp_chksum(temperature_data(s:s+nx-1,:), pelist=(/mpp_pe()/))
180  call chksum_match(out_chksum, in_chksum, "temperature", debug)
181  call mpi_barrier(mpi_comm_world, err)
182  call mpi_check(err)
183  if (mpp_pe() .eq. 0) then
184  write(output_unit,'(a/)') trim(green)//"passed."//trim(color_end)
185  endif
186  call mpi_barrier(mpi_comm_world, err)
187  call mpi_check(err)
188 end subroutine land_compressed_restart_file
