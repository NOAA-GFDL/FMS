23 
24 use mpp_mod, only : mpp_init, mpp_exit, mpp_pe, mpp_root_pe, mpp_sync

27 use mpp_domains_mod, only : mpp_get_global_domain, mpp_get_data_domain, mpp_get_compute_domain, &

30 implicit none
31 
32 type atm_type
33  type(fmsnetcdffile_t) :: fileobj_sw !< fms2io netcdf file obj for south west bc
34  type(fmsnetcdffile_t) :: fileobj_ne !< fms2io netcdf file obj for north east bc
35  logical :: bcfile_sw_open !< flag indicating if the sourth west file is
36  !! opened
37  logical :: bcfile_ne_open !< flag indicating if the sourth west file is
38  !! opened
39  type(domain2d) :: domain !< Domain with halos
40  real, allocatable, dimension(:,:) :: var2d !< 2d variable data
41  real, allocatable, dimension(:,:,:):: var3d !< 3d variable data

45 integer :: nlon !< Number of points in x axis
46 integer :: nlat !< Number of points in y axis
47 integer :: isd, jsd !< Starting x/y index (data_domain)
48 integer :: ied, jed !< Ending x/y index (data_domain)
49 integer, allocatable, dimension(:) :: all_pelist !< List of pelist associated with the test
50 integer :: n !< No description
51 type(atm_type) :: atm !< No description
52 
53 call mpp_init
54 call fms2_io_init

56 nlon = 144

59 call mpp_define_domains( (/1,nlon,1,nlat/), layout, atm%Domain, xhalo=3, yhalo=3, symmetry=.true., name='test_bc_restart')
60 call mpp_get_data_domain(atm%domain, isd, ied, jsd, jed )
61 
62 allocate(atm%var2d(isd:ied,jsd:jed))
63 allocate(atm%var3d(isd:ied,jsd:jed,5))
64 
65 atm%var2d = real(mpp_pe()+100)
66 atm%var3d = real(mpp_pe())
67 
68 !< Create array of current pelist
69 allocate (all_pelist(layout(1)*layout(2)))
70 do n = 1, layout(1)*layout(2)
71  all_pelist(n) = n - 1
72 end do
73 
74 atm%BCfile_sw_open = .false.
75 atm%BCfile_ne_open = .false.
76 
77 !< Try to write a BC restart file:
78 atm%BCfile_sw_open = open_file(atm%fileobj_sw, "BCfile_sw.nc", "overwrite", is_restart=.true., pelist=all_pelist)
79 atm%BCfile_ne_open = open_file(atm%fileobj_ne, "BCfile_ne.nc", "overwrite", is_restart=.true., pelist=all_pelist)
80 
81 call register_bcs(atm, atm%fileobj_ne, atm%fileobj_sw, "sst", layout)
82 
83 if (atm%BCfile_sw_open) then
84  call write_restart_bc(atm%fileobj_sw)
85  call close_file(atm%fileobj_sw)
86 endif
87 
88 if (atm%BCfile_ne_open) then
89  call write_restart_bc(atm%fileobj_ne)
90  call close_file(atm%fileobj_ne)
91 endif
92 
93 call mpp_sync()
94 
95 atm%var2d = real(999.)
96 atm%var3d = real(999.)
97 
98 atm%BCfile_sw_open = .false.
99 atm%BCfile_ne_open = .false.
100 
101 !< Try to read the boundary condition restart back
102 atm%BCfile_sw_open = open_file(atm%fileobj_sw, "BCfile_sw.nc", "read", is_restart=.true., pelist=all_pelist)
103 atm%BCfile_ne_open = open_file(atm%fileobj_ne, "BCfile_ne.nc", "read", is_restart=.true., pelist=all_pelist)
104 
105 call register_bcs(atm, atm%fileobj_ne, atm%fileobj_sw, "sst", layout)
106 
107 if (atm%BCfile_sw_open) then
108  call read_restart_bc(atm%fileobj_sw)
109  call close_file(atm%fileobj_sw)
110 endif
111 
112 if (atm%BCfile_ne_open) then
113  call read_restart_bc(atm%fileobj_ne)
114  call close_file(atm%fileobj_ne)
115 endif
116 
117 call mpp_sync()
118 call mpp_exit()

122 !> @brief Dummy wrapper for the register_restart_variable calls for boundary
123 !! condition restarts
124 subroutine register_bcs(atm, fileobj_ne, fileobj_sw, var_name, layout, istag, jstag)
125  type(atm_type), intent(inout) :: atm
126  type(fmsnetcdffile_t), intent(inout) :: fileobj_sw !< fms2io netcdf file obj for south west bc
127  type(fmsnetcdffile_t), intent(inout) :: fileobj_ne !< fms2io netcdf file obj for north east bc
128  character(len=*), intent(in) :: var_name !< Name of the variable
129  integer, dimension(2), intent(in) :: layout !< Domain layout
130 
131  integer, intent(in), optional :: istag, jstag
132 
133  integer :: isd, jsd !< Starting x/y index (data_domain)
134  integer :: ied, jed !< Ending x/y index (data_domain)
135  integer :: is, js !< Starting x/y index (compute_domain)
136  integer :: ie, je !< Ending x/y index (compute_domain)
137  integer :: i_stag, j_stag !< Extra x/y?
138  integer :: npx, npy !< Number of points in x/y (global_domain)
139  integer :: x_halo, y_halo !< Number of halos in x and y
140  integer :: x_halo_ns !< Halo for north and south variables
141  integer, allocatable, dimension(:) :: x1_pelist !< Pelist corresponding to the bottom region
142  !! of the domain
143  integer, allocatable, dimension(:) :: y1_pelist !< Pelist corresponding to the right region
144  !! of the domain
145  integer, allocatable, dimension(:) :: x2_pelist !< Pelist corresponding to the top region
146  !! of the domain
147  integer, allocatable, dimension(:) :: y2_pelist !< Pelist corresponding to the left region
148  !! of the domain
149  integer :: n !< No description
150  integer, dimension(3) :: global_size !< Size of the domain
151  integer, dimension(4) :: indices !< Starting/Ending indices of the current
152  logical :: is_root_pe !< Flag indicating if this is the root pe
153 
154  !< .----.----.----.----.
155  !< |PE 0|PE 1|PE 2|PE 3| <- x2_pelist
156  !< .----.----.----.----.
157  !< |PE 4|PE 5|PE 6|PE 7|
158  !< .----.----.----.----.
159  !< |PE 8|PE 9|PE10|PE11|
160  !< .----.----.----.----.
161  !< |PE12|PE13|PE14|PE15| <- x1_pelist
162  !< .----.----.----.----.
163  !< ^y2_pelist ^y1_pelist
164 
165  i_stag = 0
166  j_stag = 0
167  if (present(istag)) i_stag = i_stag
168  if (present(jstag)) j_stag = j_stag
169 
170  call mpp_get_global_domain(atm%domain, xsize = npx, ysize = npy, position=corner )
171  call mpp_get_data_domain(atm%domain, isd, ied, jsd, jed )
172  call mpp_get_compute_domain(atm%domain, is, ie, js, je )
173 
174  !< Defaults
175  x_halo = is-isd
176  y_halo = js-jsd
177  i_stag = 0
178  j_stag = 0
179 
180  allocate (x1_pelist(layout(1)))
181  allocate (y1_pelist(layout(2)))
182  allocate (x2_pelist(layout(1)))
183  allocate (y2_pelist(layout(2)))
184 
185  !< Define west and east pelist
186  do n = 1,layout(2)
187  y1_pelist(n)=mpp_root_pe()+layout(1)*n-1
188  y2_pelist(n)=mpp_root_pe()+layout(1)*(n-1)
189  enddo
190 
191  !< Define south and north pelist
192  do n = 1,layout(1)
193  x1_pelist(n)=mpp_root_pe()+layout(1)*(layout(2)-1)+(n-1)
194  x2_pelist(n)=mpp_root_pe()+(n-1)
195  enddo
196 
197  !< EAST & WEST
198  !< Set defaults for west/east halo regions
199  indices(1) = 1
200  indices(2) = x_halo
201  indices(3) = jsd
202  indices(4) = jed + j_stag
203  global_size(1) = x_halo
204  global_size(2) = npy-1+2*y_halo + j_stag
205 
206  global_size(3) = size(atm%var3d, 3)
207 
208  !< Define west root_pe
209  is_root_pe = .false.
210  if (is.eq.1 .and. js.eq.1) is_root_pe = .true.
211 
212  if (atm%BCfile_sw_open) call register_restart_field(fileobj_sw, trim(var_name)//'_west', atm%var2d, &
213  indices, global_size(1:2), y2_pelist, &
214  is_root_pe, jshift=y_halo)
215  if (atm%BCfile_sw_open) call register_restart_field(fileobj_sw, trim(var_name)//'3d_west', atm%var3d, &
216  indices, global_size(1:3), y2_pelist, &
217  is_root_pe, jshift=y_halo)
218 
219  !< Define east root_pe
220  is_root_pe = .false.
221  if (ie.eq.npx-1 .and. je.eq.npy-1) is_root_pe = .true.
222 
223  !< Reset indices for prognostic variables in the east halo
224  indices(1) = ied-x_halo+1+i_stag
225  indices(2) = ied+i_stag
226 
227  if (atm%BCfile_ne_open) call register_restart_field(fileobj_ne, trim(var_name)//'east', atm%var2d, &
228  indices, global_size(1:2), y1_pelist, &
229  is_root_pe, jshift=y_halo, &
230  x_halo=(size(atm%var2d,1)-x_halo), ishift=-(ie+i_stag))
231  if (atm%BCfile_ne_open) call register_restart_field(fileobj_ne, trim(var_name)//'3d_east', atm%var3d, &
232  indices, global_size(1:3), y1_pelist, &
233  is_root_pe, jshift=y_halo, &
234  x_halo=(size(atm%var3d,1)-x_halo), ishift=-(ie+i_stag))
235  !< NORTH & SOUTH
236  !< set defaults for north/south halo regions
237  indices(1) = isd
238  indices(2) = ied+i_stag
239  indices(3) = 1
240  indices(4) = y_halo
241  global_size(1) = npx-1+i_stag
242  global_size(2) = y_halo
243 
244  !< Modify starts and ends for certain pes
245  if (is.eq.1) indices(1) = is
246  if (ie.eq.npx-1) indices(2) = ie+i_stag
247  x_halo_ns = 0
248  if (is.eq.1) x_halo_ns=x_halo
249 
250  !define south root_pe
251  is_root_pe = .false.
252  if (is.eq.1 .and. js.eq.1) is_root_pe = .true.
253 
254  if (atm%BCfile_sw_open) call register_restart_field(fileobj_sw, trim(var_name)//'south', atm%var2d, &
255  indices, global_size(1:2), x2_pelist, &
256  is_root_pe, x_halo=x_halo_ns)
257  if (atm%BCfile_sw_open) call register_restart_field(fileobj_sw, trim(var_name)//'3d_south', atm%var3d, &
258  indices, global_size(1:3), x2_pelist, &
259  is_root_pe, x_halo=x_halo_ns)
260 
261  !< Define north root_pe
262  is_root_pe = .false.
263  if (ie.eq.npx-1 .and. je.eq.npy-1) is_root_pe = .true.
264 
265  !< Reset indices for prognostic variables in the north halo
266  indices(3) = jed-y_halo+1+j_stag
267  indices(4) = jed+j_stag
268 
269  if (atm%BCfile_ne_open) call register_restart_field(fileobj_ne, trim(var_name)//'north', atm%var2d, &
270  indices, global_size(1:2), x1_pelist, &
271  is_root_pe, x_halo=x_halo_ns, &
272  y_halo=(size(atm%var2d,2)-y_halo), jshift=-(je+j_stag))
273  if (atm%BCfile_ne_open) call register_restart_field(fileobj_ne, trim(var_name)//'3d_north', atm%var3d, &
274  indices, global_size(1:3), x1_pelist, &
275  is_root_pe, x_halo=x_halo_ns, &
276  y_halo=(size(atm%var3d,2)-y_halo), jshift=-(je+j_stag))
277 
278 end subroutine register_bcs
279 
280 end program test_bc_restart
