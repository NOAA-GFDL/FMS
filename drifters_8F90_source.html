23 !> @defgroup drifters_mod drifters_mod
24 !> @ingroup drifters
25 !! @brief <TT>Drifters_mod</TT>is a module designed to advect a set of particles, in parallel or
26 !! sequentially, given an prescribed velocity field.
27 !! @author Alexander Pletzer
28 !!
29 !> Drifters are idealized point particles with positions that evolve in time according
30 !! to a prescribed velocity field, starting from some initial conditions. Drifters have
31 !! no mass, no energy, no size, and no friction and therefore have no impact on the
32 !! dynamics of the underlying system. The only feature that distinguishes a drifter
33 !! from another is its trajectory. This makes drifters ideal for tracking pollution
34 !! clouds and probing fields (e.g. temperature, salinity) along ocean currents, to name
35 !! a few applications.
36 !! Drifters can mimic real experiments such as the Argo floats
37 !! http://www.metoffice.com/research/ocean/argo/ukfloats.html.
38 !!
39 !! When run in parallel, on a 2d decomposed domain, <TT>drifters_mod</TT> will handle all the
40 !! bookkeeping and communication transparently for the user. This involves adding/removing
41 !! drifters as they enter/leave a processor element (PE) domain. Note that the number of drifters
42 !! can vary greatly both between PE domains and within a PE domain in the course of a simulation; the drifters'
43 !! module will also manage dynamically the memory for the user.
44 !!
45 !! There are a number of basic assumptions which could make the drifters' module
46 !! ill-suited for some tasks. First and foremost, it is assumed that the motion of
47 !! drifters is not erratic but follows deterministic trajectories. Furthermore,
48 !! drifters should not cross both compute and data domain boundaries within less
49 !! than a time step. This limitation is imposed by the Runge-Kutta integration
50 !! scheme, which must be able to complete, within a time step, a trajectory
51 !! calculation that starts inside the compute domain and ends inside the data domain. Therefore, the drifters,
52 !! as they are presently modelled, are unlikely to work for very fast objects.
53 !! This constraint also puts a upper limit to the domain decomposition, although
54 !! it can often be remedied by increasing the number of ghost nodes.
55 !!
56 !! Another fundamental assumption is that the (e.g. velocity) fields are structured,
57 !! on a per PE domain basis. There is no support for locally nested or unstrucured
58 !! meshes. Meshes need not be smooth and continuous across PE domains, however.

185  !> @brief Will read positions stored in the netCDF file <TT>input_file</TT>.
186  !!
187  !> The trajectories will be saved in files <TT>output_file.PE</TT>,
188  !! one file per PE domain.
189  subroutine drifters_new(self, input_file, output_file, ermesg)
190 
191  type(drifters_type) :: self !< Opaque data structure.
192  character(len=*), intent(in) :: input_file !< NetCDF input file name containing initial positions.
193  character(len=*), intent(in) :: output_file !< NetCDF output file. Will contain trajectory
194  !! positions and interpolated fields.
195  character(len=*), intent(out) :: ermesg !< Error message (if any).
196 
197  integer nd, nf, npdim, i
198  character(len=6) :: pe_str
199 
200  ermesg = ''
201 
202  self%input_file = input_file
203  self%output_file = output_file
204 
205  call drifters_input_new(self%input, input_file, ermesg)
206  if(ermesg/='') return
207 
208  ! number of dimensions
209  nd = size(self%input%velocity_names)
210  ! estimate for the max number of particles (will resize if exceeded)
211  npdim = int(1.3*size(self%input%positions, 2))
212  call drifters_core_new(self%core, nd=nd, npdim=npdim, ermesg=ermesg)
213  if(ermesg/='') return
214 
215  ! number of fields
216  nf = size(self%input%field_names)
217 
218  ! one output file per PE
219  pe_str = ' '
220  write(pe_str, '(i6)') _mpp_pe
221  pe_str = adjustr(pe_str)
222  do i = 1, 5
223  if(pe_str(i:i)==' ') pe_str(i:i)='0'
224  enddo
225  call drifters_io_new(self%io, output_file//'.'//pe_str, nd, nf, ermesg)
226  if(ermesg/='') return
227 
228  call drifters_comm_new(self%comm)
229  if(ermesg/='') return
230 
231  ! Set meta data
232  call drifters_io_set_time_units(self%io, name=self%input%time_units, &
233  & ermesg=ermesg)
234 
235  call drifters_io_set_position_names(self%io, names=self%input%position_names, &
236  & ermesg=ermesg)
237  if(ermesg/='') return
238  call drifters_io_set_position_units(self%io, names=self%input%position_units, &
239  & ermesg=ermesg)
240  if(ermesg/='') return
241 
242  call drifters_io_set_field_names(self%io, names=self%input%field_names, &
243  & ermesg=ermesg)
244  if(ermesg/='') return
245  call drifters_io_set_field_units(self%io, names=self%input%field_units, &
246  & ermesg=ermesg)
247  if(ermesg/='') return
248 
249  self%dt = -1
250  self%time = -1
251  self%rk4_step = 0
252  self%nx = 0
253  self%ny = 0
254  self%rk4_completed = .false.
255 
256  allocate(self%rk4_k1(self%core%nd, self%core%npdim))
257  self%rk4_k1 = -huge(1.)
258  allocate(self%rk4_k2(self%core%nd, self%core%npdim))
259  self%rk4_k2 = -huge(1.)
260  allocate(self%rk4_k3(self%core%nd, self%core%npdim))
261  self%rk4_k3 = -huge(1.)
262  allocate(self%rk4_k4(self%core%nd, self%core%npdim))
263  self%rk4_k4 = -huge(1.)
264  allocate(self%remove(self%core%npdim))
265  self%remove = .false.
266  allocate(self%temp_pos(nd, self%core%npdim))
267  self%temp_pos = -huge(1.)
268 
269  allocate(self%fields(nf, self%core%npdim))
270  self%fields = -huge(1.)
271 
272  end subroutine drifters_new

392  !! <br>Example usage:
393  !! @code{.F90}
394  !! call drifters_set_domain(self, &
395  !! & xmin_comp, xmax_comp, ymin_comp, ymax_comp, &

401  & xmin_comp, xmax_comp, ymin_comp, ymax_comp, &

403  & xmin_glob, xmax_glob, ymin_glob, ymax_glob, &
404  & ermesg)
405  type(drifters_type) :: self !< Opaque data structure.
406  ! compute domain boundaries
407  real, optional, intent(in) :: xmin_comp !< Min of longitude-like axis on compute domain.
408  real, optional, intent(in) :: xmax_comp !< Max of longitude-like axis on compute domain.

427  if(present(ymax_comp)) self%comm%ycmax = ymax_comp

478  subroutine drifters_modulo(self, positions, ermesg)
479  type(drifters_type) :: self
480  real, intent(inout) :: positions(:,:)
481  character(len=*), intent(out) :: ermesg
482 
483  integer ip, np
484  real x, y
485 
486  ermesg = ''
487  np = self%core%np
488 
489  if(self%comm%xperiodic) then
490  do ip = 1, np
491  x = positions(1, ip)
492  positions(1, ip) = self%comm%xgmin + &
493  & modulo(x - self%comm%xgmin, self%comm%xgmax-self%comm%xgmin)
494  enddo
495  endif
496 
497  if(self%comm%yperiodic) then
498  do ip = 1, np
499  y = positions(2, ip)
500  positions(2, ip) = self%comm%ygmin + &
501  & modulo(y - self%comm%ygmin, self%comm%ygmax-self%comm%ygmin)
502  enddo
503  endif
504 
505  end subroutine drifters_modulo

768  !============================================================================
769  !> @brief Interpolates positions onto longitude/latitude grid.
770  !! @details In many cases, the integrated positions will not be longitudes or latitudes. This call
771  !! can be ionvoked to recover the longitude/latitude positions from the "logical" positions.
772  subroutine drifters_positions2lonlat(self, positions, &
773  & x1, y1, geolon1, &
774  & x2, y2, geolat2, &
775  & lons, lats, &
776  & ermesg)
777 
778  type(drifters_type) :: self !< Opaque data structure.
779  ! Input positions
780  real, intent(in) :: positions(:,:) !< Logical positions.
781  ! Input mesh
782  real, intent(in) :: x1(:) !< X-axis of "geolon1" field.
783  real, intent(in) :: y1(:) !< Y-axis of "geolon1" field.
784  real, intent(in) :: geolon1(:,:) !< Y-axis of "geolon1" field.
785  real, intent(in) :: x2(:) !< X-axis of "geolat2" field.
786  real, intent(in) :: y2(:) !< Y-axis of "geolat2" field.
787  real, intent(in) :: geolat2(:,:) !< Latitude field as an array of (x2, y2)
788  ! Output lon/lat
789  real, intent(out) :: lons(:) !< Returned longitudes.
790  real, intent(out) :: lats(:) !< Returned latitudes.
791  character(len=*), intent(out) :: ermesg !< Error message (if any).
792 
793  real fvals(2**self%core%nd), ts(self%core%nd)
794  integer np, ij(2), ip, ier, n1s(2), n2s(2), i, j, iertot
795  character(len=10) :: n1_str, n2_str, np_str, iertot_str
796 
797  ermesg = ''
798  lons = -huge(1.)
799  lats = -huge(1.)
800 
801  ! check dimensions
802  n1s = (/size(x1), size(y1)/)
803  n2s = (/size(x2), size(y2)/)
804  if(n1s(1) /= size(geolon1, 1) .or. n1s(2) /= size(geolon1, 2)) then
805  ermesg = 'drifters_positions2geolonlat: ERROR incompatibles dims between (x1, y1, geolon1)'
806  return
807  endif
808  if(n2s(1) /= size(geolat2, 1) .or. n2s(2) /= size(geolat2, 2)) then
809  ermesg = 'drifters_positions2geolonlat: ERROR incompatibles dims between (x2, y2, geolat2)'
810  return
811  endif
812 
813  np = size(positions, 2)
814  if(size(lons) < np .or. size(lats) < np) then
815  write(np_str, '(i10)') np
816  write(n1_str, '(i10)') size(lons)
817  write(n2_str, '(i10)') size(lats)
818  ermesg = 'drifters_positions2geolonlat: ERROR size of "lons" ('//trim(n1_str)// &
819  & ') or "lats" ('//trim(n2_str)//') < '//trim(np_str)
820  return
821  endif
822 
823  ! Interpolate
824  iertot = 0
825  do ip = 1, np
826 
827  ! get longitude
828  call cld_ntrp_locate_cell(x1, positions(1,ip), i, ier)
829  iertot = iertot + ier
830  call cld_ntrp_locate_cell(y1, positions(2,ip), j, ier)
831  iertot = iertot + ier
832  ij(1) = i; ij(2) = j;
833  call cld_ntrp_get_cell_values(n1s, _flatten(geolon1), ij, fvals, ier)
834  iertot = iertot + ier
835  ts(1) = (positions(1,ip) - x1(i))/(x1(i+1) - x1(i))
836  ts(2) = (positions(2,ip) - y1(j))/(y1(j+1) - y1(j))
837  call cld_ntrp_linear_cell_interp(fvals, ts, lons(ip), ier)
838  iertot = iertot + ier
839 
840  ! get latitude
841  call cld_ntrp_locate_cell(x2, positions(1,ip), i, ier)
842  iertot = iertot + ier
843  call cld_ntrp_locate_cell(y2, positions(2,ip), j, ier)
844  iertot = iertot + ier
845  ij(1) = i; ij(2) = j;
846  call cld_ntrp_get_cell_values(n2s, _flatten(geolat2), ij, fvals, ier)
847  iertot = iertot + ier
848  ts(1) = (positions(1,ip) - x2(i))/(x2(i+1) - x2(i))
849  ts(2) = (positions(2,ip) - y2(j))/(y2(j+1) - y2(j))
850  call cld_ntrp_linear_cell_interp(fvals, ts, lats(ip), ier)
851  iertot = iertot + ier
852 
853  enddo
854 
855  if(iertot /= 0) then
856  write(iertot_str, '(i10)') iertot
857  ermesg = 'drifters_positions2geolonlat: ERROR '//trim(iertot_str)// &
858  & ' interpolation errors (domain out of bounds?)'
859  endif
860 
861  end subroutine drifters_positions2lonlat
