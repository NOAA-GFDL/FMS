55 ! <INTERFACE NAME="time_interp">
56 
57 ! <OVERVIEW>
58 ! Returns a weight and dates or indices for interpolating between two dates. The
59 ! interface fraction_of_year is provided for backward compatibility with the
60 ! previous version.
61 ! </OVERVIEW>
62 ! <DESCRIPTION>
63 ! Returns weight by interpolating Time between Time1 and Time2.
64 ! i.e. weight = (Time-Time1)/(Time2-Time1)
65 ! Time1 and Time2 may be specified by any of several different ways,
66 ! which is the reason for multiple interfaces.
67 
68 ! If Time1 and Time2 are the begining and end of the year in which
69 ! Time falls, use first interface.
70 
71 ! If Time1 and Time2 fall on year boundaries, use second interface.
72 
73 ! If Time1 and Time2 fall on month boundaries, use third.
74 
75 ! If Time1 and Time2 fall on day boundaries, use fourth.
76 
77 ! If Time1 and Time2 are consecutive elements of an assending list, use fifth.
78 ! The fifth also returns the indices of Timelist between which Time falls.
79 
80 ! The sixth interface is for cyclical data. Time_beg and Time_end specify the
81 ! begining and end of a repeating period. In this case:
82 ! weight = (Time_adjusted - Time1) / (Time2 - Time1)
83 ! Where:
84 ! Time1 = Timelist(index1)
85 ! Time2 = Timelist(index2)
86 ! Time_adjusted = Time - N*Period
87 ! Period = Time_end-Time_beg
88 ! N is between (Time-Time_end)/Period and (Time-Time_beg)/Period
89 ! That is, N is the integer that results in Time_adjusted that is between Time_beg and Time_end.
90 !
91 ! </DESCRIPTION>
92 ! <TEMPLATE>
93 ! 1. call time_interp( Time, weight )
94 ! </TEMPLATE>
95 ! <TEMPLATE>
96 ! 2. call time_interp( Time, weight, year1, year2 )
97 ! </TEMPLATE>
98 ! <TEMPLATE>
99 ! 3. call time_interp( Time, weight, year1, year2, month1, month2 )
100 ! </TEMPLATE>
101 ! <TEMPLATE>
102 ! 4. call time_interp( Time, weight, year1, year2, month1, month2, day1, day2 )
103 ! </TEMPLATE>
104 ! <TEMPLATE>
105 ! 5. call time_interp( Time, Timelist, weight, index1, index2 [, modtime] )
106 ! </TEMPLATE>
107 ! <TEMPLATE>
108 ! 6. call time_interp( Time, Time_beg, Time_end, Timelist, weight, index1, index2 [,correct_leap_year_inconsistency])
109 ! </TEMPLATE>
110 ! <IN NAME="Time">
111 ! The time at which the the weight is computed.
112 ! </IN>
113 ! <IN NAME="Time_beg">
114 ! For cyclical interpolation: Time_beg specifies the begining time of a cycle.
115 ! </IN>
116 ! <IN NAME="Time_end">
117 ! For cyclical interpolation: Time_end specifies the ending time of a cycle.
118 ! </IN>
119 ! <IN NAME="Timelist">
120 ! For cyclical interpolation: Timelist is an array of times between Time_beg and Time_end.
121 ! Must be monotonically increasing.
122 ! </IN>
123 ! <IN NAME="modtime">
124 ! </IN>
125 ! <IN NAME="index1">
126 ! Timelist(index1) = The largest value of Timelist which is less than mod(Time,Time_end-Time_beg)
127 ! </IN>
128 ! <IN NAME="index2">
129 ! Timelist(index2) = The smallest value of Timelist which is greater than mod(Time,Time_end-Time_beg)
130 ! </IN>
131 ! <IN NAME="correct_leap_year_inconsistency">
132 ! Turns on a kluge for an inconsistency which may occur in a special case.
133 ! When the modulo time period (i.e. Time_end - Time_beg) is a whole number of years
134 ! and is not a multiple of 4, and the calendar in use has leap years, then it is
135 ! likely that the interpolation will involve mapping a common year onto a leap year.
136 ! In this case it is often desirable, but not absolutely necessary, to use data for
137 ! Feb 28 of the leap year when it is mapped onto a common year.
138 ! To turn this on, set correct_leap_year_inconsistency=.true.
139 ! </IN>
140 ! <OUT NAME="weight">
141 ! weight = (mod(Time,Time_end-Time_beg) - Timelist(index1)) / (Timelist(index2) - Timelist(index1))
142 ! </OUT>
143 ! <OUT NAME="year1"> </OUT>
144 ! <OUT NAME="year2"> </OUT>
145 ! <OUT NAME="month1"> </OUT>
146 ! <OUT NAME="month2"> </OUT>
147 ! <OUT NAME="day1"> </OUT>
148 ! <OUT NAME="day2"> </OUT>
149 ! <OUT NAME="index1"> </OUT>
150 ! <OUT NAME="index2"> </OUT>
151 ! <ERROR MSG="input time list not ascending order" STATUS="ERROR">
152 ! The list of input time types must have ascending dates.
153 ! </ERROR>
154 ! <ERROR MSG="modulo months must have same length" STATUS="ERROR">
155 ! The length of the current month for input Time and Time_list
156 ! must be the same when using the modulo month option. The
157 ! modulo month option is available but not supported.
158 ! </ERROR>
159 ! <ERROR MSG="invalid value for argument modtime" STATUS="ERROR">
160 ! The optional argument modtime must have a value set by one
161 ! of the public parameters: NONE, YEAR, MONTH, DAY. The
162 ! MONTH and DAY options are available but not supported.
163 ! </ERROR>
164 ! <ERROR MSG="period of list exceeds modulo period" STATUS="ERROR">
165 ! The difference between the last and first values in the input
166 ! Time list/array exceeds the length of the modulo period.
167 ! </ERROR>
168 ! <ERROR MSG="time before range of list or time after range of list" STATUS="ERROR">
169 ! The difference between the last and first values in the input
170 ! These errors occur when you are not using a modulo axis and
171 ! the input Time occurs before the first value in the Time
172 ! list/array or after the last value in the Time list/array.
173 ! </ERROR>
174 ! <NOTE>
175 ! Examples:
176 ! <PRE>
177 ! Time: Jan 01 00z weight = 0.0
178 ! Time: Jul 01 weight ~ 0.5
179 ! Time: Dec 31 23z weight ~ 1.0
180 ! </PRE>
181 ! </NOTE>
182 
183 !> Returns a weight and dates or indices for interpolating between two dates. The
184 !! interface fraction_of_year is provided for backward compatibility with the
185 !! previous version.\n
186 !!
187 !! Returns weight by interpolating Time between Time1 and Time2.
188 !! i.e. weight = (Time-Time1)/(Time2-Time1)
189 !! Time1 and Time2 may be specified by any of several different ways,
190 !! which is the reason for multiple interfaces.\n
191 !!
192 !! - If Time1 and Time2 are the begining and end of the year in which
193 !! Time falls, use first interface.\n
194 !!
195 !! - If Time1 and Time2 fall on year boundaries, use second interface.\n
196 !!
197 !! - If Time1 and Time2 fall on month boundaries, use third.\n
198 !!
199 !! - If Time1 and Time2 fall on day boundaries, use fourth.\n
200 !!
201 !! - If Time1 and Time2 are consecutive elements of an assending list, use fifth.
202 !! The fifth also returns the indices of Timelist between which Time falls.\n
203 !!
204 !! - The sixth interface is for cyclical data. Time_beg and Time_end specify the

206 !! weight = (Time_adjusted - Time1) / (Time2 - Time1)
207 !! <br>Where:
208 !! @code{.F90}
209 !! Time1 = Timelist(index1)
210 !! Time2 = Timelist(index2)
211 !! Time_adjusted = Time - N*Period
212 !! Period = Time_end-Time_beg
213 !! @endcode
214 !! N is between (Time-Time_end)/Period and (Time-Time_beg)/Period
215 !! That is, N is the integer that results in Time_adjusted that is between Time_beg and Time_end.
216 !!
217 !! <br>Example usages:
218 !! @code{.F90}
219 !! call time_interp( Time, weight )
220 !! call time_interp( Time, weight, year1, year2 )

222 !! call time_interp( Time, weight, year1, year2, month1, month2, day1, day2 )
223 !! call time_interp( Time, Timelist, weight, index1, index2 [, modtime] )

326 ! <SUBROUTINE NAME="time_interp_year" INTERFACE="time_interp">
327 ! <IN NAME="Time" TYPE="time_type" > </IN>
328 ! <OUT NAME="weight" TYPE="real"> </OUT>
329 ! <OUT NAME="year1" TYPE="integer"> </OUT>
330 ! <OUT NAME="year2" TYPE="integer"> </OUT>
331 ! </SUBROUTINE>
332 ! returns fractional time between mid points of consecutive years
333 
334  !> @brief Calculates fractional time between mid points of consecutive years
335  subroutine time_interp_year ( Time, weight, year1, year2 )
336 
337  type(time_type), intent(in) :: Time
338  real , intent(out) :: weight !< fractional time between midpoints of year1 and year2
339  integer , intent(out) :: year1, year2
340 
341  integer :: year, month, day, hour, minute, second
342  type (time_type) :: Mid_year, Mid_year1, Mid_year2
343 
344 
345  if ( .not. module_is_initialized ) call time_interp_init()
346 
347  call get_date (time, year, month, day, hour, minute, second)
348 
349  ! mid point of current year
350  mid_year = year_midpt(year)
351 
352  if ( time >= mid_year ) then
353  ! current time is after mid point of current year
354  year1 = year
355  year2 = year+1
356  mid_year2 = year_midpt(year2)
357  weight = real( (time - mid_year) // (mid_year2 - mid_year) )
358  else
359  ! current time is before mid point of current year
360  year2 = year
361  year1 = year-1
362  mid_year1 = year_midpt(year1)
363  weight = real( (time - mid_year1) // (mid_year - mid_year1) )
364  endif
365 
366  end subroutine time_interp_year

368 !#######################################################################
369 ! <SUBROUTINE NAME="time_interp_month" INTERFACE="time_interp">
370 ! <IN NAME="Time" TYPE="time_type" > </IN>
371 ! <OUT NAME="weight" TYPE="real"> </OUT>
372 ! <OUT NAME="year1" TYPE="integer"> </OUT>
373 ! <OUT NAME="year2" TYPE="integer"> </OUT>
374 ! <OUT NAME="month1" TYPE="integer"> </OUT>
375 ! <OUT NAME="month2" TYPE="integer"> </OUT>
376 ! </SUBROUTINE>
377  !> @brief Calculates fractional time between mid points of consecutive months
378  subroutine time_interp_month ( Time, weight, year1, year2, month1, month2 )
379 
380  type(time_type), intent(in) :: Time
381  real , intent(out) :: weight
382  integer , intent(out) :: year1, year2, month1, month2
383 
384  integer :: year, month, day, hour, minute, second, &
385  mid_month, cur_month, mid1, mid2
386 
387  if ( .not. module_is_initialized ) call time_interp_init()
388 
389  call get_date (time, year, month, day, hour, minute, second)
390 
391  ! mid point of current month in seconds
392  mid_month = days_in_month(time) * halfday
393  ! time into current month in seconds
394  cur_month = second + secmin*minute + sechour*hour + secday*(day-1)
395 
396  if ( cur_month >= mid_month ) then
397  ! current time is after mid point of current month
398  year1 = year; month1 = month
399  year2 = year; month2 = month+1
400  if (month2 > monyear) then
401  year2 = year2+1; month2 = 1
402  endif
403  mid1 = mid_month
404  mid2 = days_in_month(set_date(year2,month2,2)) * halfday
405  weight = real(cur_month - mid1) / real(mid1+mid2)
406  else
407  ! current time is before mid point of current month
408  year2 = year; month2 = month
409  year1 = year; month1 = month-1
410  if (month1 < 1) then
411  year1 = year1-1; month1 = monyear
412  endif
413  if (year1>0) then
414  mid1 = days_in_month(set_date(year1,month1,2)) * halfday
415  else
416  ! this can happen if we are at the beginning of year 1. In this case
417  ! use December 0001 to calculate the duration of December 0000.
418  ! This should work for all calendars
419  mid1 = days_in_month(set_date(1,month1,2)) * halfday
420  endif
421  mid2 = mid_month
422  weight = real(cur_month + mid1) / real(mid1+mid2)
423  endif
424 
425  end subroutine time_interp_month

427 !#######################################################################
428 ! <SUBROUTINE NAME="time_interp_day" INTERFACE="time_interp">
429 ! <IN NAME="Time" TYPE="time_type" > </IN>
430 ! <OUT NAME="weight" TYPE="real"> </OUT>
431 ! <OUT NAME="year1" TYPE="integer"> </OUT>
432 ! <OUT NAME="year2" TYPE="integer"> </OUT>
433 ! <OUT NAME="month1" TYPE="integer"> </OUT>
434 ! <OUT NAME="month2" TYPE="integer"> </OUT>
435 ! <OUT NAME="day1" TYPE="integer"> </OUT>
436 ! <OUT NAME="day2" TYPE="integer"> </OUT>
437 ! </SUBROUTINE>
438  !> @brief Calculates fractional time between mid points of consecutive days
439  subroutine time_interp_day ( Time, weight, year1, year2, month1, month2, day1, day2 )
440 
441  type(time_type), intent(in) :: Time
442  real , intent(out) :: weight
443  integer , intent(out) :: year1, year2, month1, month2, day1, day2
444 
445  integer :: year, month, day, hour, minute, second, sday
446 
447  if ( .not. module_is_initialized ) call time_interp_init()
448 
449  call get_date (time, year, month, day, hour, minute, second)
450 
451  ! time into current day in seconds
452  sday = second + secmin*minute + sechour*hour
453 
454  if ( sday >= halfday ) then
455  ! current time is after mid point of day
456  year1 = year; month1 = month; day1 = day
457  year2 = year; month2 = month; day2 = day + 1
458  weight = real(sday - halfday) / real(secday)
459 
460  if (day2 > days_in_month(time)) then
461  month2 = month2 + 1
462  day2 = 1
463  if (month2 > monyear) then
464  month2 = 1; year2 = year2+1
465  endif
466  endif
467  else
468  ! current time is before mid point of day
469  year2 = year; month2 = month; day2 = day
470  year1 = year; month1 = month; day1 = day - 1
471  weight = real(sday + halfday) / real(secday)
472 
473  if (day1 < 1) then
474  month1 = month1 - 1
475  if (month1 < 1) then
476  month1 = monyear; year1 = year1-1
477  endif
478  day1 = days_in_month(set_date(year1,month1,2))
479  endif
480  endif
481 
482  end subroutine time_interp_day

484 !#######################################################################
485 ! <SUBROUTINE NAME="time_interp_modulo" INTERFACE="time_interp">
486 ! <IN NAME="Time" TYPE="time_type" > </IN>
487 ! <IN NAME="Time_beg" TYPE="time_type"> </IN>
488 ! <IN NAME="Time_end" TYPE="time_type"> </IN>
489 ! <IN NAME="Timelist" TYPE="time_type" DIM="(:)"> </IN>
490 ! <IN NAME="correct_leap_year_inconsistency" TYPE="logical, optional" DEFAULT=".false.">
491 ! Turns on a kluge for an inconsistency which may occur in a special case.
492 ! When the modulo time period (i.e. Time_end - Time_beg) is a whole number of years
493 ! and is not a multiple of 4, and the calendar in use has leap years, then it is
494 ! likely that the interpolation will involve mapping a common year onto a leap year.
495 ! In this case it is often desirable, but not absolutely necessary, to use data for
496 ! Feb 28 of the leap year when it is mapped onto a common year.
497 ! To turn this on, set correct_leap_year_inconsistency=.true. </IN>
498 ! <OUT NAME="weight" TYPE="real"> </OUT>
499 ! <OUT NAME="index1" TYPE="real"> </OUT>
500 ! <OUT NAME="index2" TYPE="real"> </OUT>
501 ! </SUBROUTINE>
502 
503 subroutine time_interp_modulo(Time, Time_beg, Time_end, Timelist, weight, index1, index2, &
504  correct_leap_year_inconsistency, err_msg)
505 type(time_type), intent(in) :: Time, Time_beg, Time_end, Timelist(:)
506 real , intent(out) :: weight
507 integer , intent(out) :: index1, index2
508 logical, intent(in), optional :: correct_leap_year_inconsistency!< When true turns on a kluge for an
509  !! inconsistency which may occur in a special case.
510  !! When the modulo time period (i.e. Time_end - Time_beg) is a
511  !! whole number of years and is not a multiple of 4, and the calendar
512  !! in use has leap years, then it is likely that the interpolation
513  !! will involve mapping a common year onto a leap year. In this case
514  !! it is often desirable, but not absolutely necessary, to use data
515  !! for Feb 28 of the leap year when it is mapped onto a common year.
516 character(len=*), intent(out), optional :: err_msg
517 
518  type(time_type) :: Period, T
519  integer :: is, ie,i1,i2
520  integer :: ys,ms,ds,hs,mins,ss ! components of the starting date
521  integer :: ye,me,de,he,mine,se ! components of the ending date
522  integer :: yt,mt,dt,ht,mint,st ! components of the current date
523  integer :: dt1 ! temporary value for day
524  integer :: n ! size of Timelist
525  integer :: stdoutunit
526  logical :: correct_lyr, calendar_has_leap_years, do_the_lyr_correction
527 
528  if ( .not. module_is_initialized ) call time_interp_init
529  if( present(err_msg) ) err_msg = ''
530 
531  stdoutunit = stdout()
532  n = size(timelist)
533 
534  if (time_beg>=time_end) then
535  if(fms_error_handler('time_interp_modulo', &
536  'end of the specified time loop interval must be later than its beginning',err_msg)) return
537  endif
538 
539  calendar_has_leap_years = (get_calendar_type() == julian .or. get_calendar_type() == gregorian)
540 
541  period = time_end-time_beg ! period of the time axis
542 
543  if(present(correct_leap_year_inconsistency)) then
544  correct_lyr = correct_leap_year_inconsistency
545  else
546  correct_lyr = .false.
547  endif
548 
549  ! bring the requested time inside the specified time period
550  t = time
551 
552  do_the_lyr_correction = .false.
553 
554  ! Determine if the leap year correction needs to be done.
555  ! It never needs to be done unless 3 conditions are met:
556  ! 1) We are using a calendar with leap years
557  ! 2) optional argument correct_leap_year_inconsistency is present and equals .true.
558  ! 3) The modulo time period is an integer number of years
559  ! If all of these are true then set do_the_lyr_correction to .true.
560 
561  if(calendar_has_leap_years .and. correct_lyr) then
562  call get_date(time_beg,ys,ms,ds,hs,mins,ss)
563  call get_date(time_end,ye,me,de,he,mine,se)
564  if(ms==me.and.ds==de.and.hs==he.and.mins==mine.and.ss==se) then
565  ! whole number of years
566  do_the_lyr_correction = .true.
567  endif

579  ! but before the starting date. So we shift it to the last year.

589  enddo
590  endif

592  ! find indices of the first and last records in the Timelist that are within
593  ! the requested time period.
594  if (time_end<=timelist(1).or.time_beg>=timelist(n)) then

596  call print_time(time_beg, 'Time_beg' )
597  call print_time(time_end, 'Time_end' )
598  call print_time(timelist(1), 'Timelist(1)' )
599  call print_time(timelist(n), 'Timelist(n)' )
600  else
601  call print_date(time_beg, 'Time_beg' )
602  call print_date(time_end, 'Time_end' )
603  call print_date(timelist(1), 'Timelist(1)' )
604  call print_date(timelist(n), 'Timelist(n)' )
605  endif
606  write(stdoutunit,*)'where n = size(Timelist) =',n
607  if(fms_error_handler('time_interp_modulo', &
608  'the entire time list is outside the specified time loop interval',err_msg)) return
609  endif
610 
611  call bisect(timelist,time_beg,index1=i1,index2=i2)
612  if (i1 < 1) then
613  is = 1 ! Time_beg before lower boundary
614  else if (time_beg == timelist(i1)) then
615  is = i1 ! Time_beg right on the lower boundary
616  else
617  is = i2 ! Time_beg inside the interval or on upper boundary
618  endif
619  call bisect(timelist,time_end,index1=i1,index2=i2)
620  if (time_end > timelist(i1)) then
621  ie = i1
622  else if (time_end == timelist(i1)) then
623  if(time_beg == timelist(is)) then
624  ! Timelist includes time levels at both the lower and upper ends of the period.
625  ! The endpoints of Timelist specify the same point in the cycle.
626  ! This ambiguity is resolved by ignoring the last time level.
627  ie = i1-1
628  else
629  ie = i1
630  endif
631  else
632 ! This should never happen because bisect does not return i1 such that Time_end < Timelist(i1)
633  endif
634  if (is>=ie) then
635  if(get_calendar_type() == no_calendar) then
636  call print_time(time_beg, 'Time_beg =')
637  call print_time(time_end, 'Time_end =')
638  call print_time(timelist(1), 'Timelist(1)=')
639  call print_time(timelist(n), 'Timelist(n)=')
640  else
641  call print_date(time_beg, 'Time_beg =')
642  call print_date(time_end, 'Time_end =')
643  call print_date(timelist(1), 'Timelist(1)=')
644  call print_date(timelist(n), 'Timelist(n)=')
645  endif
646  write(stdoutunit,*)'where n = size(Timelist) =',n
647  write(stdoutunit,*)'is =',is,'ie =',ie
648  if(fms_error_handler('time_interp_modulo', &
649  'error in calculation of time list bounds within the specified time loop interval',err_msg)) return
650  endif
651 
652  ! handle special cases:
653  if( t>=timelist(ie) ) then
654  ! time is after the end of the portion of the time list within the requested period
655  index1 = ie; index2 = is
656  weight = real( (t-timelist(ie))//(period-(timelist(ie)-timelist(is))) )
657  else if (t<timelist(is)) then
658  ! time is before the beginning of the portion of the time list within the requested period
659  index1 = ie; index2 = is
660  weight = real( 1.0-((timelist(is)-t)//(period-(timelist(ie)-timelist(is)))) )
661  else
662  call bisect(timelist,t,index1,index2)
663  weight = real( (t-timelist(index1)) // (timelist(index2)-timelist(index1)) )
664  endif
665 
666 end subroutine time_interp_modulo

669 !> Given an array of times in ascending order and a specific time returns
670 !! values of index1 and index2 such that the Timelist(index1)<=Time and
671 !! Time<=Timelist(index2), and index2=index1+1
672 !! index1=0, index2=1 or index=n, index2=n+1 are returned to indicate that
673 !! the time is out of range
674 subroutine bisect(Timelist,Time,index1,index2)
675  type(time_type) , intent(in) :: Timelist(:)
676  type(time_type) , intent(in) :: Time
677  integer, optional, intent(out) :: index1, index2
678 
679  integer :: i,il,iu,n,i1,i2
680 
681  n = size(timelist(:))
682 
683  if (time==timelist(1)) then
684  i1 = 1 ; i2 = 2
685  else if (time==timelist(n)) then
686  i1 = n ; i2 = n+1
687  else
688  il = 0; iu=n+1
689  do while(iu-il > 1)
690  i = (iu+il)/2
691  if(timelist(i) > time) then
692  iu = i
693  else
694  il = i
695  endif
696  enddo
697  i1 = il ; i2 = il+1
698  endif
699 
700  if(PRESENT(index1)) index1 = i1
701  if(PRESENT(index2)) index2 = i2

798  endif
799  td = te-ts

801  index1 = n
802  index2 = 1
803 
804 ! time falls on ending list value
805  else if ( t == te ) then
806  if(perthlike_behavior) then
807  weight = 1.0
808  index1 = n-1
809  index2 = n
810  else
811  weight = 0.
812  index1 = n
813  if (mtime == none) then
814  index2 = n
815  else

817  endif
818  endif
819 
820 ! time falls after ending list value
821  else if ( t > te ) then
822  if (mtime == none) then
823  call time_list_error(t,terr)
824  call time_list_error(ts,tserr)

829  err_msg)) return
830  deallocate(terr,tserr,teerr)
831  endif
832  td = te-ts
833  weight = real( (t-te) // (period-td) )
834  index1 = n
835  index2 = 1
836  endif

838 end subroutine time_interp_list
