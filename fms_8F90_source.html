21 !! @brief The fms module provides routines that are commonly used
22 !! by most FMS modules.
23 !> @author Bruce Wyman
24 !!
25 !> Here is a summary of the functions performed by routines
26 !! in the fms module.
27 !!
28 !! 1. Output module version numbers to a common (<TT>log</TT>) file
29 !! using a common format.<BR/>
30 !! 2. Open specific types of files common to many FMS modules.
31 !! These include namelist files, restart files, and 32-bit IEEE
32 !! data files. There also is a matching interface to close the files.
33 !! If other file types are needed the <TT>mpp_open</TT> and <TT>mpp_close</TT>
34 !! interfaces in module @ref mpp_io_mod must be used.<BR/>
35 !! 3. Read and write distributed data to simple native unformatted files.
36 !! This type of file (called a restart file) is used to checkpoint
37 !! model integrations for a subsequent restart of the run.<BR/>
38 !! 4. For convenience there are several routines published from
39 !! the @ref mpp module. These are routines for getting processor
40 !! numbers, commonly used I/O unit numbers, error handling, and timing sections of code.

301 !> @brief Initializes the FMS module and also calls the initialization routines for all
302 !! modules in the MPP package. Will be called automatically if the user does
303 !! not call it.
304 !! @details Initialization routine for the fms module. It also calls initialization routines
305 !! for the mpp, mpp_domains, and mpp_io modules. Although this routine
306 !! will be called automatically by other fms_mod routines, users should
307 !! explicitly call fms_init. If this routine is called more than once it will
308 !! return silently. There are no arguments.
309 !!
310 !> @throws FATAL, invalid entry for namelist variable warning_level
311 !! The namelist variable warning_level must be either 'fatal' or 'warning'(case-insensitive)
312 !!
313 !> @throws FATAL, invalid entry for namelist variable clock_grain
314 !! The namelist variable clock_grain must be one of the following values:
315 !! 'NONE', 'COMPONENT', 'SUBCOMPONENT', 'MODULE_DRIVER', 'MODULE', 'ROUTINE',
316 !! 'LOOP', or 'INFRA' (case-insensitive).
317 subroutine fms_init (localcomm, alt_input_nml_path)
318 
319 !--- needed to output the version number of constants_mod to the logfile ---
320  use constants_mod, only: constants_version=>version !pjp: PI not computed
321  use fms_io_mod, only: fms_io_version
322 
323  integer, intent(in), optional :: localcomm
324  character(len=*), intent(in), optional :: alt_input_nml_path
325  integer :: ierr, io
326  integer :: logunitnum
327  integer :: stdout_unit !< Unit number for the stdout file
328 
329  if (module_is_initialized) return ! return silently if already called
330  module_is_initialized = .true.
331 !---- initialize mpp routines ----
332  if(present(localcomm)) then
333  if(present(alt_input_nml_path)) then
334  call mpp_init(localcomm=localcomm, alt_input_nml_path=alt_input_nml_path)
335  else
336  call mpp_init(localcomm=localcomm)
337  endif
338  else
339  if(present(alt_input_nml_path)) then
340  call mpp_init(alt_input_nml_path=alt_input_nml_path)
341  else
342  call mpp_init()
343  endif
344  endif
345  call mpp_domains_init()
346  call fms_io_init()
347  !! write_version_number is inaccesible from fms_io_mod so write it from here if not written
348  if(.not.fms_io_initialized) then
349  call write_version_number("FMS_IO_MOD", fms_io_version)
350  fms_io_initialized = .true.
351  endif
352  call fms2_io_init()
353  logunitnum = stdlog()
354 !---- read namelist input ----
355 
356  call nml_error_init() ! first initialize namelist iostat error codes
357 
358  read (input_nml_file, fms_nml, iostat=io)
359  ierr = check_nml_error(io,'fms_nml')
360 
361 !---- define mpp stack sizes if non-zero -----
362 
363  if ( stack_size > 0) call mpp_set_stack_size ( stack_size)
364  if (domains_stack_size > 0) call mpp_domains_set_stack_size (domains_stack_size)
365 
366 !---- set severity level for warnings ----
367 
368  select case( trim(lowercase(warning_level)) )
369  case( 'fatal' )
370  call mpp_set_warn_level ( fatal )
371  case( 'warning' )
372  call mpp_set_warn_level ( warning )
373  case default
374  call error_mesg ( 'fms_init', &
375  'invalid entry for namelist variable warning_level', fatal )
376  end select
377 
378 !--- set granularity for timing code sections ---
379 
380  select case( trim(uppercase(clock_grain)) )
381  case( 'NONE' )
382  call mpp_clock_set_grain (0)
383  case( 'COMPONENT' )
384  call mpp_clock_set_grain (clock_component)
385  case( 'SUBCOMPONENT' )
386  call mpp_clock_set_grain (clock_subcomponent)
387  case( 'MODULE_DRIVER' )
388  call mpp_clock_set_grain (clock_module_driver)
389  case( 'MODULE' )
390  call mpp_clock_set_grain (clock_module)
391  case( 'ROUTINE' )
392  call mpp_clock_set_grain (clock_routine)
393  case( 'LOOP' )
394  call mpp_clock_set_grain (clock_loop)
395  case( 'INFRA' )
396  call mpp_clock_set_grain (clock_infra)
397  case default
398  call error_mesg ( 'fms_init', &
399  'invalid entry for namelist variable clock_grain', fatal )
400  end select
401 !Balaji
402  select case( trim(uppercase(clock_flags)) )
403  case( 'NONE' )
404  clock_flag_default = 0
405  case( 'SYNC' )
406  clock_flag_default = mpp_clock_sync
407  case( 'DETAILED' )
408  clock_flag_default = mpp_clock_detailed
409  case default
410  call error_mesg ( 'fms_init', &
411  'invalid entry for namelist variable clock_flags', fatal )
412  end select
413 
414 !--- write version info and namelist to logfile ---
415 
416  call write_version_number("FMS_MOD", version)
417  if (mpp_pe() == mpp_root_pe()) then
418  stdout_unit = stdlog()
419  write (stdout_unit, nml=fms_nml)
420  write (stdout_unit,*) 'nml_error_codes=', nml_error_codes(1:num_nml_error_codes)
421  endif
422 
423  call memutils_init( print_memory_usage )
424  call print_memuse_stats('fms_init')
425 
426 !--- output version information constants to the logfile
427  call write_version_number("CONSTANTS_MOD", constants_version)
428  call grid_init
429 
430 end subroutine fms_init

466  subroutine error_mesg (routine, message, level)
467  character(len=*), intent(in) :: routine !< Routine name where the warning or error has occurred.

470  !! for FATAL, never for NOTE, and is settable for WARNING (see namelist).
471

473 ! routine name of the calling routine (character string)
474 ! message message written to output (character string)
475 ! level set to NOTE, MESSAGE, or FATAL (integer)
476 
477  if (.not.module_is_initialized) call fms_init ( )

484  !> @brief Facilitates the control of fatal error conditions
485  !! @details When err_msg is present, message is copied into err_msg
486  !! and the function returns a value of .true.
487  !! Otherwise calls mpp_error to terminate execution.
488  !! The intended use is as shown below.
489  !! @returns true when err_msg is present
490  !! @code{.F90}
491  !! if(fms_error_handler(routine, message, err_msg)) return
492  !! @endcode
493  function fms_error_handler(routine, message, err_msg)
494 
495  logical :: fms_error_handler
496  character(len=*), intent(in) :: routine !< Routine name where the fatal error has occurred.
497  character(len=*), intent(in) :: message !< fatal error message to be printed.
498  character(len=*), intent(out), optional :: err_msg !< When err_msg is present: err_msg = message
499 
500  fms_error_handler = .false.
501  if(present(err_msg)) then
502  err_msg = message
503  fms_error_handler = .true.
504  else
505  call mpp_error(trim(routine),trim(message),fatal)
506  endif
507 
508  end function fms_error_handler

510 ! used to check the iostat argument that is
511 ! returned after reading a namelist
512 ! see the online documentation for how this routine might be used
513 
514  !> @brief Checks the iostat argument that is returned after reading a namelist
515  !! and determines if the error code is valid.
516  !! @return This function returns the input iostat value (integer) if it is an
517  !! allowable error code. If the iostat error code is not
518  !! allowable, an error message is printed and the program terminated.
519  !! @details The FMS allows multiple namelist records to reside in the same file.
520  !! Use this interface to check the iostat argument that is returned after
521  !! reading a record from the namelist file. If an invalid iostat value
522  !! is detected this routine will produce a fatal error. See the NOTE below.
523  !!
524  !! Some compilers will return non-zero iostat values when reading through
525  !! files with multiple namelist. This routine
526  !! will try skip these errors and only terminate for true namelist errors.
527  !!
528  !! <br>Examples<br>
529  !!
530  !! The following example checks if a file exists, reads a namelist input
531  !! from that file, and checks for errors in that
532  !! namelist. When the correct namelist is read and it has no errors the
533  !! routine check_nml_error will return zero and the while loop will exit.
534  !! This code segment should be used to read namelist files.
535  !! @code{.F90}
536  !! integer :: ierr, io
537  !!
538  !! read (input_nml_file, fms_nml, iostat=io)
539  !! ierr = check_nml_error(io,'fms_nml')
540  !! @endcode
541  !! @throws FATAL, Unknown error while reading namelist ...., (IOSTAT = ####)
542  !! There was an error reading the namelist specified. Carefully examine all namelist and variables
543  !! for anything incorrect (e.g. malformed, hidden characters).
544  !!
545  !! @throws FATAL, Unknown namelist, or mistyped namelist variable in namelist ...., (IOSTAT = ####)
546  !! The name list given doesn't exist in the namelist file, or a variable in the namelist is
547  !! mistyped or isn't a namelist variable.
548  INTEGER FUNCTION check_nml_error(IOSTAT, NML_NAME)
549  INTEGER, INTENT(in) :: iostat !< The iostat value returned when reading a namelist record.
550  CHARACTER(len=*), INTENT(in) :: nml_name !< The name of the namelist. This name will be printed if an error is
551  !! encountered, otherwise the name is not used.
552 
553  CHARACTER(len=256) :: err_str
554 
555  IF ( .NOT.module_is_initialized) CALL fms_init()
556 
557  check_nml_error = iostat
558 
559  ! Return on valid IOSTAT values
560  IF ( iostat <= 0 .OR.&
561  & iostat == nml_errors%multipleNMLSinFile .OR.&

585  ! Also checks if the compiler returns a non-zero status if there are

598  namelist /b_nml/ i2, r2, l1

601  namelist /missingvar_nml/ i2, r2
602  namelist /not_in_file_nml/ i2, r2

614  file_opened: DO fileunit = unit_begin, unit_end

616  IF ( .NOT.opened ) EXIT file_opened
617  END DO file_opened
618 
619 #if defined(__PGI) || defined(_CRAYFTN)
620  OPEN (unit=fileunit, file='_read_error.nml', iostat=io_stat)
621 #else
622  OPEN (unit=fileunit, status='SCRATCH', iostat=io_stat)
623 #endif
624

626  WRITE (unit=fileunit, nml=a_nml, iostat=io_stat)

630  WRITE (unit=fileunit, iostat=io_stat, fmt='(/,"&missingVar_nml i2=1, r2=1.0e0, l1=.true. /",/)')
631

633  rewind(unit=fileunit)
634 
635  ! Read the second namelist from the file -- check for namelist bug
636  READ (unit=fileunit, nml=b_nml, iostat=nml_iostats(1))
637  rewind(unit=fileunit)
638

645  rewind(unit=fileunit)

648  READ (unit=fileunit, nml=missingvar_nml, iostat=nml_iostats(4))
649  rewind(unit=fileunit)

665  nml_iostats(2) = nml_iostats(4)
666  END IF
667  END IF
668  END IF

672  nml_errors%multipleNMLSinFile = nml_iostats(1)
673  nml_errors%badType1 = nml_iostats(2)
674  nml_errors%badType2 = nml_iostats(3)
675  nml_errors%missingVar = nml_iostats(4)
676  nml_errors%NotInFile = nml_iostats(5)

729 function monotonic_array ( array, direction )
730 real, intent(in) :: array(:) !< An array of real values. If the size(array) < 2 this function
731  !! assumes the array is not monotonic, no fatal error will occur.
732 integer, intent(out), optional :: direction !< If the input array is:
733  !! >> monotonic (small to large) then direction = +1.
734  !! >> monotonic (large to small) then direction = -1.

736 logical :: monotonic_array !< If the input array of real values either increases or decreases monotonically
737  !! then TRUE is returned, otherwise FALSE is returned.
738 integer :: i

740 ! initialize

742  if (present(direction)) direction = 0
743 
744 ! array too short

746 
747 ! ascending

749  do i = 2, size(array(:))
750  if (array(i-1) < array(i)) cycle

752  enddo
753  monotonic_array = .true.

755 
756 ! descending

758  do i = 2, size(array(:))
759  if (array(i-1) > array(i)) cycle
760  return
761  enddo
762  monotonic_array = .true.
763  if (present(direction)) direction = -1
764  endif
