380 !> @brief Gets the name of this variable as well as the associated parameters and adds them to fmVar_t.
381 !! Note that the length of the character arrays for the parameter names and values are allocatable.
382 !! This is why they are read twice.
383 subroutine get_name_fmvar_t(this)
384  class(fmvar_t) :: this !< variable object
385  integer :: nkeys !< numkeys
386  integer :: maxln !< max string length names
387  integer :: maxlv !< max string length values
388  integer, allocatable :: key_ids(:) !< array of key ids
389  character(len=256) :: key_name !< the name of a key
390  character(len=256) :: key_value !< the value of a key
391 
392  nkeys = get_nkeys(this%yfid, this%id)
393  allocate(key_ids(nkeys))
394  call get_key_ids(this%yfid, this%id, key_ids)
395  call get_key_value(this%yfid, key_ids(1), key_value)
396  this%name = trim(key_value)
397  if (nkeys .gt. 1) then
398  maxln = 0
399  maxlv = 0
400  do var_j=2,nkeys
401  call get_key_name(this%yfid, key_ids(var_j), key_name)
402  call get_key_value(this%yfid, key_ids(var_j), key_value)
403  maxln = max(maxln, len_trim(key_name))
404  maxlv = max(maxlv, len_trim(key_value))
405  end do
406  allocate(this%key_ids(nkeys-1))
407  allocate(character(len=maxln)::this%keys(nkeys-1))
408  allocate(character(len=maxlv)::this%values(nkeys-1))
409  do var_j=2,nkeys
410  this%key_ids(var_j-1) = key_ids(var_j)
411  call get_key_name(this%yfid, key_ids(var_j), key_name)
412  call get_key_value(this%yfid, key_ids(var_j), key_value)
413  this%keys(var_j-1) = trim(key_name)
414  this%values(var_j-1) = trim(key_value)
415  end do
416  else
417  allocate(this%key_ids(0))
418  end if
419 end subroutine get_name_fmvar_t
