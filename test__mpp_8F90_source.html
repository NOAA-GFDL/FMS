82  subroutine test_mpp_max
83 
84  allocate( a(n), b(n) )
85  a = real(pe+1)
86  print *, 'pe, pe+1 =', pe, a(1)
87  call mpp_max( a(1) )
88  print *, 'pe, max(pe+1)=', pe, a(1)
89  !pelist check
90  call mpp_sync()
91  flush(out_unit)
92  if( npes.GE.2 )then
93  if( pe.EQ.root )print *, 'Test of pelists: bcast, sum and max using PEs 0...npes-2 (excluding last PE)'
94  call mpp_declare_pelist( (/(i,i=0,npes-2)/) )
95  a = real(pe+1)
96  if( pe.NE.npes-1 )call mpp_broadcast( a, n, npes-2, (/(i,i=0,npes-2)/) )
97  print *, 'bcast(npes-1) from 0 to npes-2=', pe, a(1)
98  a = real(pe+1)
99  if( pe.NE.npes-1 )then
100  call mpp_set_current_pelist( (/(i,i=0,npes-2)/) )
101  id = mpp_clock_id( 'Partial mpp_sum' )
102  call mpp_clock_begin(id)
103  call mpp_sum( a(1:1000), 1000, (/(i,i=0,npes-2)/) )
104  call mpp_clock_end (id)
105  end if
106  if( pe.EQ.root )print *, 'sum(pe+1) from 0 to npes-2=', a(1)
107  a = real(pe+1)
108  if( pe.NE.npes-1 )call mpp_max( a(1), (/(i,i=0,npes-2)/) )
109  if( pe.EQ.root )print *, 'max(pe+1) from 0 to npes-2=', a(1)
110  end if
111  call mpp_set_current_pelist()
112 
113  end subroutine test_mpp_max

115  subroutine test_mpp_chksum()
116 
117  if( modulo(n,npes).EQ.0 )then !only set up for even division
118  n2 = 1024
119  a = 0.d0
120  if( pe.EQ.root )call random_number(a(1:n2))
121 
122  call mpp_sync()
123  call mpp_transmit( put_data=a(1), plen=n2, to_pe=all_pes, &
124  get_data=a(1), glen=n2, from_pe=root )
125  call mpp_sync_self ()
126 
127  m= n2/npes
128 
129  allocate( c(m) )
130  c = a(pe*m+1:pe*m+m)
131 
132  if( pe.EQ.root )then
133  print *
134  print *, '------------------ > Test mpp_chksum <------------------ '
135  print *, 'This test shows that a whole array and a distributed array give identical checksums.'
136  end if
137 
138  if ( mpp_chksum(a(1:n2),(/pe/)) .NE. mpp_chksum(c) ) then
139  call mpp_error(fatal, &
140  & 'Test mpp_chksum fails: a whole array and a distributed array did not give identical checksums')
141  else
142  print *, 'For pe=', pe, ' chksum(a(1:1024))=chksum(c(1:1024))='
143  endif
144 
145  else
146  call mpp_error(fatal, 'Test mpp_chksum: cannot run this test since n cannot be evenly by npes')
147  end if
148 
149  end subroutine test_mpp_chksum
