21 program test_drifters_comm
22 
23  use drifters_core_mod
24  use drifters_comm_mod
25  use mpp_mod
26  use mpp_domains_mod
27 
28  implicit none
29 
30  integer, parameter :: nd=2, npmax = 4
31  integer :: nx, ny, halox, haloy, layout(2), i, j, npes, pe, it, nt
32 ! _TYPE_DOMAIN2D :: domain
33  type(domain2d) :: domain
34  type(drifters_core_type) :: drfts
35  type(drifters_comm_type) :: drfts_com
36  real, parameter :: xmin=0., xmax=1., ymin=0., ymax=1.
37  real :: dx, dy, u0, v0, dt, Lx, Ly
38  real, allocatable :: x(:), y(:)
39  character(len=128) :: ermsg = ''
40  integer :: ids(npmax)
41  real :: positions(nd, npmax), velocity(nd, npmax)
42  integer :: io_status
43 !!$ integer :: stackmax=4000000
44 
45  namelist /drifters_comm_nml/ nx, ny, halox, haloy, u0, v0, dt, nt
46 
47  call mpp_init !(MPP_DEBUG)
48  !call mpp_set_stack_size(3145746)
49 
50  ! default input values
51  nx = 11
52  ny = 21
53  halox = 2
54  haloy = 2
55  u0 = 1.0
56  v0 = 0.0
57  dt = 0.1
58  nt = 10
59 
60  ! read input
61  read (input_nml_file, drifters_comm_nml, iostat=io_status)
62 
63  ! create global domain
64  lx = xmax - xmin
65  ly = ymax - ymin
66  dx = lx/real(nx-1)
67  dy = ly/real(ny-1)
68  allocate(x(nx), y(ny))
69  x = xmin + (/ ( i*dx, i=0, nx-1) /)
70  y = ymin + (/ ( j*dy, j=0, ny-1) /)
71 
72  ! decompose domain
73  call mpp_domains_init ! (MPP_DEBUG)
74 !!$ call mpp_domains_set_stack_size(stackmax)
75 
76  npes = mpp_npes()
77  call mpp_define_layout( (/1,nx, 1,ny/), npes, layout )
78  if(mpp_pe()==0) print *,'LAYOUT: ', layout
79  call mpp_define_domains((/1,nx, 1,ny/), layout, domain, xhalo=halox, yhalo=haloy,&
80  & xflags=cyclic_global_domain, yflags=cyclic_global_domain)
81 
82  ! set up drifters' communicator
83  call drifters_comm_new(drfts_com)
84  call drifters_comm_set_domain(drfts_com, domain, x, y, 0,0)
85  ! repeated calls just for the fun of it
86  call drifters_comm_set_domain(drfts_com, domain, x, y, 0,0)
87  call drifters_comm_set_domain(drfts_com, domain, x, y, 0,0)
88 
89  ! create drifters
90  call drifters_core_new(drfts, nd, npmax, ermsg)
91  if(ermsg /= '') then
92  print *,'ERROR',ermsg
93  call mpp_error(fatal, 'test_drifters_comm: Error')
94  endif
95 
96  pe = mpp_pe()
97  ids = (/ (i+100*pe, i=1,npmax) /)
98  call drifters_core_set_ids(drfts, ids, ermsg)
99  if(ermsg /= '') then
100  print *,'ERROR', ermsg
101  call mpp_error(fatal, 'test_drifters_comm: Error')
102  endif
103 
104  ! position particles
105  if(pe == 0) then
106  positions(:, 1) = (/ (drfts_com%xcmin + drfts_com%xcmax)/2., &
107  & (drfts_com%ycmin + drfts_com%ycmax)/2. /)
108  !positions(:, 2) = (/0.,0.01/)
109  call drifters_core_set_positions(drfts, positions(:, 1:1), ermsg)
110  if(ermsg /= '') then
111  print *,'ERROR',ermsg
112  call mpp_error(fatal, 'test_drifters_comm: Error')
113  endif
114  endif
115 
116  ! push drifters
117  velocity(:,1) = (/u0, v0/)
118  do it = 1, nt
119  positions(:,1:drfts%np) = xmin + &
120  & modulo(drfts%positions(:,1:drfts%np) + dt*velocity(:,1:drfts%np)-xmin, xmax-xmin)
121  ! this will redistribute the drifters and update the positions
122  call drifters_comm_update(drfts_com, drfts, positions(:,1:drfts%np))
123 
124  if(drfts%np > 0) then
125  do i=1,drfts%np
126  print '(a,i6,a,i3,a,i3,a, i3, a,2f10.6)', 'PE: ',pe, ' it=', it, ' np=', drfts%np, ' ip=', i, &
127  & ' x,y=', drfts%positions(1,i), drfts%positions(2,i)
128  enddo
129  endif
130 !!$ call drifters_print(drfts, ermsg)
131 !!$ if(ermsg /= '') print *,ermsg
132  enddo
133 
134  ! clean up
135  call drifters_core_del(drfts, ermsg)
136  if(ermsg /= '') then
137  print *,ermsg, '<--- THIS'
138  call mpp_error(fatal, 'test_drifters_comm: Error')
139  endif
140  call drifters_comm_del(drfts_com)
141  call mpp_domains_exit
142  call mpp_exit
143 
144 end program test_drifters_comm
