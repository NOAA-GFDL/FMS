88  !> @brief Opens the output file.
89  SUBROUTINE diag_output_init (file_name, file_title, file_unit,&
90  & domain, domainU, fileobj, fileobjU, fileobjND, fnum_domain, &
91  & attributes)
92  CHARACTER(len=*), INTENT(in) :: file_name !< Output file name
93  CHARACTER(len=*), INTENT(in) :: file_title !< Descriptive title for the file
94  INTEGER , INTENT(out) :: file_unit !< File unit number assigned to the output file.
95  !! Needed for subsuquent calls to
96  !! diag_output_mod
97  TYPE(domain2d) , INTENT(in) :: domain !< Domain associated with file, if domain decomposed
98  TYPE(domainug) , INTENT(in) :: domainu !< The unstructure domain
99  type(fmsnetcdfdomainfile_t),intent(inout),target :: fileobj !< Domain decomposed fileobj
100  type(fmsnetcdfunstructureddomainfile_t),intent(inout),target :: fileobju !< Unstructured domain fileobj
101  type(fmsnetcdffile_t),intent(inout),target :: fileobjnd !< Non domain decomposed fileobj
102  character(*),intent(out) :: fnum_domain !< String indicating the type of fileobj was used:
103  !! "2d" domain decomposed
104  !! "ug" unstrucuted domain decomposed
105  !! "nd" no domain
106  TYPE(diag_atttype), INTENT(in), OPTIONAL :: attributes(:) !< Array of global attributes to be written to file
107 
108  class(fmsnetcdffile_t), pointer :: fileob => null()
109  integer :: i !< For looping through number of attributes
110  TYPE(diag_global_att_type) :: gatt
111  integer, allocatable, dimension(:) :: current_pelist
112  integer :: mype !< The pe you are on
113  character(len=9) :: mype_string !< a string to store the pe
114  !---- initialize mpp_io ----
115  IF ( .NOT.module_is_initialized ) THEN
116  module_is_initialized = .true.
117  CALL write_version_number("DIAG_OUTPUT_MOD", version)
118  END IF
119 
120 !> Checks to make sure that only domain2D or domainUG is used. If both are not null, then FATAL
121  if (domain .NE. null_domain2d .AND. domainu .NE. null_domainug)&
122  & CALL error_mesg('diag_output_init', "Domain2D and DomainUG can not be used at the same time in "//&
123  & trim(file_name), fatal)
124 
125  !---- open output file (return file_unit id) -----
126  IF ( domain .NE. null_domain2d ) THEN
127  !> Check if there is an io_domain
128  iF ( associated(mpp_get_io_domain(domain)) ) then
129  fileob => fileobj
130  if (.not.check_if_open(fileob)) call open_check(open_file(fileobj, trim(file_name)//".nc", "overwrite", &
131  domain, is_restart=.false.))
132  fnum_domain = "2d" ! 2d domain
133  file_unit = 2
134  elSE !< No io domain, so every core is going to write its own file.
135  fileob => fileobjnd
136  mype = mpp_pe()
137  write(mype_string,'(I0.4)') mype
138  if (.not.check_if_open(fileob)) then
139  call open_check(open_file(fileobjnd, trim(file_name)//".nc."//trim(mype_string), "overwrite", &
140  is_restart=.false.))
141  !< For regional subaxis add the NumFilesInSet attribute, which is added by fms2_io for (other)
142  !< domains with sufficient decomposition info. Note mppnccombine will work with an entry of zero.
143  call register_global_attribute(fileobjnd, "NumFilesInSet", 0)
144  endif
145  fnum_domain = "nd" ! no domain
146  if (file_unit < 0) file_unit = 10
147  endiF
148  ELSE IF (domainu .NE. null_domainug) THEN
149  fileob => fileobju
150  if (.not.check_if_open(fileob)) call open_check(open_file(fileobju, trim(file_name)//".nc", "overwrite", &
151  domainu, is_restart=.false.))
152  fnum_domain = "ug" ! unstructured grid
153  file_unit=3
154  ELSE
155  fileob => fileobjnd
156  allocate(current_pelist(mpp_npes()))
157  call mpp_get_current_pelist(current_pelist)
158  if (.not.check_if_open(fileob)) then
159  call open_check(open_file(fileobjnd, trim(file_name)//".nc", "overwrite", &
160  pelist=current_pelist, is_restart=.false.))
161  endif
162  fnum_domain = "nd" ! no domain
163  if (file_unit < 0) file_unit = 10
164  deallocate(current_pelist)
165  END IF
166 
167  !---- write global attributes ----
168  IF ( file_title(1:1) /= ' ' ) THEN
169  call register_global_attribute(fileob, 'title', trim(file_title), str_len=len_trim(file_title))
170  END IF
171 
172  IF ( PRESENT(attributes) ) THEN
173  DO i=1, SIZE(attributes)
174  SELECT CASE (attributes(i)%type)
175  CASE (nf90_int)
176  call register_global_attribute(fileob, trim(attributes(i)%name), attributes(i)%iatt)
177  CASE (nf90_float)
178 
179  call register_global_attribute(fileob, trim(attributes(i)%name), attributes(i)%fatt)
180  CASE (nf90_char)
181 
182  call register_global_attribute(fileob, trim(attributes(i)%name), attributes(i)%catt, &
183  & str_len=len_trim(attributes(i)%catt))
184  CASE default
185  ! <ERROR STATUS="FATAL">
186  ! Unknown attribute type for attribute <name> to module/input_field <module_name>/<field_name>.
187  ! Contact the developers.
188  ! </ERROR>
189  CALL error_mesg('diag_output_mod::diag_output_init', 'Unknown attribute type for global attribute "'&
190  &//trim(attributes(i)%name)//'" in file "'//trim(file_name)//'". Contact the developers.', fatal)
191  END SELECT
192  END DO
193  END IF
194  !---- write grid type (mosaic or regular)
195  CALL get_diag_global_att(gatt)
196 
197  call register_global_attribute(fileob, 'grid_type', trim(gatt%grid_type), str_len=len_trim(gatt%grid_type))
198 
199  call register_global_attribute(fileob, 'grid_tile', trim(gatt%tile_name), str_len=len_trim(gatt%tile_name))
200 
201  END SUBROUTINE diag_output_init

203  !> @brief Write the axis meta data to file.
204  SUBROUTINE write_axis_meta_data(file_unit, axes, fileob, time_ops, time_axis_registered)
205  INTEGER, INTENT(in) :: file_unit !< File unit number
206  INTEGER, INTENT(in) :: axes(:) !< Array of axis ID's, including the time axis
207  class(fmsnetcdffile_t) , intent(inout) :: fileob !< FMS2_io fileobj
208  LOGICAL, INTENT(in), OPTIONAL :: time_ops !< .TRUE. if this file contains any min, max, time_rms, or time_average
209  logical, intent(inout) , optional :: time_axis_registered !< .TRUE. if the time axis was already
210  !! written to the file
211 
212  TYPE(domain1d) :: domain
213  TYPE(domainug) :: domainu
214 
215  CHARACTER(len=mxch) :: axis_name, axis_units, axis_name_current
216  CHARACTER(len=mxchl) :: axis_long_name
217  CHARACTER(len=1) :: axis_cart_name
218  INTEGER :: axis_direction, axis_edges
219  REAL, ALLOCATABLE :: axis_data(:)
220  integer :: axis_pos
221  INTEGER :: num_attributes
222  TYPE(diag_atttype), DIMENSION(:), ALLOCATABLE :: attributes
223  INTEGER :: calendar, id_axis, id_time_axis
224  INTEGER :: i, j, index, num, length, edges_index
225  INTEGER :: gend !< End index of global io_domain
226  LOGICAL :: time_ops1
227  CHARACTER(len=2048) :: err_msg
228  integer :: id_axis_current
229  logical :: is_time_axis_registered
230  integer :: istart, iend
231  integer :: gstart, cstart, cend !< Start and end of global and compute domains
232  integer :: clength !< Length of compute domain
233  character(len=32) :: type_str !< Str indicating the type of the axis data
234 
235  ! Make sure err_msg is initialized
236  err_msg = ''
237  IF ( PRESENT(time_ops) ) THEN
238  time_ops1 = time_ops
239  ELSE
240  time_ops1 = .false.
241  END IF
242  if (present(time_axis_registered)) then
243  is_time_axis_registered = time_axis_registered
244  else
245  is_time_axis_registered = .false.
246  endif
247  !---- save the current file_unit ----
248  IF ( num_axis_in_file == 0 ) current_file_unit = file_unit
249 
250  !---- dummy checks ----
251  num = SIZE(axes(:))
252  ! <ERROR STATUS="FATAL">number of axes < 1 </ERROR>
253  IF ( num < 1 ) CALL error_mesg('write_axis_meta_data', 'number of axes < 1.', fatal)
254 
255  ! <ERROR STATUS="FATAL">writing meta data out-of-order to different files.</ERROR>
256  IF ( file_unit /= current_file_unit ) CALL error_mesg('write_axis_meta_data',&
257  & 'writing meta data out-of-order to different files.', fatal)
258 
259  IF (pack_size .eq. 1) then
260  type_str = "double"
261  ELSE IF (pack_size .eq. 2) then
262  type_str = "float"
263  ENDIF
264 
265  !---- check all axes ----
266  !---- write axis meta data for new axes ----
267  DO i = 1, num
268  id_axis = axes(i)
269  index = get_axis_index( id_axis )
270 
271  !---- skip axes already written -----
272  IF ( index > 0 ) cycle
273 
274  !---- create new axistype (then point to) -----
275  num_axis_in_file = num_axis_in_file + 1
276  axis_in_file(num_axis_in_file) = id_axis
277  edge_axis_flag(num_axis_in_file) = .false.
278  length = get_axis_global_length(id_axis)
279  ALLOCATE(axis_data(length))
280 
281  CALL get_diag_axis(id_axis, axis_name, axis_units, axis_long_name,&
282  & axis_cart_name, axis_direction, axis_edges, domain, domainu, axis_data,&
283  & num_attributes, attributes, domain_position=axis_pos)
284 
285  IF ( domain .NE. null_domain1d ) THEN
286  select type (fileob)
287  type is (fmsnetcdffile_t)
288  !> If the axis is domain decomposed and the type is FmsNetcdfFile_t, this is regional diagnostic
289  !! So treat it as any other dimension
290  call mpp_get_global_domain(domain, begin=gstart, end=gend) !< Get the global indicies
291  call mpp_get_compute_domain(domain, begin=cstart, end=cend, size=clength) !< Get the compute indicies
292  iend = cend - gstart + 1 !< Get the array indicies for the axis data
293  istart = cstart - gstart + 1
294  call register_axis(fileob, axis_name, dimension_length=clength)
295  call register_field(fileob, axis_name, type_str, (/axis_name/) )
296 
297  !> For regional subaxis add the "domain_decomposition" attribute, which is added
298  !> fms2_io for (other) domains with sufficient decomposition info.
299  call register_variable_attribute(fileob, axis_name, "domain_decomposition", &
300  (/gstart, gend, cstart, cend/))
301  type is (fmsnetcdfdomainfile_t)
302  !> If the axis is domain decomposed and the type is FmsNetcdfDomainFile_t,
303  !! this is a domain decomposed dimension
304  !! so register it as one
305  call register_axis(fileob, axis_name, lowercase(trim(axis_cart_name)), domain_position=axis_pos )
306  call get_global_io_domain_indices(fileob, trim(axis_name), istart, iend)
307  call register_field(fileob, axis_name, type_str, (/axis_name/) )
308  end select
309 
310  ELSE IF ( domainu .NE. null_domainug) THEN
311  select type(fileob)
312  type is (fmsnetcdfunstructureddomainfile_t)
313  !> If the axis is in unstructured domain and the type is FmsNetcdfUnstructuredDomainFile_t,
314  !! this is an unstrucutred axis
315  !! so register it as one
316  call register_axis(fileob, axis_name )
317  end select
318  call register_field(fileob, axis_name, type_str, (/axis_name/) )
319  istart = lbound(axis_data,1)
320  iend = ubound(axis_data,1)
321  ELSE
322  !> If the axis is not in a domain, register it as a normal dimension
323  call register_axis(fileob, axis_name, dimension_length=size(axis_data))
324  call register_field(fileob, axis_name, type_str, (/axis_name/) )
325  istart = lbound(axis_data,1)
326  iend = ubound(axis_data,1)
327  ENDIF !! IF ( Domain .NE. null_domain1d )
328 
329  if (length <= 0) then
330  !> @note Check if the time variable is registered. It's possible that is_time_axis_registered
331  !! is set to true if using
332  !! time-templated files because they aren't closed when done writing. An alternative
333  !! to this set up would be to put
334  !! variable_exists into the if statement with an .or. so that it gets registered.
335  is_time_axis_registered = variable_exists(fileob,trim(axis_name),.true.)
336  if (.not. is_time_axis_registered) then
337  call register_axis(fileob, trim(axis_name), unlimited )
338  call register_field(fileob, axis_name, type_str, (/axis_name/) )
339  is_time_axis_registered = .true.
340  if (present(time_axis_registered)) time_axis_registered = is_time_axis_registered
341  endif !! if (.not. is_time_axis_registered)
342  endif !! if (length <= 0)
343 
344  !> Add the attributes
345  if(trim(axis_units) .ne. "none") call register_variable_attribute(fileob, axis_name, "units", &
346  & trim(axis_units), str_len=len_trim(axis_units))
347  call register_variable_attribute(fileob, axis_name, "long_name", trim(axis_long_name), &
348  & str_len=len_trim(axis_long_name))
349  if(trim(axis_cart_name).ne."N") call register_variable_attribute(fileob, axis_name, "axis", &
350  & trim(axis_cart_name), str_len=len_trim(axis_cart_name))
351 
352  if (length > 0 ) then
353  !> If not a time axis, add the positive attribute and write the data
354  select case (axis_direction)
355  case (1)
356  call register_variable_attribute(fileob, axis_name, "positive", "up", str_len=len_trim("up"))
357  case (-1)
358  call register_variable_attribute(fileob, axis_name, "positive", "down", str_len=len_trim("down"))
359  end select
360  call write_data(fileob, axis_name, axis_data(istart:iend) )
361  endif
362 
363  !> Write additional axis attributes, from diag_axis_add_attribute calls
364  CALL write_attribute_meta(file_unit, num_attributes, attributes, err_msg, varname=axis_name, fileob=fileob)
365  IF ( len_trim(err_msg) .GT. 0 ) THEN
366  CALL error_mesg('diag_output_mod::write_axis_meta_data', trim(err_msg), fatal)
367  END IF
368 
369  !> Write additional attribute (calendar_type) for time axis ----
370  !> @note calendar attribute is compliant with CF convention
371  !! http://www.cgd.ucar.edu/cms/eaton/netcdf/CF-current.htm#cal
372  IF ( axis_cart_name == 'T' ) THEN
373  time_axis_flag(num_axis_in_file) = .true.
374  id_time_axis = num_axis_in_file
375  calendar = get_calendar_type()
376 
377 
378  call register_variable_attribute(fileob, axis_name, "calendar_type", &
379  uppercase(trim(valid_calendar_types(calendar))), &
380  & str_len=len_trim(valid_calendar_types(calendar)) )
381  call register_variable_attribute(fileob, axis_name, "calendar", &
382  lowercase(trim(valid_calendar_types(calendar))), &
383  & str_len=len_trim(valid_calendar_types(calendar)) )
384  IF ( time_ops1 ) THEN
385  call register_variable_attribute(fileob, axis_name, 'bounds', trim(axis_name)//'_bnds', &
386  & str_len=len_trim(trim(axis_name)//'_bnds'))
387  END IF
388  call set_fileobj_time_name(fileob, axis_name)
389  ELSE
390  time_axis_flag(num_axis_in_file) = .false.
391  END IF
392 
393  DEALLOCATE(axis_data)
394 
395  !> Deallocate attributes
396  IF ( ALLOCATED(attributes) ) THEN
397  DO j=1, num_attributes
398  IF ( allocated(attributes(j)%fatt ) ) THEN
399  DEALLOCATE(attributes(j)%fatt)
400  END IF
401  IF ( allocated(attributes(j)%iatt ) ) THEN
402  DEALLOCATE(attributes(j)%iatt)
403  END IF
404  END DO
405  DEALLOCATE(attributes)
406  END IF
407 
408  !------------- write axis containing edge information ---------------
409 
410  ! --- this axis has no edges -----
411  IF ( axis_edges <= 0 ) cycle
412 
413  ! --- was this axis edge previously defined? ---
414  id_axis_current = id_axis
415  axis_name_current = axis_name
416  id_axis = axis_edges
417  edges_index = get_axis_index(id_axis)
418  IF ( edges_index > 0 ) cycle
419 
420  ! ---- get data for axis edges ----
421  length = get_axis_global_length( id_axis )
422  ALLOCATE(axis_data(length))
423  CALL get_diag_axis(id_axis, axis_name, axis_units, axis_long_name, axis_cart_name,&
424  & axis_direction, axis_edges, domain, domainu, axis_data)
425 
426  ! ---- write edges attribute to original axis ----
427  call register_variable_attribute(fileob, axis_name_current, "edges",trim(axis_name), &
428  & str_len=len_trim(axis_name))
429  ! ---- add edges index to axis list ----
430  ! ---- assume this is not a time axis ----
431  num_axis_in_file = num_axis_in_file + 1
432  axis_in_file(num_axis_in_file) = id_axis
433  edge_axis_flag(num_axis_in_file) = .true.
434  time_axis_flag(num_axis_in_file) = .false.
435 
436 !> Add edges axis with fms2_io
437  if (.not.variable_exists(fileob, axis_name)) then
438  call register_axis(fileob, axis_name, size(axis_data) )
439  call register_field(fileob, axis_name, type_str, (/axis_name/) )
440  if(trim(axis_units) .ne. "none") call register_variable_attribute(fileob, axis_name, "units", &
441  & trim(axis_units), str_len=len_trim(axis_units))
442  call register_variable_attribute(fileob, axis_name, "long_name", trim(axis_long_name), &
443  & str_len=len_trim(axis_long_name))
444  if(trim(axis_cart_name).ne."N") call register_variable_attribute(fileob, axis_name, "axis", &
445  & trim(axis_cart_name), str_len=len_trim(axis_cart_name))
446  select case (axis_direction)
447  case (1)
448  call register_variable_attribute(fileob, axis_name, "positive", "up", str_len=len_trim("up"))
449  case (-1)
450  call register_variable_attribute(fileob, axis_name, "positive", "down", str_len=len_trim("down"))
451  end select
452  call write_data(fileob, axis_name, axis_data)
453  endif !! if (.not.variable_exists(fileob, axis_name))
454 
455  DEALLOCATE (axis_data)
456  END DO
457  END SUBROUTINE write_axis_meta_data

459  !> @brief Write the field meta data to file.
460  !! @return diag_fieldtype Field
461  !! @details The meta data for the field is written to the file indicated by file_unit
462  FUNCTION write_field_meta_data ( file_unit, name, axes, units, long_name, range, pack, mval,&
463  & avg_name, time_method, standard_name, interp_method, attributes, num_attributes, &
464  & use_UGdomain, fileob) result ( Field )
465  INTEGER, INTENT(in) :: file_unit !< Output file unit number
466  INTEGER, INTENT(in) :: axes(:) !< Array of axis IDs
467  CHARACTER(len=*), INTENT(in) :: name !< Field name
468  CHARACTER(len=*), INTENT(in) :: units !< Field units
469  CHARACTER(len=*), INTENT(in) :: long_name !< Field's long name
470  REAL, OPTIONAL, INTENT(in) :: range(2) !< Valid range (min, max). If min > max, the range will be ignored
471  REAL, OPTIONAL, INTENT(in) :: mval !< Missing value, must be within valid range
472  INTEGER, OPTIONAL, INTENT(in) :: pack !< Packing flag. Only valid when range specified. Valid values:
473  !! Flag | Size
474  !! --- | ---
475  !! 1 | 64bit
476  !! 2 | 32bit
477  !! 4 | 16bit
478  !! 8 | 8bit
479  CHARACTER(len=*), OPTIONAL, INTENT(in) :: avg_name !< Name of variable containing time averaging info
