145  !> @brief Registers the time axis and opens the output file.
146  SUBROUTINE diag_output_init_fms2_io (file_name, FORMAT, file_title, file_unit,&
147  & all_scalar_or_1d, domain, domainU, fileobj, fileobjU, fileobjND, fnum_domain, &
148  & attributes)
149  CHARACTER(len=*), INTENT(in) :: file_name !< Output file name
150  CHARACTER(len=*), INTENT(in) :: file_title !< Descriptive title for the file
151  INTEGER , INTENT(in) :: FORMAT !< File format (Currently only 'NETCDF' is valid)
152  INTEGER , INTENT(out) :: file_unit !< File unit number assigned to the output file.
153  !! Needed for subsuquent calls to
154  !! diag_output_mod
155  LOGICAL , INTENT(in) :: all_scalar_or_1d
156  TYPE(domain2d) , INTENT(in) :: domain
157  TYPE(diag_atttype), INTENT(in), DIMENSION(:), OPTIONAL :: attributes
158  TYPE(domainug), INTENT(in) :: domainU !< The unstructure domain
159  type(fmsnetcdfunstructureddomainfile_t),intent(inout),target :: fileobjU
160  type(fmsnetcdfdomainfile_t),intent(inout),target :: fileobj
161  type(fmsnetcdffile_t),intent(inout),target :: fileobjND
162  class(fmsnetcdffile_t), pointer :: fileob => null()
163  character(*),intent(out) :: fnum_domain
164  INTEGER :: form, threading, fileset, i
165  TYPE(diag_global_att_type) :: gAtt
166  character(len=:),allocatable :: fname_no_tile
167  integer :: len_file_name
168  integer, allocatable, dimension(:) :: current_pelist
169  integer :: mype !< The pe you are on
170  character(len=9) :: mype_string !< a string to store the pe
171  !---- initialize mpp_io ----
172  IF ( .NOT.module_is_initialized ) THEN
173  CALL mpp_io_init ()
174  module_is_initialized = .true.
175  CALL write_version_number("DIAG_OUTPUT_MOD", version)
176  END IF
177  !---- set up output file ----
178  SELECT CASE (format)
179  CASE (netcdf1)
180  form = mpp_netcdf
181  threading = mpp_multi
182  fileset = mpp_multi
183  CASE default
184  ! <ERROR STATUS="FATAL">invalid format</ERROR>
185  CALL error_mesg('diag_output_init', 'invalid format', fatal)
186  END SELECT
187 
188  IF(all_scalar_or_1d) THEN
189  threading = mpp_single
190  fileset = mpp_single
191  END IF
192 
193  len_file_name = len(trim(file_name))
194 !> If the file name has .tileX or .tileX.nc where X is a one or two digit tile number, removes
195 !! that suffix from the time name because fms2_io will add it
196 !! \note If mpp_domains accepts more than 99 tiles, this will need to be updated
197  allocate(character(len=len_file_name) :: fname_no_tile)
198  if (len_file_name < 6) then
199  if (trim(file_name) == "tile") then
200  call error_mesg('diag_output_init', 'You can not name your history file "tile"',fatal)
201  else
202  fname_no_tile = trim(file_name)
203  endif
204  !> One-digit tile numbers example
205  !! \verbatim
206  !! filename.tile1.nc
207  !! 09876543210
208  !! ^ ^
209  !! filename.tile1
210  !! 09876543210
211  !! ^ ^
212  !! \endverbatim
213  elseif (lowercase(file_name(len_file_name-4:len_file_name-1)) .eq. "tile") then
214  fname_no_tile = file_name(1:len_file_name-6)
215  elseif (len_file_name < 9) then
216  fname_no_tile = trim(file_name)
217  elseif (lowercase(file_name(len_file_name-7:len_file_name-4)) .eq. "tile") then
218  fname_no_tile = file_name(1:len_file_name-9)
219  !> Two-digit tile numbers example
220  !! \verbatim
221  !! filename.tile10.nc
222  !! 09876543210
223  !! ^ ^
224  !! filename.tile10
225  !! 09876543210
226  !! ^ ^
227  !! \endverbatim
228  elseif (lowercase(file_name(len_file_name-5:len_file_name-2)) .eq. "tile") then
229  fname_no_tile = file_name(1:len_file_name-7)
230 
231  elseif (lowercase(file_name(len_file_name-5:len_file_name-8)) .eq. "tile") then
232  fname_no_tile = file_name(1:len_file_name-10)
233  else
234  fname_no_tile = trim(file_name)
235  endif
236 !> If there is a .nc suffix on the file name, removes the .nc
237  if (len(trim(fname_no_tile)) > 3 ) then
238  checknc: do i = 3,len(trim(fname_no_tile))
239  if (fname_no_tile(i-2:i) == ".nc") then
240  fname_no_tile(i-2:i) = " "
241  exit checknc
242  endif
243  enddo checknc
244  endif
245 
246 !> Checks to make sure that only domain2D or domainUG is used. If both are not null, then FATAL
247  if (domain .NE. null_domain2d .AND. domainu .NE. null_domainug)&
248  & CALL error_mesg('diag_output_init', "Domain2D and DomainUG can not be used at the same time in "//&
249  & trim(file_name), fatal)
250 
251  !---- open output file (return file_unit id) -----
252  IF ( domain .NE. null_domain2d ) THEN
253  !> Check if there is an io_domain
254  iF ( associated(mpp_get_io_domain(domain)) ) then
255  fileob => fileobj
256  if (.not.check_if_open(fileob)) call open_check(open_file(fileobj, trim(fname_no_tile)//".nc", "overwrite", &
257  domain, is_restart=.false.))
258  fnum_domain = "2d" ! 2d domain
259  file_unit = 2
260  elSE !< No io domain, so every core is going to write its own file.
261  fileob => fileobjnd
262  mype = mpp_pe()
263  write(mype_string,'(I0.4)') mype
264  if (.not.check_if_open(fileob)) then
265  call open_check(open_file(fileobjnd, trim(fname_no_tile)//".nc."//trim(mype_string), "overwrite", &
266  is_restart=.false.))
267  !< For regional subaxis add the NumFilesInSet attribute, which is added by fms2_io for (other)
268  !< domains with sufficient decomposition info. Note mppnccombine will work with an entry of zero.
269  call register_global_attribute(fileobjnd, "NumFilesInSet", 0)
270  endif
271  fnum_domain = "nd" ! no domain
272  if (file_unit < 0) file_unit = 10
273  endiF
274  ELSE IF (domainu .NE. null_domainug) THEN
275  fileob => fileobju
276  if (.not.check_if_open(fileob)) call open_check(open_file(fileobju, trim(fname_no_tile)//".nc", "overwrite", &
277  domainu, is_restart=.false.))
278  fnum_domain = "ug" ! unstructured grid
279  file_unit=3
280  ELSE
281  fileob => fileobjnd
282 ! if (.not.check_if_open(fileob) .and. mpp_pe() == mpp_root_pe()) then
283  allocate(current_pelist(mpp_npes()))
284  call mpp_get_current_pelist(current_pelist)
285  if (.not.check_if_open(fileob)) then
286  call open_check(open_file(fileobjnd, trim(fname_no_tile)//".nc", "overwrite", &
287  pelist=current_pelist, is_restart=.false.))
288  endif
289  fnum_domain = "nd" ! no domain
290  if (file_unit < 0) file_unit = 10
291  deallocate(current_pelist)
292  END IF
293 
294  !---- write global attributes ----
295  IF ( file_title(1:1) /= ' ' ) THEN
296  call register_global_attribute(fileob, 'title', trim(file_title), str_len=len_trim(file_title))
297  END IF
298 
299  IF ( PRESENT(attributes) ) THEN
300  DO i=1, SIZE(attributes)
301  SELECT CASE (attributes(i)%type)
302  CASE (nf90_int)
303  call register_global_attribute(fileob, trim(attributes(i)%name), attributes(i)%iatt)
304  CASE (nf90_float)
305 
306  call register_global_attribute(fileob, trim(attributes(i)%name), attributes(i)%fatt)
307  CASE (nf90_char)
308 
309  call register_global_attribute(fileob, trim(attributes(i)%name), attributes(i)%catt, str_len=len_trim(attributes(i)%catt))
310  CASE default
311  ! <ERROR STATUS="FATAL">
312  ! Unknown attribute type for attribute <name> to module/input_field <module_name>/<field_name>.
313  ! Contact the developers.
314  ! </ERROR>
315  CALL error_mesg('diag_output_mod::diag_output_init', 'Unknown attribute type for global attribute "'&
316  &//trim(attributes(i)%name)//'" in file "'//trim(file_name)//'". Contact the developers.', fatal)
317  END SELECT
318  END DO
319  END IF
320  !---- write grid type (mosaic or regular)
321  CALL get_diag_global_att(gatt)
322 
323  call register_global_attribute(fileob, 'grid_type', trim(gatt%grid_type), str_len=len_trim(gatt%grid_type))
324 
325  call register_global_attribute(fileob, 'grid_tile', trim(gatt%tile_name), str_len=len_trim(gatt%tile_name))
326 
327  END SUBROUTINE diag_output_init_fms2_io
