19  subroutine read_record_core_(unit, field, num_words, data, start, axsiz)
20  integer, intent(in) :: unit
21  type(fieldtype), intent(in) :: field
22  integer, intent(in) :: num_words
23  mpp_type_, intent(inout) :: data(num_words)
24  integer, intent(in) :: start(:), axsiz(:)
25 
26  integer(i2_kind) :: i2vals(num_words)
27 !rab used in conjunction with transfer intrinsic to determine size of a variable
28  integer(KIND=1) :: one_byte(8)
29  integer :: word_sz
30  integer(i4_kind) :: ivals(num_words)
31  real(r4_kind) :: rvals(num_words)
32 
33  real(r8_kind) :: r8vals(num_words)
34  pointer( ptr1, i2vals )
35  pointer( ptr2, ivals )
36  pointer( ptr3, rvals )
37  pointer( ptr4, r8vals )
38  if (mpp_io_stack_size < num_words) call mpp_io_set_stack_size(num_words)
39 
40 #ifdef use_netCDF
41  word_sz = size(transfer(data(1),one_byte))
42 
43  select case (field%type)
44  case(nf_byte)
45 ! use type conversion
46  call mpp_error( fatal, 'MPP_READ: does not support NF_BYTE packing' )
47  case(nf_short)
48  ptr1 = loc(mpp_io_stack(1))
49  error = nf_get_vara_int2( mpp_file(unit)%ncid, field%id, start, axsiz, i2vals )
50  call netcdf_err( error, mpp_file(unit), field=field )
51  if(field%scale == 1.0 .and. field%add == 0.0) then
52  data(:)=i2vals(:)
53  else
54  data(:)=i2vals(:)*field%scale + field%add
55  end if
56  case(nf_int)
57 
58  ptr2 = loc(mpp_io_stack(1))
59 
60  error = nf_get_vara_int( mpp_file(unit)%ncid, field%id, start, axsiz, ivals )
61  call netcdf_err( error, mpp_file(unit), field=field )
62  if(field%scale == 1.0 .and. field%add == 0.0) then
63  data(:)=ivals(:)
64  else
65  data(:)=ivals(:)*field%scale + field%add
66  end if
67  case(nf_float)
68  ptr3 = loc(mpp_io_stack(1))
69  if (size(transfer(rvals(1),one_byte)) .eq. word_sz) then
70  error = nf90_get_var( mpp_file(unit)%ncid, field%id, data, start=start, count=axsiz )
71  call netcdf_err( error, mpp_file(unit), field=field )
72  if(field%scale /= 1.0 .or. field%add /= 0.0) then
73  data(:)=data(:)*field%scale + field%add
74  end if
75  else
76  error = nf_get_vara_real( mpp_file(unit)%ncid, field%id, start, axsiz, rvals )
77  call netcdf_err( error, mpp_file(unit), field=field )
78  if(field%scale == 1.0 .and. field%add == 0.0) then
79  data(:)=rvals(:)
80  else
81  data(:)=rvals(:)*field%scale + field%add
82  end if
83  end if
84  case(nf_double)
85  ptr4 = loc(mpp_io_stack(1))
86  if (size(transfer(r8vals(1),one_byte)) .eq. word_sz) then
87  error = nf90_get_var( mpp_file(unit)%ncid, field%id, data, start=start, count=axsiz )
88  call netcdf_err( error, mpp_file(unit), field=field )
89  if(field%scale /= 1.0 .or. field%add /= 0.0) then
90  data(:)=data(:)*field%scale + field%add
91  end if
92  else
93  error = nf_get_vara_double( mpp_file(unit)%ncid, field%id, start, axsiz, r8vals )
94  call netcdf_err( error, mpp_file(unit), field=field )
95  if(field%scale == 1.0 .and. field%add == 0.0) then
96  data(:)=r8vals(:)
97  else
98  data(:)=r8vals(:)*field%scale + field%add
99  end if
100  end if
101  case default
102  call mpp_error( fatal, 'MPP_READ: invalid pack value' )
103  end select
104 #else
105  call mpp_error( fatal, 'MPP_READ currently requires use_netCDF option' )
106 #endif
107 
108  end subroutine read_record_core_

111  subroutine read_record_( unit, field, num_words, data, time_level, domain, position, tile_count, start_in, axsiz_in )
112 !routine that is finally called by all mpp_read routines to perform the read
113 !a non-netCDF record contains:
114 ! field ID
115 ! a set of 4 coordinates (is:ie,js:je) giving the data subdomain
116 ! a timelevel and a timestamp (=NULLTIME if field is static)
117 ! 3D real data (stored as 1D)
118 !if you are using direct access I/O, the RECL argument to OPEN must be large enough for the above
119 !in a global direct access file, record position on PE is given by %record.
120 
121 !Treatment of timestamp:
122 ! We assume that static fields have been passed without a timestamp.
123 ! Here that is converted into a timestamp of NULLTIME.
124 ! For non-netCDF fields, field is treated no differently, but is written
125 ! with a timestamp of NULLTIME. There is no check in the code to prevent
126 ! the user from repeatedly writing a static field.
127 
128  integer, intent(in) :: unit, num_words
129  type(fieldtype), intent(in) :: field
130  mpp_type_, intent(inout) :: data(num_words)
131  integer, intent(in), optional :: time_level
132  type(domain2D), intent(in), optional :: domain
133  integer, intent(in), optional :: position, tile_count
134  integer, intent(in), optional :: start_in(:), axsiz_in(:)
135  integer, dimension(size(field%axes(:))) :: start, axsiz
136 
137  integer :: tlevel !,subdomain(4)
138 
139 
140  integer :: i, error, is, ie, js, je, isg, ieg, jsg, jeg
141  type(domain2d), pointer :: io_domain=>null()
142 
143  if (.not.PRESENT(time_level)) then
144  tlevel = 0
145  else
146  tlevel = time_level
147  endif
148 
149  if( .NOT.module_is_initialized )call mpp_error( fatal, 'READ_RECORD: must first call mpp_io_init.' )
150  if( .NOT.mpp_file(unit)%opened )call mpp_error( fatal, 'READ_RECORD: invalid unit number.' )
151  if( .NOT.mpp_file(unit)%read_on_this_pe )return
152 
153  if( .NOT.mpp_file(unit)%initialized ) call mpp_error( fatal, 'MPP_READ: must first call mpp_read_meta.' )
154  if( mpp_file(unit)%format .NE. mpp_netcdf ) call mpp_error( fatal, 'Currently dont support non-NetCDF mpp read' )
155 
156  if (.not.PRESENT(time_level)) then
157  tlevel = 0
158  else
159  tlevel = time_level
160  endif
161 
162  if( verbose )print '(a,2i6,2i5)', 'MPP_READ: PE, unit, %id, %time_level =',&
163  pe, unit, mpp_file(unit)%id, tlevel
164  if( PRESENT(start_in) .AND. PRESENT(axsiz_in) ) then
165  if(size(start(:)) > size(start_in(:)) )call mpp_error( fatal, 'MPP_READ: size(start_in) < size(start)')
166  if(size(axsiz(:)) > size(axsiz_in(:)) )call mpp_error( fatal, 'MPP_READ: size(axsiz_in) < size(axsiz)')
167  start(:) = start_in(1:size(start(:)))
168  axsiz(:) = axsiz_in(1:size(axsiz(:)))
169  else
170 !define netCDF data block to be read:
171 ! time axis: START = time level
172 ! AXSIZ = 1
173 ! space axis: if there is no domain info
174 ! START = 1
175 ! AXSIZ = field%size(axis)
176 ! if there IS domain info:
177 ! start of domain is compute%start_index for multi-file I/O
178 ! global%start_index for all other cases
179 ! this number must be converted to 1 for NF_GET_VAR
180 ! (netCDF fortran calls are with reference to 1),
181 ! So, START = compute%start_index - <start of domain> + 1
182 ! AXSIZ = usually compute%size
183 ! However, if compute%start_index-compute%end_index+1.NE.compute%size,
184 ! we assume that the call is passing a subdomain.
185 ! To pass a subdomain, you must pass a domain2D object that satisfies the following:
186 ! global%start_index must contain the <start of domain> as defined above;
187 ! the data domain and compute domain must refer to the subdomain being passed.
188 ! In this case, START = compute%start_index - <start of domain> + 1
189 ! AXSIZ = compute%start_index - compute%end_index + 1
190 ! NOTE: passing of subdomains will fail for multi-PE single-threaded I/O,
191 ! since that attempts to gather all data on PE 0.
192  start = 1
193  do i = 1,size(field%axes(:))
194  axsiz(i) = field%size(i)
195  if( i .EQ. field%time_axis_index )start(i) = tlevel
196  end do
197  if( PRESENT(domain) )then
198  call mpp_get_compute_domain( domain, is, ie, js, je, tile_count=tile_count, position=position )
199  call mpp_get_global_domain ( domain, isg, ieg, jsg, jeg, tile_count=tile_count, position=position )
200  axsiz(1) = ie-is+1
201  axsiz(2) = je-js+1
202  if( mpp_file(unit)%fileset.EQ.mpp_single )then
203  if( npes.GT.1 )then
204  start(1) = is - isg + 1
205  start(2) = js - jsg + 1
206  else !--- z1l fix a problem related obc when npes = 1
207  if( ie-is+1.NE.ieg-isg+1 )then
208  start(1) = is - isg + 1
209  axsiz(1) = ie - is + 1
210  end if
211  if( je-js+1.NE.jeg-jsg+1 )then
212  start(2) = js - jsg + 1
213  axsiz(2) = je - js + 1
214  end if
215  end if
216  else if( mpp_file(unit)%io_domain_exist ) then
217  io_domain=>mpp_get_io_domain(domain)
218  call mpp_get_compute_domain( io_domain, is, ie, js, je, tile_count=tile_count, position=position )
219  call mpp_get_global_domain ( io_domain, isg, ieg, jsg, jeg, tile_count=tile_count, position=position )
220  start(1) = is - isg + 1
221  start(2) = js - jsg + 1
222  io_domain => null()
223  end if
224  end if
225  endif
226  if( verbose )print '(a,2i6,i6,12i4)', 'READ_RECORD: PE, unit, num_words, start, axsiz=', &
227  & pe, unit, num_words, start, axsiz
228 
229  call read_record_core_(unit, field, num_words, data, start, axsiz)
230 
231  return
232  end subroutine read_record_

247  subroutine mpp_read_2ddecomp_3d_( unit, field, domain, data, tindex, tile_count )
248 !mpp_read reads <data> which has the domain decomposition <domain>
249  integer, intent(in) :: unit
250  type(fieldtype), intent(in) :: field
251  type(domain2D), intent(in) :: domain
252  mpp_type_, intent(inout) :: data(:,:,:)
253  integer, intent(in), optional :: tindex, tile_count
254 
255  mpp_type_, allocatable :: cdata(:,:,:)
256  mpp_type_, allocatable :: gdata(:)
257  integer :: len, lenx,leny,lenz,i,j,k,n
258 !NEW: data may be on compute OR data domain
259  logical :: data_has_halos, halos_are_global, x_is_global, y_is_global
260  integer :: is, ie, js, je, isd, ied, jsd, jed, isg, ieg, jsg, jeg, ism, iem, jsm, jem
261  integer :: ioff, joff, position
262 
263  call mpp_clock_begin(mpp_read_clock)
264 
265  if (.NOT. present(tindex) .AND. mpp_file(unit)%time_level .ne. -1) &
266  call mpp_error(fatal, 'MPP_READ: need to specify a time level for data with time axis')
267 
268  if( .NOT.module_is_initialized )call mpp_error( fatal, 'MPP_READ: must first call mpp_io_init.' )
269  if( .NOT.mpp_file(unit)%opened )call mpp_error( fatal, 'MPP_READ: invalid unit number.' )
270 
271  call mpp_get_compute_domain( domain, is, ie, js, je, tile_count=tile_count )
272  call mpp_get_data_domain ( domain, isd, ied, jsd, jed, x_is_global=x_is_global, &
273  y_is_global=y_is_global, tile_count=tile_count )
274  call mpp_get_memory_domain ( domain, ism, iem, jsm, jem )
275  call mpp_get_global_domain ( domain, isg, ieg, jsg, jeg, tile_count=tile_count )
276 
277  ! when domain is symmetry, extra point is needed for some data on x/y direction
278  position = center
279  if(mpp_domain_is_symmetry(domain)) then
280  if( size(data,1).EQ.ie-is+1 .AND. size(data,2).EQ.je-js+1 ) then ! CENTER
281  data_has_halos = .false.
282  else if( size(data,1).EQ.ie-is+2 .AND. size(data,2).EQ.je-js+1 ) then ! EAST
283  data_has_halos = .false.
284  position = east
285  ie = ie + 1
286  else if( size(data,1).EQ.ie-is+1 .AND. size(data,2).EQ.je-js+2 ) then ! NORTH
287  position = north
288  data_has_halos = .false.
289  je = je + 1
290  else if( size(data,1).EQ.ie-is+2 .AND. size(data,2).EQ.je-js+2 ) then ! CORNER
291  position = corner
292  data_has_halos = .false.
293  ie = ie + 1; je = je + 1
294  else if( size(data,1).EQ.iem-ism+1 .AND. size(data,2).EQ.jem-jsm+1 )then ! CENTER
295  data_has_halos = .true.
296  else if( size(data,1).EQ.iem-ism+2 .AND. size(data,2).EQ.jem-jsm+1 )then ! EAST
297  position = east
298  data_has_halos = .true.
299  ie = ie + 1
300  else if( size(data,1).EQ.iem-ism+1 .AND. size(data,2).EQ.jem-jsm+2 )then ! NORTH
301  position = north
302  data_has_halos = .true.
303  je = je + 1
304  else if( size(data,1).EQ.iem-ism+2 .AND. size(data,2).EQ.jem-jsm+2 )then ! CORNER
305  position = corner
306  data_has_halos = .true.
307  ie = ie + 1; je = je + 1
308  else
309  call mpp_error( fatal, 'MPP_READ: when domain is symmetry, data must be either on ' &
310  //'compute domain or data domain with the consideration of shifting.' )
311  end if
312  else
313  if( size(data,1).EQ.ie-is+1 .AND. size(data,2).EQ.je-js+1 )then
314  data_has_halos = .false.
315  else if( size(data,1).EQ.iem-ism+1 .AND. size(data,2).EQ.jem-jsm+1 )then
316  data_has_halos = .true.
317  else
318  call mpp_error( fatal, 'MPP_READ: data must be either on compute domain or data domain.' )
319  end if
320  endif
321  halos_are_global = x_is_global .AND. y_is_global
322  if( npes.GT.1 .AND. mpp_file(unit)%threading.EQ.mpp_single )then
323  if( halos_are_global )then !you can read directly into data array
324  if( pe.EQ.0 )call read_record_( unit, field, size(data(:,:,:)), data, tindex )
325  else
326  lenx=size(data,1)
327  leny=size(data,2)
328  lenz=size(data,3)
329  len=lenx*leny*lenz
330  allocate(gdata(len))
331 ! read field on pe 0 and pass to all pes
332  if( pe.EQ.0 ) call read_record_( unit, field, len, gdata, tindex )
333 ! broadcasting global array, this can be expensive!
334  call mpp_transmit( put_data=gdata(1), plen=len, to_pe=all_pes, &
335  get_data=gdata(1), glen=len, from_pe=0 )
336  ioff = is; joff = js
337  if( data_has_halos )then
338  ioff = isd; joff = jsd
339  end if
340  do k=1,size(data,3)
341  do j=js,je
342  do i=is,ie
343  n=(i-isg+1) + (j-jsg)*lenx + (k-1)*lenx*leny
344  data(i-ioff+1,j-joff+1,k)=gdata(n)
345  enddo
346  enddo
347  enddo
348  deallocate(gdata)
349  call mpp_sync_self() ! ensure MPI_ISEND is done.
350  end if
351  else if( data_has_halos )then
352 ! for uniprocessor or multithreaded read
353 ! read compute domain as contiguous data
354 
355  allocate( cdata(is:ie,js:je,size(data,3)) )
356  call read_record_(unit,field,size(cdata(:,:,:)),cdata,tindex,domain,position,tile_count)
357 
358  data(is-isd+1:ie-isd+1,js-jsd+1:je-jsd+1,:) = cdata(:,:,:)
359  deallocate(cdata)
360  else
361  call read_record_(unit,field,size(data(:,:,:)),data,tindex,domain,position,tile_count)
362  end if
363 
364  call mpp_clock_end(mpp_read_clock)
365 
366  return
367  end subroutine mpp_read_2ddecomp_3d_
