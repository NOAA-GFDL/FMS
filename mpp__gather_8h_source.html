23 subroutine mpp_gather_1d_(sbuf, rbuf,pelist)
24 ! JWD: Did not create mpp_gather_2d because have no requirement for it
25 ! JWD: See mpp_gather_2dv below
26  mpp_type_, dimension(:), intent(in) :: sbuf
27  mpp_type_, dimension(:), intent(inout) :: rbuf
28  integer, dimension(:), intent(in), optional :: pelist(:)
29 
30  integer :: cnt, l, nproc, op_root
31  integer, allocatable :: pelist2(:)
32 
33 
34 ! If pelist is provided, the first position must be
35 ! the operation root
36  if(PRESENT(pelist))then
37  nproc = size(pelist)
38  allocate(pelist2(nproc))
39  pelist2 = pelist
40  else
41  nproc = mpp_npes()
42  allocate(pelist2(nproc))
43  pelist2 = (/ (l, l=root_pe, nproc-1+root_pe) /)
44  endif
45  op_root = pelist2(1)
46 
47  cnt = size(sbuf(:))
48  if(size(rbuf(:)) < cnt*nproc) call mpp_error(fatal, &
49  "MPP_GATHER_1D_: size(rbuf) must be at least npes*size(sbuf) ")
50 
51  !--- pre-post receiving
52  if(pe == op_root) then
53  rbuf(1:cnt) = sbuf
54  do l = 2, nproc
55  call mpp_recv(rbuf((l-1)*cnt+1), glen=cnt, from_pe=pelist2(l), block=.false., tag=comm_tag_1 )
56  enddo
57  else
58  call mpp_send(sbuf(1), plen=cnt, to_pe=op_root, tag=comm_tag_1)
59  endif
60 
61  call mpp_sync_self(check=event_recv)
62  call mpp_sync_self()
63  deallocate(pelist2)
64 end subroutine mpp_gather_1d_

66 subroutine mpp_gather_1dv_(sbuf, ssize, rbuf, rsize, pelist)
67  mpp_type_, dimension(:), intent(in) :: sbuf
68  mpp_type_, dimension(:), intent(inout) :: rbuf
69  integer, intent(in) :: ssize
70  integer, dimension(:), intent(in) :: rsize
71  integer, dimension(:), intent(in), optional :: pelist(:)
72 
73  integer :: l, nproc, pos, op_root
74  integer, allocatable :: pelist2(:)
75 
76 ! If pelist is provided, the first position must be
77 ! the operation root
78  if(PRESENT(pelist))then
79  nproc = size(pelist)
80  allocate(pelist2(nproc))
81  pelist2 = pelist
82  else
83  nproc = mpp_npes()
84  allocate(pelist2(nproc))
85  pelist2 = (/ (l, l=0+root_pe, nproc-1+root_pe) /)
86  endif
87  op_root = pelist2(1)
88 
89 
90  !--- pre-post receiving
91  if (pe .eq. op_root) then
92  pos = 1
93  do l = 1,nproc ! include op_root to simplify logic
94  if (rsize(l) == 0) then
95  cycle ! avoid ranks with no data
96  endif
97  call mpp_recv(rbuf(pos),glen=rsize(l),from_pe=pelist2(l), &
98  block=.false.,tag=comm_tag_2)
99  pos = pos + rsize(l)
100  enddo
101  endif
102  if (ssize .gt. 0) then
103  call mpp_send(sbuf(1),plen=ssize,to_pe=op_root,tag=comm_tag_2) !avoid ranks with no data
104  endif
105 
106  call mpp_sync_self(check=event_recv)
107  call mpp_sync_self()
108  deallocate(pelist2)
109 end subroutine mpp_gather_1dv_

135 subroutine mpp_gather_pelist_3d_(is, ie, js, je, nk, pelist, array_seg, data, is_root_pe, &
136  ishift, jshift)
137  integer, intent(in) :: is, ie, js, je, nk
138  integer, dimension(:), intent(in) :: pelist
139  mpp_type_, dimension(is:ie,js:je,1:nk), intent(in) :: array_seg
140  mpp_type_, dimension(:,:,:), intent(inout) :: data
141  logical, intent(in) :: is_root_pe
142  integer, optional, intent(in) :: ishift, jshift
143 
144  integer :: i, msgsize, root_pe, root_pe_test
145  integer :: i1, i2, j1, j2, ioff, joff
146  integer :: my_ind(4), gind(4,size(pelist))
147  type array3d
148  mpp_type_, dimension(:,:,:), allocatable :: data
149  endtype array3d
150  type(array3d), dimension(:), allocatable :: temp
151 
152  if (.not.any(mpp_pe().eq.pelist(:))) return
153 
154  if (is_root_pe) then
155  root_pe = mpp_pe()
156  root_pe_test = 999
157  if (.not.any(pelist(:).eq.root_pe)) call mpp_error(fatal, &
158  "fms_io(mpp_gather_pelist): root_pe not a member of pelist")
159  else
160  root_pe = 0
161  root_pe_test = -999
162  endif
163 ! need this check in case MPI-rank 0 is a member of the pelist
164  call mpp_max(root_pe_test, pelist)
165  if (root_pe_test.lt.0) call mpp_error(fatal, &
166  "fms_io(mpp_gather_pelist): root_pe not specified or not a member of the pelist")
167 ! need to make sure only one root_pe has been specified
168  call mpp_sum(root_pe, pelist)
169  if ((is_root_pe) .and. (mpp_pe().ne.root_pe)) call mpp_error(fatal, &
170  "fms_io(mpp_gather_pelist): too many root_pes specified")
171 
172 
173  ioff=0
174  joff=0
175  if (present(ishift)) ioff=ishift
176  if (present(jshift)) joff=jshift
177 
178  my_ind(1) = is
179  my_ind(2) = ie
180  my_ind(3) = js
181  my_ind(4) = je
182 
183 ! gather indices into global index on root_pe
184  if (is_root_pe) then
185  allocate(temp(1:size(pelist)))
186  do i = 1, size(pelist)
187 ! root_pe data copy - no send to self
188  if (pelist(i).eq.root_pe) then
189  gind(:,i) = my_ind(:)
190  else
191  call mpp_recv(gind(:,i:i), 4, pelist(i), .false., comm_tag_1)
192  endif
193  enddo
194  call mpp_sync_self(check=event_recv)
195  gind(1,:)=gind(1,:)+ioff
196  gind(2,:)=gind(2,:)+ioff
197  gind(3,:)=gind(3,:)+joff
198  gind(4,:)=gind(4,:)+joff
199 ! check indices to make sure they are within the range of "data"
200  if ((minval(gind).lt.1) .OR. (maxval(gind(1:2,:)).gt.size(data,1)) .OR. (maxval(gind(3:4,:)).gt.size(data,2))) &
201  call mpp_error(fatal,"fms_io(mpp_gather_pelist): specified indices (with shift) are outside of the &
202  &range of the receiving array")
203  else
204 ! non root_pe's send indices to root_pe
205  call mpp_send(my_ind(:), 4, root_pe, comm_tag_1)
206  call mpp_sync_self(check=event_send)
207  endif
208 
209 ! gather segments into data based on indices
210  if (is_root_pe) then
211  do i = 1, size(pelist)
212  if (pelist(i).ne.root_pe) then ! no send to self
213  i1 = gind(1,i)
214  i2 = gind(2,i)
215  j1 = gind(3,i)
216  j2 = gind(4,i)
217  msgsize = (i2-i1+1)*(j2-j1+1)*nk
218  allocate(temp(i)%data(i1:i2,j1:j2,1:nk))
219  call mpp_recv(temp(i)%data(i1:i2,j1:j2,1:nk), msgsize, pelist(i), .false., comm_tag_2)
220  endif
221  enddo
222  call mpp_sync_self(check=event_recv)
223 ! unbuffer/copy the data into the return array
224  do i = 1, size(pelist)
225  if (pelist(i).eq.root_pe) then
226 ! data copy - no send to self
227  data(is+ioff:ie+ioff,js+joff:je+joff,1:nk) = array_seg(is:ie,js:je,1:nk)
228  else
229  i1 = gind(1,i)
230  i2 = gind(2,i)
231  j1 = gind(3,i)
232  j2 = gind(4,i)
233  data(i1:i2,j1:j2,1:nk)=temp(i)%data(i1:i2,j1:j2,1:nk)
234  deallocate(temp(i)%data)
235  endif
236  enddo
237  deallocate(temp)
238  else
239 ! non root_pe's send data to root_pe
240  msgsize = (my_ind(2)-my_ind(1)+1) * (my_ind(4)-my_ind(3)+1) * nk
241  call mpp_send(array_seg, msgsize, root_pe, comm_tag_2)
242  call mpp_sync_self(check=event_send)
243  endif
244 
245  call mpp_sync_self()
246  return
247 
248 end subroutine mpp_gather_pelist_3d_
