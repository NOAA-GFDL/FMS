44  !> compare checksums of 2D 32-bit real arrays
45  subroutine compare_checksums_2d_r4( a, b, chk_str )
46  real(kind=r4_kind), intent(in), dimension(:,:) :: a, b !< 2D arrays to compare
47  character(len=*), intent(in) :: chk_str
48  integer(kind=i8_kind) :: sum1, sum2
49  integer :: i, j
50  integer :: pe
51  !> @note can't call mpp_sync here since there might be different number of tiles on each pe.
52  call mpp_sync_self()
53  pe = mpp_pe()
54 
55  if(size(a,1) .ne. size(b,1) .or. size(a,2) .ne. size(b,2) ) &
56  call mpp_error(fatal,'compare_checksums_2D_r4: sizes of a and b do not match')
57 
58  do j = 1, size(a,2)
59  do i = 1, size(a,1)
60  if(abs(a(i,j) - b(i,j)) .gt. 1e-5) then
61  print*, "a =", a(i,j)
62  print*, "b =", b(i,j)
63  write(*,'(a,i3,a,i3,a,i3,a,f20.9,a,f20.9)')"at the pe ", mpp_pe(), &
64  ", at point (",i,", ", j, "),a=", a(i,j), ",b=", b(i,j)
65  call mpp_error(fatal, trim(chk_str)//': value mismatch at data point.')
66  endif
67  enddo
68  enddo
69 
70  sum1 = mpp_chksum( a, (/pe/) )
71  sum2 = mpp_chksum( b, (/pe/) )
72 
73  if( sum1.EQ.sum2 )then
74  if( pe.EQ.mpp_root_pe() )call mpp_error( note, trim(chk_str)//': OK.' )
75  !> @note in some cases, even though the checksum agree, the two arrays
76  !! actually are different [e.g.,(1.1,-1.2) with (-1.1,1.2)].
77  !! Thus, we need to check the values point-by-point.
78  else
79  call mpp_error( fatal, trim(chk_str)//': checksums do not match.' )
80  end if
81  end subroutine compare_checksums_2d_r4

83  !> Compare the checksums of 2 3D 32-bit real arrays
84  subroutine compare_checksums_3d_r4( a, b, string )
85  real(kind=r4_kind), intent(in), dimension(:,:,:) :: a, b !< 3D 64-bit real arrays to compare
86  character(len=*), intent(in) :: string
87  integer(kind=i8_kind) :: sum1, sum2
88  integer :: i, j, k
89  integer :: pe
90  ! z1l can not call mpp_sync here since there might be different number of tiles on each pe.
91  call mpp_sync_self()
92  pe = mpp_pe()
93  if(size(a,1) .ne. size(b,1) .or. size(a,2) .ne. size(b,2) .or. size(a,3) .ne. size(b,3) ) &
94  call mpp_error(fatal,'compare_checkums_3d_r4: sizes of a and b do not match')
95 
96  do k = 1, size(a,3)
97  do j = 1, size(a,2)
98  do i = 1, size(a,1)
99  if(abs(a(i,j,k) - b(i,j,k)) .gt. 1e-5) then
100  write(*,'(a,i3,a,i3,a,i3,a,i3,a,f20.9,a,f20.9)') trim(string)//" at pe ", mpp_pe(), &
101  ", at point (",i,", ", j, ", ", k, "), a = ", a(i,j,k), ", b = ", b(i,j,k)
102  call mpp_error(fatal, trim(string)//': mismatch in values at data point.')
103  endif
104  enddo
105  enddo
106  enddo
107 
108  sum1 = mpp_chksum( a , (/pe/) )
109  sum2 = mpp_chksum( b , (/pe/) )
110 
111  if( sum1.EQ.sum2 )then
112  if( pe.EQ.mpp_root_pe() )call mpp_error( note, trim(string)//': OK.' )
113  !--- in some case, even though checksum agree, the two arrays
114  ! actually are different, like comparing (1.1,-1.2) with (-1.1,1.2)
115  !--- hence we need to check the value point by point.
116  else
117  write(stdunit, *)"sum1 =", sum1, mpp_pe()
118  write(stdunit, *)"sum2 =", sum2, mpp_pe()
119  write(stdunit,'(a,i3,a,i20,a,i20)')" at pe ", mpp_pe(), " sum(a)=", sum1, " sum(b)=", sum2
120  call mpp_error( fatal, trim(string)//': checksums do not match.' )
121  end if
122  end subroutine compare_checksums_3d_r4
