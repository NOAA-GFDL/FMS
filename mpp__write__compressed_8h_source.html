53  subroutine mpp_write_compressed_2d_(unit, field, domain, data, nelems_io, tstamp, default_data)
54  integer, intent(in) :: unit
55  type(fieldtype), intent(inout) :: field
56  type(domain2D), intent(inout) :: domain
57  mpp_type_, intent(inout) :: data(:,:)
58  integer, intent(in) :: nelems_io(:) ! number of compressed elements from each
59  ! member of the io_domain. It MUST have the
60  ! same order as the io_domain pelist.
61  mpp_type_, intent(in), optional :: tstamp
62  mpp_type_, intent(in), optional :: default_data
63 
64 !cdata is used to store the io-domain compressed data
65  mpp_type_, allocatable, dimension(:,:) :: cdata
66  mpp_type_, allocatable, dimension(:,:) :: sbuff,rbuff
67  mpp_type_ :: fill
68 
69  mpp_type_ :: sbuff1d(size(data))
70  mpp_type_ :: rbuff1d(size(data,2)*sum(nelems_io(:)))
71  pointer(sptr,sbuff1d); pointer(rptr,rbuff1d)
72  integer, allocatable :: pelist(:)
73  integer, allocatable :: nz_gather(:)
74  integer :: i,j,nz,nelems,mynelems,idx,npes
75  type(domain2d), pointer :: io_domain=>null()
76 
77  call mpp_clock_begin(mpp_write_clock)
78 
79  if( .NOT.module_is_initialized )call mpp_error( fatal, 'MPP_WRITE_COMPRESSED_2D_: must first call mpp_io_init.' )
80  if( .NOT.mpp_file(unit)%valid )call mpp_error( fatal, 'MPP_WRITE_COMPRESSED_2D_: invalid unit number.' )
81 
82  fill = 0
83  if(PRESENT(default_data)) fill = default_data
84 
85  io_domain=>mpp_get_io_domain(domain)
86  if (.not. ASSOCIATED(io_domain)) call mpp_error( fatal, 'MPP_WRITE_COMPRESSED_2D_: io_domain must be defined.' )
87  npes = mpp_get_domain_npes(io_domain)
88  allocate(pelist(npes))
89  call mpp_get_pelist(io_domain,pelist)
90 
91  mynelems = size(data,1)
92  nz = size(data,2)
93  nelems = sum(nelems_io(:))
94  ! Check that nz is consistent across all PEs in io_domain
95  allocate(nz_gather(npes))
96  call mpp_gather((/nz/), nz_gather, pelist)
97  if ( mpp_file(unit)%write_on_this_pe.and.maxloc(nz_gather,1).ne.minloc(nz_gather,1) ) then
98  call mpp_error( fatal, &
99  & 'MPP_WRITE_COMPRESSED_2D_: size(data,2) must be consistent across all PEs in io_domain' )
100  end if
101  deallocate(nz_gather)
102 
103  if(mpp_file(unit)%write_on_this_pe ) allocate(rbuff(nz,nelems))
104  allocate(sbuff(nz,mynelems))
105 
106  ! this matrix inversion makes for easy gather to the IO root
107  ! and a clear, concise unpack
108  do j=1,mynelems
109  do i=1,nz
110  sbuff(i,j) = data(j,i)
111  enddo; enddo
112 
113  ! Note that the gatherV implied here is asymmetric; only root needs to know the vector of recv size
114  sptr = loc(sbuff)
115  rptr = 0
116  if(mpp_file(unit)%write_on_this_pe ) rptr = loc(rbuff)
117  call mpp_gather(sbuff1d,size(sbuff),rbuff1d,nz*nelems_io(:),pelist)
118 
119  if(mpp_file(unit)%write_on_this_pe ) then
120  allocate(cdata(nelems,nz))
121  cdata = fill
122  do j=1,nz
123  do i=1,nelems
124  cdata(i,j) = rbuff(j,i)
125  enddo; enddo
126  ! cludge for now; need resizing accessor
127  field%size(1) = nelems
128  call write_record_( unit, field, nelems*nz, cdata, tstamp)
129  deallocate(rbuff,cdata)
130  endif
131 
132  deallocate(sbuff,pelist)
133 
134  call mpp_clock_end(mpp_write_clock)
135 
136  return
137  end subroutine mpp_write_compressed_2d_
