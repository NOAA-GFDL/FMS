19 !> @defgroup topography_mod topography_mod
20 !> @ingroup topography
21 !> @brief Routines for creating land surface topography fields and land-water masks
22 !! for latitude-longitude grids.
23 !> @author Bruce Wyman
24 !!
25 !! This module generates realistic mountains and land-water masks
26 !! on a specified latitude-longitude grid by interpolating from the
27 !! 1/6 degree Navy mean topography and percent water data sets.
28 !! The fields that can be generated are mean and standard deviation
29 !! of topography within the specified grid boxes; and land-ocean (or
30 !! water) mask and land-ocean (or water) fractional area.
31 !!
32 !! The interpolation scheme conserves the area-weighted average
33 !! of the input data by using module horiz_interp.
34 !!
35 !! The interfaces get_gaussian_topog and gaussian_topog_init are documented
36 !! in @ref gaussian_topog_mod
37

200 !-----------------------------------------------------------------------
201 ! --- resolution of the topography data set ---
202 ! <DATASET NAME="">
203 ! This module uses the 1/6 degree U.S. Navy mean topography
204 ! and percent water data sets.
205 !
206 ! These data sets have been re-formatted to separate 32-bit IEEE files.
207 ! The names of these files is specified by the <LINK SRC="#NAMELIST">namelist</LINK> input.
208 !
209 !The format for both files is as follows:
210 ! <PRE>
211 ! record = 1 nlon, nlat
212 ! record = 2 blon, blat
213 ! record = 3 data
214 ! </PRE>
215 !where:
216 ! <PRE>
217 ! nlon, nlat = The number of longitude and latitude points
218 ! in the horizontal grid. For the 1/6 degree
219 ! data sets this is 2160 x 1080. [integer]
220 ! blon, blat = The longitude and latitude grid box boundaries in degrees.
221 ! [real :: blon(nlon+1), blat(nlat+1)]
222 !
223 ! data = The topography or percent water data.
224 ! [real :: data(nlon,nlat)]
225 ! </PRE>
226 ! </DATASET>
227  integer :: ipts, jpts
228  integer, parameter :: compute_stdev = 123 ! use this flag to
229  ! compute st dev

257  !> @brief Returns a "realistic" mean surface height field.
258  !!
259  !> Returns realistic mountains on a latitude-longtude grid.
260  !! The returned field is the mean topography for the given grid boxes.
261  !! Computed using a conserving area-weighted interpolation.
262  !! The current input data set is the 1/6 degree Navy mean topography.
263  !!
264  !! @returns A logical value of true is returned if the surface height field was successfully
265  !! created. A value of false may be returned if the input topography data set was not readable.
266  !!
267  !! @throws FATAL, shape(zmean) is not equal to (/size(blon)-1,size(blat)-1/)
268  !! Check the input grid size and output field size.
269  function get_topog_mean_1d(blon, blat, zmean)
270 
271  real, intent(in), dimension(:) :: blon !< Longitude (radians) at grid box boundaries
272  real, intent(in), dimension(:) :: blat !< Latitude (radians) at grid box boundaries
273  real, intent(out), dimension(:,:) :: zmean !< Mean surface height(meters). Size must be
274  !! size(blon)-1 by size(blat)-1
275  logical :: get_topog_mean_1d
276 
277 !-----------------------------------------------------------------------
278  if (.not. module_is_initialized) call topography_init()
279 
280  if ( any(shape(zmean(:,:)) /= (/size(blon(:))-1,size(blat(:))-1/)) ) &
281  call error_mesg('get_topog_mean_1d','shape(zmean) is not&
282  & equal to (/size(blon)-1,size(blat)-1/))', fatal)
283 
284  get_topog_mean_1d = open_topog_file()
285 
286  if ( get_topog_mean_1d ) call interp_topog_1d ( blon, blat, zmean)
287 
288 !-----------------------------------------------------------------------
289 
290  end function get_topog_mean_1d

799  subroutine determine_ocean_points ( pctwater )
800  real, intent(inout) :: pctwater(:,:)
801  logical :: ocean(size(pctwater,1),size(pctwater,2))
802  integer :: i, j, m, n, im, ip, jm, jp, new
803 
804  real :: ocean_pct_crit = .500
805 
806  ! resolution of the grid
807  m = size(pctwater,1)
808  n = size(pctwater,2)
809 
810  ! the 1/6 degree navy percent water data set
811  ! designates ocean grid boxes as 100 percent water
812  ! all other grid boxes have <= 99 percent water
813 
814  ! set a mask for ocean grid boxes
815  ocean = (pctwater > .999)
816  new = count(ocean)
817 
818  ! set land grid boxes that have sufficient amount of water
819  ! to ocean grid boxes when they are adjacent to ocean points
820  ! iterate until there are no new ocean points
821  do
822  if (new == 0) exit
823  new = 0
824 
825  do j = 1, n
826  do i = 1, m
827  if (.not.ocean(i,j) .and. pctwater(i,j) > ocean_pct_crit) then
828  im = i-1; ip = i+1; jm = j-1; jp = j+1
829  if (im == 0) im = m
830  if (ip == m+1) ip = 1
831  if (jm == 0) jm = 1
832  if (jp == n+1) jp = n
833  ! check the 8 grid boxes that surround this grid box
834  if (ocean(im,j ) .or. ocean(ip,j ) .or. ocean(i ,jm) .or. ocean(i ,jp) .or. &
835  ocean(im,jm) .or. ocean(ip,jm) .or. ocean(ip,jp) .or. ocean(im,jp)) then
836  ocean(i,j) = .true.
837  new = new + 1
838  endif
839  endif
840  enddo
841  enddo
842  !print *, 'new=',new
843 
844  enddo
845 
846  ! final step is to elimate water percentage if land
847  where (.not.ocean) pctwater = 0.
848 
849  end subroutine determine_ocean_points
