93 module mersennetwister_mod
94 ! -------------------------------------------------------------
95  implicit none
96  private
97 
98  ! Algorithm parameters
99  ! -------
100  ! Period parameters
101  integer, parameter :: blocksize = 624, &
102  m = 397, &
103  matrix_a = -1727483681, & !< constant vector a (0x9908b0dfUL)
104  umask = -2147483648_8, & !< most significant w-r bits (0x80000000UL)
105  lmask = 2147483647 !< least significant r bits (0x7fffffffUL)
106  !> Tempering parameters
107  integer, parameter :: tmaskb= -1658038656, & !< (0x9d2c5680UL)
108  tmaskc= -272236544 !< (0xefc60000UL)
109  ! -------
110  !> @}

112  !> The type containing the state variable
113  !> @ingroup mersennetwister_mod
114  type randomnumbersequence
115  integer :: currentElement ! = blockSize
116  integer, dimension(0:blockSize -1) :: state ! = 0
117  end type randomnumbersequence

131  ! -------------------------------------------------------------
132  ! Private functions
133  ! ---------------------------
134  function mixbits(u, v)
135  integer, intent( in) :: u, v
136  integer :: mixbits
137 
138  mixbits = ior(iand(u, umask), iand(v, lmask))
139  end function mixbits
140  ! ---------------------------
141  function twist(u, v)
142  integer, intent( in) :: u, v
143  integer :: twist
144 
145  ! Local variable
146  integer, parameter, dimension(0:1) :: t_matrix = (/ 0, matrix_a /)
147 
148  twist = ieor(ishft(mixbits(u, v), -1), t_matrix(iand(v, 1)))
149  twist = ieor(ishft(mixbits(u, v), -1), t_matrix(iand(v, 1)))
150  end function twist
151  ! ---------------------------
152  subroutine nextstate(twister)
153  type(randomNumberSequence), intent(inout) :: twister
154 
155  ! Local variables
156  integer :: k
157 
158  do k = 0, blocksize - m - 1
159  twister%state(k) = ieor(twister%state(k + m), &
160  twist(twister%state(k), twister%state(k + 1)))
161  end do
162  do k = blocksize - m, blocksize - 2
163  twister%state(k) = ieor(twister%state(k + m - blocksize), &
164  twist(twister%state(k), twister%state(k + 1)))
165  end do
166  twister%state(blocksize - 1) = ieor(twister%state(m - 1), &
167  twist(twister%state(blocksize - 1), twister%state(0)))
168  twister%currentElement = 0
169 
170  end subroutine nextstate
171  ! ---------------------------
172  elemental function temper(y)
173  integer, intent(in) :: y
174  integer :: temper
175 
176  integer :: x
177 
178  ! Tempering
179  x = ieor(y, ishft(y, -11))
180  x = ieor(x, iand(ishft(x, 7), tmaskb))
181  x = ieor(x, iand(ishft(x, 15), tmaskc))
182  temper = ieor(x, ishft(x, -18))
183  end function temper

185  ! Public (but hidden) functions
186  ! --------------------
187  function initialize_scalar(seed) result(twister)
188  integer, intent(in ) :: seed
189  type(randomNumberSequence) :: twister
190 
191  integer :: i
192  ! See Knuth TAOCP Vol2. 3rd Ed. P.106 for multiplier. In the previous versions,
193  ! MSBs of the seed affect only MSBs of the array state[].
194  ! 2002/01/09 modified by Makoto Matsumoto
195 
196  twister%state(0) = iand(seed, -1)
197  do i = 1, blocksize - 1 ! ubound(twister%state)
198  twister%state(i) = 1812433253 * ieor(twister%state(i-1), &
199  ishft(twister%state(i-1), -30)) + i
200  twister%state(i) = iand(twister%state(i), -1) ! for >32 bit machines
201  end do
202  twister%currentElement = blocksize
203  end function initialize_scalar
204  ! -------------------------------------------------------------
205  function initialize_vector(seed) result(twister)
206  integer, dimension(0:), intent(in) :: seed
207  type(randomNumberSequence) :: twister
208 
209  integer :: i, j, k, nFirstLoop, nWraps
210 
211  nwraps = 0
212  twister = initialize_scalar(19650218)
213 
214  nfirstloop = max(blocksize, size(seed))
215  do k = 1, nfirstloop
216  i = mod(k + nwraps, blocksize)
217  j = mod(k - 1, size(seed))
218  if(i == 0) then
219  twister%state(i) = twister%state(blocksize - 1)
220  twister%state(1) = ieor(twister%state(1), &
221  ieor(twister%state(1-1), &
222  ishft(twister%state(1-1), -30)) * 1664525) + &
223  seed(j) + j ! Non-linear
224  twister%state(i) = iand(twister%state(i), -1) ! for >32 bit machines
225  nwraps = nwraps + 1
226  else
227  twister%state(i) = ieor(twister%state(i), &
228  ieor(twister%state(i-1), &
229  ishft(twister%state(i-1), -30)) * 1664525) + &
230  seed(j) + j ! Non-linear
231  twister%state(i) = iand(twister%state(i), -1) ! for >32 bit machines
232  end if
233  end do
234 
235  !
236  ! Walk through the state array, beginning where we left off in the block above
237  !
238  do i = mod(nfirstloop, blocksize) + nwraps + 1, blocksize - 1
239  twister%state(i) = ieor(twister%state(i), &
240  ieor(twister%state(i-1), &
241  ishft(twister%state(i-1), -30)) * 1566083941) - i ! Non-linear
242  twister%state(i) = iand(twister%state(i), -1) ! for >32 bit machines
243  end do
244 
245  twister%state(0) = twister%state(blocksize - 1)
246 
247  do i = 1, mod(nfirstloop, blocksize) + nwraps
248  twister%state(i) = ieor(twister%state(i), &
249  ieor(twister%state(i-1), &
250  ishft(twister%state(i-1), -30)) * 1566083941) - i ! Non-linear
251  twister%state(i) = iand(twister%state(i), -1) ! for >32 bit machines
252  end do
253 
254  twister%state(0) = umask
255  twister%currentElement = blocksize
256 
257  end function initialize_vector

259  ! Public functions
260  ! --------------------
261  !> Generate a random integer on the interval [0,0xffffffff]
262  !!
263  !> Equivalent to genrand_int32 in the C code.
264  !! Fortran doesn't have a type that's unsigned like C does,
265  !! so this is integers in the range -2**31 - 2**31
266  !! All functions for getting random numbers call this one,
267  !! then manipulate the result
268  function getrandomint(twister)
269  type(randomnumbersequence), intent(inout) :: twister
270  integer :: getrandomint
271 
272  if(twister%currentElement >= blocksize) call nextstate(twister)
273 
274  getrandomint = temper(twister%state(twister%currentElement))
275  twister%currentElement = twister%currentElement + 1
276 
277  end function getrandomint
278  ! --------------------
279  !> Generate a random integer on the interval [0,0x7fffffff]
280  !! or [0,2**31].
281  !! Equivalent to genrand_int31 in the C code.
282  function getrandompositiveint(twister)
283  type(randomnumbersequence), intent(inout) :: twister
284  integer :: getrandompositiveint
285 
286  ! Local integers
287  integer :: localint
288 
289  localint = getrandomint(twister)
290  getrandompositiveint = ishft(localint, -1)
291 
292  end function getrandompositiveint
293  ! --------------------
294  !> Generate a random number on [0,1]
295  !! Equivalent to genrand_real1 in the C code.
296  !! The result is stored as double precision but has 32 bit resolution
297  function getrandomreal(twister)
298  type(randomnumbersequence), intent(inout) :: twister
299  double precision :: getrandomreal
300 
301  integer :: localint
302 
303  localint = getrandomint(twister)
304  if(localint < 0) then
305  getrandomreal = dble(localint + 2.0d0**32)/(2.0d0**32 - 1.0d0)
306  else
307  getrandomreal = dble(localint )/(2.0d0**32 - 1.0d0)
308  end if
309  end function getrandomreal
310  ! --------------------
311  subroutine finalize_randomnumbersequence(twister)
312  type(randomnumbersequence), intent(inout) :: twister
313 
314  twister%currentElement = blocksize
315  twister%state(:) = 0
316  end subroutine finalize_randomnumbersequence
