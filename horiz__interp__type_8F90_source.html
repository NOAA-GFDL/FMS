113  !> @brief This statistics is for bilinear interpolation and spherical regrid.
114  subroutine stats ( dat, low, high, avg, miss, missing_value, mask )
115  real, intent(in) :: dat(:,:)
116  real, intent(out) :: low, high, avg
117  integer, intent(out) :: miss
118  real, intent(in), optional :: missing_value
119  real, intent(in), optional :: mask(:,:)
120 
121  real :: dsum, buffer_real(3)
122  integer :: pe, root_pe, npes, p, buffer_int(2), npts
123 
124  pe = mpp_pe()
125  root_pe = mpp_root_pe()
126  npes = mpp_npes()
127 
128  dsum = 0.0
129  miss = 0
130 
131  if (present(missing_value)) then
132  miss = count(dat(:,:) == missing_value)
133  low = minval(dat(:,:), dat(:,:) /= missing_value)
134  high = maxval(dat(:,:), dat(:,:) /= missing_value)
135  dsum = sum(dat(:,:), dat(:,:) /= missing_value)
136  else if(present(mask)) then
137  miss = count(mask(:,:) <= 0.5)
138  low = minval(dat(:,:),mask=mask(:,:) > 0.5)
139  high = maxval(dat(:,:),mask=mask(:,:) > 0.5)
140  dsum = sum(dat(:,:), mask=mask(:,:) > 0.5)
141  else
142  miss = 0
143  low = minval(dat(:,:))
144  high = maxval(dat(:,:))
145  dsum = sum(dat(:,:))
146  endif
147  avg = 0.0
148 
149  npts = size(dat(:,:)) - miss
150  if(pe == root_pe) then
151  do p = 1, npes - 1 ! root_pe receive data from other pe
152  ! Force use of "scalar", integer pointer mpp interface
153  call mpp_recv(buffer_real(1),glen=3, from_pe=p+root_pe, tag=comm_tag_1)
154  dsum = dsum + buffer_real(1)
155  low = min(low, buffer_real(2))
156  high = max(high, buffer_real(3))
157  call mpp_recv(buffer_int(1), glen=2, from_pe=p+root_pe, tag=comm_tag_2)
158  miss = miss + buffer_int(1)
159  npts = npts + buffer_int(2)
160  enddo
161  if(npts == 0) then
162  print*, 'Warning: no points is valid'
163  else
164  avg = dsum/real(npts)
165  endif
166  else ! other pe send data to the root_pe.
167  buffer_real(1) = dsum
168  buffer_real(2) = low
169  buffer_real(3) = high
170  ! Force use of "scalar", integer pointer mpp interface
171  call mpp_send(buffer_real(1),plen=3,to_pe=root_pe, tag=comm_tag_1)
172  buffer_int(1) = miss
173  buffer_int(2) = npts
174  call mpp_send(buffer_int(1), plen=2, to_pe=root_pe, tag=comm_tag_2)
175  endif
176 
177  call mpp_sync_self()
178 
179  return
180 
181  end subroutine stats
