99 !> Returns a surface height field that consists
100 !! of the sum of one or more Gaussian-shaped mountains.
101 !!
102 !> Returns a land surface topography that consists of a "set" of
103 !! simple Gaussian-shaped mountains. The height, position,
104 !! width, and elongation of the mountains can be controlled
105 !! by variables in the namelist.
106 subroutine gaussian_topog_init ( lon, lat, zsurf )
107 
108 real, intent(in) :: lon(:) !< The mean grid box longitude in radians
109 real, intent(in) :: lat(:) !< The mean grid box latitude in radians
110 real, intent(out) :: zsurf(:,:) !< The surface height (meters). Size must be size(lon) by size(lat)
111 
112 integer :: n
113 
114  if (.not.module_is_initialized) then
115  call write_version_number("GAUSSIAN_TOPOG_MOD", version)
116  endif
117 
118  if(any(shape(zsurf) /= (/size(lon(:)),size(lat(:))/))) then
119  call error_mesg ('get_gaussian_topog in topography_mod', &
120  'shape(zsurf) is not equal to (/size(lon),size(lat)/)', fatal)
121  endif
122 
123  if (do_nml) call read_namelist
124 
125 ! compute sum of all non-zero mountains
126  zsurf(:,:) = 0.
127  do n = 1, maxmts
128  if ( height(n) == 0. ) cycle
129  zsurf = zsurf + get_gaussian_topog( lon, lat, height(n), &
130  olon(n), olat(n), wlon(n), wlat(n), rlon(n), rlat(n))
131  enddo
132  module_is_initialized = .true.
133 
134 end subroutine gaussian_topog_init

138 !> @brief Returns a simple surface height field that consists of a single
139 !! Gaussian-shaped mountain.
140 !!
141 !> The height, position, width, and elongation of the mountain
142 !! is controlled by optional arguments.
143 !! @param real lon The mean grid box longitude in radians.
144 !! @param real lat The mean grid box latitude in radians.
145 !! @param real height Maximum surface height in meters.
146 !! @param real olond, olatd Position/origin of mountain in degrees longitude and latitude.
147 !! This is the location of the maximum height.
148 !! @param real wlond, wlatd Gaussian half-width of mountain in degrees longitude and latitude.
149 !! @param real rlond, rlatd Ridge half-width of mountain in degrees longitude and latitude.
150 !! This is the elongation of the maximum height.
151 !! @param real zsurf The surface height (in meters).
152 !! The size of the returned field is size(lon) by size(lat).
153 !! </OUT>
154 !!
155 !! @throws FATAL shape(zsurf) is not equal to (/size(lon),size(lat)/)
156 !! Check the input grid size and output field size.
157 !! The input grid is defined at the midpoint of grid boxes.
158 !!
159 !! @note
160 !! Mountains do not wrap around the poles.
161 !
162 !! <br>Example usage:
163 !! @code{.F90} zsurf = <B>get_gaussian_topog</B> ( lon, lat, height
164 !! [, olond, olatd, wlond, wlatd, rlond, rlatd ] )@endcode
165 function get_gaussian_topog ( lon, lat, height, &
166  olond, olatd, wlond, wlatd, rlond, rlatd ) &
167  result( zsurf )
168 
169 real, intent(in) :: lon(:), lat(:)
170 real, intent(in) :: height
171 real, intent(in), optional :: olond, olatd, wlond, wlatd, rlond, rlatd
172 real :: zsurf(size(lon,1),size(lat,1))
173 
174 integer :: i, j
175 real :: olon, olat, wlon, wlat, rlon, rlat
176 real :: tpi, dtr, dx, dy, xx, yy
177 
178  if (do_nml) call read_namelist
179 
180 ! no need to compute mountain if height=0
181  if ( height == 0. ) then
182  zsurf(:,:) = 0.
183  return
184  endif
185 
186  tpi = 2.0*pi
187  dtr = tpi/360.
188 
189 ! defaults and convert degrees to radians (dtr)
190  olon = 90.*dtr; if (present(olond)) olon=olond*dtr
191  olat = 45.*dtr; if (present(olatd)) olat=olatd*dtr
192  wlon = 15.*dtr; if (present(wlond)) wlon=wlond*dtr
193  wlat = 15.*dtr; if (present(wlatd)) wlat=wlatd*dtr
194  rlon = 0. ; if (present(rlond)) rlon=rlond*dtr
195  rlat = 0. ; if (present(rlatd)) rlat=rlatd*dtr
196 
197 ! compute gaussian-shaped mountain
198  do j=1,size(lat(:))
199  dy = abs(lat(j) - olat) ! dist from y origin
200  yy = max(0., dy-rlat)/wlat
201  do i=1,size(lon(:))
202  dx = abs(lon(i) - olon) ! dist from x origin
203  dx = min(dx, abs(dx-tpi)) ! To ensure that: -pi <= dx <= pi
204  xx = max(0., dx-rlon)/wlon
205  zsurf(i,j) = height*exp(-xx**2 - yy**2)
206  enddo
207  enddo
208 
209 end function get_gaussian_topog
