19 !> @author Tom Robinson
20 !> @description This test program is for testing the mpp_pe routine. The
21 !! test initializes mpp, then calls the integer function MPP_PE. MPP_PE
22 !! returns a positive integer value. If the value returned by MPP_PE is
23 !! less than 0, then an error is thrown.
24 program test_mpp_p5
25 
26  use mpp_mod, only : mpp_init, mpp_init_test_peset_allocated, mpp_pe, mpp_npes, &
27  stderr, stdout, mpp_error, fatal
28 
29  implicit none
30  integer :: ierr
31  integer :: total_pes !< The total number of PEs returned from mpp_npes
32  integer :: my_mpp_pe !< The unique PE identifier
33 
34 
35 !> Initialize MPI to do what mpp_init would do
36 
37  call mpp_init(test_level=mpp_init_test_peset_allocated)
38 
39 !> Get the total number of PEs
40  total_pes = mpp_npes()
41 !> Get the PE number
42  my_mpp_pe = mpp_pe()
43 !> Check that the PE number is between 0 and npes-1
44  if (my_mpp_pe < 0) then
45  call mpp_error(fatal, "The PE number is less than 0")
46  elseif (my_mpp_pe > total_pes-1) then
47  call mpp_error(fatal, "The PE number is greater than npes-1")
48  endif
49 !> Finalize mpp
50  call mpi_finalize (ierr)
51 end program test_mpp_p5
