20 !> @defgroup amip_interp_mod amip_interp_mod
21 !> @ingroup amip_interp
22 !> @brief Provides observed sea surface temperature and ice mask data sets that have been
23 !! interpolated onto your model's grid.
24 !!
25 !> @author Bruce Wyman
26 !!
27 !> When using these routines three possible data sets are available:
28 !!
29 !! 1. AMIP http://www.pcmdi.github.io/mips/amip from Jan 1979 to Jan 1989 (2 deg x 2 deg)
30 !! 2. Reynolds OI @ref amip_interp.rey_oi.txt from Nov 1981 to Jan 1999 (1 deg x 1 deg)
31 !! 3. Reynolds EOF podaac.jpl.nasa.gov/ from Jan 1950 to Dec 1998 (2 deg x 2 deg)
32 !!
33 !! All original data are observed monthly means. This module
34 !! interpolates linearly in time between pairs of monthly means.
35 !! Horizontal interpolation is done using the horiz_interp module.
36 !!
37 !! When a requested date falls outside the range of dates available
38 !! a namelist option allows for use of the climatological monthly
39 !! mean values which are computed from all of the data in a particular
40 !! data set. \n
41 !! \n AMIP 1:\n
42 !! from Jan 1979 to Jan 1989 (2 deg x 2 deg).\n\n
43 !! Reynolds OI:\n
44 !! from Nov 1981 to Jan 1999 (1 deg x 1 deg)\n
45 !! The analysis uses in situ and satellite SST's plus
46 !! SST's simulated by sea-ice cover.\n\n
47 !! Reynold's EOF:\n
48 !! from Jan 1950 to Dec 1998 (2 deg x 2 deg)\n
49 !! NCEP Reynolds Historical Reconstructed Sea Surface Temperature
50 !! The analysis uses both in-situ SSTs and satellite derived SSTs
51 !! from the NOAA Advanced Very High Resolution Radiometer.
52 !! In-situ data is used from 1950 to 1981, while both AVHRR derived
53 !! satellite SSTs and in-situ data are used from 1981 to the
54 !! end of 1998.
55 !!
56 !> @note The data set used by this module have been reformatted as 32-bit IEEE.
57 !! The data values are packed into 16-bit integers.
58 !!
59 !! The data sets are read from the following files:
60 !!
61 !! amip1 INPUT/amip1_sst.data
62 !! reynolds_io INPUT/reyoi_sst.data
63 !! reynolds_eof INPUT/reynolds_sst.data
64 !!
65 !> @var character(len=24) data_set
66 !! Name/type of SST data that will be used.
67 !! Possible values (case-insensitive) are:
68 !! 1) amip1
69 !! 2) reynolds_eof
70 !! 3) reynolds_oi
71 !! See the @ref amip_interp_oi page for more information
72 !! @var character(len=16) date_out_of_range
73 !! Controls the use of climatological monthly mean data when
74 !! the requested date falls outside the range of the data set.<BR/>
75 !! Possible values are:
76 !! <PRE>
77 !! fail - program will fail if requested date is prior
78 !! to or after the data set period.
79 !! initclimo - program uses climatological requested data is
80 !! prior to data set period and will fail if
81 !! requested date is after data set period.
82 !! climo - program uses climatological data anytime.
83 !! </PRE>
84 !! @var real tice_crit
85 !! Freezing point of sea water in degC or degK. Defaults to -1.80
86 !! @var integer verbose
87 !! Controls printed output, 0 <= verbose <= 3, default=0
88 !! additional parameters for controlling zonal prescribed sst ----
89 !! these parameters only have an effect when use_zonal=.true. ----
90 !! @var logical use_zonal
91 !! Flag to selected zonal sst or data set. Default=.false.
92 !! @var real teq
93 !! sst at the equator. Default=305
94 !! @var real tdif
95 !! Equator to pole sst difference. Default=50
96 !! @var real tann
97 !! Amplitude of annual cycle. Default=20
98 !! @var real tlag
99 !! Offset for time of year (for annual cycle). Default=0.875
100 !! @var integer amip_date
101 !! Single calendar date in integer "(year,month,day)" format
102 !! that is used only if set with year>0, month>0, day>0.
103 !! If used, model calendar date is replaced by this date,
104 !! but model time of day is still used to determine ice/sst.
105 !! Used for repeating-single-day (rsd) experiments.
106 !! Default=/-1,-1,-1/
107 !! @var real sst_pert
108 !! Temperature perturbation in degrees Kelvin added onto the SST.
109 !! The perturbation is globally-uniform (even near sea-ice).
110 !! It is only used when abs(sst_pert) > 1.e-4. SST perturbation runs
111 !! may be useful in accessing model sensitivities.
112 !! Default=0.

360 !> Retrieve sea surface temperature data and interpolated grid
361 subroutine get_amip_sst (Time, Interp, sst, err_msg, lon_model, lat_model)
362 
363  type (time_type), intent(in) :: time !< Time to interpolate
364  type (amip_interp_type), intent(inout) :: interp !< Holds data for interpolation
365  real, intent(out) :: sst(:,:) !< Sea surface temperature data
366  character(len=*), optional, intent(out) :: err_msg !< Holds error message string if present
367 
368  real, dimension(mobs,nobs) :: sice
369 
370  integer :: year1, year2, month1, month2
371  real :: fmonth
372  type (date_type) :: date1, date2, udate1, udate2
373 
374  type(time_type) :: amip_time
375  integer :: tod(3),dum(3)
376 
377 ! add by JHC
378  real, intent(in), dimension(:,:), optional :: lon_model, lat_model
379  real :: pert
380  integer :: i, j, mobs_sst, nobs_sst
381  integer :: jhctod(6)
382  type (time_type) :: udate
383  character(len=4) :: yyyy
384  integer :: nrecords, ierr, k, yr, mo, dy
385  integer, dimension(:), allocatable :: ryr, rmo, rdy
386  character(len=30) :: time_unit
387  real, dimension(:), allocatable :: timeval
388  character(len=maxc) :: ncfilename
389  type(fmsnetcdffile_t) :: fileobj
390  logical :: the_file_exists
391 ! end add by JHC
392  logical, parameter :: debug = .false. !> switch for debugging output
393  !> These are fms_io specific
394  integer :: unit
395 
396  if(present(err_msg)) err_msg = ''
397  if(.not.interp%I_am_initialized) then
398  if(fms_error_handler('get_amip_sst','The amip_interp_type variable is not initialized',err_msg)) return
399  endif
400 
401 !-----------------------------------------------------------------------
402 !----- compute zonally symetric sst ---------------
403 
404  if ( use_ncep_sst .and. forecast_mode ) no_anom_sst = .false.
405 
406  if (all(amip_date>0)) then
407  call get_date(time,dum(1),dum(2),dum(3),tod(1),tod(2),tod(3))
408  amip_time = set_date(amip_date(1),amip_date(2),amip_date(3),tod(1),tod(2),tod(3))
409  else
410  amip_time = time
411  endif
412 
413 ! add by JHC
414 if ( .not.use_daily ) then
415 ! end add by JHC
416 
417  if ( .not. allocated(temp1) ) allocate (temp1(mobs,nobs))
418  if ( .not. allocated(temp2) ) allocate (temp2(mobs,nobs))
419 
420  if (use_zonal) then
421  call zonal_sst (amip_time, sice, temp1)
422  call horiz_interp ( interp%Hintrp, temp1, sst )
423  else
424 
425 !-----------------------------------------------------------------------
426 !---------- get new observed sea surface temperature -------------------
427 
428 ! ---- time interpolation for months -----
429  call time_interp (amip_time, fmonth, year1, year2, month1, month2)
430 ! ---- force climatology ----
431  if (interp % use_climo) then
432  year1=0; year2=0
433  endif
434  if (interp % use_annual) then
435  year1=0; year2=0
436  month1=0; month2=0
437  endif
438 ! ---------------------------
439 
440  date1 = date_type( year1, month1, 0 )
441  date2 = date_type( year2, month2, 0 )
442 
443 ! -- open/rewind file --
444  unit = -1
445 !-----------------------------------------------------------------------
446 
447 
448  if (date1 /= interp % Date1) then
449 ! ---- use Date2 for Date1 ----
450  if (date1 == interp % Date2) then
451  interp % Date1 = interp % Date2
452  interp % data1 = interp % data2
453  temp1(:,:) = temp2(:,:) ! SJL BUG fix: June 24, 2011
454  else
455  call read_record ('sst', date1, udate1, temp1)
456  if ( use_ncep_sst .and. (.not. no_anom_sst) ) then
457  temp1(:,:) = temp1(:,:) + sst_anom(:,:)
458  endif
459  call horiz_interp ( interp%Hintrp, temp1, interp%data1 )
460  call clip_data ('sst', interp%data1)
461  interp % Date1 = date1
462  endif
463  endif
464 
465 !-----------------------------------------------------------------------
466 
467  if (date2 /= interp % Date2) then
468  call read_record ('sst', date2, udate2, temp2)
469  if ( use_ncep_sst .and. (.not. no_anom_sst) ) then
470  temp2(:,:) = temp2(:,:) + sst_anom(:,:)
471  endif
472  call horiz_interp ( interp%Hintrp, temp2, interp%data2 )
473  call clip_data ('sst', interp%data2)
474  interp % Date2 = date2
475  endif
476 
477 !-----------------------------------------------------------------------
478 !---------- time interpolation (between months) of sst's ---------------
479 !-----------------------------------------------------------------------
480  sst = interp % data1 + fmonth * (interp % data2 - interp % data1)
481 
482 !-------------------------------------------------------------------------------
483 ! SJL mods for NWP and TCSF ---
484 ! Nudging runs: (Note: NCEP SST updated only every 6-hr)
485 ! Compute SST anomaly from global SST datasets for subsequent forecast runs
486 !-------------------------------------------------------------------------------
487  if ( use_ncep_sst .and. no_anom_sst ) then
488  sst_anom(:,:) = sst_ncep(:,:) - (temp1(:,:) + fmonth*(temp2(:,:) - temp1(:,:)) )
489  call horiz_interp ( interp%Hintrp, sst_ncep, sst )
490  call clip_data ('sst', sst)
491  endif
492 
493 !! DEBUG CODE
494  if (debug) then
495  call get_date(amip_time,jhctod(1),jhctod(2),jhctod(3),jhctod(4),jhctod(5),jhctod(6))
496  if (mpp_pe() == 0) then
497  write (*,200) 'JHC: use_daily = F, AMIP_Time: ',jhctod(1),jhctod(2),jhctod(3),jhctod(4),jhctod(5), &
498  & jhctod(6)
499  write (*,300) 'JHC: use_daily = F, interped SST: ', sst(1,1),sst(5,5),sst(10,10)
500  endif
501  endif
502 
503 
504  endif
505 
506 ! add by JHC
507 else
508  call get_date(amip_time,jhctod(1),jhctod(2),jhctod(3),jhctod(4),jhctod(5),jhctod(6))
509  if (mpp_pe() == mpp_root_pe()) write(*,200) 'amip_interp_mod: use_daily = T, Amip_Time = ',jhctod(1), &
510  & jhctod(2),jhctod(3),jhctod(4),jhctod(5),jhctod(6)
511 
512  yr = jhctod(1); mo = jhctod(2); dy = jhctod(3)
513 
514  write (yyyy,'(i4)') jhctod(1)
515 
516  file_name_sst = 'INPUT/' // 'sst.day.mean.'//yyyy//'.v2.nc'
517  ncfilename = trim(file_name_sst)
518  time_unit = 'days since 1978-01-01 00:00:00'
519 
520  mobs_sst = 1440; nobs_sst = 720
521 
522  call set_sst_grid_edges_daily(mobs_sst, nobs_sst)
523  call horiz_interp_new ( interp%Hintrp2, lon_bnd, lat_bnd, &
524  lon_model, lat_model, interp_method="bilinear" )
525 
526  the_file_exists = fms2_io_file_exists(ncfilename)
527 
528  if ( (.NOT. the_file_exists) ) then
529  call mpp_error ('amip_interp_mod', &
530  'cannot find daily SST input data file: '//trim(ncfilename), note)
531  else
532  if (mpp_pe() == mpp_root_pe()) call mpp_error ('amip_interp_mod', &
533  'Reading NetCDF formatted daily SST from: '//trim(ncfilename), note)
534 
535  if(.not. open_file(fileobj, trim(ncfilename), 'read')) &
536  call error_mesg ('get_amip_sst', 'Error in opening file '//trim(ncfilename), fatal)
537 
538  call get_dimension_size(fileobj, 'TIME', nrecords)
539  if (nrecords < 1) call mpp_error('amip_interp_mod', &
540  'Invalid number of SST records in daily SST data file: '//trim(ncfilename), fatal)
541  allocate(timeval(nrecords), ryr(nrecords), rmo(nrecords), rdy(nrecords))
542  call fms2_io_read_data(fileobj, 'TIME', timeval)
543 !!! DEBUG CODE
544  if(debug) then
545  if (mpp_pe() == 0) then
546  print *, 'JHC: nrecords = ', nrecords
547  print *, 'JHC: TIME = ', timeval
548  endif
549  endif
550 
551  ierr = 1
552  do k = 1, nrecords
553
