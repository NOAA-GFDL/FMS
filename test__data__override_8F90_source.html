312  subroutine get_grid
313  real, allocatable, dimension(:,:,:) :: lon_vert_glo, lat_vert_glo
314  real, allocatable, dimension(:,:) :: lon_global, lat_global
315  integer, dimension(4) :: siz
316  character(len=128) :: message
317 
318 
319  if(field_exist(grid_file, 'x_T')) then
320  call field_size(grid_file, 'x_T', siz)
321  if(siz(1) /= nlon .or. siz(2) /= nlat) then
322  write(message,'(a,2i4)') 'x_T is wrong shape. shape(x_T)=',siz(1:2)
323  call error_mesg('test_data_override', trim(message), fatal)
324  endif
325  allocate(lon_vert_glo(nlon,nlat,4), lat_vert_glo(nlon,nlat,4) )
326  allocate(lon_global(nlon,nlat ), lat_global(nlon,nlat ) )
327  call read_data(trim(grid_file), 'x_vert_T', lon_vert_glo, no_domain=.true.)
328  call read_data(trim(grid_file), 'y_vert_T', lat_vert_glo, no_domain=.true.)
329  lon_global(:,:) = (lon_vert_glo(:,:,1) + lon_vert_glo(:,:,2) + lon_vert_glo(:,:,3) + lon_vert_glo(:,:,4))*0.25
330  lat_global(:,:) = (lat_vert_glo(:,:,1) + lat_vert_glo(:,:,2) + lat_vert_glo(:,:,3) + lat_vert_glo(:,:,4))*0.25
331  else if(field_exist(grid_file, "geolon_t" ) ) then
332  call field_size(grid_file, 'geolon_vert_t', siz)
333  if(siz(1) /= nlon+1 .or. siz(2) /= nlat+1) then
334  write(message,'(a,2i4)') 'geolon_vert_t is wrong shape. shape(geolon_vert_t)=',siz(1:2)
335  call error_mesg('test_data_override', trim(message), fatal)
336  endif
337  allocate(lon_vert_glo(nlon+1,nlat+1,1), lat_vert_glo(nlon+1,nlat+1,1))
338  allocate(lon_global(nlon, nlat ), lat_global(nlon, nlat ))
339  call read_data(trim(grid_file), 'geolon_vert_t', lon_vert_glo, no_domain=.true.)
340  call read_data(trim(grid_file), 'geolat_vert_t', lat_vert_glo, no_domain=.true.)
341 
342  do i = 1, nlon
343  do j = 1, nlat
344  lon_global(i,j) = (lon_vert_glo(i,j,1) + lon_vert_glo(i+1,j,1) + &
345  lon_vert_glo(i+1,j+1,1) + lon_vert_glo(i,j+1,1))*0.25
346  lat_global(i,j) = (lat_vert_glo(i,j,1) + lat_vert_glo(i+1,j,1) + &
347  lat_vert_glo(i+1,j+1,1) + lat_vert_glo(i,j+1,1))*0.25
348  enddo
349  enddo
350  else if( field_exist(grid_file, "ocn_mosaic_file") ) then ! reading from mosaic file
351  call field_size(tile_file, 'area', siz)
352  if(siz(1) /= nlon*2 .or. siz(2) /= nlat*2) then
353  write(message,'(a,2i4)') 'area is wrong shape. shape(area)=',siz(1:2)
354  call error_mesg('test_data_override', trim(message), fatal)
355  endif
356  allocate(lon_vert_glo(siz(1)+1,siz(2)+1,1), lat_vert_glo(siz(1)+1,siz(2)+1,1))
357  allocate(lon_global(nlon, nlat ), lat_global(nlon, nlat ))
358  call read_data( tile_file, 'x', lon_vert_glo, no_domain=.true.)
359  call read_data( tile_file, 'y', lat_vert_glo, no_domain=.true.)
360  do j = 1, nlat
361  do i = 1, nlon
362  lon_global(i,j) = lon_vert_glo(i*2,j*2,1)
363  lat_global(i,j) = lat_vert_glo(i*2,j*2,1)
364  end do
365  end do
366  end if
367 
368  allocate(lon(is:ie,js:je), lat(is:ie,js:je))
369  lon = lon_global(is:ie,js:je)
370  lat = lat_global(is:ie,js:je)
371 
372  deallocate(lon_vert_glo)
373  deallocate(lat_vert_glo)
374  deallocate(lon_global)
375  deallocate(lat_global)
376 
377  end subroutine get_grid

379  subroutine test_unstruct_grid( type, Time )
380 
381  character(len=*), intent(in) :: type

393  integer :: layout_tripolar(2) = (/0,0/)
394  integer :: layout_ensemble(2) = (/0,0/)
395  logical :: do_sleep = .false.

421  errunit = stderr()
422  !--- check the type
423  select case(type)
424  case ( 'Cubic-Grid' )
425  if( nx_cubic == 0 ) then
426  call mpp_error(note,'test_unstruct_update: for Cubic_grid mosaic, nx_cubic is zero, '//&
427  'No test is done for Cubic-Grid mosaic. ' )
