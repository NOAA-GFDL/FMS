98 !###############################################################################
99  !> Get exchange grid information from mosaic xgrid file.
100  !!
101  !> <br> Example usage:
102  !! @code{.F90}
103  !! call get_mosaic_xgrid(xgrid_file, nxgrid, i1, j1, i2, j2, area)
104  !! @endcode
105  subroutine get_mosaic_xgrid(xgrid_file, i1, j1, i2, j2, area, ibegin, iend)
106  character(len=*), intent(in) :: xgrid_file !< The file that contains exchange grid information.
107  integer, intent(inout) :: i1(:), j1(:) !< i and j-index in grid 1 of exchange field
108  integer, intent(inout) :: i2(:), j2(:) !< i and j-index in grid 2 of exchange field
109  real, intent(inout) :: area(:) !< area of the exchange grid. The area is sclaed to
110  !! represent unit earth area.
111  integer, optional, intent(in) :: ibegin, iend
112 
113  integer :: start(4), nread(4), istart
114  real, dimension(2, size(i1(:))) :: tile1_cell, tile2_cell
115  integer :: nxgrid, n
116  real :: garea
117  real :: get_global_area;
118 
119  garea = get_global_area();
120 
121  ! When start and nread present, make sure nread(1) is the same as the size of the data
122  if(present(ibegin) .and. present(iend)) then
123  istart = ibegin
124  nxgrid = iend - ibegin + 1
125  if(nxgrid .NE. size(i1(:))) call mpp_error(fatal, .NE."get_mosaic_xgrid: nxgrid size(i1(:))")
126  if(nxgrid .NE. size(j1(:))) call mpp_error(fatal, .NE."get_mosaic_xgrid: nxgrid size(j1(:))")
127  if(nxgrid .NE. size(i2(:))) call mpp_error(fatal, .NE."get_mosaic_xgrid: nxgrid size(i2(:))")
128  if(nxgrid .NE. size(j2(:))) call mpp_error(fatal, .NE."get_mosaic_xgrid: nxgrid size(j2(:))")
129  if(nxgrid .NE. size(area(:))) call mpp_error(fatal, .NE."get_mosaic_xgrid: nxgrid size(area(:))")
130  else
131  istart = 1
132  nxgrid = size(i1(:))
133  endif
134 
135  start = 1; nread = 1
136  start(1) = istart; nread(1) = nxgrid
137  call read_compressed(xgrid_file, 'xgrid_area', area, start=start, nread=nread, threading=mpp_multi)
138  start = 1; nread = 1
139  nread(1) = 2
140  start(2) = istart; nread(2) = nxgrid
141  call read_compressed(xgrid_file, 'tile1_cell', tile1_cell, start=start, nread=nread, threading=mpp_multi)
142  call read_compressed(xgrid_file, 'tile2_cell', tile2_cell, start=start, nread=nread, threading=mpp_multi)
143 
144  do n = 1, nxgrid
145  i1(n) = int(tile1_cell(1,n))
146  j1(n) = int(tile1_cell(2,n))
147  i2(n) = int(tile2_cell(1,n))
148  j2(n) = int(tile2_cell(2,n))
149  area(n) = area(n)/garea
150  end do
151 
152  return
153 
154  end subroutine get_mosaic_xgrid

343 function transfer_to_model_index(istart, iend, refine_ratio)
344  integer, intent(inout) :: istart, iend
345  integer :: refine_ratio
346  integer :: transfer_to_model_index
347  integer :: istart_in, iend_in
348 
349  istart_in = istart
350  iend_in = iend
351 
352  if( istart_in == iend_in ) then
353  transfer_to_model_index = 0
354  istart = (istart_in + 1)/refine_ratio
355  iend = istart
356  else
357  transfer_to_model_index = 1
358  if( iend_in > istart_in ) then
359  istart = istart_in + 1
360  iend = iend_in
361  else
362  istart = istart_in
363  iend = iend_in + 1
364  endif
365  if( mod(istart, refine_ratio) .NE. 0 .OR. mod(iend,refine_ratio) .NE. 0) call mpp_error(fatal, &
366  "mosaic_mod(transfer_to_model_index): mismatch between refine_ratio and istart/iend")
367  istart = istart/refine_ratio
368  iend = iend/refine_ratio
369 
370  endif
371 
372  return
373 
374 end function transfer_to_model_index

448  function parse_string(string, set, value)
449  character(len=*), intent(in) :: string
450  character(len=*), intent(in) :: set
451  character(len=*), intent(out) :: value(:)
452  integer :: parse_string
453  integer :: nelem, length, first, last
454 
455  nelem = size(value(:))
456  length = len_trim(string)
457 
458  first = 1; last = 0
459  parse_string = 0
460 
461  do while(first .LE. length)
462  parse_string = parse_string + 1
463  if(parse_string>nelem) then
464  call mpp_error(fatal, "mosaic_mod(parse_string) : number of element is greater than size(value(:))")
465  endif
466  last = first - 1 + scan(string(first:length), set)
467  if(last == first-1 ) then ! not found, end of string
468  value(parse_string) = string(first:length)
469  exit
470  else
471  if(last <= first) then
472  call mpp_error(fatal, "mosaic_mod(parse_string) : last <= first")
473  endif
474  value(parse_string) = string(first:(last-1))
475  first = last + 1
476  ! scan to make sure the next is not the character in the set
477  do while (first == last+1)
478  last = first - 1 + scan(string(first:length), set)
479  if(last == first) then
480  first = first+1
481  else
482  exit
483  endif
484  end do
485  endif
486  enddo
487 
488  return
489 
490  end function parse_string
