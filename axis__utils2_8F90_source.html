153  character(len=128) :: buffer
154  integer, dimension(:), allocatable :: dim_sizes
155  real(kind=r4_kind), dimension(:), allocatable :: r32

172 
173  reproduce_null_char_bug = .false.

176  buffer = ""
177  if (variable_att_exists(fileobj, name, "edges")) then
178  !! If the reproduce_null_char_bug flag is turned on fms2io will not remove the null character
179  call get_variable_attribute(fileobj, name, "edges", buffer(1:128), &
180  reproduce_null_char_bug_flag=reproduce_null_char_bug)
181 
182  !! Check for a null character here, if it exists *_bnds will be calculated instead of read in
183  if (reproduce_null_char_bug) then
184  i = 0
185  i = index(buffer, char(0))
186  if (i > 0) buffer = ""
187  endif
188  elseif (variable_att_exists(fileobj, name, "bounds")) then
189  !! If the reproduce_null_char_bug flag is turned on fms2io will not remove the null character
190  call get_variable_attribute(fileobj, name, "bounds", buffer(1:128), &
191  reproduce_null_char_bug_flag=reproduce_null_char_bug)
192 
193  !! Check for a null character here, if it exists *_bnds will be calculated instead of read in
194  if (reproduce_null_char_bug) then
195  i = 0
196  i = index(buffer, char(0))
197  if (i > 0) buffer = ""
198  endif
199  endif
200  if (trim(buffer) .ne. "") then
201  ndims = get_variable_num_dimensions(fileobj, buffer)
202  allocate(dim_sizes(ndims))
203  call get_variable_size(fileobj, buffer, dim_sizes)
204  if (size(dim_sizes) .eq. 1) then
205  if (dim_sizes(1) .ne. n+1) then
206  call mpp_error(fatal, "axis_edges: incorrect size of edge data.")
207  endif
208  call read_data(fileobj, buffer, edge_data)
209  elseif (size(dim_sizes) .eq. 2) then
210  if (dim_sizes(1) .ne. 2) then
211  call mpp_error(fatal, "axis_edges: first dimension of edge must be of size 2")
212  endif
213  if (dim_sizes(2) .ne. n) then
214  call mpp_error(fatal, "axis_edges: incorrect size of edge data.")
215  endif
216  select type (edge_data)
217  type is (real(kind=r4_kind))
218  allocate(r322d(dim_sizes(1), dim_sizes(2)))
219  call read_data(fileobj, buffer, r322d)
220  edge_data(1:dim_sizes(2)) = r322d(1,:)
221  edge_data(dim_sizes(2)+1) = r322d(2,dim_sizes(2))
222  deallocate(r322d)
223  type is (real(kind=r8_kind))
224  allocate(r642d(dim_sizes(1), dim_sizes(2)))
225  call read_data(fileobj, buffer, r642d)
226  edge_data(1:dim_sizes(2)) = r642d(1,:)
227  edge_data(dim_sizes(2)+1) = r642d(2,dim_sizes(2))
228  deallocate(r642d)
229  class default
230  call mpp_error(fatal, "axis_edges: unsupported kind.")
231  end select
232  endif
233  deallocate(dim_sizes)
234  else
235  select type (edge_data)
236  type is (real(kind=r4_kind))
237  allocate(r32(n))
238  call read_data(fileobj, name, r32)
239  do i = 2, n
240  edge_data(i) = r32(i-1) + 0.5_r4_kind*(r32(i) - r32(i-1))
241  enddo
242  edge_data(1) = r32(1) - 0.5_r4_kind*(r32(2) - r32(1))
243  if (abs(edge_data(1)) .lt. 1.e-10) then
244  edge_data(1) = 0._r4_kind
245  endif
246  edge_data(n+1) = r32(n) + 0.5_r4_kind*(r32(n) - r32(n-1))
247  deallocate(r32)
248  type is (real(kind=r8_kind))
249  allocate(r64(n))
250  call read_data(fileobj, name, r64)
251  do i = 2, n
252  edge_data(i) = r64(i-1) + 0.5_r8_kind*(r64(i) - r64(i-1))
253  enddo
254  edge_data(1) = r64(1) - 0.5_r8_kind*(r64(2) - r64(1))
255  if (abs(edge_data(1)) .lt. 1.d-10) then
256  edge_data(1) = 0._r8_kind
257  endif
258  edge_data(n+1) = r64(n) + 0.5_r8_kind*(r64(n) - r64(n-1))
259  deallocate(r64)
260  class default
261  call mpp_error(fatal, "axis_edges: unsupported kind.")
262  end select
263  endif
264 end subroutine axis_edges

306  !> @brief Returns lon_strt <= longitude <= lon_strt+360
307  !! @return real lon_in_range
308  function lon_in_range(lon, l_strt)
309  real, intent(in) :: lon, l_strt
310  real :: lon_in_range
311  real :: l_end
312 
313  lon_in_range = lon
314  l_end = l_strt+360.
315 
316  if (abs(lon_in_range - l_strt) < 1.e-4) then
317  lon_in_range = l_strt
318  return
319  endif
320 
321  if (abs(lon_in_range - l_end) < 1.e-4) then
322  lon_in_range = l_strt
323  return
324  endif
325 
326  do
327  if (lon_in_range < l_strt) then
328  lon_in_range = lon_in_range + f360
329  else if (lon_in_range > l_end) then
330  lon_in_range = lon_in_range - f360
331  else
332  exit
333  end if
334  end do
335 
336  end function lon_in_range

338  !> @brief Returns monotonic array of longitudes s.t., lon_strt <= lon(:) <= lon_strt+360.
339  !!
340  !> <br>The first istrt-1 entries are moved to the end of the array:
341  !!
342  !! e.g.
343  !! lon = 0 1 2 3 4 5 ... 358 359; lon_strt = 3 ==>
344  !! tranlon = 3 4 5 6 7 8 ... 359 360 361 362; istrt = 4
345  subroutine tranlon(lon, lon_start, istrt)
346 
347  ! returns array of longitudes s.t. lon_strt <= lon < lon_strt+360.
348  ! also, the first istrt-1 entries are moved to the end of the array
349  !
350  ! e.g.
351  ! lon = 0 1 2 3 4 5 ... 358 359; lon_strt = 3 ==>
352  ! tranlon = 3 4 5 6 7 8 ... 359 360 361 362; istrt = 4
353 
354  real, intent(inout), dimension(:) :: lon
355  real, intent(in) :: lon_start
356  integer, intent(out) :: istrt
357 
358 
359  integer :: len, i
360  real :: lon_strt, tmp(size(lon(:))-1)
361 
362  len = size(lon(:))
363 
364  do i=1,len
365  lon(i) = lon_in_range(lon(i),lon_start)
366  enddo
367 
368  istrt=0
369  do i=1,len-1
370  if (lon(i+1) < lon(i)) then
371  istrt=i+1
372  exit
373  endif
374  enddo
375 
376  if (istrt>1) then ! grid is not monotonic
377  if (abs(lon(len)-lon(1)) < epsln) then
378  tmp = cshift(lon(1:len-1),istrt-1)
379  lon(1:len-1) = tmp
380  lon(len) = lon(1)
381  else
382  lon = cshift(lon,istrt-1)
383  endif
384  lon_strt = lon(1)
385  do i=2,len+1
386  lon(i) = lon_in_range(lon(i),lon_strt)
387  lon_strt = lon(i)
388  enddo
389  endif
390 
391  return
392  end subroutine tranlon

394  !> nearest_index = index of nearest data point within "array" corresponding to
395  !! "value".
396  !!
397  !! inputs:
398  !!
399  !! value = arbitrary data...same units as elements in "array"
400  !! array = array of data points (must be monotonically increasing)
401  !!
402  !! output:
403  !!
404  !! nearest_index = index of nearest data point to "value"
405  !! if "value" is outside the domain of "array" then nearest_index = 1
406  !! or "ia" depending on whether array(1) or array(ia) is
407  !! closest to "value"
408  !!
409  !! note: if "array" is dimensioned array(0:ia) in the calling
410  !! program, then the returned index should be reduced
411  !! by one to account for the zero base.
412  !!
413  !! example:
414  !!
415  !! let model depths be defined by the following:
416  !! parameter (km=5)
417  !! dimension z(km)
418  !! data z /5.0, 10.0, 50.0, 100.0, 250.0/
419  !!
420  !! k1 = nearest_index (12.5, z, km)
421  !! k2 = nearest_index (0.0, z, km)
422  !!
423  !! k1 would be set to 2, and k2 would be set to 1 so that
424  !! z(k1) would be the nearest data point to 12.5 and z(k2) would
425  !! be the nearest data point to 0.0
426  !!
427  !! @return real frac_index
428  function frac_index (value, array)
429 
430  integer :: ia, i, ii, unit
431  real :: value !< arbitrary data...same units as elements in "array"
432  real :: frac_index
433  real, dimension(:) :: array !< array of data points (must be monotonically increasing)
434  logical keep_going
435 
436  ia = size(array(:))
437 
438  do i=2,ia
439  if (array(i) < array(i-1)) then
440  unit = stdout()
441  write (unit,*) &
442  '=> Error: "frac_index" array must be monotonically increasing when searching for nearest value to ', value
443  write (unit,*) ' array(i) < array(i-1) for i=',i
444  write (unit,*) ' array(i) for i=1..ia follows:'
445  do ii=1,ia
446  write (unit,*) 'i=',ii, ' array(i)=',array(ii)
447  enddo
448  call mpp_error(fatal,' "frac_index" array must be monotonically increasing.')
449  endif
450  enddo
451  if (value < array(1) .or. value > array(ia)) then
452 ! if (value < array(1)) frac_index = 1.
453 ! if (value > array(ia)) frac_index = float(ia)
454  frac_index = -1.0
455  else
456  i=1
457  keep_going = .true.
458  do while (i <= ia .and. keep_going)
459  i = i+1
460  if (value <= array(i)) then
461  frac_index = float(i-1) + (value-array(i-1))/(array(i)-array(i-1))
462  keep_going = .false.
463  endif
464  enddo
465  endif
466  end function frac_index

468  !> @brief Return index of nearest point along axis
469  !!
470  !> nearest_index = index of nearest data point within "array" corresponding to
471  !! "value".
472  !!
473  !! inputs:
474  !!
475  !! value = arbitrary data...same units as elements in "array"
476  !! array = array of data points (must be monotonically increasing)
477  !! ia = dimension of "array"
478  !!
479  !! output:
480  !!
481  !! nearest_index = index of nearest data point to "value"
482  !! if "value" is outside the domain of "array" then nearest_index = 1
483  !! or "ia" depending on whether array(1) or array(ia) is
484  !! closest to "value"
485  !!
486  !! note: if "array" is dimensioned array(0:ia) in the calling
487  !! program, then the returned index should be reduced
488  !! by one to account for the zero base.
489  !!
490  !! example:
491  !!
492  !! let model depths be defined by the following:
493  !! parameter (km=5)
494  !! dimension z(km)
495  !! data z /5.0, 10.0, 50.0, 100.0, 250.0/
496  !!
497  !! k1 = nearest_index (12.5, z, km)
498  !! k2 = nearest_index (0.0, z, km)
499  !!
500  !! k1 would be set to 2, and k2 would be set to 1 so that
501  !! z(k1) would be the nearest data point to 12.5 and z(k2) would
502  !! be the nearest data point to 0.0
503  !! @return integer nearest_index
504  function nearest_index (value, array)
505 
506  integer :: nearest_index
507  integer :: ia !< dimension of "array"
508  integer :: i, ii, unit
509  real :: value !< arbitrary data...same units as elements in "array"
510  real, dimension(:) :: array !< array of data points (must be monotonically increasing)
511  logical keep_going
512 
513  ia = size(array(:))
514 
515  do i=2,ia
516  if (array(i) < array(i-1)) then
517  unit = stdout()
518  write (unit,*) '=> Error: "nearest_index" array must be monotonically increasing &
519  &when searching for nearest value to ',value
520  write (unit,*) ' array(i) < array(i-1) for i=',i
521  write (unit,*) ' array(i) for i=1..ia follows:'
522  do ii=1,ia
523  write (unit,*) 'i=',ii, ' array(i)=',array(ii)
524  enddo
525  call mpp_error(fatal,' "nearest_index" array must be monotonically increasing.')
526  endif
527  enddo
528  if (value < array(1) .or. value > array(ia)) then
529  if (value < array(1)) nearest_index = 1
530  if (value > array(ia)) nearest_index = ia
531  else
532  i=1
533  keep_going = .true.
534  do while (i <= ia .and. keep_going)
535  i = i+1
536  if (value <= array(i)) then
537  nearest_index = i
538  if (array(i)-value > value-array(i-1)) nearest_index = i-1
539  keep_going = .false.
540  endif
541  enddo
542  endif
543  end function nearest_index

547  subroutine interp_1d_linear(grid1,grid2,data1,data2)
548 
549  real, dimension(:), intent(in) :: grid1, data1, grid2
550  real, dimension(:), intent(inout) :: data2
551 
552  integer :: n1, n2, i, n
553  real :: w
554 
555  n1 = size(grid1(:))
556  n2 = size(grid2(:))
557 
558 
559  do i=2,n1
560  if (grid1(i) <= grid1(i-1)) call mpp_error(fatal, 'grid1 not monotonic')
561  enddo
562 
563  do i=2,n2
564  if (grid2(i) <= grid2(i-1)) call mpp_error(fatal, 'grid2 not monotonic')
565  enddo
566 
567  if (grid1(1) > grid2(1) ) call mpp_error(fatal, 'grid2 lies outside grid1')
568  if (grid1(n1) < grid2(n2) ) call mpp_error(fatal, 'grid2 lies outside grid1')
569 
570  do i=1,n2
571  n = nearest_index(grid2(i),grid1)
572 
573  if (grid1(n) < grid2(i)) then
574  w = (grid2(i)-grid1(n))/(grid1(n+1)-grid1(n))
575  data2(i) = (1.-w)*data1(n) + w*data1(n+1)
576  else
577  if(n==1) then
578  data2(i) = data1(n)
579  else
580  w = (grid2(i)-grid1(n-1))/(grid1(n)-grid1(n-1))
581  data2(i) = (1.-w)*data1(n-1) + w*data1(n)
582  endif
583  endif
584  enddo
585 
586 
587  return
588 
589  end subroutine interp_1d_linear

592  subroutine interp_1d_cubic_spline(grid1, grid2, data1, data2, yp1, ypn)
593 
594  real, dimension(:), intent(in) :: grid1, grid2, data1
595  real, dimension(:), intent(inout) :: data2
596  real, intent(in) :: yp1, ypn
597 
598  real, dimension(size(grid1)) :: y2, u
599  real :: sig, p, qn, un, h, a ,b
600  integer :: n, m, i, k, klo, khi
601 
602  n = size(grid1(:))
603  m = size(grid2(:))
604 
605  do i=2,n
606  if (grid1(i) <= grid1(i-1)) call mpp_error(fatal, 'grid1 not monotonic')
607  enddo
608 
609  do i=2,m
610  if (grid2(i) <= grid2(i-1)) call mpp_error(fatal, 'grid2 not monotonic')
611  enddo
612 
613  if (grid1(1) > grid2(1) ) call mpp_error(fatal, 'grid2 lies outside grid1')
614  if (grid1(n) < grid2(m) ) call mpp_error(fatal, 'grid2 lies outside grid1')
615 
616  if (yp1 >.99e30) then
617  y2(1)=0.
618  u(1)=0.
619  else
620  y2(1)=-0.5
621  u(1)=(3./(grid1(2)-grid1(1)))*((data1(2)-data1(1))/(grid1(2)-grid1(1))-yp1)
622  endif
623 
624  do i=2,n-1
625  sig=(grid1(i)-grid1(i-1))/(grid1(i+1)-grid1(i-1))
626  p=sig*y2(i-1)+2.
627  y2(i)=(sig-1.)/p
628  u(i)=(6.*((data1(i+1)-data1(i))/(grid1(i+1)-grid1(i))-(data1(i)-data1(i-1)) &
629  /(grid1(i)-grid1(i-1)))/(grid1(i+1)-grid1(i-1))-sig*u(i-1))/p
630  enddo
631 
632  if (ypn > .99e30) then
633  qn=0.
634  un=0.
635  else
636  qn=0.5
637  un=(3./(grid1(n)-grid1(n-1)))*(ypn-(data1(n)-data1(n-1))/(grid1(n)-grid1(n-1)))
638  endif
639 
640  y2(n)=(un-qn*u(n-1))/(qn*y2(n-1)+1.)
641 
642  do k=n-1,1,-1
643  y2(k)=y2(k)*y2(k+1)+u(k)
644  enddo
645 
646  do k = 1, m
647  n = nearest_index(grid2(k),grid1)
648  if (grid1(n) < grid2(k)) then
649  klo = n
650  else
651  if(n==1) then
652  klo = n
653  else
654  klo = n -1
655  endif
656  endif
657  khi = klo+1
658  h = grid1(khi)-grid1(klo)
659  a = (grid1(khi) - grid2(k))/h
660  b = (grid2(k) - grid1(klo))/h
661  data2(k) = a*data1(klo) + b*data1(khi)+ ((a**3-a)*y2(klo) + (b**3-b)*y2(khi))*(h**2)/6.
662  enddo
663 
664  end subroutine interp_1d_cubic_spline
