63 !> @brief Calculates what a given calendar time would be after a interval of time
64 !!
65 !! @param time_increment A time interval
66 !! @param units
67 !! Examples of acceptable values of units:
68 !! - 'days since 1980-01-01 00:00:00'
69 !! - 'hours since 1980-1-1 0:0:0'
70 !! - 'minutes since 0001-4-12'
71 !! The first word in the string must be
72 !! 'years', 'months', 'days', 'hours', 'minutes' or 'seconds'.
73 !! The second word must be 'since'.
74 !! year number must occupy 4 spaces.
75 !! Number of months, days, hours, minutes, seconds may occupy 1 or 2 spaces
76 !! year, month and day must be separated by a '-'
77 !! hour, minute, second must be separated by a ':'
78 !! hour, minute, second are optional. If not present then zero is assumed.
79 !!
80 !! Because months are not equal increments of time, and, for julian calendar,
81 !! neither are years, the 'years since' and 'month since' cases deserve
82 !! further explaination.
83 !!
84 !! When 'years since' is used:
85 !! The year number is increased by floor(time_increment) to obtain a time T1.
86 !! The year number is increased by floor(time_increment)+1 to obtain a time T2.
87 !! The time returned is T1 + (time_increment-floor(time_increment))*(T2-T1).
88 !!
89 !! When 'months since' is used:
90 !! The month number is increased by floor(time_increment). If it falls outside
91 !! to range 1 to 12 then it is adjusted along with the year number to convert
92 !! to a valid date. The number of days in the month of this date is used to
93 !! compute the time interval of the fraction.
94 !! That is:
95 !! The month number is increased by floor(time_increment) to obtain a time T1.
96 !! delt = the number of days in the month in which T1 falls.
97 !! The time returned is T1 + ((time_increment-floor(time_increment))*delt.
98 !! Two of the consequences of this scheme should be kept in mind.
99 !! -- The time since should not be from the 29'th to 31'st of a month,
100 !! since an invalid date is likely to result, triggering an error stop.
101 !! -- When time since is from the begining of a month, the fraction of a month
102 !! will never advance into the month after that which results from only
103 !! the whole number.
104 !!
105 !! When NO_CALENDAR is in effect, units attribute must specify a starting
106 !! day and second, with day number appearing first
107 !!
108 !! Example: 'days since 100 0' Indicates 100 days 0 seconds
109 !!
110 !! @param calendar
111 !! Acceptable values of calendar are:
112 !! 'noleap'
113 !! '365_day'
114 !! '360_day'
115 !! 'julian'
116 !! 'thirty_day_months'
117 !! 'no_calendar'
118 !!
119 !! @param optional permit_calendar_conversion
120 !! It is sometimes desirable to allow the value of the intent(in) argument
121 !! "calendar" to be different than the calendar in use by time_manager_mod.
122 !! If this is not desirable, then the optional variable "permit_calendar_conversion"
123 !! should be set to .false. so as to allow an error check.
124 !! When calendar conversion is done, the time returned is the time in the
125 !! time_manager's calendar, but corresponds to the date computed using the input calendar.
126 !! For example, suppose the time_manager is using the julian calendar and
127 !! the values of the input arguments of get_cal_time are:
128 !! time_increment = 59.0
129 !! units = 'days since 1980-1-1 00:00:00'
130 !! calendar = 'noleap'
131 !! Because it will use the noleap calendar to calculate the date, get_cal_time will return
132 !! value of time for midnight March 1 1980, but it will be time in the julian calendar
133 !! rather than the noleap calendar. It will never return a value of time corresponding
134 !! to anytime during the day Feb 29.
135 !!
136 !! Another example:
137 !! Suppose the time_manager is using either the noleap or julian calendars,
138 !! and the values of the input arguments are:
139 !! time_increment = 30.0
140 !! units = 'days since 1980-1-1'
141 !! calendar = 'thirty_day_months'
142 !! In this case get_cal_time will return the value of time for Feb 1 1980 00:00:00,
143 !! but in the time_manager's calendar.
144 !!
145 !! Calendar conversion may result in a fatal error when the input calendar type is
146 !! a calendar that has more days per year than that of the time_manager's calendar.
147 !! For example, if the input calendar type is julian and the time_manager's calendar
148 !! is thirty_day_months, then get_cal_time will try to convert Jan 31 to a time in
149 !! the thirty_day_months calendar, resulting in a fatal error.
150 !!
151 !! @note This option was originally coded to allow noleap calendar as input when
152 !! the julian calendar was in effect by the time_manager.
153 function get_cal_time(time_increment, units, calendar, permit_calendar_conversion)
154 real, intent(in) :: time_increment
155 character(len=*), intent(in) :: units
156 character(len=*), intent(in) :: calendar
157 logical, intent(in), optional :: permit_calendar_conversion
158 type(time_type) :: get_cal_time
159 integer :: year, month, day, hour, minute, second
160 integer :: i1, increment_seconds, increment_days, increment_years, increment_months
161 real :: month_fraction
162 integer :: calendar_tm_i, calendar_in_i, ierr, io, logunit
163 logical :: correct_form
164 character(len=32) :: calendar_in_c
165 character(len=64) :: err_msg
166 type(time_type) :: base_time, base_time_plus_one_yr
167 real :: dt
168 logical :: permit_conversion_local
169 
170 if(.not.module_is_initialized) then
171  read (input_nml_file, get_cal_time_nml, iostat=io)
172  ierr = check_nml_error(io, 'get_cal_time_nml')
173 
174  call write_version_number("GET_CAL_TIME_MOD", version)
175  logunit = stdlog()
176  if(mpp_pe() == mpp_root_pe()) write (logunit, nml=get_cal_time_nml)
177  module_is_initialized = .true.
178 endif
179 
180 if(present(permit_calendar_conversion)) then
181  permit_conversion_local = permit_calendar_conversion
182 else
183  permit_conversion_local = allow_calendar_conversion
184 endif
185 
186 calendar_in_c = lowercase(trim(cut0(calendar)))
187 
188 correct_form = (trim(calendar_in_c)) == 'noleap' .or. (trim(calendar_in_c)) == '365_day' .or. &
189  (trim(calendar_in_c)) == '365_days' .or. &
190  (trim(calendar_in_c)) == '360_day' .or. (trim(calendar_in_c)) == 'julian' .or. &
191  (trim(calendar_in_c)) == 'no_calendar'.or. (trim(calendar_in_c)) == 'thirty_day_months' .or. &
192  (trim(calendar_in_c)) == 'gregorian'
193 
194 if(.not.correct_form) then
195  call error_mesg('get_cal_time','"'//trim(calendar_in_c)//'"'// &
196  ' is not an acceptable calendar attribute. acceptable calendars are: '// &
197  ' noleap, 365_day, 365_days, 360_day, julian, no_calendar, thirty_day_months, gregorian',fatal)
198 endif
199 
200 calendar_tm_i = get_calendar_type()
201 
202 if(.not.permit_conversion_local) then
203  correct_form = (trim(calendar_in_c) == 'noleap' .and. calendar_tm_i == noleap) .or. &
204  (trim(calendar_in_c) == '365_day' .and. calendar_tm_i == noleap) .or. &
205  (trim(calendar_in_c) == '365_days' .and. calendar_tm_i == noleap) .or. &
206  (trim(calendar_in_c) == '360_day' .and. calendar_tm_i == thirty_day_months) .or. &
207  (trim(calendar_in_c) == 'thirty_day_months' .and. calendar_tm_i == thirty_day_months) .or. &
208  (trim(calendar_in_c) == 'julian' .and. calendar_tm_i == julian) .or. &
209  (trim(calendar_in_c) == 'no_calendar' .and. calendar_tm_i == no_calendar) .or. &
210  (trim(calendar_in_c) == 'gregorian' .and. calendar_tm_i == gregorian)
211  if(.not.correct_form) then
212  call error_mesg('get_cal_time','calendar not consistent with calendar type in use by time_manager.'// &
213  ' calendar='//trim(calendar_in_c)//'. Type in use by time_manager='// &
214  & valid_calendar_types(calendar_tm_i),fatal)
215  endif
216 endif
217 
218 if (permit_conversion_local) then
219  select case (trim(calendar_in_c))
220  case ('noleap')
221  calendar_in_i = noleap
222  case ('365_day')
223  calendar_in_i = noleap
224  case ('365_days')
225  calendar_in_i = noleap
226  case ('360_day')
227  calendar_in_i = thirty_day_months
228  case ('thirty_day_months')
229  calendar_in_i = thirty_day_months
230  case ('julian')
231  calendar_in_i = julian
232  case ('no_calendar')
233  calendar_in_i = no_calendar
234  case ('gregorian')
235  calendar_in_i = gregorian
236  case default
237  call error_mesg('get_cal_time', &
238  trim(calendar_in_c)//' is an invalid calendar type (specified in call to get_cal_time)',fatal)
239  end select
240 else
241  calendar_in_i = calendar_tm_i
242 end if
243 
244 correct_form = lowercase(units(1:10)) == 'days since' .or. &
245  lowercase(units(1:11)) == 'hours since' .or. &
246  lowercase(units(1:13)) == 'minutes since' .or. &
247  lowercase(units(1:13)) == 'seconds since'
248 
249 if(calendar_in_i /= no_calendar) then
250  correct_form = correct_form .or. &
251  lowercase(units(1:11)) == 'years since' .or. &
252  lowercase(units(1:12)) == 'months since'
253 endif
254 
255 if(.not.correct_form) then
256  call error_mesg('get_cal_time',trim(units)//' is an invalid string for units.' // &
257  ' units must begin with a time unit then the word "since"' // &
258  ' Valid time units are: "seconds" "minutes", "hours", "days", and, ' // &
259  ' except when NO_CALENDAR is in effect, "months" and "years"',fatal)
260 endif
261 
262 if(calendar_in_i /= calendar_tm_i) then
263 ! switch to calendar type specified as input argument,
264 ! will switch back before returning.
265  call set_calendar_type(calendar_in_i)
266 endif
267 
268 ! index(string, substring[,back])
269 ! Returns the starting position of substring as a substring of string,
270 ! or zero if it does not occur as a substring. Default value of back is
271 ! .false. If back is .false., the starting position of the first such
272 ! substring is returned. If back is .true., the starting position of the
273 ! last such substring is returned.
274 ! Returns zero if substring is not a substring of string (regardless of value of back)
275 
276 i1 = index(units,'since') + 5
277 if(calendar_in_i == no_calendar) then
278  base_time = set_time(units(i1:len_trim(units)))
279 else
280  base_time = set_date(units(i1:len_trim(units)))
281 endif
282 
283 if(lowercase(units(1:10)) == 'days since') then
284  increment_days = floor(time_increment)
285  increment_seconds = int(86400*(time_increment - increment_days))
286 else if(lowercase(units(1:11)) == 'hours since') then
287  increment_days = floor(time_increment/24)
288  increment_seconds = int(86400*(time_increment/24 - increment_days))
289 else if(lowercase(units(1:13)) == 'minutes since') then
290  increment_days = floor(time_increment/1440)
291  increment_seconds = int(86400*(time_increment/1440 - increment_days))
292 else if(lowercase(units(1:13)) == 'seconds since') then
293  increment_days = floor(time_increment/86400)
294  increment_seconds = int(86400*(time_increment/86400 - increment_days))
295 else if(lowercase(units(1:11)) == 'years since') then
296 ! The time period between between (base_time + time_increment) and
297 ! (base_time + time_increment + 1 year) may be 360, 365, or 366 days.
298 ! This must be determined to handle time increments with year fractions.
299  call get_date(base_time, year,month,day,hour,minute,second)
300  base_time = set_date(year+floor(time_increment) ,month,day,hour,minute,second)
301  base_time_plus_one_yr = set_date(year+floor(time_increment)+1,month,day,hour,minute,second)
302  call get_time(base_time_plus_one_yr - base_time, second, day)
303  dt = (day*86400+second)*(time_increment-floor(time_increment))
304  increment_days = floor(dt/86400)
305  increment_seconds = int(dt - increment_days*86400)
306 else if(lowercase(units(1:12)) == 'months since') then
307  month_fraction = time_increment - floor(time_increment)
308  increment_years = floor(time_increment/12)
309  increment_months = floor(time_increment) - 12*increment_years
310  call get_date(base_time, year,month,day,hour,minute,second)
311  base_time = set_date(year+increment_years,month+increment_months ,day,hour,minute,second)
312  dt = 86400*days_in_month(base_time) * month_fraction
313  increment_days = floor(dt/86400)
314  increment_seconds = int(dt - increment_days*86400)
315 else
316  call error_mesg('get_cal_time','"'//trim(units)//'" is not an acceptable units attribute of time.'// &
317  & ' It must begin with: "years since", "months since", "days since", "hours since", "minutes since",'// &
318  & ' or "seconds since"',fatal)
319 endif
320 
321 if (calendar_in_i /= calendar_tm_i) then
322  if(calendar_in_i == no_calendar .or. calendar_tm_i == no_calendar) then
323  call error_mesg('get_cal_time','Cannot do calendar conversion because input calendar is '// &
324  trim(valid_calendar_types(calendar_in_i))//' and time_manager is using '// &
325  trim(valid_calendar_types(calendar_tm_i))// &
326  ' Conversion cannot be done if either is NO_CALENDAR',fatal)
327  endif
328  call get_date(base_time,year, month, day, hour, minute, second)
329  get_cal_time = set_date(year,month,day,hour,minute,second) + set_time(increment_seconds, increment_days)
330  call get_date(get_cal_time,year,month,day,hour,minute,second)
331  call set_calendar_type(calendar_tm_i)
332  get_cal_time = set_date(year,month,day,hour,minute,second, err_msg=err_msg)
333  if(err_msg /= '') then
334  call error_mesg('get_cal_time','Error in function get_cal_time: '//trim(err_msg)// &
335  ' Note that the time_manager is using the '//trim(valid_calendar_types(calendar_tm_i))//' calendar '// &
336  'while the calendar type passed to function get_cal_time is '//calendar_in_c,fatal)
337  endif
338 else
339  get_cal_time = base_time + set_time(increment_seconds, increment_days)
340 endif
341 
342 end function get_cal_time
