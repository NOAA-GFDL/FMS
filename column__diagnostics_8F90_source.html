169 !> @brief initialize_diagnostic_columns returns the (i, j, lat, lon) coord-
170 !! inates of any diagnostic columns that are located on the current
171 !! processor.
172 subroutine initialize_diagnostic_columns &
173  (module, num_diag_pts_latlon, num_diag_pts_ij, &
174  global_i , global_j , global_lat_latlon, &
175  global_lon_latlon, lonb_in, latb_in, &
176  do_column_diagnostics, &
177  diag_lon, diag_lat, diag_i, diag_j, diag_units)
178 
179 !---------------------------------------------------------------------
180 ! initialize_diagnostic_columns returns the (i, j, lat, lon) coord-
181 ! inates of any diagnostic columns that are located on the current
182 ! processor.
183 !----------------------------------------------------------------------
184 
185 !---------------------------------------------------------------------
186 character(len=*), intent(in) :: module !< module calling this subroutine
187 integer, intent(in) :: num_diag_pts_latlon !< number of diagnostic columns specified
188  !! by lat-lon coordinates
189 integer, intent(in) :: num_diag_pts_ij !< number of diagnostic columns specified
190  !! by global (i,j) coordinates
191 integer, dimension(:), intent(in) :: global_i !< specified global i coordinates
192 integer, dimension(:), intent(in) :: global_j !< specified global j coordinates
193 real , dimension(:), intent(in) :: global_lat_latlon !< specified global lat coordinates
194 real , dimension(:), intent(in) :: global_lon_latlon !< specified global lon coordinates
195 real, dimension(:,:), intent(in) :: lonb_in, latb_in
196 logical, dimension(:,:), intent(out) :: do_column_diagnostics !< is a diagnostic column in this jrow ?
197 integer, dimension(:), intent(inout) :: diag_i !< processor i indices of diagnstic columns
198 integer, dimension(:), intent(inout) :: diag_j !< processor j indices of diagnstic columns
199 real , dimension(:), intent(out) :: diag_lat !< latitudes of diagnostic columns [ degrees ]
200 real , dimension(:), intent(out) :: diag_lon !< longitudes of diagnostic columns [ degrees ]
201 integer, dimension(:), intent(out) :: diag_units !< unit number for each diagnostic column
202 !---------------------------------------------------------------------
203 
204 !---------------------------------------------------------------------
205 ! intent(in) variables:
206 !
207 ! module module calling this subroutine
208 ! num_diag_pts_latlon number of diagnostic columns specified
209 ! by lat-lon coordinates
210 ! num_diag_pts_ij number of diagnostic columns specified
211 ! by global (i,j) coordinates
212 ! global_i specified global i coordinates
213 ! global_j specified global j coordinates
214 ! global_lat_latlon specified global lat coordinates
215 ! global_lon_latlon specified global lon coordinates
216 !
217 ! intent(out) variables:
218 !
219 ! do_column_diagnostics is a diagnostic column in this jrow ?
220 ! diag_i processor i indices of diagnstic columns
221 ! diag_j processor j indices of diagnstic columns
222 ! diag_lat latitudes of diagnostic columns
223 ! [ degrees ]
224 ! diag_lon longitudes of diagnostic columns
225 ! [ degrees ]
226 ! diag_units unit number for each diagnostic column
227 !
228 !---------------------------------------------------------------------
229 
230 !--------------------------------------------------------------------
231 ! local variables:
232 
233  real, dimension(size(diag_i,1)) :: global_lat !< latitudes for all diagnostic columns [ degrees ]
234  real, dimension(size(diag_i,1)) :: global_lon !< longitudes for all diagnostic columns [ degrees ]
235  real, dimension(size(latb_in,1)-1, size(latb_in,2)-1) :: &
236  distance, distance_x, distance_y, &
237  distance_x2, distance2
238  real, dimension(size(latb_in,1), size(latb_in,2)) :: latb_deg
239  real, dimension(size(lonb_in,1), size(lonb_in,2)) :: lonb_deg
240  real :: dellat, dellon
241  real :: latb_max, latb_min, lonb_max, lonb_min
242 
243  integer :: num_diag_pts !< total number of diagnostic columns
244  integer :: i !< do loop indices
245  integer :: j !< do loop indices
246  integer :: nn !< do loop indices
247  real :: ref_lat
248  real :: current_distance
249  character(len=8) :: char !< character string for diaganostic column index
250  character(len=32) :: filename !< filename for output file for diagnostic column
251  logical :: allow_ij_input
252  logical :: open_file
253  integer :: io
254 !--------------------------------------------------------------------
255 ! local variables:
256 !
257 ! global_lat latitudes for all diagnostic columns [ degrees ]
258 ! global_lon longitudes for all diagnostic columns
259 ! [ degrees ]
260 ! num_diag_pts total number of diagnostic columns
261 ! i, j, nn do loop indices
262 ! char character string for diaganostic column index
263 ! filename filename for output file for diagnostic column
264 !
265 !---------------------------------------------------------------------
266 
267  if (.not. module_is_initialized) call column_diagnostics_init
268 
269 !--------------------------------------------------------------------
270 ! save the input lat and lon fields. define the delta of latitude
271 ! and longitude.
272 !--------------------------------------------------------------------
273  latb_deg = latb_in*radian
274  lonb_deg = lonb_in*radian
275  dellat = latb_in(1,2) - latb_in(1,1)
276  dellon = lonb_in(2,1) - lonb_in(1,1)
277  latb_max = maxval(latb_deg(:,:))
278  latb_min = minval(latb_deg(:,:))
279  lonb_max = maxval(lonb_deg(:,:))
280  lonb_min = minval(lonb_deg(:,:))
281  if (lonb_min < 10.0 .or. lonb_max > 350.) then
282  lonb_min = 0.
283  lonb_max = 360.0
284  endif
285 
286  allow_ij_input = .true.
287  ref_lat = latb_in(1,1)
288  do i =2,size(latb_in,1)
289  if (latb_in(i,1) /= ref_lat) then
290  allow_ij_input = .false.
291  exit
292  endif
293  end do
294 
295  if ( .not. allow_ij_input .and. num_diag_pts_ij /= 0) then
296  call error_mesg ('column_diagnostics_mod', &
297  'cannot specify column diagnostics column with (i,j) &
298  &coordinates when using cubed sphere -- must specify &
299  & lat/lon coordinates', fatal)
300  endif
301 
302 !----------------------------------------------------------------------
303 ! initialize column_diagnostics flag and diag unit numbers. define
304 ! total number of diagnostic columns.
305 !----------------------------------------------------------------------
306  do_column_diagnostics = .false.
307  diag_units(:) = -1
308  diag_i(:) = -99
309  diag_j(:) = -99
310  diag_lat(:) = -999.
311  diag_lon(:) = -999.
312  num_diag_pts = size(diag_i(:))
313 
314 !--------------------------------------------------------------------
315 ! define an array of lat-lon values for all diagnostic columns.
316 !--------------------------------------------------------------------
317  do nn = 1, num_diag_pts_latlon
318  global_lat(nn) = global_lat_latlon(nn)
319  global_lon(nn) = global_lon_latlon(nn)
320  end do
321 
322  do nn = 1, num_diag_pts_ij
323  global_lat(nn+num_diag_pts_latlon) = &
324  ((-0.5*acos(-1.0) + 0.5*dellat) + &
325  (global_j(nn)-1) *dellat)*radian
326  global_lon(nn+num_diag_pts_latlon) = (0.5*dellon + &
327  (global_i(nn)-1)*dellon)*radian
328  end do
329 
330 !----------------------------------------------------------------------
331 ! loop over all diagnostic points to check for their presence on
332 ! this processor.
333 !----------------------------------------------------------------------
334  do nn=1,num_diag_pts
335  open_file = .false.
336 
337 !----------------------------------------------------------------------
338 ! verify that the values of lat and lon are valid.
339 !----------------------------------------------------------------------
340  if (global_lon(nn) >= 0. .and. global_lon(nn) <= 360.0) then
341  else
342  call error_mesg ('column_diagnostics_mod', &
343  ' invalid longitude', fatal)
344  endif
345  if (global_lat(nn) >= -90.0 .and. global_lat(nn) <= 90.0) then
346  else
347  call error_mesg ('column_diagnostics_mod', &
348  ' invalid latitude', fatal)
349  endif
350 
351 !--------------------------------------------------------------------
352 ! if the desired diagnostics column is within the current
353 ! processor's domain, define the total and coordinate distances from
354 ! each of the processor's grid points to the diagnostics point.
355 !--------------------------------------------------------------------
356 
357  if (global_lat(nn) .ge. latb_min .and. &
358  global_lat(nn) .le. latb_max) then
359  if (global_lon(nn) .ge. lonb_min .and.&
360  global_lon(nn) .le. lonb_max) then
361  do j=1,size(latb_deg,2) - 1
362  do i=1,size(lonb_deg,1) - 1
363  distance_y(i,j) = abs(global_lat(nn) - latb_deg(i,j))
364  distance_x(i,j) = abs(global_lon(nn) - lonb_deg(i,j))
365  distance_x2(i,j) = abs((global_lon(nn)-360.) - &
366  lonb_deg(i,j))
367  distance(i,j) = (global_lat(nn) - latb_deg(i,j))**2 + &
368  (global_lon(nn) - lonb_deg(i,j))**2
369  distance2(i,j) = (global_lat(nn) - latb_deg(i,j))**2 + &
370  ((global_lon(nn)-360.) - &
371  lonb_deg(i,j))**2
372  end do
373  end do
374 
375 !--------------------------------------------------------------------
376 ! find the grid point on the processor that is within the specified
377 ! critical distance and also closest to the requested diagnostics
378 ! column. save the (i,j) coordinates and (lon,lat) of this model
379 ! grid point. set a flag indicating that a disgnostics file should
380 ! be opened on this processor for this diagnostic point.
381 !--------------------------------------------------------------------
382  current_distance = distance(1,1)
383  do j=1,size(latb_deg,2) - 1
384  do i=1,size(lonb_deg,1) - 1
385  if (distance_x(i,j) <= crit_xdistance .and. &
386  distance_y(i,j) <= crit_ydistance ) then
387  if (distance(i,j) < current_distance) then
388  current_distance = distance(i,j)
389  do_column_diagnostics(i,j) = .true.
390  diag_j(nn) = j
391  diag_i(nn) = i
392  diag_lon(nn) = lonb_deg(i,j)
393  diag_lat(nn) = latb_deg(i,j)
394  open_file = .true.
395  endif
396  endif
397 
398 !---------------------------------------------------------------------
399 ! check needed because of the 0.0 / 360.0 longitude periodicity.
400 !---------------------------------------------------------------------
401  if (distance_x2(i,j) <= crit_xdistance .and. &
402  distance_y(i,j) <= crit_ydistance ) then
403  if (distance2(i,j) < current_distance) then
404  current_distance = distance2(i,j)
405  do_column_diagnostics(i,j) = .true.
406  diag_j(nn) = j
407  diag_i(nn) = i
408  diag_lon(nn) = lonb_deg(i,j)
409  diag_lat(nn) = latb_deg(i,j)
410  open_file = .true.
411  endif
412  endif
413  end do
414  end do
415 
416 !--------------------------------------------------------------------
417 ! if the point has been found on this processor, open a diagnostics
418 ! file.
419 !--------------------------------------------------------------------
420  if (open_file) then
421  write (char, '(i2)') nn
422  filename = trim(module) // '_point' // &
423  trim(adjustl(char)) // '.out'
424  if(mpp_npes() > 10000) then
425  write( filename,'(a,i6.6)' )trim(filename)//'.', mpp_pe()-mpp_root_pe()
426  else
427  write( filename,'(a,i4.4)' )trim(filename)//'.', mpp_pe()-mpp_root_pe()
428  endif
429  open(newunit=diag_units(nn), file=trim(filename), action='WRITE', position='rewind', iostat=io)
430  if(io/=0) call error_mesg ('column_diagnostics_mod', 'Error in opening file '//trim(filename), fatal)
431  endif ! (open_file)
432  endif
433  endif
434  end do
435 
436 !---------------------------------------------------------------------
437 
438 
439 end subroutine initialize_diagnostic_columns
