107  !> @brief Reset the instance bounding lower and upper bounds to lower_val and upper_val, respectively.
108  SUBROUTINE reset_bounds (this, lower_val, upper_val)
109  class(fmsdiagibounds_type), target, intent(inout) :: this !< ibounds instance
110  integer, intent(in) :: lower_val !< value for the lower bounds in each dimension
111  integer, intent(in) :: upper_val !< value for the upper bounds in each dimension

120  !> @brief Update the the first three (normally x, y, and z) min and
121  !! max boundaries (array indices) of the instance bounding box
122  !! the six specified bounds values.
123  SUBROUTINE update_bounds(this, lower_i, upper_i, lower_j, upper_j, lower_k, upper_k)
124  class(fmsdiagibounds_type), intent(inout) :: this !<The bounding box of the output field buffer inindex space.
125  INTEGER, INTENT(in) :: lower_i !< Lower i bound.
126  INTEGER, INTENT(in) :: upper_i !< Upper i bound.
127  INTEGER, INTENT(in) :: lower_j !< Lower j bound.
128  INTEGER, INTENT(in) :: upper_j !< Upper j bound.
129  INTEGER, INTENT(in) :: lower_k !< Lower k bound.
130  INTEGER, INTENT(in) :: upper_k !< Upper k bound.
131  this%imin = min(this%imin, lower_i)
132  this%imax = max(this%imax, upper_i)
133  this%jmin = min(this%jmin, lower_j)
134  this%jmax = max(this%jmax, upper_j)
135  this%kmin = min(this%kmin, lower_k)
136  this%kmax = max(this%kmax, upper_k)
137  END SUBROUTINE update_bounds
