125  !> @brief Gets pe list for current ensemble or a given ensemble component.
126  subroutine get_ensemble_pelist(pelist, name)
127 
128  integer, intent(inout) :: pelist(:,:) !< Ensemble pelist
129  character(len=*), intent(in), optional :: name !< Component name.
130 
131  if (size(pelist,1) < ensemble_size) &
132  call mpp_error(fatal,'get_ensemble_pelist: size of pelist 1st index < ensemble_size')
133 
134  if(present(name)) then
135  select case(name)
136  case('ocean')
137  if (size(pelist,2) < ocean_npes_pm)&
138  call mpp_error(fatal,'get_ensemble_pelist: size of pelist 2nd index < ocean_npes_pm')
139  pelist = 0
140  pelist(1:ensemble_size,1:ocean_npes_pm) = &
141  ensemble_pelist_ocean(1:ensemble_size,1:ocean_npes_pm)
142 
143  case('atmos')
144  if (size(pelist,2) < atmos_npes_pm)&
145  call mpp_error(fatal,'get_ensemble_pelist: size of pelist 2nd index < atmos_npes_pm')
146  pelist = 0
147  pelist(1:ensemble_size,1:atmos_npes_pm) = &
148  ensemble_pelist_atmos(1:ensemble_size,1:atmos_npes_pm)
149 
150  case('land')
151  if (size(pelist,2) < land_npes_pm)&
152  call mpp_error(fatal,'get_ensemble_pelist: size of pelist 2nd index < land_npes_pm')
153  pelist = 0
154  pelist(1:ensemble_size,1:land_npes_pm) = &
155  ensemble_pelist_land(1:ensemble_size,1:land_npes_pm)
156 
157  case('ice')
158  if (size(pelist,2) < ice_npes_pm)&
159  call mpp_error(fatal,'get_ensemble_pelist: size of pelist 2nd index < ice_npes_pm')
160  pelist = 0
161  pelist(1:ensemble_size,1:ice_npes_pm) = &
162  ensemble_pelist_ice(1:ensemble_size,1:ice_npes_pm)
163 
164  case default
165  call mpp_error(fatal,'get_ensemble_pelist: unknown argument name='//name)
166  end select
167  else
168  if (size(pelist,2) < total_npes_pm)&
169  call mpp_error(fatal,'get_ensemble_pelist: size of pelist 2nd index < total_npes_pm')
170  pelist = 0
171  pelist(1:ensemble_size,1:total_npes_pm) = &
172  ensemble_pelist(1:ensemble_size,1:total_npes_pm)
173  endif
174 
175  return
176  end subroutine get_ensemble_pelist

227 !nnz: I think the following block of code should be contained in a subroutine
228 ! to consolidate and ensure the consistency of declaring the various pelists.
229 !> @brief Sets up pe list for an ensemble.
230 !!
231 !! @throw FATAL, "ensemble_manager_mod: land_npes > atmos_npes"
232 !! @throw FATAL, "ensemble_manager_mod: ice_npes > atmos_npes"
233  subroutine ensemble_pelist_setup(concurrent, atmos_npes, ocean_npes, land_npes, ice_npes, &
234  Atm_pelist, Ocean_pelist, Land_pelist, Ice_pelist)
235  logical, intent(in) :: concurrent
236  integer, intent(in) :: atmos_npes, ocean_npes
237  integer, intent(in) :: land_npes, ice_npes
238  integer, dimension(:), intent(inout) :: atm_pelist, ocean_pelist
239  integer, dimension(:), intent(inout) :: land_pelist, ice_pelist
240  integer :: atmos_pe_start, atmos_pe_end, ocean_pe_start, ocean_pe_end
241  integer :: land_pe_start, land_pe_end, ice_pe_start, ice_pe_end
242  character(len=10) :: pelist_name, text
243  integer :: npes, n, m ,i
244 
245  npes = total_npes_pm
246 
247  ! make sure land_npes and ice_npes are not greater than atmos_npes
248  if(land_npes > atmos_npes) call mpp_error(fatal, 'ensemble_manager_mod: land_npes > atmos_npes')
249  if(ice_npes > atmos_npes) call mpp_error(fatal, 'ensemble_manager_mod: ice_npes > atmos_npes')
250 
251  allocate(ensemble_pelist(ensemble_size,total_npes_pm))
252  allocate(ensemble_pelist_ocean(1:ensemble_size, 1:ocean_npes) )
253  allocate(ensemble_pelist_atmos(1:ensemble_size, 1:atmos_npes) )
254  allocate(ensemble_pelist_land(1:ensemble_size, 1:land_npes) )
255  allocate(ensemble_pelist_ice(1:ensemble_size, 1:ice_npes) )
256 
257  atmos_pe_start = 0
258  ocean_pe_start = 0
259  land_pe_start = 0
260  ice_pe_start = 0
261  if( concurrent .OR. atmos_npes+ocean_npes == npes )then
262  ocean_pe_start = ensemble_size*atmos_npes
263  endif
264  do n=1,ensemble_size
265  atmos_pe_end = atmos_pe_start + atmos_npes - 1
266  ocean_pe_end = ocean_pe_start + ocean_npes - 1
267  land_pe_end = land_pe_start + land_npes - 1
268  ice_pe_end = ice_pe_start + ice_npes - 1
269  ensemble_pelist_atmos(n, 1:atmos_npes) = (/(i,i=atmos_pe_start,atmos_pe_end)/)
270  ensemble_pelist_ocean(n, 1:ocean_npes) = (/(i,i=ocean_pe_start,ocean_pe_end)/)
271  ensemble_pelist_land(n, 1:land_npes) = (/(i,i=land_pe_start, land_pe_end)/)
272  ensemble_pelist_ice(n, 1:ice_npes) = (/(i,i=ice_pe_start, ice_pe_end)/)
273  ensemble_pelist(n, 1:atmos_npes) = ensemble_pelist_atmos(n, 1:atmos_npes)
274  if( concurrent .OR. atmos_npes+ocean_npes == npes ) &
275  ensemble_pelist(n, atmos_npes+1:npes) = ensemble_pelist_ocean(n, 1:ocean_npes)
276  if(any(ensemble_pelist(n,:) == pe)) ensemble_id = n
277  write(pelist_name,'(a,i2.2)') '_ens',n
278  call mpp_declare_pelist(ensemble_pelist(n,:), trim(pelist_name))
279  atmos_pe_start = atmos_pe_end + 1
280  ocean_pe_start = ocean_pe_end + 1
281  land_pe_start = atmos_pe_start
282  ice_pe_start = atmos_pe_start
283  enddo
284 
285  atm_pelist(:) = ensemble_pelist_atmos(ensemble_id,:)
286  ocean_pelist(:) = ensemble_pelist_ocean(ensemble_id,:)
287  land_pelist(:) = ensemble_pelist_land(ensemble_id,:)
288  ice_pelist(:) = ensemble_pelist_ice(ensemble_id,:)
289 
290  ! write(pelist_name,'(a,i2.2)') '_ocn_ens',ensemble_id
291  ! call mpp_declare_pelist(Ocean%pelist , trim(pelist_name) )
292 
293  ! write(pelist_name,'(a,i2.2)') '_atm_ens',ensemble_id
294  ! call mpp_declare_pelist(Atm%pelist , trim(pelist_name) )
295  !
296  !nnz: The above is sufficient for non-concurrent mode.
297  ! BUT
298  ! For atmosphere_init to work in ensemble, concurrent mode
299  ! the following Atm_pelist should be declared (per ensemble member)
300  ! instead of the above Atm%pelist!
301  !
302  ! allocate( Atm_pelist(1:ensemble_size, 1:atmos_npes) )
303  ! do n=1,ensemble_size
304  ! do i=1, atmos_npes
305  ! Atm_pelist(n, i) = ensemble_pelist(n, i)
306  ! enddo
307  ! write(pelist_name,'(a,i2.2)') '_atm_ens',n
308  ! call mpp_declare_pelist(Atm_pelist(n,:) , trim(pelist_name) )
309  ! enddo
310  !
311  ! The way I understand this with the help of Totalview is:
312  ! With mpp_declare_pelist(Atm%pelist)
313  ! When we are in fv_arrays_init when mp_init(comID) is called
314  ! comID is the same for the atmospheric PE's for both ensemble members
315  ! since peset(5)%id is the same (7) for those PE's, so the PE count is double what it should be inside
316  ! mp_init().
317  ! It is also true that for Ocean PE's, peset(4)%id is the same (6) for Ocean PE's in both ensemble members
318  ! but for Ocean it is not a problem because Ocean is not trying to create new communicators
319  ! from this peset whereas ATM does (vis mp_init).
320  !
321  ! Who sets peset(i)%id ? Can it be modified to assign different %id for the two subsets.
322  ! peset(i)%id = 0 for Ocean PE's on ATM pesets and for ATM PE's on Ocean pesets.
323  !
324  ! With mpp_declare_pelist(Atm_pelist(n,:)) n=1,...,ensemble_size
325  ! we get separate pesets for each ATM ensemble member and each with a different %id and mp_init is cured.
326  !
327  ! There is also a matter of precedence. If we have both calls
328  ! call mpp_declare_pelist(Atm%pelist , trim(pelist_name) )
329  ! and
330  ! call mpp_declare_pelist(Atm_pelist(n,:) , trim(pelist_name) )
331  ! then concurrent run fails because with call mpp_set_current_pelist( Atm%pelist )
332  ! peset(i) is searched for i=1,2,... and the first pelist that matches argument, its peset is set as current.
333  !
334  ! To be consistent with ATM and OCEAN we can do the following
335  ! (eventhough mpp_declare_pelist(Ocean%pelist) is adequate right now.)
336 
337  if( concurrent )then
338  do n=1,ensemble_size
339  write(pelist_name,'(a,i2.2)') 'atm_ens',n
340  call mpp_declare_pelist(ensemble_pelist_atmos(n,:) , trim(pelist_name) )
341  write(pelist_name,'(a,i2.2)') 'ocn_ens',n
342  call mpp_declare_pelist(ensemble_pelist_ocean(n,:) , trim(pelist_name) )
343  write(pelist_name,'(a,i2.2)') 'lnd_ens',n
344  call mpp_declare_pelist(ensemble_pelist_land(n,:) , trim(pelist_name) )
345  write(pelist_name,'(a,i2.2)') 'ice_ens',n
346  call mpp_declare_pelist(ensemble_pelist_ice(n,:) , trim(pelist_name) )
347  enddo
348  else
349  write(pelist_name,'(a,i2.2)') 'atm_ens',ensemble_id
350  call mpp_declare_pelist(atm_pelist , trim(pelist_name) )
351  write(pelist_name,'(a,i2.2)') 'ocn_ens',ensemble_id
352  call mpp_declare_pelist(ocean_pelist , trim(pelist_name) )
353  write(pelist_name,'(a,i2.2)') 'lnd_ens',ensemble_id
354  call mpp_declare_pelist(land_pelist , trim(pelist_name) )
355  write(pelist_name,'(a,i2.2)') 'ice_ens',ensemble_id
356  call mpp_declare_pelist(ice_pelist , trim(pelist_name) )
357  endif
358 
359  ocean_npes_pm = ocean_npes
360  atmos_npes_pm = atmos_npes
361  land_npes_pm = land_npes
362  ice_npes_pm = ice_npes
363 
364  !Declare pelist of all Ocean, Atmos, Land and Ice pes across all ensembles ( filters )
365  allocate(ensemble_pelist_ocean_filter(ensemble_size * ocean_npes_pm))
366  allocate(ensemble_pelist_atmos_filter(ensemble_size * atmos_npes_pm))
367  allocate(ensemble_pelist_land_filter(ensemble_size * land_npes_pm))
368  allocate(ensemble_pelist_ice_filter(ensemble_size * ice_npes_pm))
369  do n=1,ensemble_size
370  do m=1,ocean_npes_pm
371  i=(n-1)*ocean_npes_pm + m
372  ensemble_pelist_ocean_filter(i) = ensemble_pelist_ocean(n,m)
373  enddo
374  do m=1,atmos_npes_pm
375  i=(n-1)*atmos_npes_pm + m
376  ensemble_pelist_atmos_filter(i) = ensemble_pelist_atmos(n,m)
377  enddo
378  do m=1,land_npes_pm
379  i=(n-1)*land_npes_pm + m
380  ensemble_pelist_land_filter(i) = ensemble_pelist_land(n,m)
381  enddo
382  do m=1,ice_npes_pm
383  i=(n-1)*ice_npes_pm + m
384  ensemble_pelist_ice_filter(i) = ensemble_pelist_ice(n,m)
385  enddo
386  enddo
387 
388  write(pelist_name,'(a)') 'ocn_filter'
389  call mpp_declare_pelist(ensemble_pelist_ocean_filter, trim(pelist_name) )
390 
391  write(pelist_name,'(a)') 'atm_filter'
392  call mpp_declare_pelist(ensemble_pelist_atmos_filter, trim(pelist_name) )
393 
394  write(pelist_name,'(a)') 'lnd_filter'
395  call mpp_declare_pelist(ensemble_pelist_land_filter, trim(pelist_name) )
396 
397  write(pelist_name,'(a)') 'ice_filter'
398  call mpp_declare_pelist(ensemble_pelist_ice_filter, trim(pelist_name) )
399 
400  !
401  !Rename output files to identify the ensemble
402  !If ensemble_size=1 do not rename files so that the same coupler
403  !can be used for non-ensemble experiments
404  !
405  if (ensemble_size > 1) then
406  write( text,'(a,i2.2)' ) 'ens_', ensemble_id
407  !Append ensemble_id to the restart filenames
408 
409  !< Both calls are needed for cases where both fms2io/fmsio are used
410  call fms2_io_set_filename_appendix(trim(text))
411  call fms_io_set_filename_appendix(trim(text))
412  endif
413 
414  end subroutine ensemble_pelist_setup
