19 !> @defgroup tridiagonal_mod tridiagonal_mod
20 !> @ingroup tridiagonal
21 !> @brief Solves a tridiagonal system of equations.
22 !!
23 !> The following schematic represents the system of equations solved,
24 !! where X is the solution.
25 !! <PRE>
26 !! | B(1) A(1) 0 0 ....... 0 | |X(1)| |D(1)|
27 !! | C(2) B(2) A(2) 0 ....... 0 | |X(2)| |D(2)|
28 !! | 0 C(3) B(3) A(3) 0 ....... 0 | | .. | | .. |
29 !! | .......................................... | | .. | = | .. |
30 !! | .......................................... | | .. | | .. |
31 !! | C(N-2) B(N-2) A(N-2) 0 | | .. | | .. |
32 !! | 0 C(N-1) B(N-1) A(N-1)| | .. | | .. |
33 !! | 0 0 C(N) B(N) | |X(N)| |D(N)|
34 !!
35 !! </PRE>
36 !! To solve this system
37 !! <PRE>
38 !! call tri_invert(X,D,A,B,C)
39 !!
40 !! real, intent(out), dimension(:,:,:) :: X
41 !! real, intent(in), dimension(:,:,:) :: D
42 !! real, optional, dimension(:,:,:) :: A,B,C
43 !! </PRE>
44 !! For simplicity (?), A and C are assumed to be dimensioned the same size
45 !! as B, D, and X, although any input values for A(N) and C(1) are ignored.
46 !! (some checks are needed here)
47 !!
48 !! If A is not present, it is assumed that the matrix (A,B.C) has not been changed
49 !! since the last call to tri_invert.
50 !!
51 !! To release memory,
52 !! <PRE>
53 !! call close_tridiagonal
54 !! </PRE>
55 !! Arguments A, B, and C are optional, and are saved as module variables
56 !! if one recalls tri_invert without changing (A,B,C)
57

59 !> @{
60 module tridiagonal_mod
61 
62 !--------------------------------------------------------------------------
63 real, private, allocatable, dimension(:,:,:) :: e,g,cc
64 real, private, allocatable, dimension(:,:) :: bb
65 logical, private :: init_tridiagonal = .false.
66 !--------------------------------------------------------------------------
67 
68 contains
69

72 !> @brief Sets up and solves the tridiagonal system of equations
73 !!
74 !> For simplicity, A and C are assumed to be dimensioned the same size
75 !! as B, D, and X, although any input values for A(N) and C(1) are ignored.
76 !! There are no checks to make sure the sizes agree.
77 !!
78 !! The value of A(N) is modified on output, and B and C are unchanged.
79 subroutine tri_invert(x,d,a,b,c)
80 
81 implicit none
82 
83 real, intent(out), dimension(:,:,:) :: x !< Solution to the tridiagonal system of equations
84 real, intent(in), dimension(:,:,:) :: d !< The right-hand side term, see the schematic above.
85 real, optional, dimension(:,:,:) :: a,b,c !< Left hand side terms(see schematic above).
86  !! If not provided, values from last call are used
87 
88 real, dimension(size(x,1),size(x,2),size(x,3)) :: f
89 integer :: k
90 
91 if(present(a)) then
92 
93  !< Check if module variables are allocated
94  !$OMP SINGLE
95  init_tridiagonal = .true.
96  if(allocated(e)) deallocate(e)
97  if(allocated(g)) deallocate(g)
98  if(allocated(bb)) deallocate(bb)
99  if(allocated(cc)) deallocate(cc)
100  allocate(e(size(x,1),size(x,2),size(x,3)))
101  allocate(g(size(x,1),size(x,2),size(x,3)))
102  allocate(bb(size(x,1),size(x,2)))
103  allocate(cc(size(x,1),size(x,2),size(x,3)))
104  !$OMP END SINGLE !< There is an implicit barrier.
105 
106  e(:,:,1) = - a(:,:,1)/b(:,:,1)
107  a(:,:,size(x,3)) = 0.0
108 
109  do k= 2,size(x,3)
110  g(:,:,k) = 1.0/(b(:,:,k)+c(:,:,k)*e(:,:,k-1))
111  e(:,:,k) = - a(:,:,k)*g(:,:,k)
112  end do
113  cc = c
114  bb = 1.0/b(:,:,1)
115 
116 end if
117 
118 ! if(.not.init_tridiagonal) error
119 
120 f(:,:,1) = d(:,:,1)*bb
121 do k= 2, size(x,3)
122  f(:,:,k) = (d(:,:,k) - cc(:,:,k)*f(:,:,k-1))*g(:,:,k)
123 end do
124 
125 x(:,:,size(x,3)) = f(:,:,size(x,3))
126 do k = size(x,3)-1,1,-1
127  x(:,:,k) = e(:,:,k)*x(:,:,k+1)+f(:,:,k)
128 end do
129 
130 return
131 end subroutine tri_invert
132

135 !> @brief Releases memory used by the solver
136 subroutine close_tridiagonal
137 
138  implicit none
139 
140  !< Check if module variables are allocated
141  !$OMP SINGLE
142  if(allocated(e)) deallocate(e)
143  if(allocated(g)) deallocate(g)
144  if(allocated(bb)) deallocate(bb)
145  if(allocated(cc)) deallocate(cc)
146  !$OMP END SINGLE !< There is an implicit barrier.
147 
148 return
149 end subroutine close_tridiagonal
