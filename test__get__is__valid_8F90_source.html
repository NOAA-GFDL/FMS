20 !> @brief This programs tests calls to fms2_io get_valid and is_valid
21 
22 program test_get_is_valid
23 
24 use fms2_io_mod, only: fms2_io_init, open_file, fmsnetcdffile_t, valid_t, &
25  close_file, is_valid, get_valid
26 use mpp_mod , only: mpp_init, mpp_exit, mpp_npes, mpp_get_current_pelist, &
27  mpp_root_pe, mpp_pe, fatal, mpp_error, mpp_sync
28 use netcdf , only: nf90_create, nf90_def_var, nf90_put_att, nf90_enddef, &
29  nf90_close, nf90_clobber, nf90_64bit_offset, nf90_double
30 
31 use, intrinsic :: iso_fortran_env
32 use platform_mod
33 
34 implicit none
35 
36 type(valid_t) :: valid_type !> Fms2io valid object type
37 type(FmsNetcdfFile_t) :: fileobj !> Fms2io file obj
38 real(kind=r8_kind) :: sst(10,10) !> Data
39 integer :: mask_in(10,10) !> Array defining if sst(:,:) is valid (1) or not (0)
40 integer :: ncid, varid, err !> Netcdf integers needed
41 integer, dimension(:), allocatable :: pes !> Current pelist
42 
43 call mpp_init()
44 call fms2_io_init()
45 
46 allocate(pes(mpp_npes()))
47 call mpp_get_current_pelist(pes)
48 
49 if (mpp_root_pe() .eq. mpp_pe()) then
50 !> Create your own file with a variable to test with:
51  err = nf90_create('test_file.nc', ior(nf90_clobber, nf90_64bit_offset), ncid)
52  err = nf90_def_var(ncid, 'sst', nf90_double,varid)
53  err = nf90_put_att(ncid, varid, "_FillValue", real(999,kind=r8_kind))
54  err = nf90_put_att(ncid, varid, "missing_value", real(999,kind=r8_kind))
55  err = nf90_enddef(ncid)
56  err = nf90_close(ncid)
57 endif
58 
59 !> Wait for the root pe to catch up
60 call mpp_sync()
61 
62 !> Open the file and set up the valid type
63 if (open_file(fileobj, "test_file.nc", "read", pelist=pes)) then
64  valid_type = get_valid(fileobj, "sst")
65  call close_file(fileobj)
66 else
67  call mpp_error(fatal, "test_get_valid: Error opening file")
68 endif
69 
70 deallocate(pes)
71 
72 !> Error checking:
73 if (.not. valid_type%has_fill) call mpp_error(fatal, "test_get_valid: the fill valid_type%has_fill is not .true.")
74 if (valid_type%fill_val .ne. real(999,kind=r8_kind)) call mpp_error(fatal, &
75  "test_get_valid: the fill value is not correct")
76 
77 if (.not. valid_type%has_missing) call mpp_error(fatal, "test_get_valid: the fill valid_type%has_fill is not .true.")
78 if (valid_type%missing_val .ne. real(999,kind=r8_kind)) call mpp_error(fatal, &
79  "test_get_valid: the fill value is not correct")
80 
81 if (.not. valid_type%has_range) call mpp_error(fatal, "test_get_valid: the valid_type%has_range is not .true.")
82 if (.not. valid_type%has_max) call mpp_error(fatal, "test_get_valid: the valid_type%has_max is not .true.")
83 if (valid_type%has_min) call mpp_error(fatal, "test_get_valid: the valid_type%has_min is .true.")
84 
85 !> Now test is_valid
86 
87 !> Create some fake data
88 mask_in = 1
89 sst = real(0, kind=r8_kind)
90 sst(1,1) = real(999, kind=r8_kind)
91 
92 !> Check where the data is valid
93 !> Set the mask_in to 0 wherever sst is valid
94 where(is_valid(sst,valid_type)) mask_in = 0
95 
96 !> Error checking:
97 !> The sum of mask_in should be 1 because only one value is equal to the fill_value
98 if (sum(mask_in) .ne. 1) call mpp_error(fatal, "test_is_valid: the mask should have a 1")
99 
100 call mpp_exit()
101 
102 end program test_get_is_valid
