24 program test_io_simple
25  use, intrinsic :: iso_fortran_env, only : error_unit, output_unit
26  use mpi
27  use fms2_io_mod
28  use netcdf_io_mod
29  use mpp_domains_mod
30  use mpp_mod
31  use setup
32  use netcdf
33  use platform_mod
34  implicit none
35 
36  type(Params) :: test_params !> Some test parameters.
37  type(domain2d) :: domain !> Not sure what a domain is.
38  type(domain2d), pointer :: io_domain !> Not sure what a domain is.
39  integer :: isc !> Needed for domain.
40  integer :: iec !> Needed for domain.
41  integer :: nx !> Needed for domain.
42  integer :: jsc !> Needed for domain.
43  integer :: jec !> Needed for domain.
44  integer :: ny !> Needed for domain.
45  integer :: isd !> Needed for domain.
46  integer :: nxd !> Needed for domain.
47  integer :: jsd !> Needed for domain.
48  integer :: nyd !> Needed for domain.
49  integer :: nx_east !> Needed for domain.
50  integer :: isc_east !> Needed for domain.
51  integer :: iec_east !> Needed for domain.
52  integer :: ny_north !> Needed for domain.
53  integer :: jsc_north !> Needed for domain.
54  integer :: jec_north !> Needed for domain.
55  type(FmsNetcdfDomainFile_t) :: fileobj !> FMS file object.
56  integer, parameter :: ntiles = 6 !> Number of tiles.
57  integer :: ncchksz = 64*1024 !> Required for IO initialization.
58  character (len = 10) :: netcdf_default_format = "64bit" !> NetCDF format.
59  integer :: header_buffer_val = 16384 !> Required for IO initialization.
60  integer :: ncid !> File ID for checking file.
61  character (len = 80) :: testfile !> Base name for file created in test.
62  integer :: numfilesatt !> Value for global att in test file.
63  real (kind=r8_kind) :: att1 !> Value for global att in test file.
64  character (len = 120), dimension(3) :: my_format !> Array of formats to try.
65  character (len = 6), dimension(4) :: names !> Dim name.
66  character (len = 6) :: dimname !> Dim name we will read in.
67  integer:: dimlen !> Dim len we will read in.
68  character (len = 6) :: varname !> Name of var we will read in.
69  integer :: xtype, ndims !> More var info we will read in.
70  integer, dimension(1) :: dimids !> More var info we will read in.
71  integer :: nAtts !> More var info we will read in.
72  integer, dimension(4) :: domain_decomposition !> Domain decomposition we will read.
73  real (kind = r8_kind), dimension(96) :: double_buffer !> Data we will write.
74  real (kind = r8_kind), dimension(96) :: double_buffer_in !> Data we will read to check.
75  integer :: i !> Index for do loop.
76  integer :: j !> Index for do loop.
77  integer :: err !> Return code.
78 
79  my_format(1) = '64bit'
80  my_format(2) = 'classic'
81  my_format(3) = 'netcdf4'
82 
83  do i = 1, 96
84  double_buffer(i) = i
85  end do
86 
87  ! Initialize.
88  call init(test_params, ntiles)
89  call create_cubed_sphere_domain(test_params, domain, (/1, 1/))
90 
91  if (mpp_pe() .eq. mpp_root_pe()) then
92  write(error_unit,'(/a)') "Running simple IO test ... "
93  endif
94  call mpi_barrier(mpi_comm_world, err)
95  call mpi_check(err)
96 
97  ! Get the sizes of the I/O compute and data domains.
98  io_domain => mpp_get_io_domain(domain)
99  if (.not. associated(io_domain)) then
100  call mpp_error(fatal, "I/O domain is not associated.")
101  endif
102  call mpp_get_compute_domain(io_domain, xbegin=isc, xend=iec, xsize=nx, ybegin=jsc, yend=jec, ysize=ny)
103  call mpp_get_data_domain(io_domain, xbegin=isd, xsize=nxd, ybegin=jsd, ysize=nyd)
104  call mpp_get_compute_domain(io_domain, xbegin=isc_east, xend=iec_east, xsize=nx_east, position=east)
105  call mpp_get_compute_domain(io_domain, ybegin=jsc_north, yend=jec_north, ysize=ny_north, position=north)
106 
107  call netcdf_io_init(ncchksz, header_buffer_val, netcdf_default_format)
108 
109  do i = 1, 3
110  write(testfile,'(a,a,a)') 'test_io_simple_', trim(my_format(i)), '.nc'
111 
112  ! Open a netCDF file and initialize the file object.
113  call open_check(open_file(fileobj, testfile, "overwrite", &
114  domain, nc_format=my_format(1), is_restart=.false.))
115 
116  ! Add a global attribute.
117  call register_global_attribute(fileobj, "globalatt1", real(7., kind=r8_kind))
118 
119  ! Add a dimension.
120  call register_axis(fileobj, "lon", "x")
121 
122  ! Add a coordinate variable for the dimension.
123  names(1) = "lon"
124  call register_field(fileobj, "lon", "double", names(1:1))
125  call write_data(fileobj, "lon", double_buffer)
126 
127  ! Close the file.
128  call close_file(fileobj)
129 
130  call mpi_barrier(mpi_comm_world, err)
131  call mpi_check(err)
132 
133  ! Check for expected netcdf file.
134  if (mpp_pe() .eq. mpp_root_pe()) then
135  write(testfile,'(a,a,a)') 'test_io_simple_', trim(my_format(i)), '.tile1.nc'
136  err = nf90_open(testfile, nf90_nowrite, ncid)
137  if (err .ne. nf90_noerr) stop 2
138 
139  ! Check the atts and their values.
140  err = nf90_get_att(ncid, nf90_global, 'NumFilesInSet', numfilesatt)
141  if (err .ne. nf90_noerr) stop 10
142  if (numfilesatt .ne. 1) stop 11
143  err = nf90_get_att(ncid, nf90_global, 'globalatt1', att1)
144  if (err .ne. nf90_noerr) stop 10
145  if (att1 .ne. 7) stop 11
146 
147  ! Check the dimension.
148  err = nf90_inquire_dimension(ncid, 1, dimname, dimlen)
149  if (err .ne. nf90_noerr) stop 20
150  if (dimname .ne. "lon") stop 21
151  if (dimlen .ne. 96) stop 22
152 
153  ! Check the coordinate variable.
154  err = nf90_inquire_variable(ncid, 1, varname, xtype, ndims, dimids, natts)
155  if (err .ne. nf90_noerr) stop 30
156  if (varname .ne. "lon") stop 31
157  if (xtype .ne. nf90_double) stop 32
158  if (ndims .ne. 1 .or. dimids(1) .ne. 1) stop 33
159  if (natts .ne. 0) stop 34
160  err = nf90_get_var(ncid, 1, double_buffer_in)
161  if (err .ne. nf90_noerr) stop 38
162  do j = 1, 96
163  if (double_buffer_in(j) .ne. j) stop 39
164  end do
165 
166  ! Close the file.
167  err = nf90_close(ncid)
168  if (err .ne. nf90_noerr) stop 90
169  endif
170 
171  call mpi_barrier(mpi_comm_world, err)
172  call mpi_check(err)
173  end do
174 
175  call cleanup(test_params)
176 end program test_io_simple
