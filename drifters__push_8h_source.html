21 subroutine drifters_push_xxx(self, u, v, &
22 #if _DIMS >= 3
23  & w, &
24 #endif
25  & ermesg)
26 
27  type(drifters_type) :: self
28 #if _DIMS == 2
29  real, intent(in) :: u(:,:)
30  real, intent(in) :: v(:,:)
31 #endif
32 #if _DIMS == 3
33  real, intent(in) :: u(:,:,:)
34  real, intent(in) :: v(:,:,:)
35  real, intent(in) :: w(:,:,:)
36 #endif
37  character(len=*), intent(out) :: ermesg
38 
39  integer i, np, nf, max_add_remove
40 
41  ermesg = ''
42  np = self%core%np
43  nf = size(self%input%field_names)
44 
45  if(self%core%nd /= _dims) then
46  ermesg = 'drifters_push: wrong number of dimensions'
47  return
48  endif
49 
50  select case(self%rk4_step)
51 
52  case (0)
53 
54  ! only invoked at the first step
55 
56  call drifters_reset_rk4(self, ermesg=ermesg)
57 
58 
59  call drifters_compute_k(self, self%core%positions, &
60  & u=u, v=v, &
61 #if _DIMS >= 3
62  & w=w, &
63 #endif
64  & k=self%rk4_k1, ermesg=ermesg)
65 
66  self%temp_pos(:,1:np) = self%core%positions(:,1:np) + 0.5*self%rk4_k1(:,1:np)
67 
68  self%rk4_step = 1
69  self%rk4_completed = .false.
70 
71  case (1)
72 
73  call drifters_compute_k(self, self%temp_pos, &
74  & u=u, v=v, &
75 #if _DIMS >= 3
76  & w=w, &
77 #endif
78  & k=self%rk4_k2, ermesg=ermesg)
79 
80  self%temp_pos(:,1:np) = self%core%positions(:,1:np) + 0.5*self%rk4_k2(:,1:np)
81 
82  call drifters_compute_k(self, self%temp_pos, &
83  & u=u, v=v, &
84 #if _DIMS >= 3
85  & w=w, &
86 #endif
87  & k=self%rk4_k3, ermesg=ermesg)
88 
89  self%temp_pos(:,1:np) = self%core%positions(:,1:np) + self%rk4_k3(:,1:np)
90 
91  self%rk4_step = 2
92  self%rk4_completed = .false.
93 
94  case (2)
95 
96  call drifters_compute_k(self, self%temp_pos, &
97  & u=u, v=v, &
98 #if _DIMS >= 3
99  & w=w, &
100 #endif
101  & k=self%rk4_k4, ermesg=ermesg)
102 
103  ! This completes the RK4 steps
104  do i = 1, np
105  self%temp_pos(:,i) = self%core%positions(:,i) + &
106  & (self%rk4_k1(:,i) + 2*self%rk4_k2(:,i) + 2*self%rk4_k3(:,i) + self%rk4_k4(:,i))/6.
107  enddo
108 
109  ! correct for periodic domain, if necessary
110  call drifters_modulo(self, self%temp_pos, ermesg=ermesg)
111 
112  ! estimate of max number of particles to add/remove
113  ! this may need to be adjusted over time...
114  ! [_MPP_NPES is a macro for mpp_npes()]
115  if(self%comm%pe_end < 0) self%comm%pe_end = _mpp_npes - 1
116  max_add_remove = &
117  & max( 10, &
118  & int(0.2*self%core%npdim/ &
119  & (self%comm%pe_end-self%comm%pe_beg+1)), &
120  & int(np * 2*(self%nx+self%ny)/real(self%nx*self%ny)) &
121  & )
122 
123  ! copy/move drifters across domain boundaries, if necessary
124  call drifters_comm_update(self%comm, self%core, self%temp_pos(:,1:np), &
125  & remove=self%remove(1:np), max_add_remove=max_add_remove)
126  np = self%core%np
127 
128  ! update time
129  self%time = self%time + self%dt
130 
131  if(self%core%npdim < self%core%np) self%core%npdim = int(1.2 * self%core%np)
132 
133  ! resize local arrays if necessary
134  call drifters_reset_rk4(self, ermesg=ermesg)
135 
136  self%remove = .false. ! by definition no drifters outside data domain
137 
138  ! prepare for next step...
139  call drifters_compute_k(self, self%core%positions, &
140  & u=u, v=v, &
141 #if _DIMS >= 3
142  & w=w, &
143 #endif
144  & k=self%rk4_k1, ermesg=ermesg)
145 
146  self%temp_pos(:, 1:np) = self%core%positions(:, 1:np) + 0.5*self%rk4_k1(:, 1:np)
147 
148  self%rk4_step = 1
149  self%rk4_completed = .true.
150 
151  case default
152  ermesg = 'drifters_push: invalid rk4_step'
153 
154  end select
155 
156 end subroutine drifters_push_xxx
