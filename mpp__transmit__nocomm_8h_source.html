25 !> A message-passing routine intended to be reminiscent equally of both MPI and SHMEM
26 !! put_data and get_data are contiguous MPP_TYPE_ arrays
27 !!at each call, your put_data array is put to to_pe's get_data
28 !! your get_data array is got from from_pe's put_data
29 !!i.e we assume that typically (e.g updating halo regions) each PE performs a put _and_ a get
30 !!special PE designations:
31 !! NULL_PE: to disable a put or a get (e.g at boundaries)
32 !! ANY_PE: if remote PE for the put or get is to be unspecific
33 !! ALL_PES: broadcast and collect operations (collect not yet implemented)
34 !!ideally we would not pass length, but this f77-style call performs better (arrays passed by address, not descriptor)
35 !!further, this permits <length> contiguous words from an array of any rank to be passed (avoiding f90 rank conformance check)
36 !!caller is responsible for completion checks (mpp_sync_self) before and after
37  subroutine mpp_transmit_( put_data, put_len, to_pe, get_data, get_len, from_pe, block, tag, recv_request, &
38  & send_request )
39 
40  integer, intent(in) :: put_len, to_pe, get_len, from_pe
41  mpp_type_, intent(in) :: put_data(*)
42  mpp_type_, intent(out) :: get_data(*)
43  logical, intent(in), optional :: block
44  integer, intent(in), optional :: tag
45  integer, intent(out), optional :: recv_request, send_request
46 
47  integer :: i, outunit
48  mpp_type_, allocatable, save :: local_data(:) !local copy used by non-parallel code (no SHMEM or MPI)
49  integer(i8_kind), save :: get_data_addr=-9999
50  mpp_type_ :: get_data_local(get_len_nocomm)
51  pointer(ptr_get_data, get_data_local)
52 
53 
54  if( .NOT.module_is_initialized )call mpp_error( fatal, 'MPP_TRANSMIT: You must first call mpp_init.' )
55  if( to_pe.EQ.null_pe .AND. from_pe.EQ.null_pe )return
56 
57  outunit = stdout()
58  if( debug )then
59  call system_clock(tick)
60  write( outunit,'(a,i18,a,i5,a,2i5,2i8)' )&
61  'T=',tick, ' PE=',pe, ' MPP_TRANSMIT begin: to_pe, from_pe, put_len, get_len=', to_pe, from_pe, &
62  & put_len, get_len
63  end if
64 
65 !do put first and then get
66  if( to_pe.GE.0 .AND. to_pe.LT.npes )then
67  if( allocated(local_data) ) &
68  call mpp_error( fatal, 'MPP_TRANSMIT: local_data should have been deallocated by prior receive.' )
69  if( get_len_nocomm > 0) then ! pre-post recv
70  ptr_get_data = get_data_addr
71  do i = 1,get_len_nocomm
72  get_data_local(i) = put_data(i)
73  end do
74  get_len_nocomm = 0
75  get_data_addr = -9999
76  else
77  allocate( local_data(put_len) )
78  do i = 1,put_len
79  local_data(i) = put_data(i)
80  end do
81  endif
82  else if( to_pe.EQ.all_pes )then !this is a broadcast from from_pe
83  if( from_pe.LT.0 .OR. from_pe.GE.npes )call mpp_error( fatal, 'MPP_TRANSMIT: broadcasting from invalid PE.' )
84  if( put_len.GT.get_len )call mpp_error( fatal, 'MPP_TRANSMIT: size mismatch between put_data and get_data.' )
85  if( pe.EQ.from_pe )then
86  if( loc(get_data).NE.loc(put_data) )then
87 !dir$ IVDEP
88  do i = 1,get_len
89  get_data(i) = put_data(i)
90  end do
91  end if
92  end if
93  call mpp_broadcast( get_data, get_len, from_pe )
94  return
95 
96  else if( to_pe.EQ.any_pe )then !we don't have a destination to do puts to, so only do gets
97 
98  else if( to_pe.NE.null_pe )then !no other valid cases except NULL_PE
99  call mpp_error( fatal, 'MPP_TRANSMIT: invalid to_pe.' )
100  end if
101 
102 !do the get
103  if( from_pe.GE.0 .AND. from_pe.LT.npes )then
104  if( .NOT.allocated(local_data) ) then
105  get_data_addr = loc(get_data)
106  get_len_nocomm = get_len
107  else
108  do i = 1,get_len
109  get_data(i) = local_data(i)
110  end do
111  deallocate(local_data)
112  endif
113  else if( from_pe.EQ.any_pe )then
114 
115  else if( from_pe.EQ.all_pes )then
116  call mpp_error( fatal, 'MPP_TRANSMIT: from_pe=ALL_PES has ambiguous meaning, and hence is not implemented.' )
117 
118  else if( from_pe.NE.null_pe )then !only remaining valid choice is NULL_PE
119  call mpp_error( fatal, 'MPP_TRANSMIT: invalid from_pe.' )
120  end if
121 
122  if( debug )then
123  call system_clock(tick)
124  write( outunit,'(a,i18,a,i5,a,2i5,2i8)' )&
125  'T=',tick, ' PE=',pe, ' MPP_TRANSMIT end: to_pe, from_pe, put_len, get_len=', to_pe, from_pe, &
126  & put_len, get_len
127  end if
128  return
129  end subroutine mpp_transmit_
