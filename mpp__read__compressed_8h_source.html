34  subroutine mpp_read_compressed_2d_(unit, field, domain, data, tindex, start, nread, threading)
35  integer, intent(in) :: unit
36  type(fieldtype), intent(in) :: field
37  type(domain2D), intent(in) :: domain
38  mpp_type_, intent(inout) :: data(:,:)
39  integer, intent(in), optional :: tindex
40  integer, intent(in), optional :: start(:), nread(:)
41  integer, intent(in), optional :: threading
42 
43  integer, allocatable :: pelist(:)
44  integer :: npes, p, threading_flag
45  type(domain2d), pointer :: io_domain=>null()
46  logical :: compute_chksum,print_compressed_chksum
47  integer(i8_kind) ::chk
48 
49  call mpp_clock_begin(mpp_read_clock)
50 
51  if( .NOT.module_is_initialized )call mpp_error( fatal, 'MPP_READ_COMPRESSED_2D_: must first call mpp_io_init.' )
52  if( .NOT.mpp_file(unit)%valid )call mpp_error( fatal, 'MPP_READ_COMPRESSED_2D_: invalid unit number.' )
53 
54  print_compressed_chksum = .false.
55 
56  if(size(data) > 0) then
57  data = 0 !! zero out data so other tiles do not contribute junk to chksum
58  threading_flag = mpp_single
59  if( PRESENT(threading) )threading_flag = threading
60  if( threading_flag == mpp_multi ) then
61  call read_record_(unit,field,size(data(:,:)),data,tindex,start_in=start, axsiz_in=nread)
62  else if( threading_flag == mpp_single ) then
63 
64  io_domain=>mpp_get_io_domain(domain)
65  if(.not. ASSOCIATED(io_domain)) call mpp_error( fatal, 'MPP_READ_COMPRESSED_2D_: io_domain must be defined.' )
66  npes = mpp_get_domain_npes(io_domain)
67  allocate(pelist(npes))
68  call mpp_get_pelist(io_domain,pelist)
69 
70  if(mpp_pe() == pelist(1)) call read_record_(unit,field,size(data(:,:)),data,tindex,start_in=start, &
71  & axsiz_in=nread)
72 
73  !--- z1l replace mpp_broadcast with mpp_send/mpp_recv to avoid hang in calling MPI_COMM_CREATE
74  !--- because size(pelist) might be different for different rank.
75  !--- prepost receive
76  if( mpp_pe() == pelist(1) ) then
77  do p = 2, npes
78  call mpp_send(data(1,1), plen=size(data(:,:)), to_pe=pelist(p), tag=comm_tag_1)
79  enddo
80  call mpp_sync_self()
81  else
82  call mpp_recv(data(1,1), glen=size(data(:,:)), from_pe=pelist(1), block=.false., tag=comm_tag_1)
83  call mpp_sync_self(check=event_recv)
84  endif
85 
86  deallocate(pelist)
87  else
88  call mpp_error( fatal, 'MPP_READ_COMPRESSED_2D_: threading should be MPP_SINGLE or MPP_MULTI')
89  endif
90  endif
91 
92  compute_chksum = .false.
93  if (any(field%checksum /= default_field%checksum) ) compute_chksum = .true.
94 
95  if (compute_chksum) then
96 #ifdef use_netCDF
97  if (field%type==nf_int) then
98  if (field%fill == mpp_fill_double .or. field%fill == real(mpp_fill_int) ) then
99  chk = mpp_chksum( ceiling(data), mask_val=mpp_fill_int )
100  else
101  call mpp_error(note,"During mpp_io(mpp_read_compressed_2d) int field "//trim(field%name)// &
102  " found fill. Icebergs, or code using defaults can safely ignore. "// &
103  " If manually overriding compressed restart fills, confirm this is what you want.")
104  chk = mpp_chksum( ceiling(data), mask_val=field%fill)
105  end if
106  else !!real data
107  chk = mpp_chksum(data,mask_val=real(field%fill,kind(data)))
108  end if
109 #endif
110  !!compare
111  if ( print_compressed_chksum) then
112  if ( mpp_pe() == mpp_root_pe() ) then
113  print '(A,Z16)', "mpp_read_compressed_2d chksum: "//trim(field%name)//" = ", chk
114  !! discuss making fatal after testing/review to match other routines.
115  !! Need to do some num_word-counting and digging with pjp
116  !! this should be if ( chk /= field%checksum ) as it was tested @ulm_201505..
117  if ( mod(chk, field%checksum(1)) /= 0 ) then
118  print '(A,Z16)', "File stored checksum: "//trim(field%name)//" = ", field%checksum(1)
119  call mpp_error(note,"mpp_read_compressed_2d chksum: "//trim(field%name)//" failed!")
120  end if
121  endif
122  end if
123  end if
124 
125  call mpp_clock_end(mpp_read_clock)
126  return
127  end subroutine mpp_read_compressed_2d_
