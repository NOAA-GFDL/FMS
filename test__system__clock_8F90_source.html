20 !! @author Lauren Chilutti
21 !! @brief Test program for the system_clock routine.
22 !! @email gfdl.climate.model.info@noaa.gov
23 !! @description This test program is for testing the system_clock
24 !! routine. The test initializes mpp, then calls the function
25 !! SYSTEM_CLOCK twice. SYSTEM_CLOCK returns COUNT, COUNT_RATE, and/or
26 !! COUNT_MAX (all optional). This test returns errors if COUNT,
27 !! COUNT_RATE, and/or COUNT_MAX are undefined, if the COUNT is
28 !! greater than COUNT_MAX, that the second call returns the
29 !! same value for COUNT_MAX and COUNT_RATE as the first call, and that
30 !! the second call returns a COUNT value greater than that of the
31 !! first call.
32 module include_files_mod
33  use platform_mod
34  logical :: first_call_system_clock_mpi=.true.
35 contains
36 #include "../../mpp/include/system_clock.h"
37 end module include_files_mod
38 
39 program test_system_clock
40  use include_files_mod
41  use mpp_mod, only : mpp_init, mpp_init_test_init_true_only, stderr, stdout, mpp_error, fatal
42  implicit none
43 
44  integer(i8_kind) :: count1, count_rate1, count_max1, count2, count_rate2, count_max2
45  integer :: ierr
46 !> Initialize mpp
47  call mpp_init(test_level=mpp_init_test_init_true_only)
48 !> Call system_clock and ensure output is not undefined
49  call system_clock(count1, count_rate1, count_max1)
50  write(*,*) count1, count_rate1, count_max1
51  call system_clock(count2, count_rate2, count_max2)
52  write(*,*) count2, count_rate2, count_max2
53 !> Check that the count is less than the count_max
54  if (count1 .gt. count_max1) then
55  call mpp_error(fatal, "SYSTEM_CLOCK returns a count that is greater than count_max")
56  endif
57  if (count2 > count_max2) then
58  call mpp_error(fatal, "SYSTEM_CLOCK returns a count that is greater than count_max")
59  endif
60 !> Check that the second call to system_clock function returns the same COUNT_MAX and COUNT_RATE, but a larger COUNT
61  if (count1 .gt. count2) then
62  call mpp_error(fatal, "count2 is less than count1")
63  endif
64  if (count_rate1 .ne. count_rate2) then
65  call mpp_error(fatal, "count rates are not equal")
66  endif
67  if (count_max1 .ne. count_max2) then
68  call mpp_error(fatal, "count maxes are not equal")
69  endif
70  call mpi_finalize(ierr)
71 end program test_system_clock
