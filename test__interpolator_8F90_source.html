36 program test_interpolator
37 
38 use mpp_mod
39 use mpp_domains_mod
40 use fms_mod

47 implicit none
48 integer, parameter :: nsteps_per_day = 8, ndays = 16
49 real, parameter :: delt = 1.0/nsteps_per_day
50 integer, parameter :: nxd = 20, nyd = 40, ntsteps = nsteps_per_day*ndays, two_delt = 2*delt
51 integer :: delt_days, delt_secs
52 integer, parameter :: max_fields = 20 ! maximum number of fields to be interpolated
53 
54 integer :: i,k,n,level
55 integer :: io_status
56 integer :: ndivs
57 integer :: jscomp, jecomp, iscomp, iecomp, isd,ied,jsd,jed
58 integer :: numfields, domain_layout(2)
59 integer :: num_nbrs, nbins,axes(3), interp_diagnostic_id
60 integer :: column_diagnostic_id1, column_diagnostic_id(max_fields)
61 integer :: io, ierr
62 
63 real :: missing_value = -1.e10
64 
65 character(len=1) :: dest_grid
66 character(len=128) :: src_file, file_out, title, units, colaer
67 logical :: vector_field=.false., result
68 
69 type(domain2d) :: domain
70 type(time_type) :: model_time
71 
72 type(interpolate_type) :: o3, aerosol
73 
74 real, dimension(:,:), allocatable :: col_data
75 real, dimension(:,:,:), allocatable :: model_data, p_half, p_full
76 real, dimension(:), allocatable :: latb_mod(:,:),lonb_mod(:,:),lon_mod(:),lat_mod(:)
77 real :: dx,dy
78 real :: dtr,tpi
79 real :: p_bot,p_top,lambda
80 character(len=64) :: names(13)
81 data names(:) /"so4_anthro","so4_natural","organic_carbon","black_carbon","sea_salt",&
82 "anthro_dust_0.2","anthro_dust_0.8","anthro_dust_2.0","anthro_dust_8.0",&
83 "natural_dust_0.2","natural_dust_0.8","natural_dust_2.0","natural_dust_8.0"/
84 
85 integer :: out_of_bounds(1)
86 data out_of_bounds / constant/
87 
88 namelist /interpolator_nml/ src_file
89 
90 ! initialize communication modules
91 
92 delt_days = int(delt)
93 delt_secs = int(delt*seconds_per_day) - delt_days*seconds_per_day
94 
95 write(*,*) delt, delt_days,delt_secs
96 
97 call mpp_init
98 call mpp_domains_init
99 call set_calendar_type(julian)
100 call diag_manager_init
101 call constants_init
102 call time_interp_init
103 
104 level = 18
105 tpi = 2.0*pi !4.*acos(0.)
106 dtr = tpi/360.
107 
108 src_file = 'src_file' ! input file containing fields to be interpolated
109 
110 
111 model_time = set_date(1979,12,1,0,0,0)
112 
113 !--------------------------------------------------------------------
114 ! namelist input
115 !--------------------------------------------------------------------
116 
117 read (input_nml_file, nml=interpolator_nml, iostat=io)
118 ierr = check_nml_error(io, 'interpolator_nml')
119 
120 ! decompose model grid points
121 ! mapping can get expensive so we distribute the task at this level
122 
123 ndivs = mpp_npes()
124 
125 call mpp_define_layout ((/1,nxd,1,nyd/), ndivs, domain_layout)
126 call mpp_define_domains((/1,nxd,1,nyd/),domain_layout, domain,xhalo=0,yhalo=0)
127 call mpp_get_data_domain(domain,isd,ied,jsd,jed)
128 call mpp_get_compute_domain (domain, iscomp, iecomp, jscomp, jecomp)
129 
130 allocate(lonb_mod(nxd+1,nyd+1),lon_mod(nxd))
131 allocate(latb_mod(nxd+1,nyd+1),lat_mod(nyd))
132 allocate(col_data(isd:ied,jsd:jed)) ; col_data = 0.0
133 allocate(p_half(isd:ied,jsd:jed,level+1),p_full(isd:ied,jsd:jed,level))
134 p_top = 1.0
135 p_bot = 101325.0 !Model level in Pa
136 lambda = -1.0*log(p_top/p_bot)/(level+1)
137 
138 p_half(:,:,level+1) = p_bot
139 do i=level,1,-1
140  p_half(:,:,i)=p_half(:,:,i+1)*exp(-1.0*lambda)
141 enddo
142 do i=1,level
143  p_full(:,:,i)=(p_half(:,:,i+1)+p_half(:,:,i))/2.0
144 enddo
145 
146 allocate(model_data(isd:ied,jsd:jed,level))
147 
148 dx = 360./nxd
149 dy = 180./nyd
150 do i = 1,nxd+1
151  lonb_mod(i,:) = (i-1)*dx
152 enddo
153 do i = 1,nyd+1
154  latb_mod(:,i) = -90. + (i-1)*dy
155 enddo
156 do i=1,nxd
157  lon_mod(i)=(lonb_mod(i+1,1)+lonb_mod(i,1))/2.0
158 enddo
159 do i=1,nyd
160  lat_mod(i)=(latb_mod(1,i+1)+latb_mod(1,i))/2.0
161 enddo
162 
163 lonb_mod = lonb_mod * dtr
164 latb_mod = latb_mod * dtr
165 
166  axes(1) = diag_axis_init('x',lon_mod,units='degrees',cart_name='x',domain2=domain)
167  axes(2) = diag_axis_init('y',lat_mod,units='degrees',cart_name='y',domain2=domain)
168  axes(3) = diag_axis_init('z',p_full(isd,jsd,:),units='mb',cart_name='z')
169 
170 interp_diagnostic_id = register_diag_field('interp','ozone',axes(1:3),model_time,&
171  'interpolated_ozone_clim', 'kg/kg', missing_value)
172 column_diagnostic_id1 = register_diag_field('interp','colozone',axes(1:2),model_time,&
173  'column_ozone_clim', 'kg/m2', missing_value)
174 
175 do i=1,size(names(:))
176 colaer = 'col'//trim(names(i))
177 column_diagnostic_id(i) = register_diag_field('interp',colaer,axes(1:2),model_time,&
178  'column_aerosol_clim', 'kg/m2', missing_value)
179 enddo
180 
181 call ozone_init(o3,lonb_mod(isd:ied+1,jsd:jed+1), latb_mod(isd:ied+1,jsd:jed+1), axes, model_time, &
182  data_out_of_bounds=out_of_bounds)
183 
184 call init_clim_diag(o3, axes, model_time)
185 call sulfate_init(aerosol,lonb_mod(isd:ied+1,jsd:jed+1), latb_mod(isd:ied+1,jsd:jed+1), names, &
186  data_out_of_bounds=(/constant/) )
187 call init_clim_diag(aerosol, axes, model_time)
188 
189 do n=1,ntsteps
190  if( mpp_pe() == mpp_root_pe() ) write(*,*) n
191 
192  call get_ozone(o3,model_time,p_half,model_data)
193 
194  if(interp_diagnostic_id>0) &
195  result = send_data(interp_diagnostic_id,&
196  model_data(iscomp:iecomp,jscomp:jecomp,:),model_time)
197 
198  if(column_diagnostic_id1>0) then
199 
200  col_data(iscomp:iecomp,jscomp:jecomp)=0.0
201  do k=1,level
202  col_data(iscomp:iecomp,jscomp:jecomp)= col_data(iscomp:iecomp,jscomp:jecomp)+ &
203  model_data(iscomp:iecomp,jscomp:jecomp,k)* &
204  (p_half(iscomp:iecomp,jscomp:jecomp,k+1)-p_half(iscomp:iecomp,jscomp:jecomp,k))/grav
205  enddo
206  result = send_data(column_diagnostic_id1,col_data(:,:),model_time)
207  endif
208 
209 
210 
211  do i=1,size(names(:))
212 
213 call get_anthro_sulfate(aerosol,model_time,p_half,names(i),model_data,clim_units=units)
214 
215  if(column_diagnostic_id(i)>0) then
216 
217  col_data(iscomp:iecomp,jscomp:jecomp)=0.0
218  do k=1,level
219  if (trim(adjustl(lowercase(units))) .eq. 'kg/m^2') then
220  col_data(iscomp:iecomp,jscomp:jecomp)= col_data(iscomp:iecomp,jscomp:jecomp)+ &
221  model_data(iscomp:iecomp,jscomp:jecomp,k)
222  else
223  col_data(iscomp:iecomp,jscomp:jecomp)= col_data(iscomp:iecomp,jscomp:jecomp)+ &
224  model_data(iscomp:iecomp,jscomp:jecomp,k)* &
225  (p_half(iscomp:iecomp,jscomp:jecomp,k+1)-p_half(iscomp:iecomp,jscomp:jecomp,k))/grav
226  endif
227  enddo
228  result = send_data(column_diagnostic_id(i),&
229  col_data(iscomp:iecomp,jscomp:jecomp),model_time)
230  endif
231 
232  enddo
233 
234  model_time = model_time + set_time(delt_secs,delt_days)
235 
236  if (n.eq. ntsteps) call diag_manager_end(model_time)
237 
238 enddo
239 
240 call interpolator_end(aerosol)
241 call interpolator_end(o3)
242 
243 deallocate(lonb_mod, lon_mod, latb_mod,lat_mod, col_data, p_half, p_full, model_data)
244 
245 call mpp_exit
246 
247 contains
