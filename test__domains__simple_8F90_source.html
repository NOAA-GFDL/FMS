22 !> This is a simple test of MPP layouts and a simple 1D and 2D
23 !> domain. In testing it should be run with 4 pes.
24 
25 !> @author Ed Hartnett 6/22/20
26 program test_domains_simple
27  use mpp_mod
28  use mpp_domains_mod
29 
30  implicit none
31 #include "../../include/fms_platform.h"
32  integer :: pe, npes !> This pe and the total number of pes.
33  integer :: nx=40, ny=40 !> Size of our 2D domain.
34  integer :: layout(2) !> Layout of our 2D domain.
35  type(domain2d) :: domain_2d !> A 2D domain.
36  type(domain1d) :: domain_1d !> A 1D domain.
37  integer :: is, ie, js, je !> For checking domains.
38 
39  call mpp_init()
40 
41  pe = mpp_pe()
42  npes = mpp_npes()
43 
44  ! Initialize mpp domains.
45  call mpp_domains_init(mpp_debug)
46 
47  ! Define a layout.
48  call mpp_define_layout((/1, nx, 1, ny/), npes, layout)
49 
50  ! Check results when run on 4 pes.
51  if (npes .eq. 4) then
52  if (layout(1) .ne. 2 .or. layout(2) .ne. 2) call mpp_error(fatal, 'bad layout values')
53  endif
54 
55  ! Define a 1D domain - but this doesn't work because the pelist is missing.
56  !call mpp_define_domains((/1, nx/), 4, domain_1D)
57 
58  ! Define a 1D domain.
59  call mpp_define_domains((/1, nx/), 4, domain_1d, pelist=(/0, 1, 2, 3/))
60 
61  ! Get the values of the compute domain.
62  call mpp_get_compute_domain(domain_1d, is, ie)
63 
64  ! Check results when run on 4 pes.
65  if (npes .eq. 4) then
66  if (is .ne. pe * 10 + 1) call mpp_error(fatal, 'bad domain start value')
67  if (ie .ne. 10 * (pe + 1)) call mpp_error(fatal, 'bad domain end value')
68  endif
69 
70  ! Define a 2D domain.
71  call mpp_define_domains((/1, nx, 1, ny/), layout, domain_2d)
72 
73  ! Get the values of the 2D compute domain.
74  call mpp_get_compute_domain(domain_2d, xbegin = is, xend = ie, ybegin = js, yend = je)
75 
76  ! Check results when run on 4 pes.
77  if (npes .eq. 4) then
78  if (pe .eq. 0 .or. pe .eq. 2) then
79  if (is .ne. 1 .or. ie .ne. 20) call mpp_error(fatal, 'bad 2D domain')
80  else
81  if (is .ne. 21 .or. ie .ne. 40) call mpp_error(fatal, 'bad 2D domain')
82  endif
83  if (pe .eq. 0 .or. pe .eq. 1) then
84  if (js .ne. 1 .or. je .ne. 20) call mpp_error(fatal, 'bad 2D domain')
85  else
86  if (js .ne. 21 .or. je .ne. 40) call mpp_error(fatal, 'bad 2D domain')
87  endif
88  endif
89 
90  ! Clean up domains.
91  call mpp_domains_exit()
92 
93  ! Finalize MPP.
94  call mpp_exit()
95 
96 end program test_domains_simple
