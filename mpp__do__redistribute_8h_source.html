21  subroutine mpp_do_redistribute_3d_( f_in, f_out, d_comm, d_type )
22  integer(i8_kind), intent(in) :: f_in(:), f_out(:)
23  type(DomainCommunicator2D), intent(in) :: d_comm
24  mpp_type_, intent(in) :: d_type
25  mpp_type_ :: field_in(d_comm%domain_in%x(1)%data%begin:d_comm%domain_in%x(1)%data%end, &
26  d_comm%domain_in%y(1)%data%begin:d_comm%domain_in%y(1)%data%end,d_comm%ke)
27  pointer( ptr_field_in, field_in)
28  mpp_type_ :: field_out(d_comm%domain_out%x(1)%data%begin:d_comm%domain_out%x(1)%data%end, &
29  d_comm%domain_out%y(1)%data%begin:d_comm%domain_out%y(1)%data%end,d_comm%ke)
30  pointer( ptr_field_out, field_out)
31  type(domain2D), pointer :: domain_in, domain_out
32  integer :: i, j, k, l, n, l_size
33  integer :: is, ie, js, je
34  integer :: ke
35  integer :: list, pos, msgsize
36  integer :: to_pe, from_pe
37  mpp_type_ :: buffer(size(mpp_domains_stack(:)))
38  pointer( ptr, buffer )
39  integer :: buffer_pos, wordlen, errunit
40 
41 !fix ke
42  errunit = stderr()
43  l_size = size(f_out(:)) ! equal to size(f_in(:))
44  ke = d_comm%ke
45  domain_in =>d_comm%domain_in; domain_out =>d_comm%domain_out
46 
47  buffer_pos = 0
48  ptr = loc(mpp_domains_stack)
49  wordlen = size(transfer(buffer(1),mpp_domains_stack))
50 
51 !pre-post recv
52  n = d_comm%Rlist_size
53  do list = 0,n-1
54  if( .NOT. d_comm%R_do_buf(list) )cycle
55  from_pe = d_comm%cfrom_pe(list)
56  msgsize = d_comm%R_msize(list)*l_size
57  call mpp_recv( buffer(buffer_pos+1), glen=msgsize, from_pe=from_pe, block=.false., tag=comm_tag_1 )
58  buffer_pos = buffer_pos + msgsize
59  enddo
60 
61 !send
62  n = d_comm%Slist_size
63  do list = 0,n-1
64  if( .NOT. d_comm%S_do_buf(list) )cycle
65  to_pe = d_comm%cto_pe(list)
66  is=d_comm%sendis(1,list); ie=d_comm%sendie(1,list)
67  js=d_comm%sendjs(1,list); je=d_comm%sendje(1,list)
68  pos = buffer_pos
69  do l=1,l_size ! loop over number of fields
70  ptr_field_in = f_in(l)
71  do k = 1,ke
72  do j = js,je
73  do i = is,ie
74  pos = pos+1
75  buffer(pos) = field_in(i,j,k)
76  end do
77  end do
78  end do
79  end do
80  if( debug )write( errunit,* )'PE', pe, ' to PE ', to_pe, 'is,ie,js,je=', is, ie, js, je
81  msgsize = pos - buffer_pos
82  call mpp_send( buffer(buffer_pos+1), plen=msgsize, to_pe=to_pe, tag=comm_tag_1 )
83  buffer_pos = pos
84  end do
85 
86  call mpp_sync_self(check=event_recv)
87 
88 !unpack buffer
89  buffer_pos = 0
90  n = d_comm%Rlist_size
91  do list = 0,n-1
92  if( .NOT. d_comm%R_do_buf(list) )cycle
93  from_pe = d_comm%cfrom_pe(list)
94  is=d_comm%recvis(1,list); ie=d_comm%recvie(1,list)
95  js=d_comm%recvjs(1,list); je=d_comm%recvje(1,list)
96  if( debug )write( errunit,* )'PE', pe, ' from PE ', from_pe, 'is,ie,js,je=', is, ie, js, je
97  pos = buffer_pos
98  do l=1,l_size ! loop over number of in/out fields
99  ptr_field_out = f_out(l)
100  do k = 1,ke
101  do j = js,je
102  do i = is,ie
103  pos = pos+1
104  field_out(i,j,k) = buffer(pos)
105  end do
106  end do
107  end do
108  end do
109  buffer_pos = pos
110  end do
111 
112  call mpp_sync_self()
113  end subroutine mpp_do_redistribute_3d_
114 !> @}
