39 SUBROUTINE mpp_pass_sg_to_ug_3d_(UG_domain, field_SG, field_UG)
40  type(domainug), intent(in) :: UG_domain
41  mpp_type_, intent(inout) :: field_ug(:,:)
42  mpp_type_, intent(in) :: field_sg(:,:,:)
43  mpp_type_ :: buffer(mpp_domains_stack_size)
44  character(len=8) :: text
45  integer :: i, j, k, l, m, ke, from_pe, to_pe
46  integer :: buffer_pos, pos, msgsize, ioff, joff
47 
48  !--- check if data is on data or computing domain
49  if(size(field_sg,1) .EQ. ug_domain%SG_domain%x(1)%compute%size .AND. &
50  size(field_sg,2) .EQ. ug_domain%SG_domain%y(1)%compute%size) then
51  ioff = 1 - ug_domain%SG_domain%x(1)%compute%begin
52  joff = 1 - ug_domain%SG_domain%y(1)%compute%begin
53  else if(size(field_sg,1) .EQ. ug_domain%SG_domain%x(1)%data%size .AND. &
54  size(field_sg,2) .EQ. ug_domain%SG_domain%y(1)%data%size) then
55  ioff = 1 - ug_domain%SG_domain%x(1)%data%begin
56  joff = 1 - ug_domain%SG_domain%y(1)%data%begin
57  else
58  call mpp_error( fatal, 'mpp_pass_SG_to_UG_3D_: field_SG must match either compute domain or data domain.' )
59  endif
60 
61  ke = size(field_sg,3)
62  buffer_pos = 0
63  !---pre-post receive
64  do m = 1, ug_domain%SG2UG%nrecv
65  msgsize = ug_domain%SG2UG%recv(m)%count*ke
66  if( msgsize.GT.0 )then
67  from_pe = ug_domain%SG2UG%recv(m)%pe
68  mpp_domains_stack_hwm = max( mpp_domains_stack_hwm, (buffer_pos+msgsize) )
69  if( mpp_domains_stack_hwm.GT.mpp_domains_stack_size )then
70  write( text,'(i8)' )mpp_domains_stack_hwm
71  call mpp_error( fatal, 'mpp_pass_SG_to_UG_3D: mpp_domains_stack overflow, '// &
72  'call mpp_domains_set_stack_size('//trim(text)//') from all PEs.' )
73  end if
74  call mpp_recv( buffer(buffer_pos+1), glen=msgsize, from_pe=from_pe, block=.false., tag=comm_tag_1 )
75  buffer_pos = buffer_pos + msgsize
76  end if
77  end do
78 
79  !---pack and send data
80  do m = 1, ug_domain%SG2UG%nsend
81  pos = buffer_pos
82  msgsize = ug_domain%SG2UG%send(m)%count * ke
83  if( msgsize.GT.0 )then
84  mpp_domains_stack_hwm = max( mpp_domains_stack_hwm, (buffer_pos+msgsize) )
85  if( mpp_domains_stack_hwm.GT.mpp_domains_stack_size )then
86  write( text,'(i8)' )mpp_domains_stack_hwm
87  call mpp_error( fatal, 'mpp_pass_SG_to_UG_3D: mpp_domains_stack overflow, '// &
88  'call mpp_domains_set_stack_size('//trim(text)//') from all PEs.' )
89  end if
90 
91  do k = 1, ke
92  do l = 1, ug_domain%SG2UG%send(m)%count
93  i = ug_domain%SG2UG%send(m)%i(l)+ioff
94  j = ug_domain%SG2UG%send(m)%j(l)+joff
95  pos = pos+1
96  buffer(pos) = field_sg(i,j,k)
97  end do
98  end do
99  to_pe = ug_domain%SG2UG%send(m)%pe
100  call mpp_send( buffer(buffer_pos+1), plen=msgsize, to_pe=to_pe, tag=comm_tag_1 )
101  buffer_pos = buffer_pos + msgsize
102  end if
103  end do
104 
105  call mpp_sync_self(check=event_recv)
106  !--- unpack the buffer
107  buffer_pos = 0
108  do m = 1, ug_domain%SG2UG%nrecv
109  pos = buffer_pos
110  do k = 1, ke
111  do l = 1, ug_domain%SG2UG%recv(m)%count
112  pos = pos+1
113  i = ug_domain%SG2UG%recv(m)%i(l)
114  field_ug(i,k) = buffer(pos)
115  enddo
116  enddo
117  buffer_pos = pos
118  enddo
119 
120  call mpp_sync_self( )
121 
122 end SUBROUTINE mpp_pass_sg_to_ug_3d_
