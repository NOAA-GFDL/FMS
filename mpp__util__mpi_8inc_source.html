33 !> A very basic error handler
34 !! uses ABORT and FLUSH calls, may need to use cpp to rename
35 subroutine mpp_error_basic( errortype, errormsg )
36 #ifdef __INTEL_COMPILER
37  ! Intel module containing tracebackQQ
38  use ifcore
39 #endif
40  integer, intent(in) :: errortype
41  character(len=*), intent(in), optional :: errormsg
42  character(len=512) :: text
43  integer :: errunit
44 
45  if( .NOT.module_is_initialized )call abort()
46 
47  select case( errortype )
48  case(note)
49  text = 'NOTE' !just FYI
50  case(warning)
51  text = 'WARNING' !probable error
52  case(fatal)
53  text = 'FATAL' !fatal error
54  case default
55  text = 'WARNING: non-existent errortype (must be NOTE|WARNING|FATAL)'
56  end select
57 
58  if( npes.GT.1 )write( text,'(a,i6)' )trim(text)//' from PE', pe !this is the mpp part
59  if( PRESENT(errormsg) )text = trim(text)//': '//trim(errormsg)
60 !$OMP CRITICAL (MPP_ERROR_CRITICAL)
61  select case( errortype )
62  case(note)
63  if(pe==root_pe)write( out_unit,'(a)' )trim(text)
64  case default
65  errunit = stderr()
66  write( errunit, '(/a/)' )trim(text)
67  if(pe==root_pe)write( out_unit,'(/a/)' )trim(text)
68  if( errortype.EQ.fatal .OR. warnings_are_fatal )then
69  FLUSH(out_unit)
70 #ifdef __INTEL_COMPILER
71  ! Get traceback and return quietly for correct abort
72  call tracebackqq(user_exit_code=-1)
73 #elif __GFORTRAN__
74  call backtrace
75 #endif
76  call mpi_abort( mpi_comm_world, 1, error )
77  end if
78  end select
79 
80  error_state = errortype
81 !$OMP END CRITICAL (MPP_ERROR_CRITICAL)
82 
83 
84 end subroutine mpp_error_basic

86 !#####################################################################
87 !> Makes a PE set out of a PE list. A PE list is an ordered list of PEs
88 !! a PE set is a triad (start,log2stride,size) for SHMEM, an a communicator for MPI
89 !! if stride is non-uniform or not a power of 2,
90 !! will return error (not required for MPI but enforced for uniformity)
91 function get_peset(pelist)
92  integer :: get_peset
93  integer, intent(in), optional :: pelist(:)
94  integer :: errunit
95  integer :: i, n
96  integer, allocatable :: sorted(:)
97 
98  if( .NOT.PRESENT(pelist) )then !set it to current_peset_num
99  get_peset = current_peset_num; return
100  end if
101 
102  !--- first make sure pelist is monotonically increasing.
103  if (size(pelist(:)) .GT. 1) then
104  do n = 2, size(pelist(:))
105  if(pelist(n) <= pelist(n-1)) call mpp_error(fatal, "GET_PESET: pelist is not monotonically increasing")
106  enddo
107  endif
108 
109  allocate( sorted(size(pelist(:))) )
110  sorted = pelist
111  errunit = stderr()
112  if( debug )write( errunit,* )'pelist=', pelist
113 
114  !find if this array matches any existing peset
115  do i = 1,peset_num
116  if( debug )write( errunit,'(a,3i6)' )'pe, i, peset_num=', pe, i, peset_num
117  if( size(sorted(:)).EQ.size(peset(i)%list(:)) )then
118  if( all(sorted.EQ.peset(i)%list) )then
119  deallocate(sorted)
120  get_peset = i; return
121  end if
122  end if
123  end do
124  !not found, so create new peset
125  peset_num = peset_num + 1
126  if( peset_num > current_peset_max ) call expand_peset()
127  i = peset_num !shorthand
128  !create list
129  allocate( peset(i)%list(size(sorted(:))) )
130  peset(i)%list(:) = sorted(:)
131  peset(i)%count = size(sorted(:))
132 
133  call mpi_group_incl( peset(current_peset_num)%group, size(sorted(:)), sorted-mpp_root_pe(), peset(i)%group, error )
134  call mpi_comm_create_group(peset(current_peset_num)%id, peset(i)%group, &
135  default_tag, peset(i)%id, error )
136  deallocate(sorted)
137  get_peset = i
138 
139  return
140 
141 end function get_peset

164 !#######################################################################
165 !> This is to check if current PE's outstanding puts are complete
166 !! but we can't use shmem_fence because we are actually waiting for
167 !! a remote PE to complete its get
168 subroutine mpp_sync_self( pelist, check, request, msg_size, msg_type)
169  integer, intent(in), optional :: pelist(:)
170  integer, intent(in), optional :: check
171  integer, intent(inout), optional :: request(:)
172  integer, intent(in ), optional :: msg_size(:)
173  integer, intent(in ), optional :: msg_type(:)
174 
175  integer :: m, my_check, rsize
176 
177  if( debug .and. (current_clock.NE.0) )call system_clock(start_tick)
178  my_check = event_send
179  if(present(check)) my_check = check
180  if( my_check .NE. event_send .AND. my_check .NE. event_recv ) then
181  call mpp_error( fatal, 'mpp_sync_self: The value of optional argument check should be EVENT_SEND or EVENT_RECV')
182  endif
183 
184  if(PRESENT(request)) then
185  if( .not. present(check) ) then
186  call mpp_error(fatal, 'mpp_sync_self: check is not present when request is present')
187  endif
188  if( my_check == event_recv ) then
189  if( .not. present(msg_size) ) then
190  call mpp_error(fatal, 'mpp_sync_self: msg_size is not present when request is present and it is EVENT_RECV')
191  endif
192  if( .not. present(msg_type) ) then
193  call mpp_error(fatal, 'mpp_sync_self: msg_type is not present when request is present and it is EVENT_RECV')
194  endif
195  if(size(msg_size) .NE. size(request)) then
196  call mpp_error(fatal, 'mpp_sync_self: dimension mismatch between msg_size and request')
197  endif
198  if(size(msg_type) .NE. size(request)) then
199  call mpp_error(fatal, 'mpp_sync_self: dimension mismatch between msg_type and request')
200  endif
201 
202  do m = 1, size(request(:))
203  if( request(m) == mpi_request_null ) cycle
204  call mpi_wait(request(m), stat, error )
205  call mpi_get_count(stat, msg_type(m), rsize, error)
206  if(msg_size(m) .NE. rsize) then
207  call mpp_error(fatal, "mpp_sync_self: msg_size does not match size of data received")
208  endif
209  enddo
210  else
211  do m = 1, size(request(:))
212  if(request(m) .NE.mpi_request_null )call mpi_wait(request(m), stat, error )
213  enddo
214  endif
215  else
216  select case(my_check)
217  case(event_send)
218  do m = 1,cur_send_request
219  if( request_send(m).NE.mpi_request_null )call mpi_wait( request_send(m), stat, error )
220  end do
221  cur_send_request = 0
222  case(event_recv)
223  do m = 1,cur_recv_request
224  call mpi_wait( request_recv(m), stat, error )
225  call mpi_get_count(stat, type_recv(m), rsize, error)
226  if(size_recv(m) .NE. rsize) then
227  call mpp_error(fatal, "mpp_sync_self: size_recv does not match of data received")
228  endif
229  size_recv(m) = 0
230  end do
231  cur_recv_request = 0
232  end select
233  endif
234  if( debug .and. (current_clock.NE.0) )call increment_current_clock(event_wait)
235  return
236 end subroutine mpp_sync_self
