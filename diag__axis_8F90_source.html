103  !> @brief Initialize the axis, and return the axis ID.
104  !!
105  !> <TT>diag_axis_init</TT> initializes an axis and returns the axis ID that
106  !! is to be used with <TT>register_diag_field</TT>. This function also
107  !! increments the axis counter and fills in the axes
108  !!
109  !! @return integer axis ID
110  INTEGER FUNCTION diag_axis_init(name, DATA, units, cart_name, long_name, direction,&
111  & set_name, edges, Domain, Domain2, DomainU, aux, req, tile_count, domain_position )
112  CHARACTER(len=*), INTENT(in) :: name !< Short name for axis
113  CLASS(*), DIMENSION(:), INTENT(in) :: data !< Array of coordinate values
114  CHARACTER(len=*), INTENT(in) :: units !< Units for the axis
115  CHARACTER(len=*), INTENT(in) :: cart_name !< Cartesian axis ("X", "Y", "Z", "T")
116  CHARACTER(len=*), INTENT(in), OPTIONAL :: long_name !< Long name for the axis.
117  CHARACTER(len=*), INTENT(in), OPTIONAL :: set_name
118  INTEGER, INTENT(in), OPTIONAL :: direction !< Indicates the direction of the axis
119  INTEGER, INTENT(in), OPTIONAL :: edges !< Axis ID for the previously defined "edges axis"
120  TYPE(domain1d), INTENT(in), OPTIONAL :: domain
121  TYPE(domain2d), INTENT(in), OPTIONAL :: domain2
122  TYPE(domainug), INTENT(in), OPTIONAL :: domainu
123  CHARACTER(len=*), INTENT(in), OPTIONAL :: aux !< Auxiliary name, can only be <TT>geolon_t</TT> or <TT>geolat_t</TT>
124  CHARACTER(len=*), INTENT(in), OPTIONAL :: req !< Required field names.
125  INTEGER, INTENT(in), OPTIONAL :: tile_count
126  INTEGER, INTENT(in), OPTIONAL :: domain_position
127 
128 
129  TYPE(domain1d) :: domain_x, domain_y
130  INTEGER :: ierr, axlen
131  INTEGER :: i, set, tile
132  INTEGER :: isc, iec, isg, ieg
133  CHARACTER(len=128) :: emsg
134 
135  IF ( .NOT.module_is_initialized ) THEN
136  CALL write_version_number("DIAG_AXIS_MOD", version)
137  ENDIF
138 
139  IF ( PRESENT(tile_count)) THEN
140  tile = tile_count
141  ELSE
142  tile = 1
143  END IF
144 
145  ! Allocate the axes
146  IF (.NOT. ALLOCATED(axis_sets)) ALLOCATE(axis_sets(max_num_axis_sets))
147  IF (.NOT. ALLOCATED(axes)) ALLOCATE(axes(max_axes))
148  IF (.NOT. ALLOCATED(num_subaxes)) THEN
149  ALLOCATE(num_subaxes(max_axes))
150  num_subaxes = 0
151  END IF
152 
153  !---- is there an axis set? ----
154  IF ( PRESENT(set_name) ) THEN
155  set = get_axis_set_num(set_name)
156  !---- add new set name ----
157  IF (set == 0) THEN
158  num_axis_sets = num_axis_sets + 1
159  IF ( num_axis_sets > max_num_axis_sets ) THEN
160  WRITE (emsg, fmt='("num_axis_sets (",I2,") exceeds max_num_axis_sets (",I2,"). ")')&
161  & num_axis_sets, max_num_axis_sets
162  ! <ERROR STATUS="FATAL">
163  ! num_axis_sets (<num_axis_sets>) exceeds max_num_axis_sets(<num_axis_sets>).
164  ! Increase max_num_axis_sets via diag_manager_nml.
165  ! </ERROR>
166  CALL error_mesg('diag_axis_mod::diag_axis_init',&
167  & trim(emsg)//' Increase max_num_axis_sets via diag_manager_nml.', fatal)
168  END IF
169  set = num_axis_sets
170  axis_sets(set) = set_name
171  END IF
172  ELSE
173  set = 0
174  END IF
175 
176  !---- see if axis already exists --
177  ! if this is time axis, return the ID of a previously defined
178  ! if this is spatial axis, FATAL error
179  DO i = 1, num_def_axes
180  IF ( trim(name) == axes(i)%name ) THEN
181  IF ( trim(name) == 'Stations' .OR. trim(name) == 'Levels') THEN
182  diag_axis_init = i
183  RETURN
184  ELSE IF ( set == axes(i)%set ) THEN
185  IF ( trim(lowercase(name)) == 'time' .OR.&
186  & trim(lowercase(cart_name)) == 't' .OR.&
187  & trim(lowercase(name)) == 'nv' .OR.&
188  & trim(lowercase(cart_name)) == 'n' ) THEN
189  diag_axis_init = i
190  RETURN
191  ELSE IF ( (lowercase(cart_name) /= 'x' .AND. lowercase(cart_name) /= 'y')&
192  & .OR. tile /= axes(i)%tile_count) THEN
193  ! <ERROR STATUS="FATAL">axis_name <NAME> and axis_set already exist.</ERROR>
194  CALL error_mesg('diag_axis_mod::diag_axis_init',&
195  & 'axis_name '//trim(name)//' and axis_set already exist.', fatal)
196  END IF
197  END IF
198  END IF
199  END DO
200 
201  !---- register axis ----
202  num_def_axes = num_def_axes + 1
203  ! <ERROR STATUS="FATAL">max_axes exceeded, increase it via diag_manager_nml</ERROR>
204  IF (num_def_axes > max_axes) CALL error_mesg ('diag_axis_mod::diag_axis_init',&
205  & 'max_axes exceeded, increase via diag_manager_nml', fatal)
206  diag_axis_init = num_def_axes
207 
208  !---- check and then save cart_name name ----
209  IF ( trim(uppercase(cart_name)) == 'X' .OR.&
210  & trim(uppercase(cart_name)) == 'Y' .OR.&
211  & trim(uppercase(cart_name)) == 'Z' .OR.&
212  & trim(uppercase(cart_name)) == 'T' .OR.&
213  & trim(uppercase(cart_name)) == 'U' .OR.&
214  & trim(uppercase(cart_name)) == 'N' ) THEN
215  axes(diag_axis_init)%cart_name = trim(uppercase(cart_name))
216  ELSE
217  ! <ERROR STATUS="FATAL">Invalid cart_name name.</ERROR>
218  CALL error_mesg('diag_axis_mod::diag_axis_init', 'Invalid cart_name name. '//trim(uppercase(cart_name)), fatal)
219  END IF
220 
221  !---- allocate storage for coordinate values of axis ----
222  IF ( axes(diag_axis_init)%cart_name == 'T' ) THEN
223  axlen = 0
224  ELSE
225  axlen = SIZE(DATA(:))
226  END IF
227  ALLOCATE ( axes(diag_axis_init)%data(1:axlen) )
228 
229  ! Initialize Axes(diag_axis_init)
230  axes(diag_axis_init)%name = trim(name)
231  SELECT TYPE (data)
232  TYPE IS (real(kind=r4_kind))
233  axes(diag_axis_init)%data = DATA(1:axlen)
234  TYPE IS (real(kind=r8_kind))
235  axes(diag_axis_init)%data = real(DATA(1:axlen))
236  CLASS DEFAULT
237  CALL error_mesg('diag_axis_mod::diag_axis_init',&
238  & 'The axis data is not one of the supported types of real(kind=4) or real(kind=8)', fatal)
239  END SELECT
240  axes(diag_axis_init)%units = units
241  axes(diag_axis_init)%length = axlen
242  axes(diag_axis_init)%set = set
243  ! start and end are used in subaxes information only
244  axes(diag_axis_init)%start = -1
245  axes(diag_axis_init)%end = -1
246  axes(diag_axis_init)%subaxis_name = ""
247  axes(diag_axis_init)%shift = 0
248  axes(diag_axis_init)%num_attributes = 0
249 
250  IF ( PRESENT(long_name) ) THEN
251  axes(diag_axis_init)%long_name = long_name
252  ELSE
253  axes(diag_axis_init)%long_name = name
254  END IF
255 
256  IF ( PRESENT(aux) ) THEN
257  axes(diag_axis_init)%aux = trim(aux)
258  ELSE
259  axes(diag_axis_init)%aux = 'none'
260  END IF
261 
262  IF ( PRESENT(req) ) THEN
263  axes(diag_axis_init)%req = trim(req)
264  ELSE
265  axes(diag_axis_init)%req = 'none'
266  END IF
267  IF ( PRESENT(domain_position) ) THEN
268  if (domain_position == north .or. domain_position == east .or. domain_position == center) then
269  axes(diag_axis_init)%domain_position = domain_position
270  else
271  CALL error_mesg('diag_axis_mod::diag_axis_init', "Position must be NORTH, EAST, or CENTER" ,&
272  fatal)
273  endif
274  ELSE
275  axes(diag_axis_init)%domain_position = center
276  END IF
277 
278  !---- axis direction (-1, 0, or +1) ----
279  IF ( PRESENT(direction) )THEN
280  IF ( abs(direction) /= 1 .AND. direction /= 0 )&
281  ! <ERROR STATUS="FATAL">direction must be 0, +1, or -1</ERROR>
282  & CALL error_mesg('diag_axis_mod::diag_axis_init', 'direction must be 0, +1 or -1', fatal)
283  axes(diag_axis_init)%direction = direction
284  ELSE
285  axes(diag_axis_init)%direction = 0
286  END IF
287 
288  !---- Handle the DomainU check
289  IF (present(domainu) .AND. (PRESENT(domain2) .OR. PRESENT(domain)) ) THEN
290  ! <ERROR STATUS="FATAL">Presence of DomainU and another Domain at the same time is prohibited</ERROR>
291  CALL error_mesg('diag_axis_mod::diag_axis_init',&
292  & 'Presence of DomainU and another Domain at the same time is prohibited', fatal)
293  !---- domain2d type ----
294  ELSE IF ( PRESENT(domain2) .AND. PRESENT(domain)) THEN
295  ! <ERROR STATUS="FATAL">Presence of both Domain and Domain2 at the same time is prohibited</ERROR>
296  CALL error_mesg('diag_axis_mod::diag_axis_init',&
297  & 'Presence of both Domain and Domain2 at the same time is prohibited', fatal)
298  ELSE IF ( PRESENT(domain2) .OR. PRESENT(domain)) THEN
299  IF ( axes(diag_axis_init)%cart_name /= 'X' .AND. axes(diag_axis_init)%cart_name /= 'Y') THEN
300  ! <ERROR STATUS="FATAL">Domain must not be present for an axis which is not in the X or Y direction.</ERROR>
301  CALL error_mesg('diag_axis_mod::diag_axis_init',&
302  & 'A Structured Domain must not be present for an axis which is not in the X or Y direction', fatal)
303  END IF
304  ELSE IF (present(domainu) .AND. axes(diag_axis_init)%cart_name /= 'U') THEN
305  CALL error_mesg('diag_axis_mod::diag_axis_init',&
306  & 'In the unstructured domain, the axis cart_name must be U', fatal)
307  END IF
308 
309  axes(diag_axis_init)%tile_count = tile
310 
311  IF ( PRESENT(domain2) ) THEN
312  axes(diag_axis_init)%Domain2 = domain2
313  CALL mpp_get_domain_components(domain2, domain_x, domain_y, tile_count=tile_count)
314  IF ( axes(diag_axis_init)%cart_name == 'X' ) axes(diag_axis_init)%Domain = domain_x
315  IF ( axes(diag_axis_init)%cart_name == 'Y' ) axes(diag_axis_init)%Domain = domain_y
316  axes(diag_axis_init)%DomainUG = null_domainug
317  ELSE IF ( PRESENT(domain)) THEN
318  !---- domain1d type ----
319  axes(diag_axis_init)%Domain2 = null_domain2d ! needed since not 2-D domain
320  axes(diag_axis_init)%Domain = domain
321  axes(diag_axis_init)%DomainUG = null_domainug
322  ELSE IF (present(domainu)) THEN
323  axes(diag_axis_init)%Domain2 = null_domain2d
324  axes(diag_axis_init)%Domain = null_domain1d
325  axes(diag_axis_init)%DomainUG = domainu
326  ELSE
327  axes(diag_axis_init)%Domain2 = null_domain2d
328  axes(diag_axis_init)%Domain = null_domain1d
329  axes(diag_axis_init)%DomainUG = null_domainug
330  END IF
331 
332  !--- set up the shift value for x-y axis
333  IF ( axes(diag_axis_init)%Domain .NE. null_domain1d ) THEN
334  CALL mpp_get_compute_domain(axes(diag_axis_init)%Domain, isc, iec)
335  CALL mpp_get_global_domain(axes(diag_axis_init)%Domain, isg, ieg)
336  IF ( axes(diag_axis_init)%length == ieg - isg + 2 ) THEN
337  axes(diag_axis_init)%shift = 1
338  END IF
339  END IF
340 
341  !---- have axis edges been defined ? ----
342  axes(diag_axis_init)%edges = 0
343  IF (PRESENT(edges) ) THEN
344  IF ( edges > 0 .AND. edges < num_def_axes ) THEN
345  ierr=0
346  IF ( axes(edges)%cart_name /= axes(diag_axis_init)%cart_name) ierr=1
347  IF ( axes(edges)%length /= axes(diag_axis_init)%length+1 ) ierr=ierr+2
348  IF ( axes(edges)%set /= axes(diag_axis_init)%set ) ierr=ierr+4
349  IF ( ierr > 0 ) THEN
350  ! <ERROR STATUS="FATAL">Edges axis does not match axis (code <CODE>).</ERROR>
351  WRITE (emsg,'("Edges axis does not match axis (code ",I1,").")') ierr
352  CALL error_mesg('diag_axis_mod::diag_axis_init', emsg, fatal)
353  END IF
354  axes(diag_axis_init)%edges = edges
355  ELSE
356  ! <ERROR STATUS="FATAL">Edges axis is not defined.</ERROR>
357  CALL error_mesg('diag_axis_mod::diag_axis_init', 'Edges axis is not defined', fatal)
358  END IF
359  END IF
360 
361  ! Module is now initialized
362  module_is_initialized = .true.
363 
364  END FUNCTION diag_axis_init

366  !> @brief Create a subaxis on a parent axis.
367  !!
368  !> Given the ID of a parent axis, create a subaxis and fill it with data,
369  !! and return the ID of the corresponding subaxis.
370  !!
371  !! The subaxis is defined on the parent axis from <TT>start_indx</TT>
372  !! to <TT>end_indx</TT>.
373  !!
374  !! @return Integer ID of the corresponding subaxis.
375  INTEGER FUNCTION diag_subaxes_init(axis, subdata, start_indx, end_indx, domain_2d)
376  INTEGER, INTENT(in) :: axis !< ID of the parent axis
377  REAL, DIMENSION(:), INTENT(in) :: subdata !< Data of the subaxis
378  INTEGER, INTENT(in) :: start_indx !< Start index of the subaxis
379  INTEGER, INTENT(in) :: end_indx !< End index of the subaxis
380  TYPE(domain2d), INTENT(in), OPTIONAL :: domain_2d
381 
382  INTEGER :: i, nsub_axis, direction
383  INTEGER :: xbegin, xend, ybegin, yend
384  INTEGER :: ad_xbegin, ad_xend, ad_ybegin, ad_yend
385  CHARACTER(len=128) :: name, nsub_name
386  CHARACTER(len=128) :: units
387  CHARACTER(len=128) :: cart_name
388  CHARACTER(len=128) :: long_name
389  CHARACTER(len=128) :: emsg
390  LOGICAL :: subaxis_set, hasdomain
391 
392  ! there may be more than 1 subaxis on a parent axis, check for redundancy
393  nsub_axis = 0
394  subaxis_set = .false.
395 
396  IF ( PRESENT(domain_2d) ) THEN
397  hasdomain = .true.
398  CALL mpp_get_compute_domain(domain_2d, xbegin, xend, ybegin, yend)
399  ELSE
400  hasdomain = .false.
401  END IF
402  sa_search: DO i = 1, num_subaxes(axis)
403  IF ( start_indx == axes(axis)%start(i) .AND. end_indx == axes(axis)%end(i) ) THEN
404  IF ( hasdomain ) THEN
405  CALL mpp_get_compute_domain(axes(axis)%subaxis_domain2(i), ad_xbegin, ad_xend, ad_ybegin, ad_yend)
406  IF ( .NOT.((xbegin == ad_xbegin .AND. xend == ad_xend) .AND.&
407  & (ybegin == ad_ybegin .AND. yend == ad_yend)) ) THEN
408  cycle sa_search
409  END IF
410  END IF
411  nsub_axis = i
412  subaxis_set = .true. !subaxis already exists
413  name = trim(axes(axis)%subaxis_name(nsub_axis))
414  EXIT sa_search
415  END IF
416  END DO sa_search
417 
418  IF ( nsub_axis == 0 ) THEN ! create new subaxis
419  num_subaxes(axis) = num_subaxes(axis) + 1
420  IF (num_subaxes(axis) > max_subaxes) THEN
421  ! <ERROR STATUS="FATAL">max_subaxes (value <VALUE>) is too small. Consider increasing max_subaxes.</ERROR>
422  WRITE (emsg,'("max_subaxes (value ",I4,") is too small. Consider increasing max_subaxes.")') max_subaxes
423  CALL error_mesg('diag_axis_mod::diag_subaxes_init', emsg, fatal)
424  END IF
425  nsub_axis = num_subaxes(axis)
426  axes(axis)%start(nsub_axis) = start_indx
427  axes(axis)%end(nsub_axis) = end_indx
428  if ( hasdomain ) axes(axis)%subaxis_domain2(nsub_axis) = domain_2d
429  END IF
430 
431  ! Create new name for the subaxis from name of parent axis
432  ! If subaxis already exists, get the index and return
433  IF(subaxis_set) THEN
434  IF ( axes(axis)%set > 0 ) THEN
435  diag_subaxes_init = get_axis_num(name, set_name=trim(axis_sets(axes(axis)%set)))
436  ELSE
437  diag_subaxes_init = get_axis_num(name)
438  END IF
439  ELSE
440  ! get a new index for subaxis
441  !::sdu:: Need a check to allow larger numbers in the index number.
442  WRITE (nsub_name,'(I2.2)') nsub_axis
443  name = trim(axes(axis)%name)//'_sub'//trim(nsub_name)
444  axes(axis)%subaxis_name(nsub_axis) = name
445  long_name = trim(axes(axis)%long_name)
446  units = trim(axes(axis)%units)
447  cart_name = trim(axes(axis)%cart_name)
448  direction = axes(axis)%direction
449  IF (axes(axis)%set > 0) THEN
450  diag_subaxes_init = diag_axis_init(trim(name), subdata, trim(units), trim(cart_name), trim(long_name),&
451  & set_name=trim(axis_sets(axes(axis)%set)), direction=direction, domain2=domain_2d)
452  ELSE
453  diag_subaxes_init = diag_axis_init(trim(name), subdata, trim(units), trim(cart_name), trim(long_name),&
454  & direction=direction, domain2=domain_2d)
455  END IF
456  END IF
457  END FUNCTION diag_subaxes_init
458  !> @brief Return information about the axis with index ID
459  SUBROUTINE get_diag_axis(id, name, units, long_name, cart_name,&
460  & direction, edges, Domain, DomainU, DATA, num_attributes, attributes, domain_position)
461  CHARACTER(len=*), INTENT(out) :: name, units, long_name, cart_name
462  INTEGER, INTENT(in) :: id !< Axis ID
463  TYPE(domain1d), INTENT(out) :: domain
464  TYPE(domainug), INTENT(out) :: domainu
465  INTEGER, INTENT(out) :: direction !< Direction of data. (See <TT>@ref diag_axis_init</TT> for a description of
466  !! allowed values)
467  INTEGER, INTENT(out) :: edges !< Axis ID for the previously defined "edges axis".
468  CLASS(*), DIMENSION(:), INTENT(out) :: data !< Array of coordinate values for this axis.
469  INTEGER, INTENT(out), OPTIONAL :: num_attributes
470  TYPE(diag_atttype), ALLOCATABLE, DIMENSION(:), INTENT(out), OPTIONAL :: attributes
471  INTEGER, INTENT(out), OPTIONAL :: domain_position
472 
473  INTEGER :: i, j, istat
474 
475  CALL valid_id_check(id, 'get_diag_axis')
476  name = axes(id)%name
477  units = axes(id)%units
478  long_name = axes(id)%long_name
479  cart_name = axes(id)%cart_name
480  direction = axes(id)%direction
481  edges = axes(id)%edges
482  domain = axes(id)%Domain
483  domainu = axes(id)%DomainUG
484  if (present(domain_position)) domain_position = axes(id)%domain_position
485  IF ( axes(id)%length > SIZE(DATA(:)) ) THEN
486  ! <ERROR STATUS="FATAL">array data is too small.</ERROR>
487  CALL error_mesg('diag_axis_mod::get_diag_axis', 'array data is too small', fatal)
488  ELSE
