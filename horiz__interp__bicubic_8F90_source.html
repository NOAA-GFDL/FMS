111  !> @brief Creates a new @ref horiz_interp_type
112  !!
113  !> Allocates space and initializes a derived-type variable
114  !! that contains pre-computed interpolation indices and weights.
115  subroutine horiz_interp_bicubic_new_1d_s ( Interp, lon_in, lat_in, lon_out, lat_out, &
116  verbose, src_modulo )
117 
118  !-----------------------------------------------------------------------
119  type(horiz_interp_type), intent(inout) :: Interp !< A derived-type variable containing indices
120  !! and weights used for subsequent interpolations. To
121  !! reinitialize this variable for a different grid-to-grid
122  !! interpolation you must first use the
123  !! @ref horiz_interp_bicubic_del interface.
124  real, intent(in), dimension(:) :: lon_in !< Longitude (radians) for source data grid
125  real, intent(in), dimension(:) :: lat_in !< Latitude (radians) for source data grid
126  real, intent(in), dimension(:,:) :: lon_out !< Longitude (radians) for output data grid
127  real, intent(in), dimension(:,:) :: lat_out !< Latitude (radians) for output data grid
128  integer, intent(in), optional :: verbose !< flag for print output amount
129  logical, intent(in), optional :: src_modulo !< indicates if the boundary condition along
130  !! zonal boundary is cyclic or not. Zonal boundary condition
131  !!is cyclic when true
132  integer :: i, j, ip1, im1, jp1, jm1
133  logical :: src_is_modulo
134  integer :: nlon_in, nlat_in, nlon_out, nlat_out
135  integer :: jcl, jcu, icl, icu, jj
136  real :: xz, yz
137  integer :: unit
138 
139  if(present(verbose)) verbose_bicubic = verbose
140  src_is_modulo = .false.
141  if (present(src_modulo)) src_is_modulo = src_modulo
142 
143  if(size(lon_out,1) /= size(lat_out,1) .or. size(lon_out,2) /= size(lat_out,2) ) &
144  call mpp_error(fatal,'horiz_interp_bilinear_mod: when using bilinear ' // &
145  'interplation, the output grids should be geographical grids')
146 
147  !--- get the grid size
148  nlon_in = size(lon_in) ; nlat_in = size(lat_in)
149  nlon_out = size(lon_out,1); nlat_out = size(lat_out,2)
150  interp%nlon_src = nlon_in; interp%nlat_src = nlat_in
151  interp%nlon_dst = nlon_out; interp%nlat_dst = nlat_out
152 ! use wti(:,:,1) for x-derivative, wti(:,:,2) for y-derivative, wti(:,:,3) for xy-derivative
153  allocate ( interp%wti (nlon_in, nlat_in, 3) )
154  allocate ( interp%lon_in (nlon_in) )
155  allocate ( interp%lat_in (nlat_in) )
156  allocate ( interp%rat_x (nlon_out, nlat_out) )
157  allocate ( interp%rat_y (nlon_out, nlat_out) )
158  allocate ( interp%i_lon (nlon_out, nlat_out, 2) )
159  allocate ( interp%j_lat (nlon_out, nlat_out, 2) )
160 
161  interp%lon_in = lon_in
162  interp%lat_in = lat_in
163 
164  if ( verbose_bicubic > 0 ) then
165  unit = stdout()
166  write (unit,'(/,"Initialising bicubic interpolation, interface horiz_interp_bicubic_new_1d_s")')
167  write (unit,'(/," Longitude of coarse grid points (radian): xc(i) i=1, ",i4)') interp%nlon_src
168  write (unit,'(1x,10f10.4)') (interp%lon_in(jj),jj=1,interp%nlon_src)
169  write (unit,'(/," Latitude of coarse grid points (radian): yc(j) j=1, ",i4)') interp%nlat_src
170  write (unit,'(1x,10f10.4)') (interp%lat_in(jj),jj=1,interp%nlat_src)
171  do i=1, interp%nlat_dst
172  write (unit,*)
173  write (unit,'(/," Longitude of fine grid points (radian): xf(i) i=1, ",i4)') interp%nlat_dst
174  write (unit,'(1x,10f10.4)') (lon_out(jj,i),jj=1,interp%nlon_dst)
175  enddo
176  do i=1, interp%nlon_dst
177  write (unit,*)
178  write (unit,'(/," Latitude of fine grid points (radian): yf(j) j=1, ",i4)') interp%nlon_dst
179  write (unit,'(1x,10f10.4)') (lat_out(i,jj),jj=1,interp%nlat_dst)
180  enddo
181  endif
182 
183 
184 !---------------------------------------------------------------------------
185 ! Find the x-derivative. Use central differences and forward or
186 ! backward steps at the boundaries
187 
188  do j=1,nlat_in
189  do i=1,nlon_in
190  ip1=min(i+1,nlon_in)
191  im1=max(i-1,1)
192  interp%wti(i,j,1) = 1./(interp%lon_in(ip1)-interp%lon_in(im1))
193  enddo
194  enddo
195 
196 
197 !---------------------------------------------------------------------------
198 
199 ! Find the y-derivative. Use central differences and forward or
200 ! backward steps at the boundaries
201  do j=1,nlat_in
202  jp1=min(j+1,nlat_in)
203  jm1=max(j-1,1)
204  do i=1,nlon_in
205  interp%wti(i,j,2) = 1./(interp%lat_in(jp1)-interp%lat_in(jm1))
206  enddo
207  enddo
208 
209 !---------------------------------------------------------------------------
210 
211 ! Find the xy-derivative. Use central differences and forward or
212 ! backward steps at the boundaries
213  do j=1,nlat_in
214  jp1=min(j+1,nlat_in)
215  jm1=max(j-1,1)
216  do i=1,nlon_in
217  ip1=min(i+1,nlon_in)
218  im1=max(i-1,1)
219  interp%wti(i,j,3) = 1./((interp%lon_in(ip1)-interp%lon_in(im1))*(interp%lat_in(jp1)-interp%lat_in(jm1)))
220  enddo
221  enddo
222 !---------------------------------------------------------------------------
223 ! Now for each point at the dest-grid find the boundary points of
224 ! the source grid
225  do j=1, nlat_out
226  do i=1,nlon_out
227  yz = lat_out(i,j)
228  xz = lon_out(i,j)
229 
230  jcl = 0
231  jcu = 0
232  if( yz .le. interp%lat_in(1) ) then
233  jcl = 1
234  jcu = 1
235  else if( yz .ge. interp%lat_in(nlat_in) ) then
236  jcl = nlat_in
237  jcu = nlat_in
238  else
239  jcl = indl(interp%lat_in, yz)
240  jcu = indu(interp%lat_in, yz)
241  endif
242 
243  icl = 0
244  icu = 0
245  !--- cyclic condition, do we need to use do while
246  if( xz .gt. interp%lon_in(nlon_in) ) xz = xz - tpi
247  if( xz .le. interp%lon_in(1) ) xz = xz + tpi
248  if( xz .ge. interp%lon_in(nlon_in) ) then
249  icl = nlon_in
250  icu = 1
251  interp%rat_x(i,j) = (xz - interp%lon_in(icl))/(interp%lon_in(icu) - interp%lon_in(icl) + tpi)
252  else
253  icl = indl(interp%lon_in, xz)
254  icu = indu(interp%lon_in, xz)
255  interp%rat_x(i,j) = (xz - interp%lon_in(icl))/(interp%lon_in(icu) - interp%lon_in(icl))
256  endif
257  interp%j_lat(i,j,1) = jcl
258  interp%j_lat(i,j,2) = jcu
259  interp%i_lon(i,j,1) = icl
260  interp%i_lon(i,j,2) = icu
261  if(jcl == jcu) then
262  interp%rat_y(i,j) = 0.0
263  else
264  interp%rat_y(i,j) = (yz - interp%lat_in(jcl))/(interp%lat_in(jcu) - interp%lat_in(jcl))
265  endif
266 ! if(yz.gt.Interp%lat_in(jcu)) call mpp_error(FATAL, ' horiz_interp_bicubic_new_1d_s:
267 ! yf < ycl, no valid boundary point')
268 ! if(yz.lt.Interp%lat_in(jcl)) call mpp_error(FATAL, ' horiz_interp_bicubic_new_1d_s:
269 ! yf > ycu, no valid boundary point')
270 ! if(xz.gt.Interp%lon_in(icu)) call mpp_error(FATAL, ' horiz_interp_bicubic_new_1d_s:
271 ! xf < xcl, no valid boundary point')
272 ! if(xz.lt.Interp%lon_in(icl)) call mpp_error(FATAL, ' horiz_interp_bicubic_new_1d_s:
273 ! xf > xcu, no valid boundary point')
274  enddo
275  enddo
276  end subroutine horiz_interp_bicubic_new_1d_s

330  endif

335 ! backward steps at the boundaries

338  do i=1,nlon_in
339  ip1=min(i+1,nlon_in)

342  enddo
343  enddo
344

347

364  jm1=max(j-1,1)

366  ip1=min(i+1,nlon_in)

391  icu = 0
392  !--- cyclic condition, do we need to use do while

395  if( xz .ge. lon_in(nlon_in) ) then
396  icl = nlon_in
397  icu = 1
398  interp%rat_x(i,j) = (xz - interp%lon_in(icl))/(interp%lon_in(icu) - interp%lon_in(icl) + tpi)

428  !> @brief Perform bicubic horizontal interpolation
429  subroutine horiz_interp_bicubic( Interp, data_in, data_out, verbose, mask_in, mask_out, missing_value, &
430  & missing_permit)
431  type (horiz_interp_type), intent(in) :: interp
432  real, intent(in), dimension(:,:) :: data_in
433  real, intent(out), dimension(:,:) :: data_out
434  integer, intent(in), optional :: verbose
435  real, intent(in), dimension(:,:), optional :: mask_in
436  real, intent(out), dimension(:,:), optional :: mask_out
437  real, intent(in), optional :: missing_value
438  integer, intent(in), optional :: missing_permit
439  real :: yz, ycu, ycl
440  real :: xz, xcu, xcl
441  real :: val, val1, val2
442  real, dimension(4) :: y, y1, y2, y12
443  integer :: icl, icu, jcl, jcu
444  integer :: iclp1, icup1, jclp1, jcup1
445  integer :: iclm1, icum1, jclm1, jcum1
446  integer :: i,j
447 
448  if ( present(verbose) ) verbose_bicubic = verbose
449 ! fill_in = .false.
450 ! if ( present(fill) ) fill_in = fill
451 ! use dfc_x and dfc_y as workspace
452 ! if ( fill_in ) call fill_xy(fc(ics:ice,jcs:jce), ics, ice, jcs, jce, maxpass=2)
453 ! where ( data_in .le. missing ) data_in(:,:) = 0.
454 !!
455  do j=1, interp%nlat_dst
456  do i=1, interp%nlon_dst
457  yz = interp%rat_y(i,j)
458  xz = interp%rat_x(i,j)
459  jcl = interp%j_lat(i,j,1)
460  jcu = interp%j_lat(i,j,2)
461  icl = interp%i_lon(i,j,1)
462  icu = interp%i_lon(i,j,2)
463  if( icl > icu ) then
464  iclp1 = icu
465  icum1 = icl
466  xcl = interp%lon_in(icl)
467  xcu = interp%lon_in(icu)+tpi
468  else
469  iclp1 = min(icl+1,interp%nlon_src)
470  icum1 = max(icu-1,1)
471  xcl = interp%lon_in(icl)
472  xcu = interp%lon_in(icu)
473  endif
474  iclm1 = max(icl-1,1)
475  icup1 = min(icu+1,interp%nlon_src)
476  jclp1 = min(jcl+1,interp%nlat_src)
477  jclm1 = max(jcl-1,1)
478  jcup1 = min(jcu+1,interp%nlat_src)
479  jcum1 = max(jcu-1,1)
480  ycl = interp%lat_in(jcl)
481  ycu = interp%lat_in(jcu)
482 ! xcl = Interp%lon_in(icl)
483 ! xcu = Interp%lon_in(icu)
484  y(1) = data_in(icl,jcl)
485  y(2) = data_in(icu,jcl)
486  y(3) = data_in(icu,jcu)
487  y(4) = data_in(icl,jcu)
488  y1(1) = ( data_in(iclp1,jcl) - data_in(iclm1,jcl) ) * interp%wti(icl,jcl,1)
489  y1(2) = ( data_in(icup1,jcl) - data_in(icum1,jcl) ) * interp%wti(icu,jcl,1)
