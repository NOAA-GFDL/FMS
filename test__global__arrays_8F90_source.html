266 !> true if all pes return the same result and have a lower/higher local max/min
267 function checkresultint4(res)
268  logical :: checkResultInt4
269  integer(i4_kind),intent(in) :: res(2)
270  integer(i4_kind),allocatable :: tres(:)
271 
272  allocate(tres(2))
273  checkresultint4 = res(2).GE.maxval(datai4) .and. res(1).LE.minval(datai4)
274  if(.NOT.checkresultint4) then
275  return
276  end if
277  !> check that all pes have same results
278  if( pe.EQ.root) then
279  tres = res
280  do i=1, npes-1
281  call mpp_send(tres,2, i)
282  end do
283  checkresultint4 = .true.
284  else
285  call mpp_recv(tres,2, root)
286  checkresultint4 = checkresultint4 .and. res(1) .EQ. tres(1) .and. res(2) .eq. tres(2)
287  end if
288  call mpp_sync()
289  deallocate(tres)
290 end function checkresultint4

369 !>@brief Sum local sums from pes and compares with gsum
370 !>@return True if gsum is the global sum, false otherwise
371 function checksumreal4(gsum)
372  logical :: checkSumReal4
373  real(r4_kind),intent(in) :: gsum
374  real(r4_kind),allocatable :: recv(:) !> pe's local sum at 1, global sum at 2
375  real(r4_kind) :: nsum
376  integer :: i
377 
378  allocate(recv(2))
379  ! root receives and sums local sums from each pe
380  if(pe .eq. root) then
381  nsum = sum(datar4)
382  do i=1, npes - 1
383  call mpp_recv(recv, 2, i)
384  nsum = nsum + recv(1)
385  ! also check for matching global sum
386  if( abs((recv(2)-gsum)/gsum) .gt. 1e-4) then
387  checksumreal4 = .false.
388  deallocate(recv)
389  return
390  endif
391  end do
392  checksumreal4 = (abs((nsum-gsum)/gsum) .lt. 1e-4)
393  else
394  recv(1) = sum(datar4)
395  recv(2) = gsum
396  call mpp_send(recv, 2, root)
397  checksumreal4 = .true.
398  endif
399  call mpp_sync()
400  deallocate(recv)
401 end function checksumreal4
