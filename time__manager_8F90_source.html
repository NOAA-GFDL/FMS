20 !> @ingroup time_manager
21 !> @link http://www.gfdl.noaa.gov/fms-cgi-bin/cvsweb.cgi/FMS/
22 !> @brief A software package that provides a set of simple interfaces for
23 !! modelers to perform computations related to time and dates.
24 !!
25 !! Optional error flag can be used in calling arguments of public routines.
26 !! This allows the using routine to terminate the program. It is likely that more
27 !! diagnostic information is available from the user than from time_manager alone.
28 !! If the error flag is present then it is the responsibility of the using
29 !! routine to test it and add additional information to the error message.
30 !!
31 !! Calendar specific routines are private.
32 !! They are not used, and should not be used, by any using code.
33 !!
34 !! The module defines a type that can be used to represent discrete
35 !! times (accurate to one second) and to map these times into dates
36 !! using a variety of calendars. A time is mapped to a date by
37 !! representing the time with respect to an arbitrary base date (refer
38 !! to <B>NOTES</B> section for the base date setting).
39 !!
40 !! The time_manager provides a single defined type, time_type, which is
41 !! used to store time and date quantities. A time_type is a positive
42 !! definite quantity that represents an interval of time. It can be
43 !! most easily thought of as representing the number of seconds in some
44 !! time interval. A time interval can be mapped to a date under a given
45 !! calendar definition by using it to represent the time that has passed
46 !! since some base date. A number of interfaces are provided to operate
47 !! on time_type variables and their associated calendars. Time intervals
48 !! can be as large as n days where n is the largest number represented by
49 !! the default integer type on a compiler. This is typically considerably
50 !! greater than 10 million years (assuming 32 bit integer representation)
51 !! which is likely to be adequate for most applications. The description
52 !! of the interfaces is separated into two sections. The first deals with
53 !! operations on time intervals while the second deals with operations
54 !! that convert time intervals to dates for a given calendar.
55 !!
56 !! The smallest increment of time is referred to as a tick.
57 !! A tick cannot be larger than 1 second, which also is the default.
58 !! The number of ticks per second is set via pubic subroutine set_ticks_per_second.
59 !! For example, ticks_per_second = 1000 will set the tick to one millisecond.

273 !> Returns a time interval corresponding to this number of days, seconds, and ticks.
274 !! days, seconds and ticks may be negative, but resulting time must be positive.
275  function set_time_private(seconds, days, ticks, Time_out, err_msg)
276 
277 ! -- pjp --
278 ! To understand why inputs may be negative,
279 ! one needs to understand the intrinsic function "modulo".
280 ! The expanation below is copied from a web page on fortran 90
281 
282 ! In addition, CEILING, FLOOR and MODULO have been added to Fortran 90.
283 ! Only the last one is difficult to explain, which is most easily done with the examples from ISO (1991)
284 
285 ! MOD (8,5) gives 3 MODULO (8,5) gives 3
286 ! MOD (-8,5) gives -3 MODULO (-8,5) gives 2
287 ! MOD (8,-5) gives 3 MODULO (8,-5) gives -2
288 ! MOD (-8,-5) gives -3 MODULO (-8,-5) gives -3
289 
290 ! I don't think it is difficult to explain.
291 ! I think that is it sufficient to say this:
292 ! "The result of modulo(n,m) has the sign of m"
293 ! -- pjp --
294 
295  logical :: set_time_private
296  integer, intent(in) :: seconds, days, ticks
297  type(time_type), intent(out) :: time_out
298  character(len=*), intent(out) :: err_msg
299  integer :: seconds_new, days_new, ticks_new
300 
301  seconds_new = seconds + floor(ticks/real(ticks_per_second))
302  ticks_new = modulo(ticks,ticks_per_second)
303  days_new = days + floor(seconds_new/real(seconds_per_day))
304  seconds_new = modulo(seconds_new,seconds_per_day)
305 
306  if ( seconds_new < 0 .or. ticks_new < 0) then
307  call error_mesg('function set_time_i','Bad result for time. Contact those responsible for maintaining time_manager'&
308  & ,fatal)
309  endif
310 
311  if(days_new < 0) then
312  write(err_msg,'(a,i6,a,i6,a,i6)') 'time is negative. days=',days_new,' seconds=',seconds_new,' ticks=',ticks_new
313  set_time_private = .false.
314  else
315  time_out%days = days_new
316  time_out%seconds = seconds_new
317  time_out%ticks = ticks_new
318  err_msg = ''
319  set_time_private = .true.
320  endif
321 
322  end function set_time_private

352  !> @brief Returns a time_type set to the given amount of time via a string
353  function set_time_c(string, err_msg, allow_rounding)
354 
355  type(time_type) :: set_time_c
356  character(len=*), intent(in) :: string !< Contains days and seconds separated by a single blank.
357  !! days must be integer, seconds may be integer or real.
358  !! Examples: '100 43200' '100 43200.50'
359  character(len=*), intent(out), optional :: err_msg !< When present, and when non-blank, a fatal
360  !! error condition as been detected. The string itself is an error message.
361  !! It is recommended that, when err_msg is present in the call
362  !! to this routine, the next line of code should be something
363  !! similar to this:
364  !! if(err_msg /= '') call error_mesg('my_routine','additional info: '//trim(err_msg),FATAL)
365  logical, intent(in), optional :: allow_rounding !< When .true., any fractions of a second will be
366  !! rounded off to the nearest tick. When .false., it is a fatal error
367  !! if the second fraction cannot be exactly represented by a number of ticks.
368 
369  character(len=4) :: formt='(i )'
370  integer :: i1, i2, i3, day, second, tick, nsps
371  character(len=32) :: string_sifted_left
372  character(len=128) :: err_msg_local
373  logical :: allow_rounding_local
374 
375  if(.not.module_is_initialized) call time_manager_init
376  if(present(err_msg)) err_msg = ''
377  allow_rounding_local=.true.; if(present(allow_rounding)) allow_rounding_local=allow_rounding
378 
379  err_msg_local = 'Form of character time stamp is incorrect. The character time stamp is: '//trim(string)
380 
381  string_sifted_left = adjustl(string)
382  i1 = index(trim(string_sifted_left),' ')
383  if(i1 == 0) then
384  if(error_handler('function set_time_c', err_msg_local, err_msg)) return
385  endif
386  if(index(string,'-') /= 0 .or. index(string,':') /= 0) then
387  if(error_handler('function set_time_c', err_msg_local, err_msg)) return
388  endif
389 
390  i2 = index(trim(string_sifted_left),'.')
391  i3 = len_trim(cut0(string_sifted_left))
392 
393  if(i2 /= 0) then ! There is no decimal point
394  ! Check that decimal is on seconds (not days)
395  if(i2 < i1) then
396  if(error_handler('function set_time_c', err_msg_local, err_msg)) return
397  endif
398  endif
399  write(formt(3:3),'(i1)') i1-1
400  read(string_sifted_left(1:i1-1),formt) day
401 
402  if(i2 == 0) then ! There is no decimal point
403  write(formt(3:3),'(i1)') i3-i1
404  read(string_sifted_left(i1+1:i3),formt) second
405  tick = 0
406  else ! There is a decimal point
407  ! nsps = spaces occupied by whole number of seconds
408  nsps = i2-i1-1
409  if(nsps == 0) then
410  second = 0
411  else
412  write(formt(3:3),'(i1)') nsps
413  read(string_sifted_left(i1+1:i2-1),formt) second
414  endif
415 
416  if(.not.get_tick_from_string(string_sifted_left(i2:i3), err_msg_local, allow_rounding_local, tick)) then
417  if(error_handler('function set_time_c', err_msg_local, err_msg)) return
418  endif
419  ! If tick has been rounded up to ticks_per_second, then bump up second.
420  if(tick == ticks_per_second) then
421  second = second + 1
422  tick = 0
423  endif
424  endif
425 
426  if(.not.set_time_private(second, day, tick, set_time_c, err_msg_local)) then
427  if(error_handler('function set_time_c', err_msg_local, err_msg)) return
428  endif
429 
430  end function set_time_c

434 
435  logical :: get_tick_from_string
436  character(len=*), intent(in) :: string
437  character(len=*), intent(out) :: err_msg
438  logical, intent(in) :: allow_rounding
439  integer, intent(out) :: tick
440 
441  character(len=4) :: formt='(i )'
442  integer :: i3, nspf, fraction, magnitude, tpsfrac
443 
444  err_msg = ''
445  get_tick_from_string = .true.
446  i3 = len_trim(string)
447  nspf = i3 - 1 ! nspf = spaces occupied by fractional seconds, excluding decimal point
448  if(nspf == 0) then
449  tick = 0 ! Nothing to the right of the decimal point
450  else
451  write(formt(3:3),'(i1)') nspf
452  read(string(2:i3),formt) fraction
453  if(fraction == 0) then
454  tick = 0 ! All zeros to the right of the decimal point
455  else
456  magnitude = 10**nspf
457  tpsfrac = ticks_per_second*fraction
458  if(allow_rounding) then
459  tick = nint((real(tpsfrac)/magnitude))
460  else
461  if(modulo(tpsfrac,magnitude) == 0) then
462  tick = tpsfrac/magnitude
463  else
464  write(err_msg,'(a,i6)') 'Second fraction cannot be exactly represented with ticks. '// &
465  'fraction='//trim(string)//' ticks_per_second=',ticks_per_second
466  get_tick_from_string = .false.
467  endif
468  endif
469  endif
470  endif
471 
472  end function get_tick_from_string
