225 !###############################################################################
226 !> @brief diag_integral_init is the constructor for diag_integral_mod.
227 !!
228 !! <b> Template: </b>
229 !!
230 !! @code{.f90}
231 !! call diag_integral_init (Time_init, Time, blon, blat)
232 !! @endcode
233 !!
234 !! <b> Parameters: </b>
235 !!
236 !! @code{.f90}
237 !! type (time_type), intent(in), optional :: Time_init, Time
238 !! real,dimension(:,:), intent(in), optional :: blon, blat, area_in
239 !! @endcode
240 !!
241 !! @param [in] <Time_init> Initial time to start the integral
242 !! @param [in] <Time> current time
243 !! @param [in] <latb> array of model latitudes at cell boundaries [radians]
244 !! @param [in] <lonb> array of model longitudes at cell boundaries [radians]
245 !!
246 subroutine diag_integral_init (Time_init, Time, blon, blat, area_in)
247 
248 type (time_type), intent(in), optional :: time_init !< Initial time to start the integral
249 type (time_type), intent(in), optional :: time !< current time
250 real,dimension(:,:), intent(in), optional :: blon !< array of model latitudes at cell boundaries [radians]
251 real,dimension(:,:), intent(in), optional :: blat !< array of model longitudes at cell boundaries [radians]
252 real,dimension(:,:), intent(in), optional :: area_in
253 
254 !-------------------------------------------------------------------------------
255 ! local variables:
256 ! r2
257 ! rsize
258 ! unit
259 ! io
260 ! ierr
261 ! seconds
262 ! nc
263 ! i,j
264 !-------------------------------------------------------------------------------
265  real :: rsize
266  integer :: io, ierr, nc, logunit
267  integer :: field_size_local
268  real :: sum_area_local
269  integer :: ensemble_size(6)
270 !-------------------------------------------------------------------------------
271 ! if routine has already been executed, exit.
272 !-------------------------------------------------------------------------------
273  if (module_is_initialized) return
274 
275 !-------------------------------------------------------------------------------
276 ! verify that modules used by this module that are not called later
277 ! have already been initialized.
278 !-------------------------------------------------------------------------------
279  call fms_init
280  call mpp_init
281  call constants_init
282  call time_manager_init
283 
284 !-------------------------------------------------------------------------------
285 ! if this is the initialization call, proceed. if this was simply
286 ! a verification of previous initialization, return.
287 !-------------------------------------------------------------------------------
288  if (present(time_init) .and. present(time) .and. &
289  present(blon) .and. present(blat) ) then
290 
291 !-------------------------------------------------------------------------------
292 ! read namelist.
293 !-------------------------------------------------------------------------------
294  read (input_nml_file, nml=diag_integral_nml, iostat=io)
295  ierr = check_nml_error(io,'diag_integral_nml')
296 
297 !-------------------------------------------------------------------------------
298 ! write version number and namelist to logfile.
299 !-------------------------------------------------------------------------------
300  call write_version_number ('DIAG_INTEGRAL_MOD', version)
301  logunit = stdlog()
302  if (mpp_pe() == mpp_root_pe() ) &
303  write (logunit, nml=diag_integral_nml)
304 
305 !-------------------------------------------------------------------------------
306 ! save the initial time to time-stamp the integrals which will be
307 ! calculated.
308 !-------------------------------------------------------------------------------
309  time_init_save = time_init
310 
311 !-------------------------------------------------------------------------------
312 ! define the model grid sizes and the total number of columns on
313 ! the processor. sum over all processors and store the global
314 ! number of columns in field_size.
315 !-------------------------------------------------------------------------------
316  idim = size(blon,1) - 1
317  jdim = size(blon,2) - 1
318  field_size_local = idim*jdim
319  rsize = real(field_size_local)
320  call mpp_sum (rsize)
321  field_size = nint(rsize)
322 
323 !-------------------------------------------------------------------------------
324 ! define an array to hold the surface area of each grid column
325 ! so that the integrals may be weighted properly. sum over the
326 ! processor, and then over all processors, storing the total
327 ! global surface area in sum_area.
328 !-------------------------------------------------------------------------------
329  allocate (area(idim,jdim))
330 
331  area = area_in
332 
333  sum_area_local = sum(area)
334  sum_area = sum_area_local
335  call mpp_sum (sum_area)
336 
337 !-------------------------------------------------------------------------------
338 ! if integral output is to go to a file, open the file on unit
339 ! diag_unit.
340 !-------------------------------------------------------------------------------
341  if (file_name(1:1) /= ' ' ) then
342  ensemble_size = get_ensemble_size()
343  if (ensemble_size(1) > 1) then
344  file_name = ensemble_file_name(file_name)
345  endif
346  nc = len_trim(file_name)
347  open(newunit=diag_unit, file=file_name(1:nc), action='write')
348  endif
349 
350 !-------------------------------------------------------------------------------
351 ! define the variables needed to control the time interval of
352 ! output. Zero time is a flag indicating that the alarm is not set,
353 ! i.e., integrals are not desired. otherwise set the next time to
354 ! output integrals to be at the value of nml variable
355 ! output_interval from now.
356 !-------------------------------------------------------------------------------
357  zero_time = set_time(0,0)
358  if (output_interval >= -0.01) then
359  alarm_interval = set_axis_time(output_interval, time_units)
360  next_alarm_time = time + alarm_interval
361  else
362  alarm_interval = zero_time
363  endif
364  next_alarm_time = time + alarm_interval
365 
366 !-------------------------------------------------------------------------------
367 ! deallocate the local array and mark the module as initialized.
368 !-------------------------------------------------------------------------------
369  module_is_initialized = .true.
370  endif ! (present optional arguments)
371 
372 end subroutine diag_integral_init

376 !###############################################################################
377 !> @brief diag_integral_field_init registers and intializes an integral field
378 !!
379 !! <b> Template: </b>
380 !!
381 !! @code{.f90}
382 !! call diag_integral_field_init (name, format)
383 !! @endcode
384 !!
385 !! <b> Parameters: </b>
386 !!
387 !! @code{.f90}
388 !! character(len=*), intent(in) :: name, format
389 !! @endcode
390 !!
391 !! @param [in] <name> Name of the field to be integrated
392 !! @param [in] <format> Output format of the field to be integrated
393 !!
394 subroutine diag_integral_field_init (name, format)
395 
396 character(len=*), intent(in) :: name !< Name of the field to be integrated
397 character(len=*), intent(in) :: format !< Output format of the field to be integrated
398 
399 !-------------------------------------------------------------------------------
400 ! local variables:
401 !-------------------------------------------------------------------------------
402  integer :: field !< index assigned to the current integral
403 
404 !-------------------------------------------------------------------------------
405 ! note: no initialization is required for this interface. all needed
406 ! variables are initialized in the source.
407 !-------------------------------------------------------------------------------
408 
409 !-------------------------------------------------------------------------------
410 ! make sure the integral name is not too long.
411 !-------------------------------------------------------------------------------
412  if (len(name) > max_len_name ) then
413  call error_mesg ('diag_integral_mod', &
414  ' integral name too long', fatal)
415  endif
416 
417 !-------------------------------------------------------------------------------
418 ! check to be sure the integral name has not already been
419 ! initialized.
420 !-------------------------------------------------------------------------------
421  field = get_field_index(name)
422  if (field /= 0) then
423  call error_mesg ('diag_integral_mod', &
424  'integral name already exists', fatal)
425  endif
426 
427 !-------------------------------------------------------------------------------
428 ! prepare to register the integral. make sure that there are not
429 ! more integrals registered than space was provided for; if so, exit.
430 !-------------------------------------------------------------------------------
431  num_field = num_field + 1
432  if (num_field > max_num_field) then
433  call error_mesg ('diag_integral_mod', &
434  'too many fields initialized', fatal)
435  endif
436 
437 !-------------------------------------------------------------------------------
438 ! register the name and output format desired for the given integral.
439 ! initialize its value and the number of grid points that have been
440 ! counted to zero.
441 !-------------------------------------------------------------------------------
442  field_name(num_field) = name
443  field_format(num_field) = format
444  field_sum(num_field) = 0.0
445  field_count(num_field) = 0
446 
447 end subroutine diag_integral_field_init

451 !###############################################################################
452 !> @brief Perform a 2 dimensional summation of named field
453 !!
454 !! @implements sum_diag_integral_field
455 !!
456 !! <b> Template: </b>
457 !!
458 !! @code{.f90}
459 !! call sum_field_2d (name, data, is, js)
460 !! @endcode
461 !!
462 !! <b> Parameters: </b>
463 !!
464 !! @code{.f90}
465 !! character(len=*), intent(in) :: name
466 !! real, intent(in) :: data(:,:)
467 !! integer, optional, intent(in) :: is, js
468 !! @endcode
469 !!
470 !! @param [in] <name> Name of the field to be integrated
471 !! @param [in] <data> field of integrands to be summed over
472 !! @param [in] <is, js> starting i,j indices over which summation is to occur
473 !!
474 subroutine sum_field_2d (name, data, is, js)
475 
476 character(len=*), intent(in) :: name !< Name of the field to be integrated
477 real, intent(in) :: data(:,:) !< field of integrands to be summed over
478 integer, optional, intent(in) :: is !< starting i indices over which summation is to occur
479 integer, optional, intent(in) :: js !< starting j indices over which summation is to occur
480 
481 !-------------------------------------------------------------------------------
482 ! local variables:
483 !-------------------------------------------------------------------------------
484  integer :: field !< index of desired integral
485  integer :: i1 !< location indices of current data in
486  !! processor-global coordinates
487  integer :: j1 !< location indices of current data in
488  !! processor-global coordinates
489  integer :: i2 !< location indices of current data in
490  !! processor-global coordinates
491  integer :: j2 !< location indices of current data in
492  !! processor-global coordinates
493 
494 
495 !-------------------------------------------------------------------------------
496 ! be sure module has been initialized.
497 !-------------------------------------------------------------------------------
498  if (.not. module_is_initialized ) then
499  call error_mesg ('diag_integral_mod', &
500  'module has not been initialized', fatal )
501  endif
502 
503 !-------------------------------------------------------------------------------
504 ! obtain the index of the current integral. make certain it is valid.
505 !-------------------------------------------------------------------------------
506  field = get_field_index(name)
507  if (field == 0) then
508  call error_mesg ('diag_integral_mod', &
509  'field does not exist', fatal)
510  endif
511 
512 !-------------------------------------------------------------------------------
513 ! define the processor-global indices of the current data. use the
514 ! value 1 for the initial grid points, if is and js are not input.
515 !-------------------------------------------------------------------------------
516  i1 = 1; if (present(is)) i1 = is
517  j1 = 1; if (present(js)) j1 = js
518  i2 = i1 + size(data,1) - 1
519  j2 = j1 + size(data,2) - 1
520 
521 !-------------------------------------------------------------------------------
522 ! increment the count of points toward this integral and add the
523 ! values at this set of grid points to the accumulation array.
524 !-------------------------------------------------------------------------------
525 !$OMP CRITICAL
526  field_count(field) = field_count(field) + &
527  size(data,1)*size(data,2)
528  field_sum(field) = field_sum(field) + &
529  sum(data * area(i1:i2,j1:j2))
530 
531 !$OMP END CRITICAL
532 
533 end subroutine sum_field_2d

540 !! @implements sum_diag_integral_field
541 !!
542 !! <b> Template: </b>
543 !!
544 !! @code{.f90}
545 !! call sum_field_3d (name, data, is, js)

548 !! <b> Parameters: </b>
549 !!
550 !! @code{.f90}
551 !! character(len=*), intent(in) :: name
552 !! real, intent(in) :: data(:,:,:)
553 !! integer, optional, intent(in) :: is, js
554 !! @endcode
555 !!
556 !! @param [in] <name> Name of the field to be integrated
557 !! @param [in] <data> field of integrands to be summed over
558 !! @param [in] <is, js> starting i,j indices over which summation is to occur
559 !!
560 subroutine sum_field_3d (name, data, is, js)
561 
562 character(len=*), intent(in) :: name !< Name of the field to be integrated
563 real, intent(in) :: data(:,:,:) !< field of integrands to be summed over
564 integer, optional, intent(in) :: is !< starting i,j indices over which summation is to occur
565 integer, optional, intent(in) :: js !< starting i,j indices over which summation is to occur
566 
567 !-------------------------------------------------------------------------------
568 ! local variables:
569 ! data2
570 ! field ! index of desired integral
571 ! i1, j1, i2, j2 ! location indices of current data in
572 ! processor-global coordinates
573 !-------------------------------------------------------------------------------
574  real, dimension (size(data,1), & size(data,2)) :: data2
575 
576  integer :: field !< index of desired integral
577  integer :: i1 !< location indices of current data in
578  !! processor-global coordinates
579  integer :: j1 !< location indices of current data in
580  !! processor-global coordinates
581  integer :: i2 !< location indices of current data in
582  !! processor-global coordinates
583  integer :: j2 !< location indices of current data in
584  !! processor-global coordinates
585 
586 
587 !-------------------------------------------------------------------------------
588 ! be sure module has been initialized.
589 !-------------------------------------------------------------------------------
590  if (.not. module_is_initialized ) then
591  call error_mesg ('diag_integral_mod', &
592  'module has not been initialized', fatal )
593  endif
594 
595 !-------------------------------------------------------------------------------
596 ! obtain the index of the current integral. make certain it is valid.
597 !-------------------------------------------------------------------------------
598  field = get_field_index(name)
599  if (field == 0) then
600  call error_mesg ('diag_integral_mod', &
601  'field does not exist', fatal)
602  endif
603 
604 !-------------------------------------------------------------------------------
605 ! define the processor-global indices of the current data. use the
606 ! value 1 for the initial grid points, if is and js are not input.
607 !-------------------------------------------------------------------------------
608  i1 = 1; if (present(is)) i1 = is
609  j1 = 1; if (present(js)) j1 = js
610  i2 = i1 + size(data,1) - 1
611  j2 = j1 + size(data,2) - 1
612 
613 !-------------------------------------------------------------------------------
614 ! increment the count of points toward this integral. sum first
615 ! in the vertical and then add the values at this set of grid points
616 ! to the accumulation array.
617 !-------------------------------------------------------------------------------
618 !$OMP CRITICAL
619  field_count(field) = field_count(field) + &
620  size(data,1)*size(data,2)
621  data2 = sum(data,3)
622  field_sum(field) = field_sum(field) + &
623  sum(data2 * area(i1:i2,j1:j2))
624 
625 !$OMP END CRITICAL
626 
627 end subroutine sum_field_3d
