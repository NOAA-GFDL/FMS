38  function MPP_GLOBAL_REDUCE_3D_( domain, field, locus, position )
39  MPP_TYPE_ :: MPP_GLOBAL_REDUCE_3D_
40  type(domain2D), intent(in) :: domain
41  MPP_TYPE_, intent(in) :: field(0:,0:,:)
42  integer, intent(out), optional :: locus(3)
43  integer, intent(in), optional :: position
44 
45  MPP_TYPE_ :: local
46  integer, save :: l_locus(3)
47  MPP_TYPE_, save :: g_val ! need storage class w/ global address; not sure whether fn result has required class
48  integer, save :: here ! need storage class w/ global address
49  integer :: ioff, joff, isc, iec, jsc, jec, ishift, jshift
50 
51  if( .NOT.module_is_initialized )call mpp_error( FATAL, 'MPP_GLOBAL_REDUCE: You must first call mpp_domains_init.' )
52 
53  call mpp_get_compute_domain(domain, isc, iec, jsc, jec )
54  call mpp_get_domain_shift(domain, ishift, jshift, position)
55  iec = iec + ishift
56  jec = jec + jshift
57 
58  if( size(field,1).EQ. iec-isc+1 .AND. size(field,2).EQ. jec-jsc+1 )then
59 !field is on compute domain
60  ioff = isc
61  joff = jsc
62  else if( size(field,1).EQ.domain%x(1)%memory%size+ishift .AND. size(field,2).EQ.domain%y(1)%memory%size+jshift )then
63 !field is on data domain
64  ioff = domain%x(1)%data%begin
65  joff = domain%y(1)%data%begin
66  else
67  call mpp_error( FATAL, 'MPP_GLOBAL_REDUCE_: incoming field array must match either compute domain or data domain.' )
68  end if
69 
70 !get your local max/min
71  local = REDUCE_VAL_(field(isc-ioff: iec-ioff, jsc-joff: jec-joff,:))
72 !find the global
73  g_val = local
74  call MPP_REDUCE_( g_val, domain%list(:)%pe )
75 !find locus of the global max/min
76  if( PRESENT(locus) )then
77 !which PE is it on? min of all the PEs that have it
78  here = mpp_npes()+1
79  if( g_val == local )here = pe
80  call mpp_min( here, domain%list(:)%pe )
81 !find the locus here
82  if( pe.EQ.here )l_locus = REDUCE_LOC_(field(isc-ioff: iec-ioff, jsc-joff: jec-joff,:))
83  l_locus(1) = l_locus(1) + ioff
84  l_locus(2) = l_locus(2) + joff
85  call mpp_broadcast( l_locus, 3, here, domain%list(:)%pe )
86  locus = l_locus
87  end if
88  MPP_GLOBAL_REDUCE_3D_ = g_val
89  return
90  end function MPP_GLOBAL_REDUCE_3D_

92  function MPP_GLOBAL_REDUCE_4D_( domain, field, locus, position )
93  MPP_TYPE_ :: MPP_GLOBAL_REDUCE_4D_
94  type(domain2D), intent(in) :: domain
95  MPP_TYPE_, intent(in) :: field(:,:,:,:)
96  integer, intent(out), optional :: locus(4)
97  integer, intent(in), optional :: position
98 
99  MPP_TYPE_ :: field3D(size(field,1),size(field,2),size(field,3)*size(field,4))
100  integer :: locus3D(3)
101  pointer( ptr, field3D )
102  ptr = LOC(field)
103  if( PRESENT(locus) )then
104  MPP_GLOBAL_REDUCE_4D_ = MPP_GLOBAL_REDUCE_3D_( domain, field3D, locus3D, position )
105  locus(1:2) = locus3D(1:2)
106  locus(3) = modulo(locus3D(3),size(field,3))
107  locus(4) = (locus3D(3)-locus(3))/size(field,3) + 1
108  if( locus(3).EQ.0 )then
109  locus(3) = size(field,3)
110  locus(4) = locus(4) - 1
111  end if
112  else
113  MPP_GLOBAL_REDUCE_4D_ = MPP_GLOBAL_REDUCE_3D_( domain, field3D, position = position )
114  end if
115  return
116  end function MPP_GLOBAL_REDUCE_4D_
