216 !> Return grid cell area for the specified model component and tile
217 subroutine get_grid_cell_area_sg(component, tile, cellarea, domain)
218  character(len=*), intent(in) :: component
219  integer , intent(in) :: tile
220  real , intent(inout) :: cellarea(:,:)
221  type(domain2d) , intent(in), optional :: domain
222 
223  ! local vars
224  integer :: nlon, nlat
225  real, allocatable :: glonb(:,:), glatb(:,:)
226 
227  select case(get_grid_version())
228  case(version_0,version_1)
229  select case(trim(component))
230  case('LND')
231  call read_data(grid_file, 'AREA_LND_CELL', cellarea, &
232  no_domain=.not.present(domain), domain=domain)
233  case('ATM','OCN')
234  call read_data(grid_file, 'AREA_'//trim(uppercase(component)),cellarea,&
235  no_domain=.not.present(domain),domain=domain)
236  case default
237  call mpp_error(fatal, module_name//'/get_grid_cell_area: Illegal component name "'//trim(component) &
238  & //'": must be one of ATM, LND, or OCN')
239  end select
240  ! convert area to m2
241  cellarea = cellarea*4.*pi*radius**2
242  case(version_2)
243  if (present(domain)) then
244  call mpp_get_compute_domain(domain,xsize=nlon,ysize=nlat)
245  else
246  call get_grid_size(component,tile,nlon,nlat)
247  endif
248  allocate(glonb(nlon+1,nlat+1),glatb(nlon+1,nlat+1))
249  call get_grid_cell_vertices(component, tile, glonb, glatb, domain)
250  if (great_circle_algorithm) then
251  call calc_mosaic_grid_great_circle_area(glonb*pi/180.0, glatb*pi/180.0, cellarea)
252  else
253  call calc_mosaic_grid_area(glonb*pi/180.0, glatb*pi/180.0, cellarea)
254  end if
255  deallocate(glonb,glatb)
256  end select
257 
258 end subroutine get_grid_cell_area_sg

260 ! ============================================================================
261 ! ============================================================================
262 !> Get the area of the component per grid cell
263 subroutine get_grid_comp_area_sg(component,tile,area,domain)
264  character(len=*) :: component
265  integer, intent(in) :: tile
266  real, intent(inout) :: area(:,:)
267  type(domain2d), intent(in), optional :: domain
268  ! local vars
269  integer :: n_xgrid_files ! number of exchange grid files in the mosaic
270  integer :: siz(4), nxgrid
271  integer :: i,j,m,n
272  integer, allocatable :: i1(:), j1(:), i2(:), j2(:)
273  real, allocatable :: xgrid_area(:)
274  real, allocatable :: rmask(:,:)
275  character(len=MAX_NAME) :: &
276  xgrid_name, & ! name of the variable holding xgrid names
277  tile_name, & ! name of the tile
278  xgrid_file, & ! name of the current xgrid file
279  mosaic_name,& ! name of the mosaic
280  mosaic_file,&
281  tilefile
282  character(len=4096) :: attvalue
283  character(len=MAX_NAME), allocatable :: nest_tile_name(:)
284  integer :: is,ie,js,je ! boundaries of our domain
285  integer :: i0, j0 ! offsets for x and y, respectively
286  integer :: num_nest_tile, ntiles
287  logical :: is_nest
288  integer :: found_xgrid_files ! how many xgrid files we actually found in the grid spec
289  integer :: ibegin, iend, bsize, l
290 
291  select case (get_grid_version())
292  case(version_0,version_1)
293  select case(component)
294  case('ATM')
295  call read_data(grid_file,'AREA_ATM',area, no_domain=.not.present(domain),domain=domain)
296  case('OCN')
297  allocate(rmask(size(area,1),size(area,2)))
298  call read_data(grid_file,'AREA_OCN',area, no_domain=.not.present(domain),domain=domain)
299  call read_data(grid_file,'wet', rmask,no_domain=.not.present(domain),domain=domain)
300  area = area*rmask
301  deallocate(rmask)
302  case('LND')
303  call read_data(grid_file,'AREA_LND',area,no_domain=.not.present(domain),domain=domain)
304  case default
305  call mpp_error(fatal, module_name// &
306  & '/get_grid_comp_area: Illegal component name "'//trim(component)//'": must be one of ATM, LND, or OCN')
307  end select
308  case(version_2) ! mosaic gridspec
309  select case (component)
310  case ('ATM')
311  ! just read the grid cell area and return
312  call get_grid_cell_area(component,tile,area)
313  return
314  case ('LND')
315  xgrid_name = 'aXl_file'
316  call read_data(grid_file, 'lnd_mosaic', mosaic_name)
317  tile_name = trim(mosaic_name)//'_tile'//char(tile+ichar('0'))
318  case ('OCN')
319  xgrid_name = 'aXo_file'
320  call read_data(grid_file, 'ocn_mosaic', mosaic_name)
321  tile_name = trim(mosaic_name)//'_tile'//char(tile+ichar('0'))
322  case default
323  call mpp_error(fatal, module_name// &
324  & '/get_grid_comp_area: Illegal component name "'//trim(component)//'": must be one of ATM, LND, or OCN')
325  end select
326  ! get the boundaries of the requested domain
327  if(present(domain)) then
328  call mpp_get_compute_domain(domain,is,ie,js,je)
329  i0 = 1-is ; j0=1-js
330  else
331  call get_grid_size(component,tile,ie,je)
332  is = 1 ; i0 = 0
333  js = 1 ; j0 = 0
334  endif
335  if (size(area,1)/=ie-is+1.or.size(area,2)/=je-js+1) &
336  call mpp_error(fatal, module_name// &
337  & '/get_grid_comp_area: size of the output argument "area" is not consistent with the domain')
338 
339  ! find the nest tile
340  call read_data(grid_file, 'atm_mosaic', mosaic_name)
341  call read_data(grid_file,'atm_mosaic_file',mosaic_file)
342  mosaic_file = grid_dir//trim(mosaic_file)
343  ntiles = get_mosaic_ntiles(trim(mosaic_file))
344  allocate(nest_tile_name(ntiles))
345  num_nest_tile = 0
346  do n = 1, ntiles
347  call read_data(mosaic_file, 'gridfiles', tilefile, level=n)
348  tilefile = grid_dir//trim(tilefile)
349  if( get_global_att_value(tilefile, "nest_grid", attvalue) ) then
350  if(trim(attvalue) == "TRUE") then
351  num_nest_tile = num_nest_tile + 1
352  nest_tile_name(num_nest_tile) = trim(mosaic_name)//'_tile'//char(n+ichar('0'))
353  else if(trim(attvalue) .NE. "FALSE") then
354  call mpp_error(fatal, module_name//'/get_grid_comp_area: value of global attribute nest_grid in file'// &
355  & trim(tilefile)//' should be TRUE of FALSE')
356  endif
357  end if
358  end do
359  area(:,:) = 0.
360  if(field_exist(grid_file,xgrid_name)) then
361  ! get the number of the exchange-grid files
362  call field_size(grid_file,xgrid_name,siz)
363  n_xgrid_files = siz(2)
364  found_xgrid_files = 0
365  ! loop through all exchange grid files
366  do n = 1, n_xgrid_files
367  ! get the name of the current exchange grid file
368  call read_data(grid_file,xgrid_name,xgrid_file,level=n)
369  ! skip the rest of the loop if the name of the current tile isn't found
370  ! in the file name, but check this only if there is more than 1 tile
371  if(n_xgrid_files>1) then
372  if(index(xgrid_file,trim(tile_name))==0) cycle
373  endif
374  found_xgrid_files = found_xgrid_files + 1
375  !---make sure the atmosphere grid is not a nested grid
376  is_nest = .false.
377  do m = 1, num_nest_tile
378  if(index(xgrid_file, trim(nest_tile_name(m))) .NE. 0) then
379  is_nest = .true.
380  exit
381  end if
382  end do
383  if(is_nest) cycle
384 
385  ! finally read the exchange grid
386  nxgrid = get_mosaic_xgrid_size(grid_dir//xgrid_file)
387  if(nxgrid < bufsize) then
388  allocate(i1(nxgrid), j1(nxgrid), i2(nxgrid), j2(nxgrid), xgrid_area(nxgrid))
389  else
390  allocate(i1(bufsize), j1(bufsize), i2(bufsize), j2(bufsize), xgrid_area(bufsize))
391  endif
392  ibegin = 1
393  do l = 1,nxgrid,bufsize
394  bsize = min(bufsize, nxgrid-l+1)
395  iend = ibegin + bsize - 1
396  call get_mosaic_xgrid(grid_dir//xgrid_file, i1(1:bsize), j1(1:bsize), i2(1:bsize), j2(1:bsize), &
397  xgrid_area(1:bsize), ibegin, iend)
398  ! and sum the exchange grid areas
399  do m = 1, bsize
400  i = i2(m); j = j2(m)
401  if (i<is.or.i>ie) cycle
402  if (j<js.or.j>je) cycle
403  area(i+i0,j+j0) = area(i+i0,j+j0) + xgrid_area(m)
404  end do
405  ibegin = iend + 1
406  enddo
407  deallocate(i1, j1, i2, j2, xgrid_area)
408  enddo
409  if (found_xgrid_files == 0) &
410  call mpp_error(fatal, 'get_grid_comp_area: no xgrid files were found for component '// &
411  & trim(component)//' (mosaic name is '//trim(mosaic_name)//')')
412 
413  endif
414  deallocate(nest_tile_name)
415  end select ! version
416  ! convert area to m2
417  area = area*4.*pi*radius**2
418 end subroutine get_grid_comp_area_sg

456 ! ============================================================================
457 ! ============================================================================
458 !> Returns arrays of global grid cell boundaries for given model component and
459 !! mosaic tile number.
460 !!
461 !> @note In the case of non-lat-lon grid the returned coordinates may have be not so
462 !! meaningful, by the very nature of such grids. But presumably these 1D coordinate
463 !! arrays are good enough for diag axis and such.
464 subroutine get_grid_cell_vertices_1d(component, tile, glonb, glatb)
465  character(len=*), intent(in) :: component
466  integer, intent(in) :: tile
467  real, intent(inout) :: glonb(:),glatb(:)
468 
469  integer :: nlon, nlat
470  integer :: start(4), nread(4)
471  real, allocatable :: tmp(:,:), x_vert_t(:,:,:), y_vert_t(:,:,:)
472  character(len=MAX_FILE) :: filename1, filename2
473 
474  call get_grid_size_for_one_tile(component, tile, nlon, nlat)
475  if (size(glonb(:))/=nlon+1) &
476  call mpp_error (fatal, module_name// &
477  & '/get_grid_cell_vertices_1D: Size of argument "glonb" is not consistent with the grid size')
478  if (size(glatb(:))/=nlat+1) &
479  call mpp_error (fatal, module_name// &
480  & '/get_grid_cell_vertices_1D: Size of argument "glatb" is not consistent with the grid size')
481  if(trim(component) .NE. 'ATM' .AND. component .NE. 'LND' .AND. component .NE. 'OCN') then
482  call mpp_error(fatal, module_name//'/get_grid_cell_vertices_1D: Illegal component name "'// &
483  & trim(component)//'": must be one of ATM, LND, or OCN')
484  endif
485 
486  select case(get_grid_version())
487  case(version_0)
488  select case(trim(component))
489  case('ATM','LND')
490  call read_data(grid_file, 'xb'//lowercase(component(1:1)), glonb, no_domain=.true.)
491  call read_data(grid_file, 'yb'//lowercase(component(1:1)), glatb, no_domain=.true.)
492  case('OCN')
493  call read_data(grid_file, "gridlon_vert_t", glonb, no_domain=.true.)
494  call read_data(grid_file, "gridlat_vert_t", glatb, no_domain=.true.)
495  end select
496  case(version_1)
497  select case(trim(component))
498  case('ATM','LND')
499  call read_data(grid_file, 'xb'//lowercase(component(1:1)), glonb, no_domain=.true.)
500  call read_data(grid_file, 'yb'//lowercase(component(1:1)), glatb, no_domain=.true.)
501  case('OCN')
502  allocate (x_vert_t(nlon,1,2), y_vert_t(1,nlat,2) )
503  start = 1; nread = 1
504  nread(1) = nlon; nread(2) = 1; start(3) = 1
505  call read_data(grid_file, "x_vert_T", x_vert_t(:,:,1), start, nread, no_domain=.true.)
506  nread(1) = nlon; nread(2) = 1; start(3) = 2
507  call read_data(grid_file, "x_vert_T", x_vert_t(:,:,2), start, nread, no_domain=.true.)
508 
509  nread(1) = 1; nread(2) = nlat; start(3) = 1
510  call read_data(grid_file, "y_vert_T", y_vert_t(:,:,1), start, nread, no_domain=.true.)
511  nread(1) = 1; nread(2) = nlat; start(3) = 4
512  call read_data(grid_file, "y_vert_T", y_vert_t(:,:,2), start, nread, no_domain=.true.)
513  glonb(1:nlon) = x_vert_t(1:nlon,1,1)
514  glonb(nlon+1) = x_vert_t(nlon,1,2)
515  glatb(1:nlat) = y_vert_t(1,1:nlat,1)
516  glatb(nlat+1) = y_vert_t(1,nlat,2)
517  deallocate(x_vert_t, y_vert_t)
518  end select
519  case(version_2)
520  ! get the name of the mosaic file for the component
521  call read_data(grid_file, trim(lowercase(component))//'_mosaic_file', filename1)
522  filename1=grid_dir//trim(filename1)
523  ! get the name of the grid file for the component and tile
524  call read_data(filename1, 'gridfiles', filename2, level=tile)
525  filename2 = grid_dir//trim(filename2)
526 
527  start = 1; nread = 1
528  nread(1) = 2*nlon+1
529  allocate( tmp(2*nlon+1,1) )
